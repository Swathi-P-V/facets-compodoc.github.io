import { Injectable } from '@angular/core';
import * as facetsIconMap from 'src/app/constants/facetsIconMap.json';
import { Subject } from 'rxjs';
import { Graph } from '@antv/x6';

export enum highlightStates {
  ACTIVE,
  NEIGHBOR,
  DISCONNECTED,
  DEFAULT,
}

@Injectable({
  providedIn: 'root',
})
export class BlueprintDesignerUtilService {
  public configurationWrapperState: Subject<any> = new Subject<any>(); // Subject which configuration component subscribes to
  public connectionHighlights: Subject<any> = new Subject<any>();
  public resourceMapReady: Subject<boolean> = new Subject<boolean>();
  public clickNodeSubscriber: Subject<any> = new Subject<any>();
  // the conf tool opens and closes with the change of state
  // in this subject
  openNode: any; // The node for which the configuration wrapper is supposed to open up
  graph: Graph; // A copy of the
  connectedHighlightedEdges: any[] = [];
  connectedHighlightedNodes: Set<any> = new Set();
  ORIGINAL_EDGE_COLOR = '#A2B1C3';
  HIGHLIGHTED_EDGE_COLOR = '#645df6';
  allNodesMap: Map<string, any[]>;
  highlightState: Map<string, highlightStates>; // Stores the highlight state at all times
  allNodeIDs: string[];
  configWrapperOpen: boolean = false;
  facetsIconNames: Set<string>;
  mode: 'TABLE' | 'GRAPH';

  constructor() {
    this.facetsIconNames = new Set(Object.keys(facetsIconMap || {}));
  }

  // Sends an update to the configuration component
  // through a change in the configurationWrapperState subject
  // so that it can open the node data inside of it
  openConfigurationTool(node: any) {
    this.configWrapperOpen = true;
    this.configurationWrapperState.next(node);
  }

  onClickNode(node: any) {
    this.clickNodeSubscriber.next(node);
  }

  // Sends an update to the configuration component
  // with the 'close' string for it to close the component
  closeConfigurationTool() {
    this.removeExistingHighlights();
    if (this.configWrapperOpen) {
      this.setActiveNode(null);
    }
    this.configurationWrapperState.next('close');
  }

  // NOT IN USE
  // Highlight connections incoming & outgoing to & from this node
  // Give the edges and nodes a slightly lighter tone of highlight than the node in focus
  // [TO-DO]: Before highlighting anything, we need to remove the highlight state from the
  // already highlighted nodes and edges
  // highlightConnections(node: any) {

  //   this.connectedHighlightedEdges = this.getConnectedEdges(node);

  // Highlighted all the edges which are passed to this function
  highlightEdges(edges) {
    edges.forEach((edge) => {
      edge.attr({
        line: {
          stroke: this.HIGHLIGHTED_EDGE_COLOR,
        },
      });
    });
  }

  removeHighlightFromEdges(edges) {
    edges.forEach((edge) => {
      edge.attr({
        line: {
          stroke: this.ORIGINAL_EDGE_COLOR,
        },
      });
    });
  }

  // Removes existing highlights from nodes and edges
  removeExistingHighlights() {
    if (this.connectedHighlightedEdges?.length) {
      this.removeHighlightFromEdges(this.connectedHighlightedEdges);
    }
    // this.connectionHighlights.next({
    //   nodes: new Set(),
    // });
  }

  // Get a list of incoming and outgoing edges
  getConnectedEdges(node: any) {
    let edges = this.graph?.model?.getConnectedEdges(this.getResourceID(node));
    return edges;
  }

  // Get a list of all the connected nodes
  getConnectedNodes(node: any, edges: any) {
    let currentNode = this.getNodeUsingID(this.getResourceID(node));
    let neighbors = edges.map((edge) => {
      const sourceNode = edge.getSourceNode();
      const targetNode = edge.getTargetNode();
      return sourceNode === currentNode ? targetNode : sourceNode;
    });
    neighbors = neighbors.map((neighbor) => neighbor.id);
    return neighbors;
  }

  // Takes in the node and returns the compound key which is used as the ID
  getResourceID(node: any) {
    return node?.resourceType + ':' + node?.resourceName;
  }

  // Set the current instance of graph
  setGraph(graph: any) {
    this.graph = graph;
  }

  // Get node from the graph object based on its ID
  getNodeUsingID(nodeID) {
    return this.graph.getCellById(nodeID);
  }

  // Setter for all nodes map, main blueprint component sets this value
  // while it's initializing the graph
  // Also marks all the nodes' highlight states to DEFAULT
  setAllNodesMap(resourceNodes) {
    this.allNodesMap = new Map<string, any[]>();
    resourceNodes.forEach((resource) => {
      if (!this.allNodesMap.has(resource.resourceType)) {
        this.allNodesMap.set(resource.resourceType, []);
      }
      this.allNodesMap.get(resource.resourceType).push(resource);
    });
    this.resourceMapReady.next(true);
    this.setDefaultHighlights();
  }

  // Getter for all nodes map, used currently in: search
  getAllNodesMap() {
    return this.allNodesMap;
  }

  // Sets all the nodes to be at default highlight state
  setDefaultHighlights() {
    this.highlightState = new Map<string, highlightStates>();
    this.allNodeIDs = [];
    for (let resourceType of this.allNodesMap?.keys()) {
      this.allNodesMap.get(resourceType).forEach((resource) => {
        this.allNodeIDs.push(this.getResourceID(resource));
      });
    }
    this.setHighlightStateForNodes(this.allNodeIDs, highlightStates.DEFAULT);
    if (this.connectedHighlightedEdges?.length) {
      this.removeHighlightFromEdges(this.connectedHighlightedEdges);
    }
  }

  setHighlightStateForNodes(nodeIDs: string[], state: highlightStates) {
    nodeIDs.forEach((id) => {
      this.highlightState.set(id, state);
    });
  }

  // Should take in a node object and:
  // Mark (node) as ACTIVE
  // Mark (neighbors of node) to NEIGHBOR
  // Mark (all nodes - (node + neighbor of node)) to DISCONNECTED
  setActiveNode(node) {
    if (!node) {
      this.setDefaultHighlights();
      return;
    }

    this.setHighlightStateForNodes(
      [this.getResourceID(node)],
      highlightStates.ACTIVE,
    );

    const neighboringEdges = this.getConnectedEdges(node);
    const neighboringNodes = this.getConnectedNodes(node, neighboringEdges);
    this.setHighlightStateForNodes(neighboringNodes, highlightStates.NEIGHBOR);
    this.removeHighlightFromEdges(this.connectedHighlightedEdges);
    this.highlightEdges(neighboringEdges);
    this.connectedHighlightedEdges = neighboringEdges;

    const allNodesSet: Set<string> = new Set(this.allNodeIDs);
    [this.getResourceID(node), ...neighboringNodes].forEach((ID) => {
      allNodesSet.delete(ID);
    });
    const otherNodes = Array.from(allNodesSet);
    this.setHighlightStateForNodes(otherNodes, highlightStates.DISCONNECTED);
    this.openConfigurationTool(node);
  }

  bringNodeInFocus(resourceNode: any, containerName?: string) {
    if (!containerName) {
      containerName = 'graph-container';
    }
    this.centreNode(this.getResourceID(resourceNode), containerName);
  }

  /*
    The function uses an if-else statement to determine which of the two quadratic equations to use,
    depending on the value of t. If t is less than 0.5, the function returns the equation 2 * t * t,
    which produces a smooth quadratic curve that starts out slowly and then accelerates. 
    If t is greater than or equal to 0.5, the function returns the equation -2 * t * t + 4 * t - 1, 
    which produces a smooth quadratic curve that starts out fast and then decelerates.
  */
  quadraticEaseInOut(t: number) {
    if (t < 0.5) {
      return 2 * t * t;
    } else {
      return -2 * t * t + 4 * t - 1;
    }
  }

  // Bring the node with the passed nodeID to the centre of the
  // graph with some animation
  centreNode(nodeID: string, containerName?: string) {
    const container = document.getElementById(containerName);
    const node = this.graph.getCellById(nodeID);

    if (!node) {
      return;
    }

    const containerWidth = container.clientWidth;
    const containerHeight = container.clientHeight;
    const nodeBBox = node.getBBox();

    let scaleFactor = Math.min(
      containerWidth / nodeBBox.width,
      containerHeight / nodeBBox.height,
    );

    const maxScaleFactor = 1;
    const minScaleFactor = 0.8;

    if (scaleFactor > maxScaleFactor) {
      scaleFactor = maxScaleFactor;
    }

    if (scaleFactor < minScaleFactor) {
      scaleFactor = minScaleFactor;
    }

    const currentTranslate = this.graph.translate();
    const startXOffset = currentTranslate.tx;
    const startYOffset = currentTranslate.ty;
    const endXOffset =
      (containerWidth - nodeBBox.width * scaleFactor) / 2 -
      nodeBBox.x * scaleFactor -
      300;
    const endYOffset =
      (containerHeight - nodeBBox.height * scaleFactor) / 2 -
      nodeBBox.y * scaleFactor;

    const duration = 300;

    let animationStartTime = performance.now();
    let animationId: number;
    function animate() {
      const elapsedTime = performance.now() - animationStartTime;

      if (elapsedTime >= duration) {
        this.graph.translate(endXOffset, endYOffset);
        cancelAnimationFrame(animationId);
        return;
      }

      const progress = this.quadraticEaseInOut(elapsedTime / duration);
      const currentXOffset =
        startXOffset + (endXOffset - startXOffset) * progress;
      const currentYOffset =
        startYOffset + (endYOffset - startYOffset) * progress;
      this.graph.translate(currentXOffset, currentYOffset);
      animationId = requestAnimationFrame(animate.bind(this));
    }

    animationId = requestAnimationFrame(animate.bind(this));
    this.graph.scale(scaleFactor);
  }

  setCurrentMode(mode: 'TABLE' | 'GRAPH') {
    this.mode = mode;
  }

  getCurrentMode() {
    return this.mode;
  }
}
