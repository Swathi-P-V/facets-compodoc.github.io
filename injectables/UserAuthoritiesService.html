import { Injectable } from '@angular/core';
import { Observable, Observer } from 'rxjs';
import { ApplicationControllerService } from '../cc-api/services';
import { actionDisabled } from '../utils/action-disabled-utils';

@Injectable({
  providedIn: 'root',
})
export class UserAuthoritiesService {
  private authoritiesList = [];
  private clusterRoleBinding = {};
  buttonDisabledTooltip = 'Access is forbidden';

  constructor(
    private applicationControllerService: ApplicationControllerService,
  ) {}

  initAuthorities(userData) {
    this.authoritiesList = [];
    this.clusterRoleBinding = userData?.clusterRoleBinding;
    userData.authorities.forEach((ele) => {
      this.authoritiesList.push(ele.authority);
    });
  }

  isActionDisabled(action?: string, clusterId?: string, actionsList?: []) {
    return actionDisabled(
      this.clusterRoleBinding,
      this.authoritiesList,
      action,
      clusterId,
      actionsList,
    );
  }
  resetData() {
    this.authoritiesList = [];
    this.clusterRoleBinding = {};
  }
}
