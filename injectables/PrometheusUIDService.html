import { Injectable } from '@angular/core';
import { map, Observable, of, flatMap } from 'rxjs';
import { UiSettingsControllerService } from '../cc-api/services';

@Injectable({
  providedIn: 'root',
})
export class PrometheusUIDService {
  prometheusUIDMap: Map<string, string> = new Map();
  callCompleted: boolean = false;

  constructor(private uiSettings: UiSettingsControllerService) {}

  getPrometheusUID(clusterID: string): Observable<any> {
    let _this = this;
    let keyPresent = false;

    //check if environment ID is already present as key in the map.
    //If not, set the callCompleted to false so that function does not try to return cached value
    //which will be undefined in this case.

    if (this.prometheusUIDMap.has(clusterID)) {
      keyPresent = true;
    }

    if (!keyPresent) {
      this.callCompleted = false;
    }

    //this sends the cached key if the environment and source ID is already part of the map, otherwise api below is called
    if (this.callCompleted) {
      return of(this.prometheusUIDMap.get(clusterID));
    }

    function handleMissingUID(clusterId) {
      _this.prometheusUIDMap.set(clusterId, '1');
    }

    //check if cluster id is valid, or return '1' as default value.
    if (clusterID?.length > 0) {
      return this.uiSettings
        .getSettingValueUsingGET({
          entityType: 'CLUSTER',
          entityId: clusterID,
        })
        .pipe(
          map(
            (settings) => {
              if (settings.hasOwnProperty('CLUSTER_PROMETHEUS_DATA_SOURCE')) {
                this.prometheusUIDMap.set(
                  clusterID,
                  settings['CLUSTER_PROMETHEUS_DATA_SOURCE'].toString(),
                );
              } else {
                handleMissingUID(clusterID);
              }
              this.callCompleted = true;
              return this.prometheusUIDMap.get(clusterID);
            },
            (error) => {
              handleMissingUID(clusterID);
              this.callCompleted = true;
              return this.prometheusUIDMap.get(clusterID);
            },
          ),
        );
    } else {
      return of('1');
    }
  }
}
