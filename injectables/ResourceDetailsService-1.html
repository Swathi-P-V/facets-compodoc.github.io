import { Injectable } from '@angular/core';
import { UiDropdownsControllerService } from 'src/app/cc-api/services';

@Injectable({
  providedIn: 'root',
})
export class ResourceDetailsService {
  resource: any;
  contextURLStack = [];

  constructor(
    private uiDropdownsControllerService: UiDropdownsControllerService,
  ) {}

  getLastContext() {
    const prevURL = this.contextURLStack.pop();
    return prevURL;
  }

  pushContext(contextURL) {
    this.contextURLStack.push(contextURL);
  }

  isLastContextPresent() {
    return this.contextURLStack.length > 0;
  }

  clearContextStack() {
    this.contextURLStack = [];
  }

  getResource() {
    return this.resource;
  }

  getResourceID() {
    return this.resource?.resourceType + ':' + this.resource?.resourceName;
  }

  setResource(resource) {
    this.resource = resource;
  }

  clearResource() {
    this.resource = null;
  }

  getResourceContent() {
    try {
      return JSON.parse(this.resource?.content);
    } catch (error) {
      return {};
    }
  }

  geResourceBuildInfo() {
    return this.getResourceContent()?.spec?.release?.build;
  }

  getCIName() {
    return this.resource?.info?.ciName;
  }

  async fetchResource(blueprint, resourceID, branchName) {
    const resourceName = resourceID.split(':')[1];
    const resourceType = resourceID.split(':')[0];
    return new Promise((resolve) => {
      this.uiDropdownsControllerService
        .getResourceByStackUsingGET({
          stackName: blueprint,
          resourceName: resourceName,
          resourceType: resourceType,
          branch: branchName,
        })
        .subscribe(
          (rsp) => {
            const resource = rsp;
            this.setResource(resource);
            resolve(true);
          },
          (err) => {
            resolve(false);
          },
        );
    });
  }

  async fetchAllResources(blueprint) {
    return new Promise((resolve) => {
      this.uiDropdownsControllerService
        .getAllResourcesByStackUsingGET({
          stackName: blueprint,
          includeContent: false,
        })
        .subscribe(
          (rsp) => {
            resolve(rsp);
          },
          (error) => {
            resolve(false);
          },
        );
    });
  }

  async fetchAllConnectedResources(blueprint) {
    const allResources = await this.fetchAllConnectedResources(blueprint);
  }
}
