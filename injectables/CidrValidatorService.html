import { Injectable } from '@angular/core';

@Injectable({
  providedIn: 'root',
})
export class CidrValidatorService {
  private readonly privateRanges = [
    { start: '10.0.0.0', end: '10.255.255.255' },
    { start: '172.16.0.0', end: '172.31.255.255' },
    { start: '192.168.0.0', end: '192.168.255.255' },
  ];

  constructor() {}

  isCidrValid(cidr: string): boolean {
    const parts = cidr.split('/');
    if (parts.length !== 2) {
      return false;
    }

    const ip = parts[0];
    const prefix = parseInt(parts[1]);

    return this.isIpValid(ip) && !isNaN(prefix) && prefix <= 24 && prefix >= 16;
  }

  isCidrPrivate(cidr: string): boolean {
    if (!this.isCidrValid(cidr)) {
      return false;
    }

    const ip = cidr.split('/')[0];
    return this.privateRanges.some((range) =>
      this.isIpInRange(ip, range.start, range.end),
    );
  }

  private isIpValid(ip: string): boolean {
    const octets = ip.split('.').map((octet) => parseInt(octet));
    return (
      octets.length === 4 &&
      octets.every((octet) => !isNaN(octet) && octet >= 0 && octet <= 255)
    );
  }

  private isIpInRange(ip: string, start: string, end: string): boolean {
    const ipValue = this.ipToLong(ip);
    const startValue = this.ipToLong(start);
    const endValue = this.ipToLong(end);
    return ipValue >= startValue && ipValue <= endValue;
  }

  private ipToLong(ip: string): number {
    return ip.split('.').reduce((acc, octet) => acc * 256 + parseInt(octet), 0);
  }
}
