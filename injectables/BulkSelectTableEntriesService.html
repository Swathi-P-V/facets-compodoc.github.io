import { Injectable } from '@angular/core';
import cloneDeep from 'lodash/cloneDeep';
import groupBy from 'lodash/groupBy';
import { LocalDataSource } from 'ng2-smart-table';

@Injectable({
  providedIn: 'root',
})
export class BulkSelectTableEntriesService {
  constructor() {}

  /**
   * Updates the 'selected' property of all currently displayed entities in the table based on the 'checked' value and returns the updated source data as an array.
   * @param source - The source data of the ng2-smart-table (LocalDataSource).
   * @param filteredEntities - The currently displayed entities in the table.
   * @param checked - The value to set for the 'selected' property (true or false).
   * @param mapKey - The unique key used to access entities efficiently.
   * @returns An array of the updated source data after modifying the 'selected' property.
   */
  updateSelectedPropOnSelectAllClick(
    source,
    filteredEntities,
    checked,
    mapKey,
  ) {
    // Step 1: Map over 'filteredEntities' and add/update the 'selected' property
    let sortedAndFilteredEntities = filteredEntities?.map((entity) => ({
      ...entity,
      selected: checked,
    }));

    // Step 2: Create a map ('entityMapByUniqueKey') for efficient access by unique key
    const entityMapByUniqueKey = new Map(
      source?.['data'].map((item) => [`${item[mapKey]}`, item]),
    );

    // Step 3: Update the 'selected' property of corresponding entities in 'entityMapByUniqueKey'
    sortedAndFilteredEntities?.forEach((entity) => {
      const item = entityMapByUniqueKey.get(`${entity[mapKey]}`);
      if (item) {
        item['selected'] = entity?.selected;
      }
    });

    // Return the updated source as an array
    return {
      updatedSource: Array.from(entityMapByUniqueKey.values()),
      filteredEntities,
    };
  }

  /**
   * Manually updates the 'selected' property of a specific entity based on checkbox interaction.
   *
   * @param data - The data object containing the checkbox value and row data.
   * @param source - The ng2-smart-table data source.
   * @param findResource - The condition function to find the target entity.
   * @returns A new array of cloned entities with the 'selected' property of the target entity updated.
   */
  updateSelectedPropOnCheckboxClick(
    data: { val: boolean; rowData: any },
    source: LocalDataSource,
    findResource: (entity: any) => boolean,
  ) {
    if (typeof data.val !== 'boolean') {
      return;
    }
    const clonedResources = cloneDeep(source?.['data']);
    clonedResources.find(findResource).selected = data?.val;
    return clonedResources;
  }

  /**
   * Determines the select all checkbox state (indeterminate or selectAll) based on updated entity selections.
   *
   * @param source - The ng2-smart-table data source.
   * @param mapKey - The unique key to efficiently access entities in the source.
   * @param filteredEntities - The list of currently filtered entities.
   * @returns An object with 'isIndeterminate' and 'isSelectAll' properties indicating checkbox state.
   */

  getSelectAllCheckboxState(source, mapKey, filteredEntities: any[]) {
    // Step 1: Create a map ('entityMapByUniqueKey') for efficient access by unique key
    const entityMapByUniqueKey = new Map(
      source?.['data']?.map((item) => [`${item[mapKey]}`, item]),
    );
    // Step 2: Filter the 'filteredEntities' to get currently selected entities
    const currentSelectedEntities = filteredEntities?.filter(
      (entity) => entityMapByUniqueKey.get(`${entity[mapKey]}`)['selected'],
    );
    // Step 3: Update 'isIndeterminate' based on the number of selected entities
    const isIndeterminate = currentSelectedEntities?.length
      ? currentSelectedEntities?.length !== filteredEntities.length
      : false;
    // Step 4: Update 'isSelectAll' based on whether there are selected entities
    const isSelectAll = currentSelectedEntities?.length
      ? currentSelectedEntities?.length === filteredEntities?.length
      : false;
    return { isIndeterminate, isSelectAll };
  }

  /**
   * Determines the state of selected entities based on a grouping parameter.
   *
   * @param source - The ng2-smart-table data source.
   * @param groupByParam - The parameter by which entities are grouped.
   * @returns An object with 'selectedEntitiesForBulkOps' and 'statusOfSelectedEntity' properties
   *          indicating the state of selected entities, or `false` if entities of different types are selected.
   */
  determineSelectedEntityState(
    source,
    groupByParam,
  ):
    | {
        selectedEntitiesForBulkOps: any[];
        statusOfSelectedEntities: string;
      }
    | boolean {
    // Step 1: Filter currently selected entities from the source data
    const currentSelectedEntities = source?.['data']?.filter(
      (entity) => entity?.selected,
    );

    // Step 2: Group the selected entities based on the specified parameter
    const groupedEntities = groupBy(
      currentSelectedEntities,
      (entity: any) => entity[groupByParam],
    );

    // Step 3: Return false
    // if no entities are selected, source data is empty,
    // or groupBy parameter is not found in any entities
    // entities of different types are selected, return false if true
    if (Object.keys(groupedEntities).length > 1) {
      //User has selected resources of both type, return false from this function
      return false;
    }
    // Step 4: If entities are of the same type, update the selectedEntitiesForBulkOps
    let selectedEntitiesForBulkOps = cloneDeep(currentSelectedEntities);
    let statusOfSelectedEntities = Object.keys(groupedEntities)?.[0];

    // Step 5: Return an object with the state of selected entities
    return { selectedEntitiesForBulkOps, statusOfSelectedEntities };
  }
}
