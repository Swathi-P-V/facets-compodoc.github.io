import { Injectable } from '@angular/core';
import { Observable, Subject } from 'rxjs';
import { Branch, Environment } from '../cc-api/models';
import { BreadCrumbConfig } from '../models/breadcrumb.model';
import { CookieUtilWrapperService } from './cookie-util-wrapper.service';

@Injectable({
  providedIn: 'root',
})
export class ContextUpdatesService {
  successfulContextUpdates: Subject<any> = new Subject();

  optimisticContextUpdates: Subject<any> = new Subject();

  refreshContextSwitcher: Subject<any> = new Subject();

  contextSwitcherUpdated: Subject<any> = new Subject();

  allBlueprints: any[];

  allEnvironments: any[];

  blueprintResourceMap: Map<string, Map<string, string>>;

  currentContext: BreadCrumbConfig = {} as BreadCrumbConfig;

  // This map stores the last selected branch for a blueprint.
  // Would be initialized from the data stored in cookies as soon as
  // the service is created on load.
  blueprintBranchMap: any = {};
  BRANCH_COOKIE_NAME = 'lastBranchData';

  constructor(private cookieService: CookieUtilWrapperService) {
    this.initializeBranchData();
  }

  initializeBranchData() {
    this.blueprintBranchMap = this.getLastOpenedBranchMapFromCookies();
  }

  setContextUpdates(
    contexts: {
      currentContext: BreadCrumbConfig;
      previousContext: BreadCrumbConfig;
    },
    status: 'optimistic' | 'successful',
  ) {
    status === 'optimistic'
      ? this.optimisticContextUpdates.next(contexts)
      : this.successfulContextUpdates.next(contexts);
  }

  setCurrentContext(context: BreadCrumbConfig) {
    this.currentContext = context;
  }

  getCurrentContext() {
    return this.currentContext;
  }

  requestContextSwitcherRefresh(extras?: any) {
    this.refreshContextSwitcher.next({ refresh: true, extras: extras });
  }

  notifyContextSwitcherUpdate(extras: any) {
    this.contextSwitcherUpdated.next({ updated: true, extras: extras });
  }

  setCurrentEnvironments(environment: any[]) {
    this.allEnvironments = environment;
  }

  getCurrentEnvironments() {
    return this.allEnvironments;
  }

  setBlueprintResourceMapping(blueprint, resourceMap) {
    if (!this.blueprintResourceMap) {
      this.blueprintResourceMap = new Map<string, Map<string, string>>();
    }
    this.blueprintResourceMap.set(blueprint, resourceMap);
  }

  getResourceMapForBlueprint(blueprint) {
    return this.blueprintResourceMap?.get(blueprint);
  }

  getAllResourcesForBlueprint(blueprint) {
    if (!this.blueprintResourceMap) {
      return null;
    }
    let allResources = [];
    for (const [key, values] of Object.entries(
      this.blueprintResourceMap?.get(blueprint),
    )) {
      allResources.push(...values);
    }
    return allResources;
  }

  setAllBlueprints(blueprints) {
    this.allBlueprints = blueprints;
  }

  getAllBlueprints() {
    return this.allBlueprints;
  }

  getContextFromCookies(key: 'currentContext' | 'previousContext') {
    let value;
    try {
      value = JSON.parse(this.cookieService.get(key)) as BreadCrumbConfig;
    } catch {
      value = {} as BreadCrumbConfig;
    }
    return value;
  }

  setLastOpenedBranchInCookies(blueprint: string, branchName: string) {
    this.blueprintBranchMap[blueprint] = branchName;
    this.cookieService.set(
      this.BRANCH_COOKIE_NAME,
      JSON.stringify(this.blueprintBranchMap),
      undefined,
      '/',
    );
  }

  getLastOpenedBranchMapFromCookies() {
    const branchData = this.cookieService.get(this.BRANCH_COOKIE_NAME);
    if (branchData) {
      try {
        return JSON.parse(branchData);
      } catch (err) {
        return {};
      }
    } else {
      return {};
    }
  }

  getLastBranchForBlueprint(blueprint: string) {
    return this.getLastOpenedBranchMapFromCookies()[blueprint];
  }
}
