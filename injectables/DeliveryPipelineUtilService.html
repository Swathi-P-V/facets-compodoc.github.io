import { Injectable } from '@angular/core';
import { Graph } from '@antv/x6';
import { BehaviorSubject, Subject, of } from 'rxjs';
import { delay } from 'rxjs';

export class EnvironmentConfigPipeline {
  id: string;
  clusterName: string;
  clusterId: string;
  parentClusterId: string;
  config: {
    requiresSignOff: boolean;
    autoPromoteArtifact: boolean;
  };
}

export class PipelineTreeNode {
  width: number = 210;
  height: number = 192;
  id: string;
  name: string;
  clusterId: string;
  parentClusterId: string;
  config: {
    requiresSignOff: boolean;
    autoPromoteArtifact: boolean;
  };
  isRoot: boolean = false;
  isLeafNode: boolean = true;
  children: PipelineTreeNode[] = [];

  constructor(
    id: string,
    name: string,
    clusterId: string,
    parentClusterId: string,
    config: { requiresSignOff: boolean; autoPromoteArtifact: boolean },
    width?: number,
    height?: number,
  ) {
    this.id = id;
    this.name = name;
    this.clusterId = clusterId;
    this.parentClusterId = parentClusterId;
    this.config = config;
    this.isRoot = !id;
    if (width && height) {
      this.width = width;
      this.height = height;
    }
  }
}

@Injectable({
  providedIn: 'root',
})
export class DeliveryPipelineUtilService {
  private pipelineTree: PipelineTreeNode;
  private nodeMap: Map<string, PipelineTreeNode> = new Map<
    string,
    PipelineTreeNode
  >(); // Stores the node against an ID
  private environmentMap: Map<string, any> = new Map<string, any>();
  private allEnvironments: any[];
  private graph: Graph;
  private editModeOn: boolean = false;
  public graphConfigUpdates: Subject<any> = new Subject<any>();
  private cellToCenter: any;

  constructor() {}

  setEditModeTo(mode: 'on' | 'off') {
    this.editModeOn = mode === 'on';
  }

  isEditMode() {
    return this.editModeOn;
  }

  updateEnvironmentTransitionConfig(
    environment: PipelineTreeNode,
    oldParent: PipelineTreeNode,
    newParent: PipelineTreeNode,
  ) {
    // Run the logic for updating the parent

    let currentPipeline = this.pipelineTree;

    let _oldParent = this.findNodeInPipeline(
      this.pipelineTree,
      oldParent?.clusterId,
    );
    let _newParent =
      this.findNodeInPipeline(this.pipelineTree, newParent?.clusterId) ||
      this.pipelineTree;

    _oldParent.children = _oldParent.children.filter(
      (child) => child.clusterId != environment.clusterId,
    );
    _newParent.children.push(environment);
    environment.parentClusterId = _newParent.clusterId;
    this.nodeMap.set(environment.clusterId, environment);

    // Emit signal from the graphConfigUpdates and let the parent component
    // catch that signal and re-render the graph

    this.setCellToCenter(environment.id);

    // console.log('Updated pipeline => ', this.pipelineTree);

    this.graphConfigUpdates.next(this.pipelineTree);
  }

  getCellToCenter() {
    return this.cellToCenter;
  }

  setCellToCenter(cell) {
    this.cellToCenter = cell;
  }

  findNodeInPipeline(pipelineTree: PipelineTreeNode, id: string) {
    let stack = [pipelineTree];
    while (stack.length) {
      let top = stack.pop();
      if (top.clusterId == id) {
        return top;
      } else {
        let children = top.children || [];
        children.forEach((env) => {
          stack.push(env);
        });
      }
    }
  }

  getPossibleParents(environment: PipelineTreeNode) {
    let root = this.pipelineTree;
    let dfsStack = [root];
    let possibleParents: PipelineTreeNode[] = [];
    while (dfsStack.length) {
      let top = dfsStack.pop();
      if (top.id != environment.id) {
        let children = top.children;
        possibleParents.push(top);
        if (children?.length) {
          children.forEach((env) => {
            dfsStack.push(env);
          });
        }
      }
    }
    return possibleParents;
  }

  setPipelineTree(root: PipelineTreeNode) {
    this.pipelineTree = root;
  }

  getParentNode(environment: PipelineTreeNode) {
    if (environment.isRoot) {
      return environment;
    }
    return environment.parentClusterId
      ? this.nodeMap.get(environment.parentClusterId)
      : this.pipelineTree;
  }

  getNodeUsingEnvironmentID(id: string) {
    if (id == 'root') {
      return this.pipelineTree;
    }
    return this.nodeMap.get(id);
  }

  getAllEnvironments() {
    return this.allEnvironments;
  }

  setAllEnvironments(allEnvironments: any[], environmentMap: Map<string, any>) {
    this.allEnvironments = allEnvironments;
    this.environmentMap = environmentMap;
  }

  addToEnvironmentMap(id, environment) {
    if (
      this.environmentMap?.get(id)?.clusterName?.length ||
      this.environmentMap?.get(id)?.name?.length
    ) {
      return;
    }
    this.environmentMap?.set(id, environment);
  }

  getPipeline_unset(blueprintName: string) {
    return blueprintName.includes('spring')
      ? of(this.shuffleArray(configDevQAParallel)).pipe(delay(1000))
      : of(this.shuffleArray(configA)).pipe(delay(1000));
  }

  setGraph(graph: Graph) {
    this.graph = graph;
  }

  setNodeMap(nodeMap) {
    this.nodeMap = nodeMap;
  }

  // 'name' attribute should definitely be present inside the all values
  getEnvironmentMap() {
    return this.environmentMap;
  }

  shuffleArray(array) {
    let i = array.length;
    while (i--) {
      const ri = Math.floor(Math.random() * i);
      [array[i], array[ri]] = [array[ri], array[i]];
    }
    console.log('Shuffled array => ', array);
    return array;
  }
}
