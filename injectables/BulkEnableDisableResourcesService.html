import { Injectable } from '@angular/core';
import { Subject } from 'rxjs';
import { bulkResourceOperationButtons } from '../pages/resource-center/resource-list/enable-disable-footer-buttons';

@Injectable({
  providedIn: 'root',
})
export class BulkEnableDisableResourcesService {
  constructor() {}

  bulkResourceOpsComplete = new Subject<boolean>();
  updateFooterData = new Subject<boolean>();

  bulkResourceOperationButtons = bulkResourceOperationButtons;

  sendMessage(message: boolean) {
    this.bulkResourceOpsComplete.next(message);
  }

  sendFooterData(data) {
    this.updateFooterData.next(data);
  }

  findResource(eventData: any): (resource: any) => boolean {
    return (resource) =>
      resource.resourceType === eventData.rowData.resourceType &&
      resource.resourceName === eventData.rowData.resourceName;
  }

  getBulkEnableDisableActionButtons(status) {
    const validStatuses = ['true', 'false', 'Enabled', 'Disabled'];
    if (!validStatuses.includes(status)) {
      return undefined; // Unknown status
    }

    const cancelBtn = this.bulkResourceOperationButtons.find(
      (btn) => btn.id === 'cancel-bulk-enable-disable-resources',
    );
    if (['false', 'Enabled'].includes(status)) {
      const disableButton = this.bulkResourceOperationButtons.find(
        (btn) => btn.id === 'disable-resources',
      );
      disableButton.disabled = false;
      return [disableButton, cancelBtn];
    } else if (['true', 'Disabled'].includes(status)) {
      const enableButton = this.bulkResourceOperationButtons.find(
        (btn) => btn.id === 'enable-resources',
      );
      enableButton.disabled = false;
      return [enableButton, cancelBtn];
    }
  }
}
