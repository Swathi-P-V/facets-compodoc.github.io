import { Injectable } from '@angular/core';
import { Subject } from 'rxjs';
import { BlueprintFile, ResourceInfo } from 'src/app/cc-api/models';
import { UiApplicationControllerService } from 'src/app/cc-api/services';

@Injectable({
  providedIn: 'root',
})
export class ResourceDetailsService {
  resourceInfo: BlueprintFile;
  resourceInfoContent: any;
  resourceStaticImage: string;
  getResourceInfoObserver: Subject<boolean> = new Subject();
  showK8sExplorer: boolean = false;
  podsObserver: Subject<{ isBlueGreenInPreview: boolean }> = new Subject();
  resourceHistoryObserver: Subject<{ isResourceQueued: boolean }> =
    new Subject();

  constructor(
    private uiApplicationControllerService: UiApplicationControllerService,
  ) {}

  setResourceInfo(resourceInfo: ResourceInfo) {
    this.resourceInfo = resourceInfo;
    try {
      this.resourceInfoContent = JSON.parse(this.resourceInfo?.content || '{}');
    } catch (error) {
      this.resourceInfoContent = {};
    }
    this.resourceStaticImage = this.resourceInfo?.info?.staticImage;

    this.setK8sExplorerVisibleStatus();
  }
  isAlphaModule() {
    return this.resourceInfo?.alpha;
  }
  isSubstackResource() {
    return this.resourceInfo?.substack;
  }

  setK8sExplorerVisibleStatus = () => {
    this.showK8sExplorer = ['default', 'k8s'].includes(
      this.resourceInfo?.info?.['flavor'] || this.resourceInfo?.info?.flavour,
    );
  };

  setHideK8sExplorer = () => {
    this.showK8sExplorer = false;
  };

  setBlueGreenPreviewStatus = (pods) => {
    const isInPreview = pods?.find((pod) => pod?.role === 'preview');
    this.podsObserver.next({ isBlueGreenInPreview: !!isInPreview });
  };

  isResourceQueued({
    artifactHistory,
    overridesHistory,
    blueprintHistory,
  }): boolean {
    const allVersions = [
      ...(artifactHistory || []),
      ...(overridesHistory || []),
      ...(blueprintHistory || []),
    ];
    const isResourceQueuedStatus = allVersions?.find(
      (version) => version?.state?.toLowerCase() === 'queued',
    );

    return !!isResourceQueuedStatus;
  }

  triggerIsResourceQueued({
    artifactHistory,
    overridesHistory,
    blueprintHistory,
  }) {
    const isResourceQueuedStatus = this.isResourceQueued({
      artifactHistory,
      overridesHistory,
      blueprintHistory,
    });
    this.resourceHistoryObserver.next({
      isResourceQueued: isResourceQueuedStatus,
    });
  }
}
