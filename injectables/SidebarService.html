import { Injectable } from '@angular/core';
import { UserService } from 'src/app/services/user.service';
import { FeatureFlagsService } from 'src/app/services/feature-flags.service';
import { Subject } from 'rxjs';

@Injectable({
  providedIn: 'root',
})
export class SidebarService {
  BASE_URL: string = '';
  sidebarLoadingComplete = new Subject<any[]>();

  constructor(
    private userService: UserService,
    private featureFlagsService: FeatureFlagsService,
  ) {}

  getSidebarConfig(
    currentStack: string,
    currentCluster: string,
    currentClusterURL: string,
    branchName: string,
    allStacks: any[],
    noClustersPresent: boolean,
    currentURL: string,
  ) {
    const sidebarItems = [];

    const blueprintExpanded = this.isBlueprintMenuActive(
      currentURL,
      currentStack,
      currentCluster,
    );

    // If there are any blueprints available
    // Add the 'Artifacts' & 'Blueprint' menu items.
    if (allStacks?.length > 0) {
      let artifactsBaseURL =
        this.BASE_URL + '/stack/' + currentStack + '/artifacts/';

      sidebarItems.push({
        title: 'Artifacts',
        icon: { icon: 'book-close-outline', pack: 'fci' },
        home: true,
        selected: currentURL === artifactsBaseURL + 'ci',
        link: artifactsBaseURL + 'ci',
        pathMatch: 'full',
        expanded: currentURL.includes('/artifacts/'),
        children: [
          {
            title: 'CI Integration',
            selected: currentURL === artifactsBaseURL + 'ci',
            link: artifactsBaseURL + 'ci',
            pathMatch: 'full',
          },
          {
            title: 'CI Rules',
            selected: currentURL === artifactsBaseURL + 'ci-rules',
            link: artifactsBaseURL + 'ci-rules',
            pathMatch: 'full',
          },
          {
            title: 'Promotion Workflow',
            selected:
              currentURL === artifactsBaseURL + 'promotion-workflow/workflows',
            link: artifactsBaseURL + 'promotion-workflow/workflows',
            pathMatch: 'full',
          },
        ],
      });

      sidebarItems.push({
        title: 'Blueprints',
        icon: { icon: 'three-layers-outline', pack: 'fci' },
        link: this.BASE_URL + '/home',
        pathMatch: 'prefix',
        expanded: blueprintExpanded,
        selected: blueprintExpanded,
        children: [
          {
            title: 'List',
            selected: currentURL === this.BASE_URL + '/home',
            link: this.BASE_URL + '/home',
            pathMatch: 'full',
          },
          {
            title: 'Designer',
            selected: currentURL.includes(
              this.BASE_URL +
                '/stack/' +
                currentStack +
                `/blueprint-designer/${branchName}/`,
            ),
            link:
              this.BASE_URL +
              '/stack/' +
              currentStack +
              `/blueprint-designer/${branchName}/resource-list`,
            pathMatch: 'full',
            badge: {
              text: 'BETA',
              status: 'success',
            },
          },
          {
            title: 'Environments',
            selected: currentURL === this.BASE_URL + '/stack/' + currentStack,
            link: this.BASE_URL + '/stack/' + currentStack,
            pathMatch: 'full',
          },
          {
            title: 'Pipeline',
            selected:
              currentURL ===
              this.BASE_URL + '/stack/' + currentStack + '/pipeline',
            link: this.BASE_URL + '/stack/' + currentStack + '/pipeline',
            pathMatch: 'full',
          },
        ],
      });
    } else {
      // If there aren't any blueprints available
      // Just show the blueprints listing page
      // so that the user can create one
      sidebarItems.push({
        title: 'Blueprint',
        icon: { icon: 'three-layers-outline', pack: 'fci' },
        link: this.BASE_URL + '/home',
        pathMatch: 'prefix',
        expanded: true,
        children: [
          {
            title: 'List',
            selected: currentURL === this.BASE_URL + '/home',
            link: this.BASE_URL + '/home',
            pathMatch: 'full',
          },
        ],
      });
    }

    // Add the 'Environment' menu
    if (currentClusterURL && allStacks?.length > 0) {
      const environmentMenuChildren = [
        {
          title: 'Overview',
          selected:
            currentURL === this.BASE_URL + currentClusterURL + '/overview',
          link: this.BASE_URL + currentClusterURL + '/overview',
          pathMatch: 'prefix',
          visible: true,
        },
        {
          title: 'Secrets & Variables',
          selected:
            currentURL === this.BASE_URL + currentClusterURL + '/variables',
          link: this.BASE_URL + currentClusterURL + '/variables',
          pathMatch: 'prefix',
          visible: true,
        },
        {
          title: 'Releases',
          selected: currentURL.includes(
            this.BASE_URL + currentClusterURL + '/releases',
          ),
          link: this.BASE_URL + currentClusterURL + '/releases',
          pathMatch: 'prefix',
          visible: true,
        },
        {
          title: 'Tools',
          selected: currentURL.includes(
            this.BASE_URL + currentClusterURL + '/tools',
          ),
          link: this.BASE_URL + currentClusterURL + '/tools',
          pathMatch: 'prefix',
          visible: true,
        },
        {
          title: 'Resource Center',
          selected: currentURL.includes(
            this.BASE_URL + currentClusterURL + '/resources',
          ),
          link: this.BASE_URL + `${currentClusterURL}/resources`,
          pathMatch: 'prefix',
          visible: true,
        },
        {
          title: 'Alerts',
          selected:
            currentURL === this.BASE_URL + currentClusterURL + '/alerts',
          link: this.BASE_URL + currentClusterURL + '/alerts',
          pathMatch: 'prefix',
          visible: true,
        },
        {
          title: 'Backups',
          selected:
            currentURL ===
            this.BASE_URL + currentClusterURL + '/disaster-recovery',
          link: this.BASE_URL + currentClusterURL + '/disaster-recovery',
          pathMatch: 'prefix',
          visible: false,
        },
        {
          title: 'Provided Resources',
          selected:
            currentURL === this.BASE_URL + currentClusterURL + '/provided',
          link: this.BASE_URL + currentClusterURL + '/provided',
          pathMatch: 'prefix',
          visible: window?.location?.origin?.includes('moveinsync'),
        },
        {
          title: 'Information',
          selected:
            currentURL ===
            this.BASE_URL + currentClusterURL + '/resource-details',
          link: this.BASE_URL + currentClusterURL + '/resource-details',
          pathMatch: 'prefix',
          visible: true,
        },
        {
          title: 'Template Inputs',
          selected:
            currentURL ===
            this.BASE_URL + currentClusterURL + '/template-inputs',
          link: this.BASE_URL + currentClusterURL + '/template-inputs',
          pathMatch: 'prefix',
          visible: true,
        },
      ];
      sidebarItems.push({
        title: 'Environment',
        icon: { icon: 'cloud-outline', pack: 'fci' },
        expanded: true,
        hidden: noClustersPresent,
        link: this.BASE_URL + currentClusterURL,
        selected: currentURL === this.BASE_URL + currentClusterURL,
        pathMatch: 'prefix',
        children: environmentMenuChildren?.filter((menu) => menu?.visible),
      });

      if (!this.featureFlagsService?.getHideAssistantStatus()) {
        sidebarItems.push({
          title: 'AI Assistant',
          icon: { icon: 'ai', pack: 'fci' },
          selected: currentURL.includes('/ai-assistant/'),
          link: `${this.BASE_URL}/${currentClusterURL}/ai-assistant`,
          pathMatch: 'full',
          expanded: currentURL.includes(`${currentClusterURL}/ai-assistant/`),
          badge: {
            text: 'BETA',
            status: 'success',
          },
          children: [
            {
              title: 'Alerts Assistant',
              selected: currentURL.includes(`/ai-assistant/alerts`),
              link: `${this.BASE_URL}/${currentClusterURL}/ai-assistant/alerts`,
              pathMatch: 'full',
            },
            {
              title: 'Kubernetes Copilot',
              selected: currentURL.includes(`/ai-assistant/kubernetes`),
              link: `${this.BASE_URL}/${currentClusterURL}/ai-assistant/kubernetes`,
              pathMatch: 'full',
            },
          ],
        });
      }
    }

    // Add the 'Dashboard' menu
    if (allStacks?.length > 0) {
      sidebarItems.push({
        title: 'Dashboards',
        icon: { icon: 'bar-chart-outline', pack: 'fci' },
        selected: currentURL.includes('/dashboards/'),
        link: this.BASE_URL + '/stack/' + currentStack + '/dashboards',
        pathMatch: 'full',
        expanded: currentURL.includes('/dashboards/'),
        children: [
          {
            title: 'Cloud Cost Explorer',
            selected: currentURL.includes(
              this.BASE_URL +
                '/stack/' +
                currentStack +
                '/dashboards/cloud-cost-explorer',
            ),
            link:
              this.BASE_URL +
              '/stack/' +
              currentStack +
              '/dashboards/cloud-cost-explorer',
            pathMatch: 'full',
            badge: {
              text: 'BETA',
              status: 'success',
            },
          },
        ],
      });
    }

    // Add the 'Settings' menu
    const settingsSubMenu = this.userService.getSidebarSubMenuItemsOf(
      'SETTINGS',
      this.BASE_URL,
    );

    if (currentURL.includes('/settings/')) {
      settingsSubMenu.forEach((menuItem) => {
        if (currentURL.includes(menuItem.link)) {
          menuItem['selected'] = true;
        }
      });
    }

    if (settingsSubMenu?.length) {
      sidebarItems.push({
        title: 'Settings',
        icon: { icon: 'settings-outline', pack: 'fci' },
        link: `${this.BASE_URL}/settings`,
        pathMatch: 'prefix',
        selected: currentURL.includes('/settings'),
        expanded: currentURL.includes('/settings'),
        children: settingsSubMenu,
      });
    }

    // Add the 'Trash' menu
    sidebarItems.push({
      title: 'Trash',
      icon: { icon: 'trash-3-outline', pack: 'fci' },
      expanded: false,
      link: this.BASE_URL + '/trash',
      pathMatch: 'full',
      selected: currentURL === this.BASE_URL + '/trash',
    });

    let customSidebarItems = [];
    try {
      customSidebarItems = JSON.parse(
        this.featureFlagsService.getSidebarItems(),
      );
    } catch (error) {}
    if (customSidebarItems?.length) {
      customSidebarItems.forEach((parentItem) => {
        parentItem?.children?.forEach((child) => {
          (child.target = '_blank'), (child.skipLocationChange = true);
        });
        sidebarItems.push({
          title: parentItem?.title,
          icon: { icon: parentItem?.icon, pack: 'fci' },
          children: parentItem?.children,
        });
      });
    }
    this.sidebarLoadingComplete.next(sidebarItems);
    return sidebarItems;
  }

  isBlueprintMenuActive(currentURL, currentStack, currentCluster) {
    // Artifact pages and Dashboard pages have the stack name but not the environment but is not a part of blueprints sub-menu
    // Below condition checks for the same
    if (
      currentURL.includes('/artifacts/') ||
      currentURL.includes('/dashboards/')
    ) {
      return false;
    }
    return (
      (currentURL.includes(currentStack) &&
        !currentURL.includes(currentCluster)) ||
      currentURL.includes('/home') ||
      currentURL.includes('/createStack')
    );
  }
}
