import { Injectable } from '@angular/core';
import { UiCommonClusterControllerService } from '../cc-api/services';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { AccessDeniedService } from './access-denied.service';
import { OnAPIErrorService } from './on-api-error.service';

@Injectable({
  providedIn: 'root',
})
export class ReleaseScheduleCrudService {
  constructor(
    private commonClusterController: UiCommonClusterControllerService,
    private toastService: CustomToastService,
    private accessDeniedService: AccessDeniedService,
    private onAPIErrorService: OnAPIErrorService,
  ) {}

  pauseResumeReleaseSchedule(releaseScheduleData, clusterId, ref) {
    return new Promise((resolve, reject) => {
      let reqObj = {
        ...releaseScheduleData,
        pauseReleaseSchedule: !releaseScheduleData?.isPaused,
      };

      this.commonClusterController
        .updateClusterSchedulesUsingPUT({
          request: reqObj,
          clusterScheduleId: releaseScheduleData?.id,
          clusterId: clusterId,
        })
        .subscribe(
          (res) => {
            ref?.close();
            this.toastService.success(
              `Release Schedule has been ${
                res?.[0]?.isPaused ? 'paused' : 'resumed'
              } successfully.`,
            );
            resolve(res);
          },
          (error) => {
            ref?.close();
            this.onAPIErrorService.onAPIError(
              `Failed to ${
                releaseScheduleData?.isPaused ? 'resume' : 'pause'
              } release.`,
              error,
            );
          },
        );
    });
    // isPaused is the state of release schedule, when pausing/resuming send pauseReleaseSchedule
  }
}
