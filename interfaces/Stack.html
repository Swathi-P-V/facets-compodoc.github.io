    | 'CONTROL_PLANE'

    | 'ARTIFACT_CI'

    | 'ARTIFACTORY';
  githubOrganizationName?: string;
  id?: string;
  lastModifiedBy?: string;
  lastModifiedDate?: string;
  monoRepo?: boolean;
  name?: string;
  numberOfVersions?: number;
  pauseReleases?: boolean;
  primaryCloud?: 'AWS' | 'AZURE' | 'LOCAL' | 'GCP' | 'KUBERNETES';

  relativePath?: string;
  stackVars?: { [key: string]: string };
  template?: boolean;
  templateId?: string;
  templateName?: string;
  user?: string;
  vcs?: 'GITHUB' | 'BITBUCKET' | 'GITLAB';

  versioningKey?: string;
