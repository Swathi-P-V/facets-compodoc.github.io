import { Component, OnInit, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import {
  UiCustomRoleControllerService,
  UiDropdownsControllerService,
} from 'src/app/cc-api/services';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { SmartTableLongStringEllipsisComponent } from 'src/app/components/smart-table-long-string-ellipsis/smart-table-long-string-ellipsis.component';
import { SmartTableRouterlinkComponent } from 'src/app/components/smart-table-routerlink/smart-table-routerlink.component';
import { SmartTableWithBadgeComponent } from 'src/app/components/smart-table-with-badge/smart-table-with-badge.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import cloneDeep from 'lodash/cloneDeep';
import { FormatDateService } from 'src/app/services/format-date.service';
import {
  CUSTOM_ROLE_DELETE,
  CUSTOM_ROLE_WRITE,
} from 'src/app/constants/permissions';
import { sortPermissionsInPermissionGroups } from 'src/app/utils/permissions/permissions.utils';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const errorStateIcon = './assets/error.svg';
const emptyStateIcon = './assets/empty.svg';
const noAccessIcon = './assets/alert-empty.svg';

export interface Permission {
  permission: string;
  description: string;
  key: string;
}

@Component({
  selector: 'app-user-roles',
  templateUrl: './user-roles.component.html',
  styleUrls: ['./user-roles.component.scss'],
})
export class UserRolesComponent extends FacetsPage implements OnInit {
  customRoleWrite = CUSTOM_ROLE_WRITE;
  // Roles table
  smartTableCustomActionButtons: any = [
    {
      id: 'edit',
      icon: 'edit-2-outline',
      toolTip: 'Edit Role',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: CUSTOM_ROLE_WRITE,
      onClick: (data: any) => {
        this.editCustomRole(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete Role',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: CUSTOM_ROLE_DELETE,
      onClick: (data: any) => {
        this.requestDeletePopup(data);
      },
    },
  ];

  rolesTablesConfig: any = {
    columns: {
      label: {
        title: 'Role',
        filter: true,
      },
      description: {
        title: 'Description',
        filter: false,
        type: 'custom',
        valuePrepareFunction: (cell: any) => {
          return JSON.stringify({
            value: cell,
            truncateLength: 35,
          });
        },
        renderComponent: SmartTableLongStringEllipsisComponent,
      },
      roleType: {
        title: 'Type',
        filter: false,
        type: 'custom',
        valuePrepareFunction: (cell) => {
          return JSON.stringify({
            severity:
              cell === 'SYSTEM_DEFINED' ? 'System defined' : 'Custom role',
            background: cell === 'SYSTEM_DEFINED' ? '#DEE2FE' : '#DAF6F5',
            color: cell === 'SYSTEM_DEFINED' ? '#040849' : '#02A19B',
          });
        },
        renderComponent: SmartTableWithBadgeComponent,
      },
      // createdBy: {
      //   title: 'Created by',
      //   filter: false,
      //   type: 'custom',
      //   renderComponent: UserRendererComponent,
      // },
      lastModifiedBy: {
        title: 'Last modified by',
        filter: false,
        valuePrepareFunction: (cell: any) => {
          return cell || 'NA';
        },
      },
      lastModifiedDate: {
        title: 'Last modified date',
        filter: false,
        valuePrepareFunction: (cell: Date) => {
          return cell ? this.formatDateService.getFormattedDate(cell) : 'NA';
        },
      },
      entities: {
        title: 'Permissions',
        filter: false,
        type: 'custom',
        valuePrepareFunction: (value, row, cell) => {
          const data = {
            url: null,
            display: 'View Permissions',
            linkUnderline: true,
          };
          return JSON.stringify(data);
        },
        renderComponent: SmartTableRouterlinkComponent,
        onComponentInitFunction: (instance: any) => {
          instance.byLinkClick.subscribe((data: any) => {
            this.openViewPermissionsDialog(data);
          });
        },
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          let formattedSmartTableCustomActionButtons;
          //Considering roleName as uniquie column here
          if (cell?.row?.data?.roleType != 'SYSTEM_DEFINED') {
            formattedSmartTableCustomActionButtons = cloneDeep(
              this.smartTableCustomActionButtons,
            );
            formattedSmartTableCustomActionButtons.forEach((button) => {
              button['uniqueId'] = button.id + '_' + row.roleName;
            });
          }
          const actionButtons =
            cell?.row?.data?.roleType != 'SYSTEM_DEFINED'
              ? {
                  smartTableCustomActionButtons:
                    formattedSmartTableCustomActionButtons,
                }
              : {};
          return JSON.stringify(actionButtons);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance: any) => {
          instance.byCustomActionButtonClick.subscribe((data: any) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '12%',
        filter: false,
      },
    },
    noDataMessage: 'No roles available to display',
    pager: {
      display: false,
      perPage: 10,
    },
    mode: 'external',
    actions: false,
  };

  noDataIcon = emptyStateIcon; // Empty state icon from assets

  // Fetching
  fetchingRoles: boolean = false; // Status indicator for fetching roles from the backend
  fetchingError: boolean = false; // Error indicator for API response
  primaryMessage = 'No roles found'; // Primary message for no roles, will be overridden in case of fetch error
  secondaryMessage = 'Click below to create your first custom role'; // Secondary message for no roles, will be overridden in case of fetch error
  allRoles: any[];

  // Creating
  creatingRole: boolean = false;

  // Deleting
  deletingRole: boolean = false;

  // Permissions
  @ViewChild('viewPermissions', { static: true })
  viewPermissionsDialog: TemplateRef<any>;
  @ViewChild('deleteCustomRole', { static: true })
  deleteCustomRoleDialog: TemplateRef<any>;
  permissionGroups: Map<string, Permission[]> = new Map();
  permissionGroupsArray: { categoryName: any; categoryLabel: any }[] = [];
  currentRoleInPermissionsView: any;
  fetchingPermissions: boolean = false;
  showError: boolean = false;
  permissionsErrorMessage: string;
  permissionsErrorIcon: any;
  noAccessIcon = noAccessIcon;

  constructor(
    private uiCustomRoleControllerService: UiCustomRoleControllerService,
    private uiDropDownControllerService: UiDropdownsControllerService,
    private dialogService: NbDialogService,
    private router: Router,
    private activatedRouter: ActivatedRoute,
    private formatDateService: FormatDateService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRouter);
  }

  ngOnInit(): void {
    this.fetchAllRoles();
  }

  OnPageLoad(p: Params): void {
    throw new Error('Method not implemented.');
  }

  fetchAllRoles() {
    function setErrorData(error: any, _this: any) {
      _this.fetchingError = true;
      _this.fetchingRoles = false;
      (_this.primaryMessage = 'Failed to fetch roles'),
        (_this.secondaryMessage =
          error?.error?.message ||
          'Something went wrong, please try again later or contact your admin');
      _this.noDataIcon = errorStateIcon;
    }

    this.fetchingRoles = true;
    this.fetchingError = false;
    this.uiCustomRoleControllerService
      .getAllRolesUsingGET()
      .subscribe(
        (allRoles: any[]) => {
          this.allRoles = allRoles?.filter((roles) => roles?.baseRole);
        },
        (error: any) => {
          setErrorData(error, this);
        },
      )
      .add(() => {
        this.fetchingRoles = false;
      });
  }

  async openViewPermissionsDialog(role: any) {
    this.permissionGroups.clear();
    this.permissionGroupsArray = [];
    this.showError = false;
    this.currentRoleInPermissionsView = role;

    this.dialogService.open(this.viewPermissionsDialog);

    //Permissions for all roles is now available in entities key of each role
    //If the entities is empty array then show an empty state
    if (role?.entities?.length) {
      role.entities.forEach((p) => {
        let permissionGroup = p?.category?.categoryName;
        if (!this.permissionGroups.has(permissionGroup)) {
          this.permissionGroups.set(permissionGroup, []);
          this.permissionGroupsArray.push(p?.category);
        }
        this.permissionGroups.get(permissionGroup).push(p);
      });
      this.permissionGroupsArray.sort((a, b) =>
        a?.categoryLabel?.localeCompare(b?.categoryLabel),
      );

      this.permissionGroups = sortPermissionsInPermissionGroups(
        this.permissionGroups,
      );
    } else {
      this.showError = true;
      this.permissionsErrorIcon = errorStateIcon;
      this.permissionsErrorMessage = '';
    }
  }

  async getPermissionsForRole(roleName: string) {
    return new Promise((resolve) => {
      this.uiDropDownControllerService
        .getRolePermissionsUsingGET(roleName)
        .subscribe(
          (permissions: any[]) => {
            resolve(permissions);
          },
          (error: any) => {
            resolve(error?.error?.message || 'Error occurred');
          },
        );
    });
  }

  createCustomRole() {
    this.router.navigate(['../../user-custom-role/create'], {
      relativeTo: this.route,
      state: {
        allRoles: this.allRoles,
      },
    });
  }

  editCustomRole(role: any, ref?: any) {
    if (ref) {
      ref.close();
    }
    // console.log('Edit -> ', role);
    this.router.navigate([`../../user-custom-role/edit/${role.roleName}`], {
      relativeTo: this.route,
      state: {
        allRoles: this.allRoles,
        currentRole: role,
      },
    });
  }

  requestDeletePopup(role: any) {
    this.dialogService.open(this.deleteCustomRoleDialog, {
      context: {
        role: role,
      },
    });
  }

  confirmRoleDelete(ref: any, roleName: any) {
    this.deletingRole = true;
    this.uiCustomRoleControllerService
      .deleteCustomRoleUsingDELETE(roleName)
      .subscribe(
        (rsp) => {
          this.customToastService.success('Role deleted - ' + roleName);
          if (ref) {
            ref.close();
          }
          this.fetchAllRoles();
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            'Could not delete role - ' + error?.error?.message,
            error,
          );
        },
      )
      .add(() => {
        this.deletingRole = false;
      });
  }
}
