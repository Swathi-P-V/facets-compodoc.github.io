  cloud?: 'AWS' | 'AZURE' | 'LOCAL' | 'GCP' | 'KUBERNETES';
  cloudAccountId?: string;
  cloudAccountSecretId?: string;
  clusterCode?: string;
  clusterState?:
    | 'STOPPED'
    | 'RUNNING'
    | 'LAUNCHING'
    | 'DESTROYING'
    | 'LAUNCH_FAILED'
    | 'DESTROY_FAILED'
    | 'UNKNOWN'
    | 'SCALE_DOWN'
    | 'SCALING_DOWN'
    | 'SCALE_DOWN_FAILED'
    | 'SCALING_UP'
    | 'SCALE_UP_FAILED';
  commonEnvironmentVariables?: { [key: string]: string };
  componentVersions?: { [key: string]: string };
  createdBy?: string;
  creationDate?: string;
  deleted?: boolean;
  enableAutoSignOff?: boolean;
  entityType?:
    | 'CLUSTER'
    | 'BLUE_PRINT'
    | 'TEMPLATE_INPUT'
    | 'CONTROL_PLANE'
    | 'IAC'
    | 'ARTIFACT_CI'
    | 'USER_GROUP'
    | 'ACCOUNT'
    | 'ARTIFACTORY';
  globalVariables?: { [key: string]: string };
  id?: string;
  isEphemeral?: boolean;
  k8sRequestsToLimitsRatio?: number;
  lastModifiedBy?: string;
  lastModifiedDate?: string;
  name?: string;
  namespace?: string;
  numberOfVersions?: number;
  pauseReleases?: boolean;
  releaseStream?: string;
  requireSignOff?: boolean;
  schedules?: { [key: string]: string };
  secrets?: { [key: string]: string };
  secretsUid?: string;
  stackName?: string;
  tz?: string;
  variables?: { [key: string]: Variables };
  versioningKey?: string;
}
