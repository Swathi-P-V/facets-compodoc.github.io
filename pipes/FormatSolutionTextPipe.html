import { Pipe, PipeTransform } from '@angular/core';
import { DomSanitizer, SafeHtml } from '@angular/platform-browser';

/**
 * This pipe is used to format the plain string and get solutions in proper format. This is currently being used in K8s-assistant component.
 * Input:
    Error: The Kubernetes service does not have any endpoints and is expecting a label component called 'kube-controller-manager'. Solution: 1. Check if the service is correctly configured. 2. Ensure that the 'kube-controller-manager' component is correctly labeled in your Kubernetes setup.
  * Output:
    Solution:
      1. Check if the service is correctly configured.
      2. Ensure that the 'kube-controller-manager' component is correctly labeled in your Kubernetes setup.
**/
@Pipe({ name: 'formatSolutionText' })
export class FormatSolutionTextPipe implements PipeTransform {
  constructor() {}

  // Add a new line before number followed by dot(.)
  addNewLineBeforeNumbers(inputString) {
    const formattedString = inputString
      ?.replace(/(\d+)(\.)/g, '\n$1$2')
      ?.replace(/\n{2,}/g, '\n')
      .trim();

    return formattedString;
  }

  transform(errorMsg) {
    if (!errorMsg) {
      return '';
    }
    let formattedSolution = errorMsg;
    const solutionIndex = formattedSolution?.indexOf('Solution:');
    if (solutionIndex !== -1) {
      formattedSolution = formattedSolution?.slice(solutionIndex)?.trim();
    }

    return this.addNewLineBeforeNumbers(formattedSolution);
  }
}
