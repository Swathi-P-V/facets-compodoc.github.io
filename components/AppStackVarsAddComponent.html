import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { UiBlueprintDesignerControllerService } from 'src/app/cc-api/services';
import { Component } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import {
  whiteSpaceInputPattern,
  whiteSpaceAllPattern,
} from 'src/app/constants/inputPatterns';
import { LocalDataSource } from 'ng2-smart-table';
import cloneDeep from 'lodash/cloneDeep';
import { variableSettings } from 'src/app/constants/secrets-variables-config';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-stack-vars-add',
  templateUrl: './stack-vars-add.component.html',
  styleUrls: ['./stack-vars-add.component.scss'],
})
export class AppStackVarsAddComponent extends FacetsPage {
  stackName: string = '';
  secretsAndVariablesForm: FormGroup = this.fb.group({
    name: ['', [Validators.required, Validators.pattern(whiteSpaceAllPattern)]],
    value: [null, [Validators.required]],
    description: [''],
    type: ['SECRET', Validators.required],
    global: [false, Validators.required],
  });
  radioTypeOptions: { label: string; value: string; disabled: boolean }[] = [
    { label: 'Secret', value: 'SECRET', disabled: false },
    { label: 'Variable', value: 'VARIABLE', disabled: false },
  ];

  settings = variableSettings;
  stackVarsSource: LocalDataSource = new LocalDataSource();
  variablesList: any = [];
  creationInProgress: boolean = false;
  constructor(
    route: ActivatedRoute,
    private fb: FormBuilder,
    private router: Router,
    private blueprintDesignerService: UiBlueprintDesignerControllerService,
    private customToast: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(route);
  }

  get name() {
    return this.secretsAndVariablesForm.get('name');
  }

  get value() {
    return this.secretsAndVariablesForm.get('value');
  }

  get description() {
    return this.secretsAndVariablesForm.get('description');
  }

  get type() {
    return this.secretsAndVariablesForm.get('type');
  }

  get global() {
    return this.secretsAndVariablesForm.get('global');
  }

  OnPageLoad(p: Params): void {
    if (p?.stackName) {
      this.stackName = p.stackName;
    }
  }

  handleBack = () => {
    this.router.navigate(['/', 'stack', this.stackName, 'variables']);
  };

  addVariableEntry() {
    const isVariablePresent = this.variablesList.find(
      (variable) => variable?.name === this.name?.value,
    );
    if (!isVariablePresent) {
      this.variablesList.push(this.secretsAndVariablesForm.value);
      this.stackVarsSource.load(this.variablesList);
      this.resetForm();
    } else {
      this.customToast.error('Variable name should be unique');
    }
  }

  resetForm() {
    this.secretsAndVariablesForm.reset();
    this.type.setValue('SECRET');
    this.global.setValue(false);
  }

  validateSecrets(variableChangeData) {
    /*
      Check if event data's new value is not white space, then only process the value as valid
    */
    if (whiteSpaceInputPattern.test(variableChangeData?.newData?.value)) {
      variableChangeData.confirm.resolve(variableChangeData?.newData);
    } else {
      variableChangeData.confirm.reject();
    }
  }

  deleteSecretsConfirm(event) {
    this.variablesList = cloneDeep(this.variablesList).filter(
      (variable) => variable?.name !== event?.data?.name,
    );
    this.stackVarsSource.load(this.variablesList);
  }

  createVariables() {
    this.stackVarsSource?.getAll().then((variableList) => {
      this.creationInProgress = true;
      if (variableList?.length > 0) {
        const addVarsParams = {
          stackName: this.stackName,
          variables: {},
        };
        variableList?.forEach((variable) => {
          addVarsParams.variables[variable['name']] = {
            value: variable['value'],
            global: variable['global'],
            description: variable['description'],
            secret: variable['type'] === 'SECRET',
          };
        });
        this.blueprintDesignerService
          .addVariablesUsingPOST(addVarsParams)
          .subscribe(
            (data) => {
              this.customToast.success(
                `${this.variablesList?.length} Variables created successfully`,
              );
              this.creationInProgress = false;
              this.handleBack();
            },
            (error) => {
              this.onAPIErrorService.onAPIError(
                error?.error?.message || `Variable creation failed`,
                error,
              );
              this.creationInProgress = false;
            },
          );
      }
    });
  }
}
