import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import { NotificationChannel, Subscription } from 'src/app/cc-api/models';
import { UiNotificationControllerService } from 'src/app/cc-api/services';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { SmartTableHyperlinkComponent } from 'src/app/components/smart-table-hyperlink/smart-table-hyperlink.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CHANNEL_WRITE, CHANNEL_DELETE } from 'src/app/constants/permissions';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

const errorIcon = './assets/error.svg';
const emptyStateChannel = './assets/channel-empty.svg';

@Component({
  selector: 'app-channel-list',
  templateUrl: './channel-list.component.html',
  styleUrls: ['./channel-list.component.scss'],
})
export class ChannelListComponent extends FacetsPage {
  channels: Array<NotificationChannel>;
  subscriptions: Array<Subscription>;
  nc: NotificationChannel = {};
  sub: Subscription = {};
  notificationError: any = '';
  channelMap = {};
  testingResult: string;
  loadingChannel: boolean = true;
  emptyLogoChannel = emptyStateChannel;
  errorLogo = errorIcon;
  primaryChannel: string = 'No channels created!';
  secondaryChannel: string = 'Create channel to send and manage notifications';
  showEmptyChannel: boolean = false;
  statusChannel: 'info' | 'error' = 'info';
  showBtnChannel: boolean = true;
  buttonTextChannel: string = 'Add Channel';
  tooltipChannel: string = 'Add Channel';
  deleteInProgress: boolean;
  channelWritePermission = CHANNEL_WRITE;
  channelTypes = [];

  actionButtons = [
    {
      id: 'edit',
      icon: 'edit-2-outline',
      toolTip: 'Edit Channel',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: CHANNEL_WRITE,
      onClick: (data) => {
        this.openChannelForm(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete Channel',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: CHANNEL_DELETE,
      onClick: (data) => {
        this.deleteChannel(data);
      },
    },
  ];

  channelsSettings: any = {
    columns: {
      channelType: {
        title: 'Type',
        width: '20%',
        filter: {
          type: 'list',
          config: {
            selectText: 'Type',
            list: this.channelTypes,
          },
        },
      },
      name: {
        title: 'Channel Name',
        filter: true,
        width: '20%',
      },
      channelAddress: {
        title: 'URL',
        filter: true,
        width: '48%',
        type: 'custom',
        valuePrepareFunction: (cell) => {
          return JSON.stringify({ data: cell, showPlainText: true });
        },
        renderComponent: SmartTableHyperlinkComponent,
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {
            smartTableCustomActionButtons: this.actionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.actionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '12%',
        filter: false,
      },
    },
    pager: {
      display: false,
    },
    actions: false,
  };

  @ViewChild('deleteChannelConfirmUI', { static: true })
  private deleteChannelConfirmUI: TemplateRef<any>;

  constructor(
    activatedRoute: ActivatedRoute,
    private uiNotificationControllerService: UiNotificationControllerService,
    private router: Router,
    private dialogService: NbDialogService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.getAllChannelTypes().then(
      (res: []) => {
        this.channelTypes = res;
        this.channelsSettings.columns.channelType.filter.config.list =
          this.channelTypes;

        this.uiNotificationControllerService.getAllChannelsUsingGET().subscribe(
          (res) => {
            if (res?.length <= 0) {
              this.showEmptyChannel = true;
            } else {
              this.populateChannels(res);
            }
            this.loadingChannel = false;
          },
          (err) => {
            this.loadingChannel = false;
            this.showEmptyChannel = true;
            this.setErrorVars(err);
          },
        );
      },
      (error) => {
        this.loadingChannel = false;
        this.showEmptyChannel = true;
        this.setErrorVars(error);
      },
    );
  }

  setErrorVars(error: any) {
    this.primaryChannel = '';
    this.secondaryChannel =
      error?.error?.message ||
      ' There is some error fetching channels information';
    this.showBtnChannel = false;
    this.statusChannel = 'error';
    this.emptyLogoChannel = this.errorLogo;
  }

  populateChannels(res: Array<NotificationChannel>) {
    this.channels = res;
    this.channels.forEach((channel) => {
      this.channelMap[channel.id] = channel.name;
    });
  }

  openChannelForm(rowData?) {
    let url = rowData ? `../channel/edit/${rowData.id}` : '../channel/create';
    this.router.navigate([url], {
      relativeTo: this.route,
      state: {
        channelData: rowData,
      },
    });
  }

  deleteChannel(data: any) {
    this.dialogService.open(this.deleteChannelConfirmUI, { context: data });
  }

  confirmChannelDelete(ref, channelData) {
    try {
      this.deleteInProgress = true;
      this.uiNotificationControllerService
        .deleteNotificationChannelUsingDELETE(channelData.id)
        .subscribe(
          (remainingChannels) => {
            this.populateChannels(remainingChannels);
            this.customToastService.success('Channel successfully deleted ');
            this.deleteInProgress = false;
            ref.close();
          },
          (error) => {
            this.deleteInProgress = false;
            ref.close();
            this.onAPIErrorService.onAPIError(
              'Channel delete failed. ' + error?.error?.message,
              error,
            );
          },
        );
    } catch (err) {
      this.deleteInProgress = false;
      this.customToastService.error('Channel delete failed');
      ref.close();
    }
  }

  getAllChannelTypes() {
    return new Promise((resolve, reject) => {
      this.uiNotificationControllerService
        .getAllChannelTypesUsingGET()
        .subscribe(
          (res) => {
            const types = res.map((channel) => ({
              title: channel?.label,
              value: channel?.type,
            }));
            resolve(types);
          },
          (error) => {
            reject(error);
          },
        );
    });
  }
}
