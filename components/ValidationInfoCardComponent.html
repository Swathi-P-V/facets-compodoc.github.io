import {
  Component,
  Input,
  OnChanges,
  OnDestroy,
  SimpleChanges,
} from '@angular/core';
import { NbMenuService } from '@nebular/theme';
import { filter, map } from 'rxjs';
import { BlueprintValidationError } from 'src/app/cc-api/models';
import * as iconMapping from 'src/app/constants/facetsIconMap.json';
import { ContextUpdatesService } from 'src/app/services/context-updates.service';

class ValidationResponseExtended {
  directory?: string;
  errors?: Array<BlueprintValidationError>;
  filename?: string;
  resourceName?: string;
  resourceType?: string;
  error: BlueprintValidationError;
}

@Component({
  selector: 'app-validation-card',
  templateUrl: './validation-info-card.component.html',
  styleUrls: ['./validation-info-card.component.scss'],
})
export class ValidationInfoCardComponent implements OnChanges, OnDestroy {
  @Input() validation: ValidationResponseExtended;
  @Input() stackName: string;
  @Input() clusterId: string;
  @Input() showSeverity: boolean = true;

  errorBadge: any;
  warningBadge: any;
  iconName: string;
  menuItemSubscriber: any;
  menuItems = [
    {
      id: 'blueprint-spec',
      title: 'Blueprint spec',
    },
    {
      id: 'environment-spec',
      title: 'Environment spec',
    },
  ];

  constructor(
    private menuService: NbMenuService,
    private contextUpdatesSerivce: ContextUpdatesService,
  ) {
    this.errorBadge = JSON.stringify({
      severity: 'error',
      badgeClass: 'p-2 text-capitalize mb-0 critical',
    });

    this.warningBadge = JSON.stringify({
      severity: 'warning',
      badgeClass: 'p-2 text-capitalize mb-0',
    });

    this.menuItemSubscriber = this.menuService
      .onItemClick()
      .pipe(
        filter((data) => {
          return (
            data.tag ===
            `open-resource-spec-${this.validation?.resourceName}-${this.validation?.resourceType}`
          );
        }),
      )
      .subscribe((menuItem) => {
        this.openInNewTab(menuItem, this.validation);
      });
  }

  ngOnChanges(changes: SimpleChanges): void {
    if (changes.validation?.currentValue) {
      this.iconName = iconMapping[
        changes?.validation?.currentValue?.resourceType
      ]
        ? changes.validation?.currentValue?.resourceType
        : 'default';
    }
  }

  openInNewTab = (menuItem, validation) => {
    let baseURL = '';
    let urlToOpen = '';
    if (menuItem?.item?.id === 'blueprint-spec') {
      baseURL = `${window?.location?.origin}/capc/stack/${
        this.stackName
      }/blueprint-designer/${this.contextUpdatesSerivce.getLastBranchForBlueprint(
        this.stackName,
      )}/resource/type/${this.validation?.resourceType}/name/${
        this.validation?.resourceName
      }`;
      urlToOpen =
        this.validation?.error?.category === 'Syntax Error'
          ? `${baseURL}/overview`
          : `${baseURL}/spec`;
    } else if (menuItem?.item?.id === 'environment-spec') {
      baseURL = `${window?.location?.origin}/capc/${this.stackName}/cluster/${this.clusterId}/resources/type/${this.validation?.resourceType}/resource/${this.validation?.resourceName}/configuration`;
      urlToOpen = baseURL;
    }

    window.open(urlToOpen, '_blank');
  };

  ngOnDestroy(): void {
    this.menuItemSubscriber?.unsubscribe();
  }
}
