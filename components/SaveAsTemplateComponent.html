import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import {
  Component,
  ViewChild,
  TemplateRef,
  Input,
  OnChanges,
  SimpleChanges,
} from '@angular/core';
import { NbDialogService } from '@nebular/theme';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { UiStackControllerService } from 'src/app/cc-api/services';
import { UserAuthoritiesService } from 'src/app/services/user-authorities.service';
import { ContextUpdatesService } from '../../../services/context-updates.service';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { BLUEPRINT_TEMPLATE_WRITE } from './../../../constants/permissions';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-save-as-template',
  templateUrl: './save-as-template.component.html',
  styleUrls: ['./save-as-template.component.scss'],
})
export class SaveAsTemplateComponent extends FacetsPage implements OnChanges {
  @ViewChild('templateCreationDialog') templateCreationDialog: TemplateRef<any>;
  @Input() vscType: string = 'GITHUB';

  templateCreationForm: FormGroup = this.fb.group({
    name: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    description: [
      null,
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
  });
  stackName: string = '';
  templateCreationInProgress: boolean = false;

  tertiaryButtons: any = [];

  templateBtn = {
    id: 'save-as-template',
    title: 'Save as Template',
    buttonType: 'primary',
    icon: 'template-icon',
    iconPack: 'fci',
    hidden: false,
    action: BLUEPRINT_TEMPLATE_WRITE,
    loading: false,
    iconStatus: 'primary',
  };
  variablesBtn: any = {
    id: 'variables',
    title: 'Secrets and Variables',
    buttonType: 'primary',
    icon: 'lock-outline',
    iconPack: 'eva',
    hidden: false,
    loading: false,
    iconStatus: 'primary',
    action: BLUEPRINT_TEMPLATE_WRITE,
  };
  toastersLinks = [
    {
      id: 'blueprint-list-page',
      label: 'See blueprints',
      icon: 'arrow-ios-forward-outline',
      click: () => {
        this.routeToBlueprintsListPage();
      },
    },
  ];
  toasterLinkSubscription: any;

  constructor(
    private dialogService: NbDialogService,
    private fb: FormBuilder,
    route: ActivatedRoute,
    private uiStackControllerService: UiStackControllerService,
    public userAuthoritiesService: UserAuthoritiesService,
    private contextUpdatesService: ContextUpdatesService,
    private customToast: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
    private router: Router,
  ) {
    super(route);
  }

  get name() {
    return this.templateCreationForm.get('name');
  }

  get description() {
    return this.templateCreationForm.get('description');
  }

  OnPageInit(): void {
    this.toasterLinkSubscription = this.customToast.buttonClick$.subscribe(
      (data) => {
        this.toastersLinks.find((link) => link?.id === data?.link?.id)?.click();
      },
    );
  }
  OnPageLoad(p: Params) {
    this.stackName = p?.stackName;
  }

  onButtonClick(button) {
    if (button?.id === 'save-as-template') {
      this.dialogService.open(this.templateCreationDialog);
    } else if (button?.id === 'variables') {
      this.openStackSecrets();
    }
  }

  openStackSecrets() {
    this.router.navigate(['/', 'stack', this.stackName, 'variables']);
  }

  ngOnChanges(changes: SimpleChanges): void {
    this.tertiaryButtons = [];
    if (changes?.vscType && changes?.vscType?.currentValue?.length) {
      let vcsVal = changes?.vscType?.currentValue;
      if (vcsVal === 'GITHUB') {
        this.tertiaryButtons = [this.templateBtn, this.variablesBtn];
      } else {
        this.tertiaryButtons.push(this.variablesBtn);
      }
    } else {
      this.tertiaryButtons.push(this.variablesBtn);
    }
  }

  createTemplate(ref) {
    this.templateCreationInProgress = true;
    this.uiStackControllerService
      .saveAsTemplateUsingPOST({
        stackName: this.stackName,
        createTemplateRequest: this.templateCreationForm.value,
      })
      .subscribe(
        (data) => {
          this.contextUpdatesService.requestContextSwitcherRefresh();
          this.customToast.show({
            type: 'success',
            title: 'Success',
            content:
              'Blueprint template created! You can find the template in the template gallery while creating a new blueprint.',
            link: this.toastersLinks[0],
          });

          this.templateCreationInProgress = false;
          ref.close();
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            error?.error?.message ||
              error?.message ||
              'Blueprint template could not be created.',
            error,
          );
          this.templateCreationInProgress = false;
          // ref.close();
        },
      );
  }

  routeToBlueprintsListPage() {
    this.router.navigate(['/', 'home']);
  }

  OnDestroy() {
    this.toasterLinkSubscription?.unsubscribe();
  }
}
