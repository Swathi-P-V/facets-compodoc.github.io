import { Component, TemplateRef, ViewChild } from '@angular/core';
import { Params, ActivatedRoute, Router } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ResourceHistoryService } from '../../services/resource-history.service';
import cloneDeep from 'lodash/cloneDeep';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { NbDialogService } from '@nebular/theme';
import { UiVersioningControllerService } from 'src/app/cc-api/services';
import { CustomToastService } from '../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

const emptyIcon = './assets/empty.svg';

@Component({
  selector: 'app-resource-override-history',
  templateUrl: './resource-override-history.component.html',
  styleUrls: ['./resource-override-history.component.scss'],
})
export class ResourceOverrideHistoryComponent extends FacetsPage {
  blueprintName: string = '';
  clusterId: string;
  resourceType: string;
  resourceName: string;

  loadingResourceHistory: boolean = false;
  currentVersion: any;
  previousVersions: any;
  currentVersionTableSettings: any;
  previousVersionTableSettings: any;
  allOverrideHistory: any;
  editorOptions = { theme: 'vs-light', language: 'javascript' };
  versionRollbackInProgress: boolean = false;
  errorLoadingResourceHistory: string = '';
  errorStateLogo: any = './assets/error.svg';
  fromQueryParam: string = '';

  previousVersionsAction = [
    {
      id: 'compare-current',
      icon: 'current-compare',
      toolTip: 'Compare with current version',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: null,
      isFciPack: true,
      onClick: (rowData) => this.openOverrideDiff(rowData, 'compare-current'),
    },
    {
      id: 'compare-last',
      icon: 'last-compare',
      toolTip: 'View changes',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: null,
      isFciPack: true,
      onClick: (rowData) => this.openOverrideDiff(rowData, 'compare-last'),
    },
    {
      id: 'rollback',
      icon: 'refresh-outline',
      toolTip: 'Configuration rollback',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: 'ENVIRONMENT_CONFIGURE',
      isFciPack: true,
      onClick: (rowData) => this.openConfirmationPopup(rowData),
    },
  ];
  currentVersionActions = [
    {
      id: 'compare-last',
      icon: 'last-compare',
      toolTip: 'View changes',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: null,
      isFciPack: true,
    },
  ];

  @ViewChild('openDiffEditor') openDiffEditor: TemplateRef<any>;
  @ViewChild('openRollbackConfirmPopup')
  openRollbackConfirmPopup: TemplateRef<any>;
  emptyIcon = emptyIcon;

  constructor(
    private activatedRoute: ActivatedRoute,
    private resourceHistoryService: ResourceHistoryService,
    private router: Router,
    private dialogService: NbDialogService,
    private versioningControllerService: UiVersioningControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {
    this.activatedRoute.queryParams.subscribe((params) => {
      if (params && params['from']) {
        this.fromQueryParam = params['from'];
      }
    });
  }

  OnPageLoad(params: Params): void {
    this.blueprintName = params?.stackName;
    this.clusterId = params?.clusterId;
    this.resourceType = params?.resourceType;
    this.resourceName = params?.resourceName;

    this.getResourceHistory();
  }

  async getResourceHistory() {
    this.loadingResourceHistory = true;
    this.errorLoadingResourceHistory = '';
    try {
      const apiResponse =
        await this.resourceHistoryService.getResourceHistoryDataToRender({
          requestParams: {
            clusterId: this.clusterId,
            resourceType: this.resourceType,
            resourceName: this.resourceName,
            blueprintName: this.blueprintName,
          },
          includeBlueprintHistory: false,
        });
      this.setRenderData(apiResponse);
      this.loadingResourceHistory = false;
    } catch (error) {
      this.errorLoadingResourceHistory =
        error?.message || error?.error?.message;
      this.loadingResourceHistory = false;
    }
  }

  setRenderData(apiResponse) {
    const { overrideProps } = cloneDeep(apiResponse || {});
    this.currentVersionTableSettings = cloneDeep(overrideProps?.tableSettings);
    this.previousVersionTableSettings = cloneDeep(overrideProps?.tableSettings);
    this.currentVersionTableSettings.columns.actions =
      this.getActionColumn('current');

    this.allOverrideHistory = cloneDeep(overrideProps?.historyData?.versions);
    this.currentVersion = overrideProps?.historyData?.versions?.slice(0, 1);
    this.previousVersions = overrideProps?.historyData?.versions?.slice(1);

    this.previousVersionTableSettings.columns['actions'] =
      this.getActionColumn('previous');
    this.loadingResourceHistory = false;
  }

  getActionColumn(tableType = 'current') {
    return {
      title: 'Actions',
      filter: false,
      type: 'custom',
      valuePrepareFunction: (cell, row) => {
        let actionButtons =
          tableType === 'current'
            ? cloneDeep(this.currentVersionActions)
            : cloneDeep(this.previousVersionsAction);

        if (this.allOverrideHistory?.length - 1 === row.index) {
          actionButtons = actionButtons.filter(
            (action) => action?.id !== 'compare-last',
          );
        }
        const displayVal = {
          smartTableCustomActionButtons: actionButtons,
        };
        return JSON.stringify(displayVal);
      },
      renderComponent: CustomActionsComponent,
      onComponentInitFunction: (instance) => {
        instance.byCustomActionButtonClick.subscribe((data) => {
          let customActionButton = this.previousVersionsAction.find(
            (actionButton) => {
              return actionButton.id === data.actionButton.id;
            },
          );
          customActionButton?.onClick(data.rowData);
        });
      },
    };
  }

  openOverrideDiff = (rowData, compareType) => {
    const clickedVersionIndex = this.allOverrideHistory.findIndex(
      (version) => version.versionId === rowData?.versionId,
    );
    let compareVersionIndex = clickedVersionIndex + 1;
    let originalVersion;
    let modifiedVersion;
    if (compareType === 'compare-current') {
      compareVersionIndex = 0;
      originalVersion = this.allOverrideHistory[clickedVersionIndex];
      modifiedVersion = this.allOverrideHistory[compareVersionIndex];
    } else {
      originalVersion = this.allOverrideHistory[compareVersionIndex];
      modifiedVersion = this.allOverrideHistory[clickedVersionIndex];
    }

    this.dialogService.open(this.openDiffEditor, {
      context: {
        modifiedJson: JSON.stringify(
          modifiedVersion?.entity?.overrides,
          null,
          '\t',
        ),
        originalJson: JSON.stringify(
          originalVersion?.entity?.overrides,
          null,
          '\t',
        ),
        originalVersionId: originalVersion?.version,
        modifiedVersionId: modifiedVersion?.version,
      },
    });
  };
  openConfirmationPopup = (rowData) => {
    this.dialogService.open(this.openRollbackConfirmPopup, {
      context: {
        version: rowData?.version,
        versionId: rowData?.versionId,
      },
    });
  };
  versionRollback(versionDetails, ref) {
    this.versionRollbackInProgress = true;
    this.versioningControllerService
      .restoreUsingPOST(versionDetails?.versionId)
      .subscribe(
        (data) => {
          this.versionRollbackInProgress = false;
          this.customToastService.success(
            `Configuration Rollback was successful`,
          );
          ref.close();
          this.getResourceHistory();
        },
        (error) => {
          this.versionRollbackInProgress = false;
          ref.close();
          this.onAPIErrorService.onAPIError(
            error?.error?.message || `Configuration Rollback failed`,
            error,
          );
        },
      );
  }
  handleBack = () => {
    this.router.navigate([
      '/',
      this.blueprintName,
      'cluster',
      this.clusterId,
      'resources',
      'type',
      this.resourceType,
      'resource',
      this.resourceName,
      this.fromQueryParam || 'configuration',
    ]);
  };
}
