import {
  Component,
  Input,
  Output,
  EventEmitter,
  SimpleChanges,
  OnChanges,
  OnDestroy,
} from '@angular/core';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params } from '@angular/router';
import { FormControl, FormGroup, FormGroupDirective } from '@angular/forms';
import { GroupedOptions } from './grouped-options.model';

@Component({
  selector: 'app-multi-select-search-v2',
  templateUrl: './multi-select-search-v2.component.html',
  styleUrls: ['./multi-select-search-v2.component.scss'],
  providers: [FormGroupDirective],
})
export class MultiSelectSearchV2Component implements OnChanges {
  //events
  @Output() selectedOptionEvent = new EventEmitter<any>(); //event emitter for selected option
  //@Output() removeOptionEvent = new EventEmitter<string>(); //event emitter for removed option

  //inputs
  @Input() options: GroupedOptions[] = [];
  @Input() noOptionsMessage: string = 'No options available';
  optionsOriginal: GroupedOptions[] = [];
  @Input() searchInputId: string = '';
  @Input() selectInputPlaceholder: string = 'Select options';
  @Input() searchInputPlaceholder: string = 'Search...';
  @Input() selectInputLabel: string = '';
  @Input() showRequiredIcon: boolean = false;
  @Input() showClearAllOptionsIcon: boolean = true;
  @Input() tooltipText: string = '';
  @Input() showLimitedTags: boolean = false;
  @Input() tagLimit: number = 3;
  @Input() showSelectionsOutside: boolean = true;
  @Input() inputColSize: number = 5;
  @Input() labelColSize: number = 3;
  @Input() showSpinner: boolean = false;
  @Input() selectionHeight: string = '15vh'; ///check percentage height instead of vh
  @Input() showRemovableOptionTags: boolean = true;
  @Input() showCombinedText: boolean = true;
  @Input() disableSelectInput: boolean = false;
  @Input() selectionTextLabel: string = '';
  @Input() showSelectAll: boolean = false;
  @Input() showParentOptionAsCheckbox: boolean = true;
  @Input() disabledInputTooltip: string = '';
  @Input() showDisabledInputTooltip: boolean = false;
  @Input() showNoOptionsMessage: boolean = true; //Set this boolean to display "No options available" as a placeholder text when the dropdown options array is empty, overriding the value of the selectInputPlaceholder variable.

  //local variables
  openOptionsPopover: boolean = false;
  selectedOptionsList: {
    label: string;
    value: string;
    children: { label: string; value: string }[];
    disabled?: boolean;
  }[] = [];
  searchString: FormControl = new FormControl('');
  selectAllControl: FormControl = new FormControl(false);
  tagOptionsList: { label: string; value: string }[] = [];

  constructor(private activatedRoute: ActivatedRoute) {}

  ngOnChanges(changes: SimpleChanges) {
    //show spinner for dropdown before data loads
    this.showSpinner = changes?.showSpinner?.currentValue || false;
    /*
     * set options data for comparision purpose
     */
    this.optionsOriginal = [...this.options];
    if (changes?.options) {
      this.setSelectedOptionValuesInitial(changes?.options?.currentValue);
    }
  }

  setSelectedOptionValuesInitial(optionsValue) {
    this.selectedOptionsList = [];
    optionsValue?.forEach((option) => {
      if (option?.checked) {
        option?.children.forEach((childOption) => (childOption.checked = true));
      }
      if (
        option?.children?.every((childOption) => childOption?.checked) &&
        !option?.checked
      ) {
        option.checked = true;
      }
    });
    this.constructSelectedOptionsList(optionsValue);
    this.searchString.setValue('');
  }

  constructSelectedOptionsList(options) {
    this.tagOptionsList = [];
    this.selectedOptionsList = [];

    options?.forEach((parent) => {
      parent?.children?.forEach((child) => {
        if (child?.checked) {
          const parentValue = parent?.value;
          const childValue = child?.value;

          // Add the parent to the selected options list if it is not already added
          let selectedParent = this.selectedOptionsList?.find(
            (option) => option?.value === parentValue,
          );
          if (!selectedParent) {
            selectedParent = {
              label: parent?.label,
              value: parentValue,
              children: [],
            };
            this.selectedOptionsList.push(selectedParent);
          }

          // Add the child to the selected parent's children array
          selectedParent?.children?.push({
            label: child?.label,
            value: childValue,
          });

          // Add the child to the tag options list if it is not already added
          const tagOptionExists = this.tagOptionsList?.some(
            (option) =>
              option?.label === parentValue && option?.value === childValue,
          );
          if (!tagOptionExists) {
            this.tagOptionsList.push({ label: parentValue, value: childValue });
          }
        }
      });
    });
    this.setSelectAllCheckboxStatus();
  }

  /*
   * Check parent option if all child are checked
   * construct selected options list to send with the event
   * send the event
   */
  changeChildOptionValue(parentOption) {
    parentOption.checked = parentOption?.children?.every(
      (item) => item?.checked,
    );
    this.constructSelectedOptionsList(this.optionsOriginal);
    this.emitSelectedEvents();
  }

  /*
    Filter function
  */
  getFilteredOptions() {
    if (this.searchString.value?.length > 0) {
      this.options = [
        ...this.searchOptions(
          this.optionsOriginal,
          'label',
          this.searchString.value,
        ),
      ];
    } else {
      this.options = [...this.optionsOriginal];
    }
    this.checkAllChildChecked();
  }

  checkAllChildChecked() {
    this.options.forEach((option) => {
      option.checked = option?.children?.every((item) => item?.checked);
    });
  }

  searchOptions(optionsList, key, searchStr) {
    return optionsList?.reduce((result, parent) => {
      if (parent?.[key]?.includes(searchStr)) {
        result.push({ ...parent, children: parent?.children || [] });
      } else {
        const matchingChildren = (parent?.children || []).filter((child) =>
          child[key]?.includes(searchStr),
        );
        if (matchingChildren?.length) {
          result.push({ ...parent, children: matchingChildren });
        }
      }
      return result;
    }, []);
  }

  /*
   * Remove clicked option when clicked on cross icon on visual results
   * Selected values are either shown inside input or outside of it
   * Clicking on close button for each result will result in calling this function and option getting removed.
   */
  removeClicked(filterOption: { label: string; value: string }) {
    this.options.forEach((parentOption) => {
      if (parentOption.value === filterOption.label) {
        const childOptionToUpdate = parentOption?.children?.find(
          (child) => child?.value === filterOption?.value,
        );
        if (childOptionToUpdate) childOptionToUpdate.checked = false;
        parentOption.checked = parentOption?.children?.every(
          (item) => item.checked,
        );
      }
    });
    this.constructSelectedOptionsList(this.optionsOriginal);
    this.emitSelectedEvents();
  }

  /*
   * Clear all selected options when user clicks on close button on top level select input
   */
  clearAllSelectedOptions() {
    this.selectedOptionsList = [];
    this.optionsOriginal.forEach((parentOption) => {
      parentOption.checked = false;
      parentOption?.children?.forEach((childOption) => {
        childOption.checked = false;
      });
    });
    this.options = [...this.optionsOriginal];
    this.emitSelectedEvents();
  }

  emitSelectedEvents() {
    this.selectedOptionEvent.emit({
      selectedOptionsList: this.selectedOptionsList,
    });
  }

  /*
   * Keeps track of parent options that are open or close when user clicks on dropdown or up arrow
   */
  openChildOptions(event, option) {
    event.stopPropagation();
    option.expand = !option?.expand;
  }

  onChangeParentChecked(event, parentOption) {
    if (!parentOption.checked) {
      parentOption?.children?.forEach((child) => {
        child.checked = false;
      });
    } else {
      parentOption?.children?.forEach((child) => {
        child.checked =
          !child.hasOwnProperty('disabled') || child.disabled !== true;
      });
    }
    this.constructSelectedOptionsList(this.optionsOriginal);
    this.emitSelectedEvents();
  }

  setSelectAllCheckboxStatus() {
    this.selectAllControl.setValue(
      this.options.every((option) => option.checked),
    );
  }

  showStateChange(event) {
    this.openOptionsPopover = event.isShown;
    if (!event.isShown) {
      this.searchString.setValue('');
      this.options = [...this.optionsOriginal];
      this.checkAllChildChecked();
      this.setSelectAllCheckboxStatus();
    } else {
      this.openOptionsPopover = true;
      let selectBoundingRect = document
        .querySelector('.input-parent')
        .getBoundingClientRect();
      let popoverElem = document.querySelector('nb-popover');
      let dropdownElement = popoverElem?.parentElement;
      if (dropdownElement) {
        dropdownElement.style.width = selectBoundingRect.width + 'px';
        dropdownElement.style.left = selectBoundingRect.left + 'px';
      }
    }
  }

  onSelectAllClick(event) {
    this.selectedOptionsList = [];
    this.options?.forEach((option) => {
      option.checked = event;
      option?.children?.forEach((childOption) => (childOption.checked = event));
    });
    this.constructSelectedOptionsList(this.options);
    this.emitSelectedEvents();
    this.searchString.setValue('');
  }
}
