import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { ActivatedRoute, Router } from '@angular/router';
import { UiNoAuthUserControllerService } from 'src/app/cc-api/services';
import { FormBuilder, FormGroup, NgForm, Validators } from '@angular/forms';
import { ApplicationControllerService } from 'src/app/cc-api/services';
import { SimpleOauth2User } from '../../../cc-api/models/simple-oauth-2user';
import {
  SavePasswordResetPasswordRequest,
  TokenValidationRequest,
} from 'src/app/cc-api/models';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-reset-password',
  templateUrl: './reset-password.component.html',
  styleUrls: ['./reset-password.component.scss'],
})
export class ResetPasswordComponent implements OnInit {
  errorMsg: string = '';
  logo: any;
  isCustomIntegrationPresent: boolean = false;
  redirectTo: string = '';
  token: string = '';
  resetPasswordInProgress: boolean = false;
  showPassword: boolean = false;
  showConfirmPassword: boolean = false;
  invalidTokenMessage: boolean = false;
  resetPwdForm: FormGroup;
  loading: boolean = true;
  loginInProgress: boolean = false;
  tokenInvalidMsg: string = 'Provided link is expired';
  email: string = '';

  constructor(
    private http: HttpClient,
    private route: ActivatedRoute,
    private router: Router,
    private readonly fb: FormBuilder,
    private uiNoAuthService: UiNoAuthUserControllerService,
    private applicationControllerService: ApplicationControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {}

  ngOnInit(): void {
    this.loading = true;
    this.resetPwdForm = this.fb.group(
      {
        password: [
          '',
          [
            Validators.required,
            Validators.minLength(8),
            Validators.maxLength(20),
            Validators.pattern(
              '^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.*[!@#$%^&*])([a-zA-Z0-9!@#$%^&*]+)',
            ),
          ],
        ],
        confirmPassword: ['', Validators.required],
      },
      {
        validator: this.confirmedValidator('password', 'confirmPassword'),
      },
    );
    //If the user is already logged-in then redirect to home page
    this.applicationControllerService.meUsingGET().subscribe(
      (x: SimpleOauth2User) => {
        this.router.navigate(['/home']);
      },
      (error) => {},
    );
    this.route.queryParams.subscribe((params) => {
      this.token = params?.token || '';
      this.invalidTokenMessage = this.token?.length > 0 ? true : false;
      if (this.token?.length > 0) {
        this.verifyToken();
      } else {
        this.loading = false;
        this.invalidTokenMessage = true;
        this.errorMsg = this.tokenInvalidMsg;
      }
    });
    this.http
      .get('/public/v1/logo', { responseType: 'text' })
      .subscribe((x) => {
        this.logo = x;
      });
  }

  confirmedValidator(controlName: string, matchingControlName: string) {
    return (formGroup: FormGroup) => {
      const control = formGroup?.controls[controlName];
      const matchingControl = formGroup?.controls[matchingControlName];
      if (
        matchingControl?.errors &&
        !matchingControl?.errors?.confirmedValidator
      ) {
        return;
      }
      if (control?.value !== matchingControl?.value) {
        matchingControl.setErrors({ confirmedValidator: true });
      } else {
        matchingControl.setErrors(null);
      }
    };
  }

  verifyToken() {
    this.loading = true;
    this.errorMsg = '';
    let params: TokenValidationRequest = {
      token: this.token,
    };
    this.uiNoAuthService.validateTokenUsingPOSTResponse(params).subscribe(
      (response) => {
        this.loading = false;
        this.invalidTokenMessage = !response?.body?.valid;
        this.errorMsg = !response?.body?.valid ? this.tokenInvalidMsg : '';
        this.email = response?.body?.username;
      },
      (error) => {
        this.loading = false;
        this.invalidTokenMessage = true;
        this.errorMsg =
          error?.error?.message || 'Error while validating token.';
      },
    );
  }

  get password() {
    return this.resetPwdForm.get('password');
  }

  get confirmPassword() {
    return this.resetPwdForm.get('confirmPassword');
  }

  formResetPwd() {
    this.errorMsg = '';
    this.resetPasswordInProgress = true;
    let resetPwdParams: SavePasswordResetPasswordRequest = {
      email: this.email,
      password: this.password.value,
      token: this.token,
    };
    this.uiNoAuthService.savePasswordUsingPOST(resetPwdParams).subscribe(
      (response) => {
        this.resetPasswordInProgress = false;
        this.customToastService.success(
          response?.message || 'Password updated successfully',
        );
        if (this.email?.length > 0 && this.password?.value?.length > 0) {
          this.loginUser();
        } else {
          this.returnToLogin();
        }
      },
      (error) => {
        this.onAPIErrorService.onAPIError(
          error?.error?.message || 'Reset password failed',
          error,
        );
      },
    );
  }

  loginUser() {
    this.loginInProgress = true;
    let data = new FormData();
    data.append('username', this.email);
    data.append('password', this.password.value);

    this.http.post('/perform_login', data, { responseType: 'text' }).subscribe(
      () => {
        //If redirectTo is not null navigate user to redirectTo route on successful login else navigate to home page
        this.loginInProgress = false;
        this.router.navigate(['/home']);
      },
      (err) => {
        this.loginInProgress = false;
        this.returnToLogin();
      },
    );
  }

  returnToLogin() {
    this.router.navigate(['/', 'login']);
  }

  togglePassword() {
    this.showPassword = !this.showPassword;
  }
  toggleConfirmPassword() {
    this.showConfirmPassword = !this.showConfirmPassword;
  }
}
