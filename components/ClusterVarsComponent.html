import { Component, ViewChild } from '@angular/core';
import { LocalDataSource } from 'ng2-smart-table';
import { NbDialogService } from '@nebular/theme';
import { ActivatedRoute, Params } from '@angular/router';
import { AwsCluster } from '../../../cc-api/models/aws-cluster';
import { AbstractCluster } from '../../../cc-api/models/abstract-cluster';
import { UiStackControllerService } from '../../../cc-api/services/ui-stack-controller.service';
import { UiCommonClusterControllerService } from '../../../cc-api/services/ui-common-cluster-controller.service';
import { OverrideStatusBadgeComponent } from 'src/app/components/override-status-badge/override-status-badge.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CanDeactivateComponent } from 'src/app/can-deactivate.guard';
import { SaveChangesFooterComponent } from 'src/app/components/save-changes-footer/save-changes-footer.component';
import { ENVIRONMENT_CONFIGURE } from 'src/app/constants/permissions';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { SmartTableMultiTypeComponent } from 'src/app/components/smart-table-multi-type/smart-table-multi-type.component';
import { SmartTableTextareaEditorComponent } from 'src/app/components/smart-table-textarea-editor/smart-table-textarea-editor.component';
import { SmartTableTextTooltipComponent } from 'src/app/components/smart-table-text-tooltip/smart-table-text-tooltip.component';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

const emptyStateLogo = './assets/secret-empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-cluster-vars',
  templateUrl: './cluster-vars.component.html',
  styleUrls: ['./cluster-vars.component.scss'],
})
export class ClusterVarsComponent
  extends FacetsPage
  implements CanDeactivateComponent
{
  enableSubmitForClusterOverrides = true;
  clusterSource: LocalDataSource = new LocalDataSource();
  secretsChanged = {};
  settings = {
    columns: {
      name: {
        title: 'Variable name',
        filter: true,
        width: '15%',
        editable: false,
      },
      value: {
        title: 'Variable value',
        filter: false,
        width: '20%',
        editable: true,
        editor: {
          type: 'custom',
          component: SmartTableTextareaEditorComponent,
        },
        type: 'custom',
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            maxWidth: '250px',
            truncateText: true,
            label: cell,
            tooltipText: cell,
            allowCopy: true,
          });
        },
        renderComponent: SmartTableTextTooltipComponent,
      },
      description: {
        title: 'Description',
        filter: false,
        width: '15%',
        editable: false,
      },
      type: {
        title: 'Variable type',
        width: '12%',
        editable: false,
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [
              { value: 'Secret', title: 'Secret' },
              { value: 'Variable', title: 'Variable' },
            ],
          },
        },
      },
      global: {
        title: 'Global',
        width: '10%',
        editable: false,
        type: 'custom',
        renderComponent: SmartTableMultiTypeComponent,
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            type: 'icon',
            props: {
              icon: cell ? 'checkmark-outline' : '',
            },
          });
        },
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [
              { value: true, title: 'True' },
              { value: false, title: 'False' },
            ],
          },
        },
      },
      status: {
        title: 'Override Status',
        width: '15%',
        editable: false,
        type: 'custom',
        renderComponent: OverrideStatusBadgeComponent,
        filter: {
          type: 'list',
          config: {
            selectText: 'Status',
            list: [
              { value: 'DEFAULT', title: 'No change' },
              { value: 'OVERRIDDEN', title: 'Overridden' },
            ],
          },
        },
        valuePrepareFunction(value, row) {
          if (value === 'DEFAULT') {
            return 'NO CHANGE';
          }
          return value;
        },
      },
    },
    noDataMessage: '',
    pager: {
      display: true,
      perPage: 10,
    },
    actions: {
      add: false,
      edit: true,
      delete: false,
      position: 'right',
    },
    edit: {
      inputClass: '',
      editButtonContent:
        '<img class="custom-table-action-icon" src="assets/edit-2-outline.svg"/>',
      saveButtonContent:
        '<img class="custom-table-action-icon" src="assets/checkmark-outline-success.svg"/>',
      cancelButtonContent:
        '<img class="custom-table-action-icon" src="assets/close-outline-danger.svg"/>',
      confirmSave: true,
    },
  };
  addOverrideSpinner: boolean = false;
  cluster: AwsCluster;
  extraEnvVars = ['TZ', 'CLUSTER', 'AWS_REGION'];
  private stackName: any;
  originalData: any = {};
  objectKeys = Object.keys;
  varsChanged: any = {};
  emptyStateLogo = emptyStateLogo;
  errorLogo = errorIcon;
  primaryMsg: string = 'No secrets or variables found';
  secondaryMsg: string =
    'There are no secrets or variables registered with envioronment';
  showEmptyState: boolean = false;
  status: 'info' | 'error' = 'info';
  private stackVars: any;
  clusterId: any;
  reviewComments: string = '';
  fetchingData = false;
  actionButtons = [
    {
      id: 'sync-git',
      title: 'Sync with Git',
      buttonType: 'secondary',
      status: 'primary',
      nbTooltip:
        'Sync to view the newly added secrets and variables in the blueprint',
      icon: 'sync-outline',
      loading: false,
      hidden: false,
      onClick: (id) => {
        this.refreshStack(id);
      },
    },
  ];
  environmentConfigurePermission = ENVIRONMENT_CONFIGURE;

  @ViewChild(SaveChangesFooterComponent, { static: false })
  private footerComponent: SaveChangesFooterComponent;
  constructor(
    route: ActivatedRoute,
    private dialogService: NbDialogService,
    private uiStackControllerService: UiStackControllerService,
    private uiCommonClusterControllerService: UiCommonClusterControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(route);
  }

  OnPageLoad(p: Params): void {
    this.showEmptyState = false;
    if (p?.clusterId && p?.stackName) {
      this.stackName = p.stackName;
      this.clusterId = p.clusterId;
      this.getSecrets();
    }
  }

  getSecrets() {
    this.fetchingData = true;
    this.uiStackControllerService.getStackUsingGET(this.stackName).subscribe(
      (s) => {
        this.stackVars = s.stackVars;
        this.uiCommonClusterControllerService
          .getVarsUsingGET(this.clusterId)
          .subscribe(
            (t) => {
              this.cluster = t;
              this.loadDataInTables(this.cluster, this.stackVars);
              this.fetchingData = false;
            },
            (error) => {
              this.clusterSource.load([]);
              this.fetchingData = false;
              this.showEmptyState = true;
              this.secondaryMsg =
                error?.error?.message ||
                ' There is some error fetching Secrets and Variables';
              this.primaryMsg = '';
              this.emptyStateLogo = this.errorLogo;
              this.status = 'error';
            },
          );
      },
      (err) => (this.fetchingData = false),
    );
  }

  validateSecrets(event) {
    /*
      Check if event data's new value is not white space, then only process the value as valid
    */
    if (whiteSpaceInputPattern.test(event?.newData?.value)) {
      if (event?.newData?.type === 'Secret') {
        if (event?.newData?.value != this.originalData[event?.newData?.name]) {
          this.secretsChanged[event?.newData?.name] = event?.newData?.value;
        } else {
          delete this.secretsChanged[event?.newData?.name];
        }
      } else if (event?.newData?.type === 'Variable') {
        if (event?.newData?.value != this.originalData[event?.newData?.name]) {
          this.varsChanged[event?.newData?.name] = event?.newData?.value;
        } else {
          delete this.varsChanged[event?.newData?.name];
        }
      }
      event.confirm.resolve(event?.newData);
    } else {
      event.confirm.reject();
    }
  }

  /**
   * Load data in table datasources.
   * @param cluster
   */
  loadDataInTables(cluster: AbstractCluster, stackVars) {
    let dataSource = [];
    Object.entries(cluster.variables).forEach(([key, value]) => {
      if (value?.['secret']) {
        dataSource.push({
          name: key,
          status: value?.['status'],
          type: 'Secret',
          value: '*',
          description: value?.['description'] || '',
          global: value?.['global'],
        });
        this.originalData[key] = value?.['value'];
      } else if (
        !value?.['secret'] &&
        !(this.extraEnvVars.includes(key) || key in stackVars)
      ) {
        dataSource.push({
          name: key,
          status: value?.['status'],
          type: 'Variable',
          value: value?.['value'],
          description: '',
          global: value?.['global'],
        });
        this.originalData[key] = value?.['value'];
      }
    });
    this.showEmptyState = dataSource.length <= 0;
    this.clusterSource.load(dataSource);
  }

  submitClusterOverrides = async (ref: any) => {
    this.addOverrideSpinner = true;
    this.uiCommonClusterControllerService
      .upsertVarsUsingPOST1({
        clusterId: this.cluster.id,
        clusterVars: { ...this.secretsChanged, ...this.varsChanged },
      })
      .subscribe(
        (c) => {
          this.loadDataInTables(c, this.stackVars);
          this.cluster = c;
          this.addOverrideSpinner = false;
          this.secretsChanged = {};
          this.varsChanged = {};
          this.customToastService.success('Updated environment variables');
          ref.close();
        },
        (error) => {
          this.addOverrideSpinner = false;
          this.onAPIErrorService.onAPIError(
            'Update environment variables Failed: ' + error?.error?.message,
            error,
          );
          ref.close();
        },
      );
  };

  openConfirmation(confirmation, ref = null) {
    if (ref) {
      ref.close('confirm');
    }
    this.dialogService.open(confirmation, {});
  }

  openConfirmRemove(confirmRemove) {
    this.dialogService.open(confirmRemove, {});
  }

  clearValue($event: any) {
    console.log($event);
  }

  cleanSecVar = (ref?) => {
    ref?.close('discard');
    this.secretsChanged = {};
    this.varsChanged = {};
    this.clusterSource.load([]);
    this.loadDataInTables(this.cluster, this.stackVars);
  };

  async canDeactivate(): Promise<boolean> {
    if (
      this.objectKeys(this.secretsChanged)?.length == 0 &&
      this.objectKeys(this.varsChanged)?.length == 0
    ) {
      return true;
    }

    return this.footerComponent.onDeactivate();
  }

  handlePageActions(event) {
    try {
      this.actionButtons.find((button) => button?.id === event)?.onClick(event);
    } catch (error) {
      console.log('Error while handling button click', error);
    }
  }

  refreshStack(id) {
    let gitSyncButton = this.actionButtons.find((button) => button?.id === id);
    gitSyncButton.loading = true;
    this.uiStackControllerService
      .reloadStackUsingGET1(this.stackName)
      .subscribe(
        (stack) => {
          gitSyncButton.loading = false;
          this.customToastService.success('Synced with git');
          this.cleanSecVar();
          this.getSecrets();
        },
        (error) => {
          gitSyncButton.loading = false;
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Syncing with git failed',
            error,
          );
        },
      );
  }
}
