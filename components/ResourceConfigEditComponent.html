import { Component, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { View } from '@antv/x6';
import { CanDeactivateComponent } from 'src/app/can-deactivate.guard';
import { whiteListedResources } from 'src/app/constants/resourceTypes';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { AppConfigEditComponent } from './components/app-config-edit/app-config-edit.component';
import { SimpleResourceConfigEditComponent } from './components/simple-resource-config-edit/simple-resource-config-edit.component';
@Component({
  selector: 'app-resource-config-edit',
  templateUrl: './resource-config-edit.component.html',
  styleUrls: ['./resource-config-edit.component.scss'],
})
export class ResourceConfigEditComponent
  extends FacetsPage
  implements CanDeactivateComponent
{
  stackName: any;
  clusterId: any;
  resourceType: any;
  appName: any;
  jsonMode: boolean = false;
  jsonPageActions = [
    {
      id: 'editor',
      title: 'Form Mode',
      buttonType: 'primary',
      nbTooltip: 'Switch to Form editor mode',
    },
  ];

  formPageActions = [
    {
      id: 'editor',
      title: 'JSON Mode',
      buttonType: 'primary',
      nbTooltip: 'Switch to JSON editor mode',
    },
  ];

  whiteListedResources = whiteListedResources;
  showAppConfig = false;

  @ViewChild(AppConfigEditComponent)
  private appConfigEditComponent: AppConfigEditComponent;
  @ViewChild(SimpleResourceConfigEditComponent)
  private simpleteResourceConfigEditComponent: SimpleResourceConfigEditComponent;

  constructor(_route: ActivatedRoute, private router: Router) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p.clusterId;
    this.resourceType = p.resourceType;
    this.appName = p.resourceName;
    this.stackName = p.stackName;
    this.showAppConfig = [
      'application',
      'cronjob',
      'jobs',
      'statefulsets',
    ].includes(this.resourceType);
  }

  handleBack = () => {
    this.router.navigate([
      '/',
      this.stackName,
      'cluster',
      this.clusterId,
      'resources',
      'type',
      this.resourceType,
      'resource',
      this.appName,
      'configuration',
    ]);
  };

  handlePageActions(event) {
    switch (event) {
      case 'editor':
        this.jsonMode = !this.jsonMode;
        break;
      default:
        break;
    }
  }
  async canDeactivate(): Promise<boolean> {
    if (this.appConfigEditComponent) {
      return this.appConfigEditComponent.canDeactivate();
    }
    if (this.simpleteResourceConfigEditComponent) {
      return this.simpleteResourceConfigEditComponent.canDeactivate();
    }
    return true;
  }
}

<div class="mb-2 row">
  <div class="col-sm-12">
    <app-page-header
      [onBack]="handleBack"
      heading="Override Configuration Variables"
      [subheading]="appName"
      [actionButtons]="
        showAppConfig ? (jsonMode ? jsonPageActions : formPageActions) : []
      "
      (click)="handlePageActions($event)"
    ></app-page-header>
  </div>
</div>
<ng-container *ngIf="!jsonMode && showAppConfig">

</ng-container>

</ng-container>
