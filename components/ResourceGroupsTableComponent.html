import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { UiResourceGroupControllerService } from 'src/app/cc-api/services';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import {
  RESOURCE_GROUP_DELETE,
  RESOURCE_GROUP_WRITE,
} from 'src/app/constants/permissions';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { SmartTableRouterlinkComponent } from 'src/app/components/smart-table-routerlink/smart-table-routerlink.component';
import { NbDialogService } from '@nebular/theme';
import { SmartTableCellWithEllipsisComponent } from './../../../../components/smart-table-cell-with-ellipsis/smart-table-cell-with-ellipsis.component';
import { SmartTableTimeagoTooltipComponent } from '../../../../components/smart-table-timeago-tooltip/smart-table-timeago-tooltip.component';
import { ResourceGroup } from '../../../../cc-api/models/resource-group';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const noAccessIcon = './assets/alert-empty.svg';
const errorStateIcon = './assets/error.svg';
const emptyStateIcon = './assets/empty_resource_groups.svg';

@Component({
  selector: 'app-resource-groups-table',
  templateUrl: './resource-groups-table.component.html',
  styleUrls: ['./resource-groups-table.component.scss'],
})
export class ResourceGroupsTableComponent extends FacetsPage {
  noAccessIcon = noAccessIcon;
  noDataIcon = emptyStateIcon;
  errorIcon = errorStateIcon;
  botIcon = './assets/bot.svg';

  resourceGroupWritePermission = RESOURCE_GROUP_WRITE;

  @ViewChild('deleteResourceGroupTemplate', { static: true })
  private deleteResourceGroupTemplate: TemplateRef<any>;

  resourceToBeDeleted: string = '';
  deletingResGrp: boolean = false;

  fetchingResourceGroups: boolean = false;
  allResourceGroups: Array<ResourceGroup>;
  fetchingError: boolean = false;

  userSubscription: any;

  smartTableCustomActionButtons: any = [
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete resource group',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: RESOURCE_GROUP_DELETE,
      onClick: (data: any) => {
        this.promptDeleteResourceGroup(data);
      },
    },
  ];

  resourceGroupTableConfig: any = {
    columns: {
      name: {
        title: 'Name',
        filter: true,
        width: '20%',
        valuePrepareFunction: (value) => {
          const displayVal = {
            url: ['/', 'settings', 'resource-groups', value],
            display: value,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: SmartTableRouterlinkComponent,
        type: 'custom',
      },
      resources: {
        title: 'Resources',
        filter: true,
        width: '40%',
        type: 'custom',
        valuePrepareFunction: (cell) => {
          return JSON.stringify({
            data: cell,
            decorate: true,
            noDataMessage: 'N/A',
          });
        },
        renderComponent: SmartTableCellWithEllipsisComponent,
      },
      lastModifiedBy: {
        title: 'Modified By',
        width: '20%',
        filter: true,
        type: 'custom',
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            user: row.lastModifiedBy,
            label: row.lastModifiedDate,
          });
        },
        renderComponent: SmartTableTimeagoTooltipComponent,
      },
      custom: {
        type: 'custom',
        width: '10%',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          const cols = {
            smartTableCustomActionButtons: this.smartTableCustomActionButtons,
          };
          return JSON.stringify(cols);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        filter: false,
      },
    },
    pager: {
      display: false,
      perPage: 10,
    },
    mode: 'external',
    actions: false,
  };

  constructor(
    private uiResourceGroupController: UiResourceGroupControllerService,
    private router: Router,
    private _route: ActivatedRoute,
    private dialogService: NbDialogService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.getAllResourceGroups();
  }

  getAllResourceGroups() {
    this.fetchingResourceGroups = true;
    this.fetchingError = false;
    this.uiResourceGroupController
      .findAllUsingGET()
      .subscribe(
        (resourceGrpInfo) => {
          this.allResourceGroups = resourceGrpInfo;
          this.allResourceGroups.forEach((resourceGrp: any) => {
            let allResources = [];
            resourceGrp.resourceInfo?.forEach((stack: any) => {
              let resources = [];
              stack.resources?.forEach((resourceType: any) => {
                for (let i = 0; i < resourceType?.resourceNames?.length; i++) {
                  resources.push(resourceType.resourceNames[i]);
                }
              });
              allResources.push(...resources);
            });
            resourceGrp['resources'] = allResources;
          });
        },
        (error) => {
          this.fetchingError = true;
        },
      )
      .add(() => {
        this.fetchingResourceGroups = false;
      });
  }

  promptDeleteResourceGroup(rowData: any) {
    this.dialogService.open(this.deleteResourceGroupTemplate, {
      context: { resourceGroup: rowData },
    });
  }

  deleteResourceGroup(data, ref) {
    this.deletingResGrp = true;
    this.uiResourceGroupController
      .deleteUsingDELETE1(data.id)
      .subscribe(
        (rsp: any) => {
          this.customToastService.success(
            `Successfully deleted resource group: ${data.name}`,
          );
          this.resourceToBeDeleted = '';
          this.getAllResourceGroups();
          ref.close();
        },
        (error: any) => {
          this.onAPIErrorService.onAPIError(
            `Could not delete resource group: ${data.name}`,
            error,
          );
        },
      )
      .add(() => {
        this.deletingResGrp = false;
      });
  }

  navigateToCreateResourceGroup() {
    this.router.navigate(['../../create-resource-group'], {
      relativeTo: this._route,
    });
  }
}

<ng-template #deleteResourceGroupTemplate let-data let-ref="dialogRef">
  <nb-card>
    <nb-card-header>
      <ng-container> Delete {{ data?.resourceGroup?.name }} </ng-container>
      <nb-icon
        icon="close-outline"
        class="float-right cursor-pointer"
        (click)="ref.close()"
      ></nb-icon>
    </nb-card-header>

    <nb-card-body>
      <div>
        <p>
          This action is going to delete the resource group
          <strong>
            {{ data?.resourceGroup?.name }}
          </strong>
          permanently.
        </p>
        <p>
          Please type
          <strong>
            {{ data?.resourceGroup?.name }}
          </strong>
          to confirm your action.
        </p>
        <input
          nbInput
          fullWidth
          fieldSize="small"
          type="text"
          placeholder="Enter the resource group name to be deleted"
          [(ngModel)]="resourceToBeDeleted"
        />
      </div>
    </nb-card-body>
    <nb-card-footer>
      <button
        nbButton
        status="danger"
        size="small"
        (click)="deleteResourceGroup(data?.resourceGroup, ref)"
        class="cc-margin-right"
        [disabled]="
          resourceToBeDeleted?.toLowerCase() !==
            data?.resourceGroup?.name?.toLowerCase() || deletingResGrp
        "
        [nbSpinner]="deletingResGrp"
      >
        Delete
      </button>
      <button
        nbButton
        size="small"
        (click)="ref.close(); resourceToBeDeleted = ''"
        [disabled]="deletingResGrp"
      >
        Cancel
      </button>
    </nb-card-footer>
  </nb-card>
</ng-template>
