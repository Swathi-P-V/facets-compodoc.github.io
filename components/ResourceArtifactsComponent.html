import { Component, Input, OnInit } from '@angular/core';
import { ResourceDetailsService } from '../resource-details/resource-details.service';
import {
  UiArtifactCiControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
const emptyStateLogo = './assets/artifact-empty.svg';
const errorLogo = './assets/error.svg';

@Component({
  selector: 'app-resource-artifacts',
  templateUrl: './resource-artifacts.component.html',
  styleUrls: ['./resource-artifacts.component.scss'],
})
export class ResourceArtifactsComponent implements OnInit {
  @Input() blueprint;
  ciName: any;

  allArtifacts: any[];

  errorStateIcon = errorLogo;
  emptyStateIcon = emptyStateLogo;

  dataLoaded: boolean = false;

  ciID: string;

  errorWhileFetchingArtifacts: boolean = false;
  errorMessage: string;
  fetchingArtifacts: boolean = true;

  constructor(
    private resourceDetailsService: ResourceDetailsService,
    private uiStackControllerService: UiStackControllerService,
    private uiArtifactCIControllerService: UiArtifactCiControllerService,
  ) {}

  async ngOnInit() {
    this.ciID = '';
    this.ciName = this.resourceDetailsService.getCIName();
    await this.getCiID(this.ciName);
  }

  fetchingCiID: boolean = true;
  errorWhileFetchingCiID: boolean = false;
  async getCiID(reqCiName: string) {
    return new Promise((resolve) => {
      this.fetchingCiID = true;
      this.errorWhileFetchingArtifacts = false;
      this.uiArtifactCIControllerService
        .getArtifactCisByStackUsingGET(this.blueprint)
        .subscribe(
          (artifactCIs) => {
            this.ciID = artifactCIs?.find(
              (artifactCI) => artifactCI.ciName === reqCiName,
            )?.id;
          },
          (error) => {
            this.errorWhileFetchingCiID = true;
            this.errorMessage =
              error?.error?.message || 'There was some error fetching CI Id';
          },
        )
        .add(() => {
          this.fetchingCiID = false;
        });
    });
  }

  setDataLoaded() {
    this.dataLoaded = true;
  }
}
