import { Component, ViewChild } from '@angular/core';
import { Router, ActivatedRoute, Params } from '@angular/router';
import { UiKubernetesExplorerControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ContainerDTO } from 'src/app/cc-api/models';
import { camelToPascal } from 'src/app/utils/string-util';
import { secondsToDuration } from 'src/app/utils/calendar-utils';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { NbDialogService } from '@nebular/theme';
import { HttpClient } from '@angular/common/http';

const emptyIcon = './assets/pods-empty.svg';
const errorIcon = './assets/error.svg';

const tableColumnComponentMap = {
  actions: CustomActionsComponent,
};

const nameToTitleMapping = {
  ageInSeconds: 'Age',
};

@Component({
  selector: 'app-resource-pod-containers',
  templateUrl: './resource-pod-containers.component.html',
  styleUrls: ['./resource-pod-containers.component.scss'],
})
export class ResourcePodContainersComponent extends FacetsPage {
  @ViewChild('actionsModal') actionsModal;
  tailLogsParams = {
    tail: 100,
    podName: '',
    clusterId: '',
  };
  logsUrl: string = '';
  stackName: string = '';
  clusterId: string = '';
  resourceType: string = '';
  resourceName: string = '';
  podName: string = '';

  containersTableLoader: boolean = false;
  containersTableData = [];
  containerTableSettings = {
    actions: false,
    columns: {},
  };
  emptyMessage: string = '';
  errorMessage: string = '';

  actionsColumn = [
    {
      id: 'actions',
      icon: 'file-text-outline',
      status: 'primary',
      appearance: 'ghost',
      // isFciPack: true,
      onClick: (data) => {
        this.onClickActionsIcon(data);
      },
      iconStatus: 'primary',
    },
  ];

  emptyIcon = emptyIcon;
  errorIcon = errorIcon;

  selectedContainerName: string = '';

  private rootUrl: string = window.location.origin;

  constructor(
    private activatedRoute: ActivatedRoute,
    private k8sService: UiKubernetesExplorerControllerService,
    private router: Router,
    private dialogService: NbDialogService,
    private httpClient: HttpClient,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(params: Params) {
    this.stackName = params?.stackName;
    this.clusterId = params?.clusterId;
    this.resourceType = params?.resourceType;
    this.resourceName = params?.resourceName;
    this.podName = params?.podName;
    this.getPodContainers();
  }

  setTableSettings = (apiResponse) => {
    const tableSettings = {
      actions: false,
      columns: {},
    };
    let columns = [...Object.keys(apiResponse?.[0]), 'actions'];
    columns.forEach((column) => {
      tableSettings.columns[column] = {
        title: nameToTitleMapping[column] || camelToPascal(column),
        filter: false,
        valuePrepareFunction: (val, row) => {
          const valueMap = {
            ageInSeconds: secondsToDuration(val),
            actions: JSON.stringify({
              smartTableCustomActionButtons: this.actionsColumn,
            }),
          };

          if (valueMap[column]) {
            return valueMap[column];
          }

          return val;
        },
        onComponentInitFunction: (instance: any) => {
          instance.byCustomActionButtonClick?.subscribe((data) => {
            let customActionButton = this.actionsColumn.filter(
              (actionButton) => {
                return actionButton?.id === data?.actionButton?.id;
              },
            )[0];
            customActionButton.onClick(data?.rowData);
          });
        },
        renderComponent: tableColumnComponentMap[column],
        type: Object.keys(tableColumnComponentMap).includes(column)
          ? 'custom'
          : undefined,
      };
    });
    this.containerTableSettings = Object.assign({}, tableSettings);
  };

  getPodContainers() {
    this.containersTableLoader = true;
    const url = `${this.rootUrl}/cc-ui/v1/clusters/${this.clusterId}/k8s-explorer/${this.podName}/v2/containers?resourceName=${this.resourceName}&resourceType=${this.resourceType}`;
    // this.k8sService
    //   .listContainersInPodV2UsingGET({
    //     clusterId: this.clusterId,
    //     podName: this.podName,
    //     labels: {},
    //   })
    this.httpClient.get(url).subscribe(
      (response: Array<ContainerDTO>) => {
        this.containersTableLoader = false;
        const apiResponse = response.slice();

        if (!apiResponse?.length) {
          this.emptyMessage = `No containers found`;
          return;
        }

        this.setTableSettings(apiResponse);
        this.containersTableData = apiResponse?.slice();
      },
      (error) => {
        this.containersTableLoader = false;
        this.errorMessage =
          error?.error?.message ||
          error?.message ||
          'Error fetching pod container';
      },
    );
  }
  onClickActionsIcon(rowData) {
    this.selectedContainerName = rowData.name;
    this.logsUrl = `${this.rootUrl}/cc-ui/v1/clusters/${this.clusterId}${
      this.selectedContainerName ? '/k8s-explorer' : ''
    }/pods/${this.podName}/${
      this.selectedContainerName ? this.selectedContainerName + '/' : ''
    }logs?tail=${this.tailLogsParams?.tail}&resourceName=${
      this.resourceName
    }&resourceType=${this.resourceType}`;
    this.dialogService.open(this.actionsModal, {
      context: {
        ...rowData,
      },
    });
  }
  onChangeDropdown(event) {
    this.selectedContainerName = event;
  }

  handleBack = () => {
    this.router.navigate([
      '/',
      this.stackName,
      'cluster',
      this.clusterId,
      'resources',
      'type',
      this.resourceType,
      'resource',
      this.resourceName,
      'kubernetes',
    ]);
  };
}
