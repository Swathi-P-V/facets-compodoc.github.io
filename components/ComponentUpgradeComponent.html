import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import { LocalDataSource } from 'ng2-smart-table';
import {
  AbstractCluster,
  AwsClusterRequest,
  AzureClusterRequest,
} from 'src/app/cc-api/models';
import {
  UiAwsClusterControllerService,
  UiAzureClusterControllerService,
  UiCommonClusterControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-component-upgrade',
  templateUrl: './component-upgrade.component.html',
  styleUrls: ['./component-upgrade.component.scss'],
})
export class ComponentUpgradeComponent extends FacetsPage {
  cloud: 'AWS' | 'AZURE' | 'LOCAL' | 'GCP' | 'KUBERNETES';
  cluster: AbstractCluster;
  clusterId: string;
  stackName: string;
  fetchingConfig = false;
  componentUpgradeInProgress: boolean;
  componentUpgradeSource: LocalDataSource = new LocalDataSource();

  smartTableCustomActionButtons = [
    {
      id: 'upgrade',
      icon: 'arrow-upward-outline',
      toolTip: 'Upgrade',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: 'ENVIRONMENT_WRITE',
      onClick: (data) => {
        this.openComponentUpgradePopup(data);
      },
    },
  ];

  settingsComponentUpgrades = {
    hideSubHeader: true,
    columns: {
      component: {
        title: 'Component',
        filter: false,
        width: '30%',
        editable: false,
      },
      currentValue: {
        title: 'Current Version',
        filter: false,
        width: '30%',
        editable: false,
        editor: { type: 'text' },
      },
      newValue: {
        title: 'Available Version',
        filter: false,
        width: '30%',
        editable: false,
        editor: { type: 'text' },
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: () => {
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {
            smartTableCustomActionButtons: this.smartTableCustomActionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '12%',
        filter: false,
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
      perPage: 10,
    },
    actions: false,
  };

  @ViewChild('componentUpgradeConfirmUI', { static: true })
  private componentUpgradeConfirmUI: TemplateRef<any>;

  constructor(
    private aWSClusterService: UiAwsClusterControllerService,
    private uiCommonClusterControllerService: UiCommonClusterControllerService,
    activatedRoute: ActivatedRoute,
    private dialogService: NbDialogService,
    private stackService: UiStackControllerService,
    private azureClusterService: UiAzureClusterControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p.clusterId;
    this.stackName = p.stackName;

    if (this.clusterId) {
      this.fetchingConfig = true;
      this.uiCommonClusterControllerService
        .getClusterCommonUsingGET(this.clusterId)
        .subscribe(
          (t) => {
            this.cluster = t;
            this.cloud = t.cloud;
            this.populateComponentUpgrades(t);
            this.fetchingConfig = false;
          },
          () => {
            this.fetchingConfig = false;
          },
        );
    }
  }

  populateComponentUpgrades(cluster: AbstractCluster) {
    this.stackService.getStackUsingGET(cluster.stackName).subscribe((stack) => {
      this.uiCommonClusterControllerService
        .getClusterTaskUsingGET(cluster.id)
        .subscribe((clusterTasks) => {
          if (clusterTasks?.tasks?.length > 0) {
            this.componentUpgradeSource.reset();
            this.componentUpgradeSource.load([]);
            return;
          }

          let componentUpgrades = [];
          for (let clusterKey in cluster.componentVersions) {
            let clusterValue = parseFloat(
              cluster.componentVersions[clusterKey],
            );

            let stackValue = stack?.componentVersions[clusterKey];
            if (clusterKey === 'KUBERNETES' && this.cloud === 'AWS') {
              let k8sStackValue = parseFloat(stackValue);
              if (k8sStackValue > clusterValue) {
                componentUpgrades.push({
                  component: clusterKey,
                  currentValue: clusterValue.toString(),
                  newValue: k8sStackValue.toString(),
                });
              }
            }
          }
          this.componentUpgradeSource.load(componentUpgrades);
        });
    });
  }

  private async constructUpdateClusterRequest() {
    let clusterRequest: AwsClusterRequest | AzureClusterRequest = {};
    clusterRequest.cloud = this.cluster.cloud;
    clusterRequest.clusterName = this.cluster.name;
    clusterRequest.stackName = this.cluster.stackName;
    clusterRequest.componentVersions = this.cluster.componentVersions;
    clusterRequest.cdPipelineParent = this.cluster.cdPipelineParent;
    return clusterRequest;
  }

  private async updateCluster(clusterRequest: any) {
    this.componentUpgradeInProgress = true;
    try {
      let httpReq;
      if (this.cloud === 'AWS') {
        httpReq = this.aWSClusterService.updateClusterUsingPUT2({
          request: clusterRequest,
          clusterId: this.cluster.id,
        });
      } else if (this.cloud === 'AZURE') {
        this.azureClusterService.updateAzureClusterUsingPUT({
          request: clusterRequest,
          clusterId: this.cluster.id,
        });
      }
      httpReq.subscribe(
        (c) => {
          this.componentUpgradeInProgress = false;
          this.customToastService.success('Component update is successful');
          location.reload();
        },
        (error) => {
          this.componentUpgradeInProgress = false;
          this.onAPIErrorService.onAPIError(error?.error?.message, error);
        },
      );
    } catch (err) {
      this.componentUpgradeInProgress = false;
      this.customToastService.error('Component update failed');
    }
  }

  openComponentUpgradePopup(rowData) {
    this.dialogService.open(this.componentUpgradeConfirmUI, {
      context: {
        component: rowData['component'],
        version: rowData['newValue'],
      },
    });
  }

  async proceedUpgradeComponent(contextData) {
    const clusterRequest: AwsClusterRequest | AzureClusterRequest =
      await this.constructUpdateClusterRequest();
    clusterRequest.componentVersions[contextData['component']] =
      contextData['version'];
    await this.updateCluster(clusterRequest);
  }
}
