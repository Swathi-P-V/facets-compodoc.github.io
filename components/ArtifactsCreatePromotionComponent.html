import { Component, ViewChild, ElementRef } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  UiStackControllerService,
  UiPromotionWorkflowControllerService,
  UiDropdownsControllerService,
} from 'src/app/cc-api/services';
import {
  CdkDragDrop,
  moveItemInArray,
  transferArrayItem,
} from '@angular/cdk/drag-drop';
import {
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import { NbDialogService } from '@nebular/theme';
import {
  AbstractCluster,
  PromotionWorkflow,
  Stack,
} from 'src/app/cc-api/models';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { PROMOTIONAL_WORKFLOW_WRITE } from 'src/app/constants/permissions';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-artifacts-create-promotion',
  templateUrl: './artifacts-create-promotion.component.html',
  styleUrls: ['./artifacts-create-promotion.component.scss'],
})
export class ArtifactsCreatePromotionComponent extends FacetsPage {
  //Non used properties
  @ViewChild('dropListContainer') dropListContainer?: ElementRef;
  @ViewChild('overrideDefault') overrideDefault: any;
  @ViewChild('invalidOverride') invalidOverride: any;
  dropListReceiverElement?: HTMLElement;
  dragDropInfo?: {
    dragIndex: number;
    dropIndex: number;
  };
  selectedResource: any = [];
  stackName: string = ''; //stackName passed down from last component
  stacks: Stack[] = []; //blueprint list
  formType: string = 'create'; //form type 'create' or 'edit
  loadingEnvironments: boolean = false; //environments list api loader
  loadingPromotionData: boolean = false; //promotion data (in case of edit) api loader
  loadingReleaseStreams: boolean = false; //release stream api loader
  workflowId: string = ''; //workflowId in case of edit workflow
  releaseStreams: string[] = []; //release streams to show while setting hierarchy
  allEnvs: AbstractCluster[] = []; //environments to show while setting hierarchy
  callInProgress: boolean = false; //progress loader variable for create and edit promotion workflow
  promotionWorkflowForm: FormGroup = this.fb.group({
    name: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    registrationType: ['RELEASE_STREAM', Validators.required],
    blueprint: [false],
    default: [false, Validators.required],
  }); //promotion workflow form
  typeSelectOptions: { label: string; value: string }[] = [
    { label: 'Environment', value: 'ENVIRONMENT' },
    { label: 'Release Stream', value: 'RELEASE_STREAM' },
  ]; //registration type select menu options
  advanceOpenLogo = './assets/advance-open.svg'; //advanced options open options logo
  advanceCloseLogo = './assets/advance-close.svg'; //advanced options close options logo
  connector = './assets/env-connector.svg'; //connector logo for connecting 2 environments/release streams in hierarchy
  filterMove = './assets/filter-move.svg'; //moving hierarchy list item logo
  showAdvance: boolean = true; //advance option conditional variable... true = show options, false = no options
  hierarchyList: any = [null, null]; //list containing release streams and environments available
  blueprintOptionsList: { label: string; value: string }[] = []; //blueprint select menu options list
  hierarchyOptionsList: { label: string; value: string }[] = []; //hierarchy selection options list
  hierarchyOptionsObj: any = {}; //hierarchy options object to store selected hierarchy values
  envIdNameMapping: any = {}; //environment id to name mapping
  envNameIdMapping: any = {}; //environment name to id mapping
  currentPromotionalWorkflow: PromotionWorkflow = {}; //existing promotion workflow while editing
  hierarchyFormGroup: FormGroup = new FormGroup({
    env0: new FormControl(''),
    env1: new FormControl(''),
  }); //environment/release streams hierarchy select menus collection formgroup
  showShortOptionsMsg: boolean = false; //message condition variable when environments are less than 2 for a stack.
  totalViableEnv: string[] = [];
  promotionWorkflowWrite = PROMOTIONAL_WORKFLOW_WRITE;

  constructor(
    private activatedRoute: ActivatedRoute,
    private uiStackController: UiStackControllerService,
    private router: Router,
    private fb: FormBuilder,
    private promotionWorkflowController: UiPromotionWorkflowControllerService,
    private uiDropdownController: UiDropdownsControllerService,
    private dialogService: NbDialogService,
    private customToastService: CustomToastService,
    private onApiErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  //implementation of drag and drop using cdk drag and drop
  drop(event: CdkDragDrop<string[]>) {
    if (event.previousContainer === event.container) {
      //rearranging items in the same container
      moveItemInArray(
        event.container.data,
        event.previousIndex,
        event.currentIndex,
      );
    } else {
      //transfer the items from one container to another after drop
      transferArrayItem(
        event.previousContainer.data,
        event.container.data,
        event.previousIndex,
        event.currentIndex,
      );
    }
  }

  /*
    promotion hierarchy name getter
  */
  get name() {
    return this.promotionWorkflowForm.get('name');
  }

  /*
    promotion hierarchy registration type getter
  */
  get registrationType() {
    return this.promotionWorkflowForm.get('registrationType');
  }

  /*
    promotion hierarchy stack name getter
  */
  get blueprint() {
    return this.promotionWorkflowForm.get('blueprint');
  }

  /*
    promotion hierarchy stack default getter
  */
  get default() {
    return this.promotionWorkflowForm.get('default');
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.showShortOptionsMsg = false;
    this.resetForm();
    this.selectedResource = [];
    this.totalViableEnv = [];
    this.hierarchyList = [];
    this.workflowId = p?.workflowId || '';
    this.stackName = p?.stackName || '';
    //if workflow id present, then get stacks first which will be needed for constructing form.
    //otherwise get the environments and release streams
    if (this.workflowId?.length > 0) {
      this.formType = 'edit';
      this.getWorkflowInfo();
    } else {
      this.getAllEnvironments();
      this.getAllReleaseStreams();
    }
  }

  resetForm() {
    this.blueprint.setValue(false);
    this.name.reset();
    this.registrationType.setValue('RELEASE_STREAM');
    this.default.setValue(false);
  }

  getAllReleaseStreams(populateValues = false) {
    this.loadingReleaseStreams = true;
    this.uiDropdownController.getReleaseStreamsUsingGET().subscribe(
      (res) => {
        this.releaseStreams = res;
        //this.constructSelectForStreams(res)
        this.loadingReleaseStreams = false;

        //populate values indicate the edit promotion workflow type form
        if (populateValues) {
          //set the streams and then set the values
          this.constructSelectForStreams(res, true);
          //set current workflow values in release streams
          this.setCurrentWorkflowValues(this.currentPromotionalWorkflow);
        } else if (this.registrationType?.value === 'RELEASE_STREAM') {
          this.constructSelectForStreams(this.releaseStreams);
        }
      },
      (err) => {
        this.loadingReleaseStreams = false;
      },
    );
  }

  constructSelectForStreams(releaseStreams, populate = false) {
    this.hierarchyList = [];
    this.totalViableEnv = [];
    if (releaseStreams?.length > 1) {
      releaseStreams.forEach((stream, index) => {
        this.hierarchyList.push(stream);
        if (!populate) {
          this.totalViableEnv.push(stream);
        }
      });
    }
  }

  getWorkflowInfo() {
    this.loadingPromotionData = true;
    this.promotionWorkflowController
      .getWorkflowUsingGET(this.workflowId)
      .subscribe(
        (res) => {
          if (res) {
            this.currentPromotionalWorkflow = res;
            this.loadingPromotionData = false;
            //get either release streams or environments based on promotion workflow registration type
            //Getting either of them is proper here as we can't change registration type while editing.
            //only one type of hierarchy values can be edited while in edit mode.
            if (res?.registrationType === 'ENVIRONMENT') {
              this.getAllEnvironments(true);
            } else if (res?.registrationType === 'RELEASE_STREAM') {
              this.getAllReleaseStreams(true);
            }
          }
        },
        (err) => {
          this.loadingPromotionData = false;
          this.getAllEnvironments();
          this.getAllReleaseStreams();
        },
      );
  }

  setCurrentWorkflowValues(res) {
    this.totalViableEnv = [];
    //set default workflow name
    if (res?.workflowName) {
      this.name.setValue(res.workflowName);
    }

    //set default value
    if (res?.defaultWorkflow) {
      this.default.setValue(res.defaultWorkflow);
    }

    //set registration value
    if (res?.registrationType) {
      this.registrationType.setValue(res?.registrationType);
    }

    //if registration type is release stream and stack name is there,
    //set the global value to false, or set it to true
    if (
      res?.registrationType === 'RELEASE_STREAM' &&
      res?.stackName?.length > 0
    ) {
      this.blueprint.setValue(false);
    } else if (res?.registrationType === 'RELEASE_STREAM' && !res?.stackName) {
      this.blueprint.setValue(true);
    }
    this.name.disable();
    this.blueprint.disable();
    this.registrationType.disable();
    this.hierarchyList = [];
    if (
      res.hierarchies?.length > 0 &&
      res?.registrationType === 'ENVIRONMENT'
    ) {
      //total viable environment is the list of all possible environments
      //for current stack.
      this.allEnvs?.forEach((env, index) => {
        if (res?.stackName === env?.stackName) {
          this.totalViableEnv.push(env.name);
        }
      });

      //store selected environments values in a hierarchy variable selectedResource
      res?.hierarchies?.forEach((hierarchy) => {
        this.selectedResource[hierarchy?.sequence - 1] =
          this.envIdNameMapping[hierarchy?.registrationValue];
      });

      //transfer the values from total environments which are not present in selected resource
      //to non selected hierarchy list
      this.totalViableEnv.forEach((env) => {
        if (!this.selectedResource.includes(env)) {
          this.hierarchyList.push(env);
        }
      });
    } else if (
      res?.hierarchies?.length > 0 &&
      res?.registrationType === 'RELEASE_STREAM'
    ) {
      res?.hierarchies?.forEach((hierarchy) => {
        this.selectedResource[hierarchy?.sequence - 1] =
          hierarchy?.registrationValue;
      });
      this.releaseStreams.forEach((stream) => {
        this.totalViableEnv.push(stream);
        if (!this.selectedResource.includes(stream)) {
          this.hierarchyList.push(stream);
        }
      });
    }

    //Eliminate null elements from the selected environments in hierarchy
    //this is used when environment is deleted and if it is still in the promotion workflow
    //Ideally, this condition should not occur as we don't allow user to delete environment if it is a part of hierarchy
    this.selectedResource = this.selectedResource.filter(
      (elem) => elem?.length,
    );
  }

  goBack = () => {
    if (!this.stackName?.length) {
      this.stackName = this.stacks[0].name;
    }
    this.router.navigate([
      '/',
      'stack',
      this.stackName,
      'artifacts',
      'promotion-workflow',
      'workflows',
    ]);
  };

  getAllEnvironments(populateVals = false) {
    this.loadingEnvironments = true;
    this.uiStackController
      .getAllClustersUsingGET({
        page: 0,
        perPage: 10000,
        sortBy: 'ID',
      })
      .subscribe(
        (envs) => {
          this.allEnvs = envs?.content;
          this.allEnvs.forEach((env) => {
            this.envNameIdMapping[env?.name] = env?.id;
            this.envIdNameMapping[env?.id] = env?.name;
          });
          //if edit promotion workflow, then set the values
          //or select the default value in hierarchy options
          if (populateVals) {
            this.setCurrentWorkflowValues(this.currentPromotionalWorkflow);
          } else if (this.registrationType.value === 'ENVIRONMENT') {
            this.selectedOption(this.stackName, 'ENVIRONMENT');
          }
          this.loadingEnvironments = false;
        },
        (err) => {
          this.loadingEnvironments = false;
        },
      );
  }

  selectedOption(event = '', registration = '') {
    this.showShortOptionsMsg = false;
    if (
      (event?.length > 0 || registration?.length > 0) &&
      this.workflowId?.length === 0
    ) {
      if (
        this.registrationType?.value === 'ENVIRONMENT' ||
        registration === 'ENVIRONMENT'
      ) {
        this.hierarchyList = [];
        this.totalViableEnv = [];
        if (event?.length > 0) {
          this.allEnvs.forEach((env, index) => {
            if (event === env?.stackName) {
              this.hierarchyList.push(env?.name);
              this.totalViableEnv.push(env?.name);
            }
          });
        }
        if (this.hierarchyList.length < 2) {
          this.showShortOptionsMsg = true;
        }
      }
    }
  }

  async getDefaultWorkflow(
    stack: string,
    registrationType: 'ENVIRONMENT' | 'RELEASE_STREAM' | 'HYBRID',
  ) {
    return new Promise((resolve) => {
      this.promotionWorkflowController
        .getDefaultWorkflowUsingGET(
          stack?.length
            ? {
                stackName: stack,
                registrationType: registrationType,
              }
            : {
                registrationType: registrationType,
              },
        )
        .subscribe(
          (data) => {
            resolve(data);
          },
          (error) => {
            resolve(false);
          },
        );
    });
  }

  // If (default == true) and (registrationType == 'RELEASE_STREAM')
  // STEP 1: Check if there already is a local default (make getDefaultWorkflow API with the stackName)
  // STEP 2: Check if there already is a global default (make defaultWorkflow API without the stackName)
  // Follow these combinations:
  //  -> Exists: Global default, Setting: New global default => Let the user override the default
  //  -> Exists: Global default, Setting: New local default => Tell the user that it's an invalid operation
  //  -> Exists: Local default, Setting: New global default => Tell the user that it's an invalid operation
  //  -> Exists: Local default, Setting: New Local default => Let the user override the default
  async confirmOverrideForPromotionWorkflow() {
    let hierarchies: {
      registrationValue?: string;
      sequence?: number;
    }[] = [];
    let registrationType = this.registrationType.value;

    //Iterate over selected resource list and create a hierarchy object with value and sequence number
    this.selectedResource.forEach((resource, index) => {
      hierarchies.push({
        registrationValue:
          registrationType === 'ENVIRONMENT'
            ? this.envNameIdMapping[resource]
            : resource,
        sequence: index + 1,
      });
    });
    hierarchies.sort((a, b) => {
      return a.sequence - b.sequence;
    });

    const stackName =
      this.registrationType.value === 'ENVIRONMENT' ||
      (this.registrationType.value === 'RELEASE_STREAM' &&
        !this.blueprint.value)
        ? this.stackName
        : null;
    let workflowParams: PromotionWorkflow = {
      workflowName: this.name.value ? this.name.value.trim() : '',
      stackName,
      registrationType: this.registrationType.value,
      hierarchies: hierarchies,
      defaultWorkflow: this.default.value,
    };
    this.checkForOverride(
      workflowParams,
      registrationType,
      'create',
      this.createPromotionWorkflow.bind(this),
    );
  }

  async checkForOverride(
    workflowParams,
    registrationType,
    type,
    registerWorkflowCallback: Function,
  ) {
    this.callInProgress = true;
    if (this.default.value) {
      // We need to warn the user and block them at this stage
      const globalDefault = await this.getDefaultWorkflow('', registrationType);
      const localDefault = await this.getDefaultWorkflow(
        this.stackName,
        registrationType,
      );
      this.callInProgress = false;
      const isCurrentGlobal = !workflowParams?.stackName?.length;
      if (
        (globalDefault && !isCurrentGlobal) ||
        (localDefault && isCurrentGlobal)
      ) {
        this.dialogService.open(this.invalidOverride, {
          hasBackdrop: true,
          closeOnBackdropClick: false,
          context: {
            currentGlobalDefault: globalDefault,
            header: globalDefault ? 'Global' : 'Environment specific',
            message:
              globalDefault && !isCurrentGlobal
                ? 'You need to unset the current global default first before setting any other workflow as default workflow for this registration type.'
                : 'Marking a global workflow as default would override all the currently set default release stream workflows for all the blueprints. Please perform this operation manually.',
          },
        });
        return;
      }
      if (localDefault || globalDefault) {
        this.dialogService.open(this.overrideDefault, {
          hasBackdrop: true,
          closeOnBackdropClick: false,
          context: {
            prev: localDefault || globalDefault,
            new: workflowParams,
            type: type,
          },
        });
      } else {
        registerWorkflowCallback(workflowParams, localDefault);
      }
    } else {
      registerWorkflowCallback(workflowParams);
    }
  }

  async createPromotionWorkflow(newWorkflowParams, oldWorkflowParams?, ref?) {
    if (ref) {
      ref.close();
    }
    this.callInProgress = true;
    let unsetActionRsp: any;
    if (oldWorkflowParams) {
      unsetActionRsp = await this.unsetCurrentDefaultWorkflow(
        oldWorkflowParams,
      );
    }
    if (!oldWorkflowParams || unsetActionRsp?.id == oldWorkflowParams?.id) {
      this.promotionWorkflowController
        .createWorkflowUsingPOST(newWorkflowParams)
        .subscribe(
          (res) => {
            this.callInProgress = false;
            this.customToastService.success(
              `Promotion workflow ${res?.workflowName} created successfully`,
            );
            this.callInProgress = true;
            this.goBack();
          },
          (error) => {
            this.onApiErrorService.onAPIError(
              error?.error?.message ||
                `Failed to create promotion workflow ${newWorkflowParams?.workflowName}`,
              error,
            );
            this.callInProgress = false;
          },
        );
    } else {
      this.onApiErrorService.onAPIError(
        unsetActionRsp?.error?.message ||
          `Failed to set/unset promotion workflow ${newWorkflowParams?.workflowName} as default`,
        unsetActionRsp,
      );
    }
  }

  async unsetCurrentDefaultWorkflow(workflow) {
    workflow.defaultWorkflow = false;
    return new Promise((resolve) => {
      this.promotionWorkflowController
        .updateWorkflowUsingPUT({
          workflowId: workflow.id,
          promotionWorkflow: workflow,
        })
        .subscribe(
          (data) => {
            resolve(data);
          },
          (error) => {
            resolve(error);
          },
        );
    });
  }

  async editPromotionWorkflow() {
    let hierarchies: {
      registrationValue?: string;
      sequence?: number;
    }[] = [];
    this.selectedResource.forEach((resource, index) => {
      hierarchies.push({
        registrationValue:
          this.currentPromotionalWorkflow?.registrationType === 'ENVIRONMENT'
            ? this.envNameIdMapping[resource]
            : resource,
        sequence: index + 1,
      });
    });
    hierarchies.sort((a, b) => {
      return a.sequence - b.sequence;
    });
    let workflowParams: PromotionWorkflow = {
      workflowName: this.currentPromotionalWorkflow?.workflowName,
      stackName: this.currentPromotionalWorkflow?.stackName,
      registrationType: this.currentPromotionalWorkflow?.registrationType,
      hierarchies: hierarchies,
      defaultWorkflow: this.default.value,
    };
    this.callInProgress = true;
    this.checkForOverride(
      workflowParams,
      this.currentPromotionalWorkflow?.registrationType,
      'edit',
      this.saveChangesForPromotionWorkflow.bind(this),
    );
  }

  async saveChangesForPromotionWorkflow(
    newWorkflowParams,
    oldWorkflowParams?,
    ref?,
  ) {
    if (ref) {
      ref.close();
    }
    let unsetActionRsp;
    this.callInProgress = true;
    if (oldWorkflowParams) {
      unsetActionRsp = await this.unsetCurrentDefaultWorkflow(
        oldWorkflowParams,
      );
    }
    this.promotionWorkflowController
      .updateWorkflowUsingPUT({
        workflowId: this.currentPromotionalWorkflow.id,
        promotionWorkflow: newWorkflowParams,
      })
      .subscribe(
        (res) => {
          this.customToastService.success(
            `Promotion workflow ${newWorkflowParams?.workflowName} edited successfully`,
          );
          this.goBack();
        },
        (error) => {
          this.onApiErrorService.onAPIError(
            error?.error?.message ||
              `Failed to edit promotion workflow ${newWorkflowParams?.workflowName}`,
            error,
          );
        },
      )
      .add(() => {
        this.callInProgress = false;
      });
  }

  typeChange(event) {
    this.showShortOptionsMsg = false;
    this.selectedResource = [];
    this.name.markAsUntouched();
    this.name.markAsPristine();
    if (event === 'RELEASE_STREAM') {
      this.constructSelectForStreams(this.releaseStreams, false);
    } else if (event === 'ENVIRONMENT') {
      this.selectedOption(this.stackName, 'ENVIRONMENT');
    }
  }

  isLoading() {
    return (
      this.loadingEnvironments ||
      this.loadingPromotionData ||
      this.loadingReleaseStreams
    );
  }

  //add resource from hierarchy list to selected list
  addResource(env) {
    if (!this.selectedResource.includes(env)) {
      this.selectedResource.push(env);
    }
    this.hierarchyList = this.hierarchyList.filter((item) => item != env);
  }

  //remove resource from selected list and add to hierarchy list
  removeResource(env) {
    this.selectedResource = this.selectedResource.filter((item) => item != env);
    this.hierarchyList.push(env);
  }
}
