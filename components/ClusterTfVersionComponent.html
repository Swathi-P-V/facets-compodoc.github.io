import {
  Component,
  ViewChild,
  Input,
  Output,
  EventEmitter,
  OnDestroy,
} from '@angular/core';
import { ActivatedRoute, Params } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { TfVersion } from 'src/app/cc-api/models';
import { UiTfVersionControllerService } from 'src/app/cc-api/services';
import { ClusterTfVersionMapping } from 'src/app/cc-api/models';
import { Subject } from 'rxjs';
import { debounceTime, distinctUntilChanged } from 'rxjs';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
const emptyStateLogo = './assets/template-empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-cluster-tf-version',
  templateUrl: './cluster-tf-version.component.html',
  styleUrls: ['./cluster-tf-version.component.scss'],
})
export class ClusterTfVersionComponent extends FacetsPage implements OnDestroy {
  clusterId: any;
  stackName: any;
  versionForm: FormGroup = this.fb.group({
    stream: ['', [Validators.required]],
    major: [null, Validators.required],
    minor: ['', Validators.required],
  });
  currentVal: {
    stream: string;
    major: number;
    minor: string;
  } = { stream: '', major: null, minor: '' };
  streamList: string[] = [];
  majorList: number[] = [];
  minorList: string[] = [];
  selectStream: { value: string; label: string }[] = [];
  selectMajor: {
    value: number;
    label: string;
    disabled: boolean;
  }[] = [];
  selectMinor: { value: string; label: string }[] = [];
  availableVersions: TfVersion[] = [];
  callInProgress: boolean;
  fetchingTfData: boolean;
  emptyStateLogo: any = emptyStateLogo;
  errorIcon: any = errorIcon;
  primaryMsg: string = '';
  secondaryMsg: string = '';
  status: 'info' | 'error' = 'info';
  @Input()
  ref: any;
  @Input() iacActionText: string = 'Edit';
  tfVersionMap: ClusterTfVersionMapping;
  @Output()
  versionChange = new EventEmitter();
  @ViewChild('streamInput') inputRT;
  showEmptyState: boolean = false;
  disabledMsg: string = '';
  disableMajorSelect: boolean = false;
  streamNameUpdate = new Subject<any>();
  versionTimeout: any;
  minorTimeout: any;
  loadStatus: boolean = false;
  loadTimeout: any;

  constructor(
    activatedRoute: ActivatedRoute,
    private uiTfVersionController: UiTfVersionControllerService,
    private fb: FormBuilder,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  get stream() {
    return this.versionForm.get('stream');
  }

  get major() {
    return this.versionForm.get('major');
  }

  get minor() {
    return this.versionForm.get('minor');
  }

  OnPageLoad(p: Params): void {
    this.stackName = p.stackName;
    this.clusterId = p.clusterId;
    this.getTfInfo();
    this.streamNameUpdate
      .pipe(
        debounceTime(1000), //debounce is used such that api request won't be made on each button press in search input
        distinctUntilChanged(),
      ) //this line ensures change function to fire only if incoming search string is changed
      .subscribe((event) => {
        //subscribing to streamNameUpdate subject gives the search string result as event value here
        this.versionForm.patchValue({ stream: event?.target?.value });
        if (this.stream?.value?.length > 0) {
          this.getTfStreamData(); //get IaC stream data based on search string
        } else {
          this.setEmptyMsg();
        }
      });
  }

  ngOnDestroy() {
    this.versionTimeout ? clearTimeout(this.versionTimeout) : null;
    this.minorTimeout ? clearTimeout(this.minorTimeout) : null;
    this.loadTimeout ? clearTimeout(this.loadTimeout) : null;
  }

  setEmptyMsg() {
    this.showEmptyState = true;
    this.major.reset();
    this.minor.reset();
    this.primaryMsg = 'No major and minor versions found for the stream';
    this.secondaryMsg = 'Please enter a valid stream name';
    this.status = 'info';
  }

  //Function connected to predefined options of streams. if user selects certain stream from predefined option, this fires.
  onSelectionChangeRT($event) {
    if ($event?.length > 0 && !this.loadStatus) {
      this.getTfStreamData();
    }
  }

  getTfInfo() {
    this.streamList = [];
    this.availableVersions = [];
    this.selectStream = [];
    this.fetchingTfData = true;

    //set 3 default stream values.
    this.selectStream.push(
      {
        label: 'production',
        value: 'production',
      },
      {
        label: 'stage',
        value: 'stage',
      },
      {
        label: 'tfdev',
        value: 'tfdev',
      },
    );

    //get latest stream, major and minor version values
    this.uiTfVersionController
      .getTfStreamForClusterUsingGET(this.clusterId)
      .subscribe(
        (version) => {
          this.tfVersionMap = version;
          this.currentVal = {
            stream: this.tfVersionMap?.version?.tfStream, //current stream name
            major: this.tfVersionMap?.version?.majorVersion, //current major version
            minor: this.tfVersionMap?.version?.minorVersion, //current minor version
          };

          /*  
            Setting stream into input value causes the input change event to be emitted,
            Which in turn can recursively call the fetch tf version api again and again
            The below loadStatus variable make sure that when we set stream value, we 
            don't fire any input change event
          */
          this.loadStatus = true;
          this.loadTimeout = setTimeout(() => {
            this.loadStatus = false;
          }, 10);

          //set the latest stream value from the api
          this.versionForm.patchValue({
            stream: this.tfVersionMap?.version?.tfStream,
          });

          //If the latest stream is not one of production, tfdev or stage, then push it in the stream list.
          if (
            this.tfVersionMap?.version?.tfStream !== 'production' &&
            this.tfVersionMap?.version?.tfStream !== 'tfdev' &&
            this.tfVersionMap?.version?.tfStream !== 'stage'
          ) {
            this.selectStream.push({
              label: this.tfVersionMap?.version?.tfStream,
              value: this.tfVersionMap?.version?.tfStream,
            });
          }
          this.getTfStreamData();
        },
        (error) => {
          this.setErrorVals(error);
        },
      );
  }

  setErrorVals(error) {
    this.iacActionText = error?.error?.code === '404' ? 'Set' : 'Edit';
    this.primaryMsg = '';
    this.secondaryMsg =
      error?.error?.message || ' There is some error fetching stream versions';
    this.showEmptyState = true;
    this.fetchingTfData = false;
    this.major.reset();
    this.minor.reset();
    this.emptyStateLogo = this.errorIcon;
    this.status = 'error';
  }

  //get latest major and minor versions data for selected stream
  getTfStreamData() {
    this.fetchingTfData = true;
    this.showEmptyState = false;
    //get data based on the stream value
    this.uiTfVersionController
      .getTfVersionsForStreamUsingGET(this.stream?.value)
      .subscribe(
        (versions) => {
          this.availableVersions = versions;
          if (versions?.length > 0) {
            this.setStreamData();
          } else {
            //error while fetching stream data
            this.setEmptyMsg();
          }
          this.fetchingTfData = false;
        },
        (error) => {
          this.setErrorVals(error);
        },
      );
  }

  resetFieldData() {
    this.major?.enable();
    this.major?.reset();
    this.minor?.reset();
    this.majorList = [];
    this.minorList = [];
    this.selectMajor = [];
    this.selectMinor = [];
    this.disableMajorSelect = false;
  }

  setStreamData() {
    this.resetFieldData();
    //create major versions list based on stream data.
    this.availableVersions.forEach((version) => {
      if (!this.majorList?.includes(version.majorVersion)) {
        this.majorList.push(version.majorVersion);
      }
    });
    this.majorList?.sort().reverse();

    //for major data, if only 1 version is present, send disabled input status, otherwise normal input.
    let tempMajorList = [];
    this.majorList.forEach((major) => {
      if (
        major < this.currentVal['major'] &&
        this.stream?.value === this.currentVal?.stream
      ) {
        tempMajorList.push({
          label: major.toString(),
          value: major,
          disabled: true,
        });
      } else {
        tempMajorList.push({
          label: major.toString(),
          value: major,
          disabled: false,
        });
      }
    });
    this.selectMajor = [...tempMajorList];

    /*
      * If only 1 version is present for stream, then disable the input field, 
        select the version and show disabled message
      * Else set major stream values
      * Here, using settimeout makes sure the ui is updated properly after we set the major stream input values  
    */
    this.versionTimeout = setTimeout(() => {
      if (this.majorList?.length === 1) {
        this.setMajorStreamSingleValue();
      } else {
        this.setMajorStreamValues();
      }
      this.filterMinorVersions();
    });
  }

  //set single major version value, disable input, show single version message
  setMajorStreamSingleValue() {
    this.versionForm.patchValue({ major: this.majorList?.[0] });
    this.disableMajorSelect = true;
    this.major.disable();
    this.disabledMsg = 'Only version ' + this.major?.value + ' available';
  }

  /*
    * If current set stream is different from current selected stream, 
      then set selected value as first item from major list
    * Else, set major version as current selected major version (current stream = selected stream)  
  */
  setMajorStreamValues() {
    const majorVersion =
      this.stream?.value !== this.currentVal?.stream
        ? this.majorList?.[0]
        : this.currentVal?.major;
    this.versionForm.patchValue({
      major: majorVersion,
    });
  }

  /*
   * Filter minor versions from the available versions based on major version selected
   * Build select input fields based on filtered minor versions
   * Set minor input value
   */
  filterMinorVersions() {
    this.availableVersions.forEach((version) => {
      if (version?.majorVersion === this.major?.value) {
        this.minorList.push(version?.minorVersion);
      }
    });
    this.selectMinor.push({
      label: 'latest',
      value: 'latest',
    });
    this.minorList.forEach((minor) => {
      this.selectMinor.push({
        label: minor,
        value: minor,
      });
    });
    this.minorTimeout = setTimeout(() => {
      this.setMinorInputValues();
    });
  }

  /*
    * If current selected stream is different from set stream, then set minor value as latest
    * If current stream is similar to set stream, but set major version is not similar to selected major version
      then set minor value to latest
    * Else set it to the current value, as in this case the both selected stream and major values will be similar to 
      set values for the environment.   
  */
  setMinorInputValues() {
    let tempMinorValue;
    if (
      this.stream?.value !== this.currentVal?.stream ||
      (this.stream?.value === this.currentVal?.stream &&
        this.major?.value !== this.currentVal?.major)
    ) {
      tempMinorValue = this.minorList?.length > 0 ? 'latest' : '';
    } else {
      tempMinorValue = this.currentVal?.minor;
    }
    this.versionForm.patchValue({
      minor: tempMinorValue,
    });
    this.fetchingTfData = false;
  }

  //When major version is changed,filter minor versions based on major version
  majorChanged(event) {
    this.minorList = [];
    this.selectMinor = [];
    this.filterMinorVersions();
  }

  changeVersion(ref) {
    this.callInProgress = true;
    let changeParams: {
      tfVersion: TfVersion;
      clusterId: string;
    } = {
      tfVersion: {
        lastModified: this.tfVersionMap?.lastModifiedDate,
        majorVersion: this.major?.value,
        minorVersion: this.minor?.value,
        tfStream: this.stream?.value,
      },
      clusterId: this.clusterId,
    };
    this.uiTfVersionController
      .setTfVersionForClusterUsingPUT(changeParams)
      .subscribe(
        (response) => {
          this.callInProgress = false;
          this.customToastService.success('IaC version updated successfully');
          this.versionChange.emit('success');
          this.getTfInfo();
          ref.close();
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Failed to update IaC version',
            error,
          );
          this.versionChange.emit('failure');
          this.callInProgress = false;
        },
      );
  }
}
