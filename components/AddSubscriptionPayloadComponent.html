import { Component, OnDestroy } from '@angular/core';
import { FormGroup } from '@angular/forms';
import { ActivatedRoute, Params, Router } from '@angular/router';
import {
  MonacoEditorConstructionOptions,
  MonacoEditorLoaderService,
} from '@materia-ui/ngx-monaco-editor';
import { SUBSCRIPTION_WRITE } from './../../../../constants/permissions';
import { Subscription } from 'rxjs';
import { filter, map, take } from 'rxjs';
import {
  Subscription as _Subscription,
  TestNotificationRequest,
} from 'src/app/cc-api/models';
import { UiNotificationControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';

var suggestions: any[];

function createDependencyProposals(range) {
  suggestions?.forEach((suggestion) => {
    suggestion.range = range;
  });
  return suggestions;
}

@Component({
  selector: 'app-add-subscription-payload',
  templateUrl: './add-subscription-payload.component.html',
  styleUrls: ['./add-subscription-payload.component.scss'],
})
export class AddSubscriptionPayloadComponent
  extends FacetsPage
  implements OnDestroy
{
  headingOptions = {
    // Payload can either be added or edited
    add: 'Add Payload',
    edit: 'Edit Payload',
  };
  mode: 'add' | 'edit' = 'add'; // By default, the component is initialized in 'add' mode
  subscription: _Subscription;
  payload: any; // Json config which would be sent as the data-type for the webhook
  channelName: string; // Since subscription only contains the ID of the channel, we need the channel name to be sent to this component
  // through the parent directly
  optionalFilters: {
    // Caching in the selected optional filters
    id: string;
    label: string;
    tooltip: string;
    multiSelect: boolean;
    selectedValues: [];
    selectValues: [];
  }[] = [];
  editorOptions: MonacoEditorConstructionOptions = {
    // Editor options for the monaco editor
    theme: 'vs-light',
    language: 'json',
    minimap: {
      enabled: false,
    },
  };
  code: string = '{}';
  isJsonCodeValid: boolean = false;
  allFieldAttributes: any[]; // List of all the field attributes
  subscriptions: Subscription[] = [];
  addedAttributes: Set<string> = new Set<string>();
  editID: string;
  monaco: any;
  completionProvider: any;
  testingResult: string;
  notificationError: string;
  additionalFilters: FormGroup;
  selectedFilters: [];
  subscriptionWrite = SUBSCRIPTION_WRITE;

  constructor(
    private router: Router,
    private activatedRoute: ActivatedRoute,
    private notificationControllerService: UiNotificationControllerService,
    private monacoLoaderService: MonacoEditorLoaderService,
  ) {
    super(activatedRoute);
    this.setRouterStateListener();

    this.monaco = (window as any).monaco;
    this.monacoLoaderService.isMonacoLoaded$
      .pipe(
        filter((isLoaded) => isLoaded),
        take(1),
      )
      .subscribe(() => {
        this.monaco = (window as any).monaco;
        this.initAllAttributes();
      });
  }

  OnPageLoad(params: Params): void {
    if (params) {
      this.editID = params.editId;
      if (this.editID) {
        this.mode = 'edit';
      }
    }
  }

  ngOnDestroy(): void {
    this.subscriptions?.forEach((subscription) => {
      subscription.unsubscribe();
    });
    if (this.completionProvider) {
      this.completionProvider.dispose();
      if (this.monaco) {
        this.monaco = undefined;
      }
    }
  }

  // Initializes the fields list
  initAllAttributes() {
    if (!this.subscription?.notificationType) {
      this.handleBack(false);
    }
    this.subscriptions?.push(
      this.notificationControllerService
        .getSubscriptionAttributesUsingGET(this.subscription?.notificationType)
        .subscribe(
          (rsp) => {
            this.allFieldAttributes = rsp;
            suggestions = this.createDynamicSuggestions(
              this.allFieldAttributes,
            );
            this.setAutoComplete(this.allFieldAttributes);
            this.isJsonValid(this.code);
          },
          (error) => {
            this.handleBack(false);
          },
        ),
    );
  }

  setAutoComplete(suggestions: string[]) {
    this.completionProvider =
      this.monaco.languages.registerCompletionItemProvider('json', {
        provideCompletionItems: function (model, position) {
          var word = model.getWordUntilPosition(position);
          var range = {
            startLineNumber: position.lineNumber,
            endLineNumber: position.lineNumber,
            startColumn: word.startColumn,
            endColumn: word.endColumn,
          };
          return {
            suggestions: createDependencyProposals(range),
          };
        },
      });
  }

  createDynamicSuggestions(suggestions: string[]) {
    let suggestionsConfig: any[] = [];
    suggestions?.forEach((suggestion) => {
      let suggestionConfig = {
        label: `"${suggestion}"`,
        kind: monaco.languages.CompletionItemKind.Function,
        documentation: '',
        insertText: `"{{${suggestion}}}"`,
        range: null,
      };
      suggestionsConfig.push(suggestionConfig);
    });
    return suggestionsConfig;
  }

  handleBack = (cancel: boolean = true) => {
    let url = this.editID
      ? `/settings/notification-center/subscription/edit/${this.editID}`
      : `/settings/notification-center/subscription/create`;
    this.router.navigate([url], {
      state: {
        subscription: this.subscription,
        payload: cancel ? this.subscription?.payloadJson : this.payload,
        additionalFilters: this.additionalFilters,
        selectedFilters: this.selectedFilters,
        previousComponent: this.editID
          ? 'edit-subscription'
          : 'create-subscription',
      },
    });
  };

  setRouterStateListener() {
    this.activatedRoute.paramMap
      .pipe(map(() => window.history.state))
      .subscribe((state) => {
        this.subscription = state.subscription;
        this.channelName = state.channelName;
        this.additionalFilters = state.additionalFilters;
        this.selectedFilters = state.selectedFilters;
        if (this.subscription?.payloadJson) {
          this.code = this.subscription.payloadJson;
          this.payload = this.subscription.payloadJson;
        }
      });
  }

  // // Adds the attribute from the field list to code
  // addAttributeToCode(attribute: string) {
  //   this.code += attribute;
  // }

  savePayload() {
    this.payload = this.code;
    this.handleBack(false);
  }

  isJsonValid(content) {
    if (!content || !content?.length) {
      this.addedAttributes = new Set<string>();
    }
    this.isJsonCodeValid = this.isJson(content);
    if (this.isJsonCodeValid) {
      this.processAddedAttributes(content);
    }
  }

  async processAddedAttributes(content: string) {
    this.allFieldAttributes?.forEach((attribute) => {
      if (content.includes(attribute)) {
        this.addedAttributes.add(attribute);
      } else {
        this.addedAttributes.delete(attribute);
      }
    });
  }

  isJson(item) {
    item = typeof item !== 'string' ? JSON.stringify(item) : item;
    try {
      item = JSON.parse(item);
    } catch (e) {
      return false;
    }
    if (typeof item === 'object' && item !== null) {
      return true;
    }
    return false;
  }

  cancel() {
    this.handleBack();
  }

  testNotification() {
    this.testingResult = '';
    this.notificationError = '';
    let payload: TestNotificationRequest = {
      channelAddress: this.subscription?.channelAddress,
      channelType: this.subscription?.channelType,
      integrationKey: null,
      notificationType: this.subscription?.notificationType
        .toUpperCase()
        .replace(' ', '_') as any,
      payload: this.code,
    };

    this.notificationControllerService
      .testNotificationChannelUsingPOSTResponse(payload)
      .subscribe(
        (response) => {
          if (response) {
            this.testingResult = 'Test Notification succeeded';
          }
        },
        (error) => {
          try {
            error = JSON.parse(error.error);
          } catch (error) {
            error = {};
          }

          this.notificationError =
            'Error: ' + (error?.message || 'Test Notification failed');
        },
      );
  }
}
