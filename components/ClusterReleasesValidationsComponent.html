import {
  animate,
  state,
  style,
  transition,
  trigger,
} from '@angular/animations';
import {
  Component,
  EventEmitter,
  Input,
  OnChanges,
  OnDestroy,
  OnInit,
  Output,
  SimpleChanges,
  TemplateRef,
  ViewChild,
} from '@angular/core';
import { FormBuilder, FormControl, FormGroup } from '@angular/forms';
import { NbMenuItem, NbMenuService } from '@nebular/theme';
import cloneDeep from 'lodash/cloneDeep';
import debounce from 'lodash/debounce';
import groupBy from 'lodash/groupBy';
import { filter, map } from 'rxjs';
import {
  BlueprintValidationError,
  ValidationResponse,
} from 'src/app/cc-api/models';
import { UiApplicationControllerService } from 'src/app/cc-api/services';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
const emptyStateLogo = './assets/empty.svg';

type CategoryKeys = {
  [category in BlueprintValidationError['category']]: string;
};

@Component({
  selector: 'app-cluster-releases-validations',
  templateUrl: './cluster-releases-validations.component.html',
  styleUrls: ['./cluster-releases-validations.component.scss'],
  animations: [
    trigger('menuSlide', [
      state('hidden', style({ transform: 'translateX(-100%)' })),
      state('visible', style({ transform: 'translateX(0)' })),
      transition('hidden => visible', animate('300ms ease-in')),
      transition('visible => hidden', animate('300ms ease-out')),
    ]),
  ],
})
export class ClusterReleasesValidationsComponent
  implements OnInit, OnChanges, OnDestroy
{
  @Input() clusterId: string;
  @Input() stackName: string;
  @Input() fetchingValidations: boolean = false;
  @Input() validations: ValidationResponse[];

  errorCategoryMap: Map<string, ValidationResponse[]>;
  runningValidationCallInProgress: boolean = false;
  errorMenuItems: NbMenuItem[] = [];
  validationsAutoRefreshingIn: Date;
  showCountdown: boolean = true;
  selectedMenuItemId: string;
  menuItemSubscriber: any;
  emptyStateLogo: any = emptyStateLogo;
  categoryTotalValidationsMap: Map<string, number>;

  filterForm: FormGroup;
  @ViewChild('filterOptions', { read: TemplateRef })
  filterOptions: TemplateRef<any>;

  categoryIconMapping: CategoryKeys = {
    'Disabled Resource References': 'file',
    'Invalid Reference Expression': 'file',
    'GuardRails Compliance Issues': 'file-guard',
    'Non-Existent Resource Reference': 'file',
    'Schema Compliance Error': 'flower-brackets',
    'Syntax Error': 'flower-brackets',
  };

  constructor(
    private uiApplicationControllerService: UiApplicationControllerService,
    private customToasterService: CustomToastService,
    private formBuilder: FormBuilder,
    private menuService: NbMenuService,
  ) {
    this.menuItemSubscriber = this.menuService
      .onItemClick()
      .pipe(filter(({ tag }) => tag === 'validations-category-side-menu'))
      .subscribe(this.onMenuItemClick);
    this.initFilterFormAndListener();
  }

  ngOnChanges(changes: SimpleChanges): void {
    if (changes?.validations?.currentValue) {
      this.updateValidations();
    }
  }

  ngOnInit(): void {
    this.beginAutoRefreshing();
  }

  initFilterFormAndListener() {
    this.filterForm = this.formBuilder.group({
      showWarnings: [true],
      showErrors: [true],
      searchKey: new FormControl(''),
    });
    this.filterForm.valueChanges.subscribe(
      debounce((state) => {
        this.applyFilter(state);
      }, 300),
    );
  }

  showingAllData: boolean = true;
  applyFilter(state) {
    if (!state.searchKey?.length && state.showWarnings && state.showErrors) {
      this.errorCategoryMap = this.originalDataBackup;
      this.showingAllData = true;
    } else {
      this.errorCategoryMap = this.getFilteredResults(
        state,
        this.originalDataBackup,
      );
      this.showingAllData = false;
    }
  }

  onMenuItemClick = (event) => {
    this.selectedMenuItemId = event?.item?.data?.id;
    this.errorMenuItems.forEach((menuItem) => {
      menuItem.selected = menuItem.data?.id === this.selectedMenuItemId;
    });
    this.applyFilter(this.filterForm?.value);
  };

  beginAutoRefreshing() {
    this.validationsAutoRefreshingIn = this.getTime5MinutesFromNow(new Date());
    this.showCountdown = false;
    setTimeout(() => {
      this.showCountdown = true;
    }, 0);
  }

  getTime5MinutesFromNow(currentTime: Date) {
    const FIVE_MINUTES = 5 * 60 * 1000;
    return new Date(currentTime.getTime() + FIVE_MINUTES);
  }

  originalDataBackup: Map<string, ValidationResponse[]> = new Map<
    string,
    ValidationResponse[]
  >();
  updateValidations() {
    this.errorCategoryMap = this.categorizeErrors(this.validations);
    this.originalDataBackup = this.errorCategoryMap;
    this.categoryTotalValidationsMap = this.getCategoryTotalValidationLength(
      this.originalDataBackup,
    );
    this.applyFilter(this.filterForm.value);
    this.updateSidebarMenu();
    this.beginAutoRefreshing();
  }

  updateSidebarMenu() {
    this.errorMenuItems = this.getErrorMenuItems();
    if (
      !this.errorMenuItems?.find(
        (menuItem) => menuItem.data.id === this.selectedMenuItemId,
      )
    ) {
      this.selectedMenuItemId = this.errorMenuItems?.length
        ? this.errorMenuItems[0].data.id
        : undefined;
    }
    this.onMenuItemClick({ item: { data: { id: this.selectedMenuItemId } } });
  }

  @Output() setTotalValidations: EventEmitter<number> =
    new EventEmitter<number>();
  totalValidations: number;
  categorizeErrors(validations: any[]): any {
    this.totalValidations = 0;

    const updatedValidations = validations?.map((validation) => {
      this.totalValidations += validation?.errors?.length;
      return {
        ...validation,
        errorGroupedByCategory: groupBy(validation?.errors, 'category'),
      };
    });
    this.setTotalValidations.emit(this.totalValidations);

    const errorCategoryMap = new Map<string, ValidationResponse[]>();
    updatedValidations?.forEach((resource) => {
      const errorGroups = cloneDeep(resource?.errorGroupedByCategory);
      delete resource?.errorGroupedByCategory;
      Object.entries(errorGroups)?.forEach(([key, value]) => {
        if (!errorCategoryMap.has(key)) {
          errorCategoryMap.set(key, []);
        }
        errorCategoryMap.get(key).push({
          ...resource,
          errors: value,
        });
      });
    });
    return errorCategoryMap;
  }

  getErrorMenuItems(): NbMenuItem[] {
    const menuItems: NbMenuItem[] = [];

    Array.from(this.originalDataBackup?.keys()).forEach((errorCategory) => {
      menuItems.push({
        title:
          errorCategory +
          ` (${this.categoryTotalValidationsMap.get(errorCategory)})`,
        data: { id: errorCategory },
        selected: false,
        icon: {
          icon: this.categoryIconMapping[errorCategory] || 'settings-outline',
          pack: 'fci',
        },
      });
    });
    if (menuItems?.length && !this.selectedMenuItemId) {
      this.selectedMenuItemId = menuItems[0].data?.id;
    }
    return menuItems;
  }

  runValidation() {
    this.runningValidationCallInProgress = true;
    this.uiApplicationControllerService
      .runValidationUsingPOST(this.clusterId)
      .subscribe(
        (rsp) => {
          this.customToasterService.success(
            'Syncing and fetching latest validations',
          );
          this.fetchValidations();
        },
        (error) => {
          this.customToasterService.error(
            'Validation request encountered an issue. Please try again later or raise a ClickUp ticket if the problem persists.',
          );
        },
      )
      .add(() => {
        this.runningValidationCallInProgress = false;
      });
  }

  errorsBeingDisplayed: number;
  getFilteredResults(
    state: any,
    errorsMap: Map<string, ValidationResponse[]>,
  ): Map<string, ValidationResponse[]> {
    let filteredErrorsMap = cloneDeep(errorsMap);
    this.errorsBeingDisplayed = 0;
    const selectedCategory = this.selectedMenuItemId;

    const filteredValidations = this.filteredValidationsForSeverity(
      filteredErrorsMap.get(selectedCategory),
      state,
    );

    filteredErrorsMap?.set(selectedCategory, filteredValidations);

    const filteredValidationsForFilter = this.filterValidationsForSearchString(
      filteredErrorsMap?.get(selectedCategory),
      state,
    );
    filteredErrorsMap.set(selectedCategory, filteredValidationsForFilter);

    const categoryValidationLength =
      this.getCategoryTotalValidationLength(filteredErrorsMap);
    this.errorsBeingDisplayed = categoryValidationLength?.get(selectedCategory);

    if (!filteredErrorsMap?.get(selectedCategory)?.length) {
      filteredErrorsMap?.delete(selectedCategory);
    }
    return filteredErrorsMap;
  }

  filteredValidationsForSeverity(
    validations: ValidationResponse[],
    state: any,
  ) {
    const clonedValidations = cloneDeep(validations);
    if (!state?.showErrors && !state?.showWarnings) {
      return [];
    }
    if (state?.showErrors && state?.showWarnings) {
      return validations;
    }
    clonedValidations?.forEach((validation) => {
      validation.errors = validation.errors?.filter((error) => {
        return (
          (state?.showWarnings && error?.severity === 'WARNING') ||
          (state?.showErrors && error?.severity === 'ERROR')
        );
      });
      return validation;
    });
    const filteredValidations = clonedValidations?.filter(
      (validation) => validation?.errors?.length,
    );
    return filteredValidations;
  }

  filterValidationsForSearchString(
    validations: ValidationResponse[],
    state: any,
  ) {
    const clonedValidations = cloneDeep(validations);

    const searchedString = state?.searchKey?.trim();
    if (searchedString) {
      const filteredValidations = [];
      clonedValidations?.forEach((validation) => {
        const matchingValidation =
          validation?.resourceName
            ?.toLowerCase()
            ?.includes(searchedString?.toLowerCase()) ||
          validation?.directory
            ?.toLowerCase()
            ?.includes(searchedString?.toLowerCase()) ||
          validation?.filename
            ?.toLowerCase()
            ?.includes(searchedString?.toLowerCase());

        if (matchingValidation) {
          filteredValidations?.push(validation);
        } else {
          const errors = validation?.errors?.filter((error) =>
            error?.message?.includes(searchedString),
          );
          validation.errors = errors;
          if (validation?.errors?.length) {
            filteredValidations.push(validation);
          }
        }
      });
      return filteredValidations;
    } else {
      return validations;
    }
  }

  @Output() fetchValidationsEmitter: EventEmitter<any> =
    new EventEmitter<any>();
  fetchValidations() {
    this.fetchValidationsEmitter.emit();
  }

  get menuState(): string {
    return this.errorCategoryMap?.size ? 'visible' : 'hidden';
  }

  getCategoryTotalValidationLength(
    errorCategoryMap: Map<string, ValidationResponse[]>,
  ) {
    const categoryLengthMap = new Map<string, number>();
    Array.from(errorCategoryMap?.keys())?.forEach((errorCategory) => {
      const categoryTotalValidations = errorCategoryMap
        ?.get(errorCategory)
        ?.reduce((acc, currentValue) => {
          return currentValue?.errors?.length + acc;
        }, 0);
      categoryLengthMap.set(errorCategory, categoryTotalValidations);
    });
    return categoryLengthMap;
  }

  ngOnDestroy(): void {
    this.menuItemSubscriber?.unsubscribe();
  }
}
