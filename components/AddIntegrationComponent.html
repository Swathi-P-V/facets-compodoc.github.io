import { HttpErrorResponse } from '@angular/common/http';
import { Component } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Params, Router } from '@angular/router';
import {
  CustomOAuth2ClientRegistration,
  GenericOAuth2ClientRegistration,
} from 'src/app/cc-api/models';
import { UiOAuthControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  HEADING,
  SUBHEADING,
  CLIENT_ID_TOOLTIP,
  CLIENT_SECRET_TOOLTIP,
  ISSUER_URL_TOOLTIP,
  AZURE_AD_PREFIX_ISSUER_URL,
} from './integration';
import { OAUTH_INTEGRATION_WRITE } from 'src/app/constants/permissions';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { getToastSubTexts } from 'src/app/utils/toast.utils';
import { trimObjectValues } from 'src/app/utils/object-utils';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-add-integration',
  templateUrl: './add-integration.component.html',
  styleUrls: ['./add-integration.component.scss'],
})
export class AddIntegrationComponent extends FacetsPage {
  oauthIntegrationWrite = OAUTH_INTEGRATION_WRITE;
  azureADStaticIssuerUrlPrefix = AZURE_AD_PREFIX_ISSUER_URL;
  oauthIntegrationForm: FormGroup = new FormGroup({
    provider: new FormControl('', [Validators.required]),
    registrationId: new FormControl('', [
      Validators.required,
      Validators.pattern('[a-zA-Z0-9-_]*'),
      Validators.minLength(4),
      Validators.maxLength(20),
      Validators.pattern(whiteSpaceInputPattern),
    ]),
    clientId: new FormControl('', [
      Validators.required,
      Validators.pattern(whiteSpaceInputPattern),
    ]),
    clientSecret: new FormControl('', [
      Validators.required,
      Validators.pattern(whiteSpaceInputPattern),
    ]),
    loginButtonText: new FormControl('', [
      Validators.required,
      Validators.pattern('[a-zA-Z0-9 ]*'),
      Validators.minLength(4),
      Validators.maxLength(20),
      Validators.pattern(whiteSpaceInputPattern),
    ]),
    issuerUrl: new FormControl('', [
      Validators.required,
      Validators.pattern(whiteSpaceInputPattern),
    ]),
    scope: new FormControl('email'),
    isSystemConfigured: new FormControl(false),
    authorizationUri: new FormControl(
      '',
      Validators.pattern(whiteSpaceInputPattern),
    ),
    jwkSetUri: new FormControl('', Validators.pattern(whiteSpaceInputPattern)),
    tokenUri: new FormControl('', Validators.pattern(whiteSpaceInputPattern)),
    userInfoUri: new FormControl(
      '',
      Validators.pattern(whiteSpaceInputPattern),
    ),
    clientAuthenticationMethod: new FormControl('POST'),
  });

  providerOptions = [
    { label: 'Google', value: 'GOOGLE' },
    { label: 'Okta', value: 'OKTA' },
    { label: 'OneLogin', value: 'ONE_LOGIN' },
    { label: 'Azure AD', value: 'AZURE_AD' },
    { label: 'OIDC', value: 'OPEN_ID' },
    { label: 'JumpCloud', value: 'JUMP_CLOUD' },
  ];

  clientAuthenticationMethods = [
    { label: 'Post', value: 'POST' },
    { label: 'Basic', value: 'BASIC' },
  ];

  addIntegrationInProgress: boolean = false;
  mode: string = 'create';
  loading: boolean = true;
  integration: CustomOAuth2ClientRegistration;

  heading = HEADING;
  subHeading = SUBHEADING;
  clientIdTooltip = CLIENT_ID_TOOLTIP;
  clientSecretTooltip = CLIENT_SECRET_TOOLTIP;
  issuerUrlTooltip = ISSUER_URL_TOOLTIP;
  customer_cp_host = '';

  constructor(
    activatedRoute: ActivatedRoute,
    private router: Router,
    private oAuthControllerService: UiOAuthControllerService,
    private customToasterService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  get provider() {
    return this.oauthIntegrationForm.get('provider');
  }

  get issuerUrl() {
    return this.oauthIntegrationForm.get('issuerUrl');
  }

  get registrationId() {
    return this.oauthIntegrationForm.get('registrationId');
  }

  get clientId() {
    return this.oauthIntegrationForm.get('clientId');
  }

  get clientSecret() {
    return this.oauthIntegrationForm.get('clientSecret');
  }

  get loginButtonText() {
    return this.oauthIntegrationForm.get('loginButtonText');
  }

  get authorizationUri() {
    return this.oauthIntegrationForm.get('authorizationUri');
  }

  get jwkSetUri() {
    return this.oauthIntegrationForm.get('jwkSetUri');
  }
  get tokenUri() {
    return this.oauthIntegrationForm.get('tokenUri');
  }
  get userInfoUri() {
    return this.oauthIntegrationForm.get('userInfoUri');
  }

  get clientAuthenticationMethod() {
    return this.oauthIntegrationForm.get('clientAuthenticationMethod');
  }

  get scope() {
    return this.oauthIntegrationForm?.get('scope');
  }

  OnPageLoad(p: Params): void {
    this.loading = true;
    if (p?.registrationId) {
      this.mode = 'edit';
      this.getIntegration(p.registrationId);
    } else {
      this.loading = false;
    }
    this.customer_cp_host = window.location.hostname;
  }

  patchOauthIntegrationForm(
    integration:
      | CustomOAuth2ClientRegistration
      | GenericOAuth2ClientRegistration,
  ): void {
    this.oauthIntegrationForm.patchValue({
      provider: integration?.provider,
      registrationId: integration?.registrationId,
      clientId: integration?.clientId,
      clientSecret: integration?.clientSecret,
      loginButtonText: integration?.loginButtonText,
      issuerUrl:
        integration?.provider === 'AZURE_AD'
          ? this.getAzureADIssuerPrefixUrl(integration?.issuerUrl)
          : integration?.issuerUrl,
      authorizationUri: (integration as GenericOAuth2ClientRegistration)
        .authorizationUri,
      jwkSetUri: (integration as GenericOAuth2ClientRegistration)?.jwkSetUri,
      tokenUri: (integration as GenericOAuth2ClientRegistration)?.tokenUri,
      userInfoUri: (integration as GenericOAuth2ClientRegistration)
        ?.userInfoUri,
      clientAuthenticationMethod: (
        integration as GenericOAuth2ClientRegistration
      )?.clientAuthenticationMethod?.value,
    });
    this.registrationId.disable();
    this.provider.disable();
    this.providerChange(this.provider?.value);
  }

  getAzureADIssuerPrefixUrl(issuerUrl) {
    return issuerUrl?.split(this.azureADStaticIssuerUrlPrefix)?.slice(1)?.[0];
  }

  getIntegration(registrationId) {
    this.oAuthControllerService.getAllIntegrationsUsingGET().subscribe(
      (response) => {
        this.integration = response.find(
          (integration) => integration?.registrationId === registrationId,
        );
        this.patchOauthIntegrationForm(this.integration);
        this.loading = false;
      },
      (error) => {
        this.loading = false;
      },
    );
  }

  onSaveOAuthIntegration() {
    if (this.provider?.value === 'OPEN_ID') {
      this.saveOpenIdOAuthIntegration();
      return;
    }

    this.addIntegrationInProgress = true;
    const originalIssuerUrl = this.oauthIntegrationForm?.value?.issuerUrl;
    this.mergeAzureADIssuerUrl();

    const httpReq =
      this.mode === 'create'
        ? this.oAuthControllerService.addIntegrationsUsingPOST(
            trimObjectValues(this.oauthIntegrationForm.value),
          )
        : this.oAuthControllerService.updateIntegrationsUsingPUT({
            client: trimObjectValues(this.oauthIntegrationForm.getRawValue()),
            registrationId: this.registrationId?.value,
          });

    httpReq?.subscribe(
      (response) => {
        this.onSaveIntegrationSuccess();
      },
      (error) => {
        this.onSaveIntegrationFailure(error, originalIssuerUrl);
      },
    );
  }

  saveOpenIdOAuthIntegration() {
    this.addIntegrationInProgress = true;
    const requestPayload = {
      ...this.oauthIntegrationForm?.getRawValue(),
    };
    const httpReq =
      this.mode === 'create'
        ? this.oAuthControllerService?.addGenericOAuthIntegrationUsingPOST(
            requestPayload,
          )
        : this.oAuthControllerService?.editGenericOAuthIntegrationUsingPUT({
            client: this.oauthIntegrationForm.getRawValue(),
            registrationId: this.registrationId?.value,
          });

    httpReq.subscribe(
      () => {
        this.onSaveIntegrationSuccess();
      },
      (error) => {
        this.onSaveIntegrationFailure(error);
      },
    );
  }

  onSaveIntegrationSuccess() {
    this.addIntegrationInProgress = false;
    const { successSubText } = getToastSubTexts(this.mode);
    this.customToasterService.success(
      `${this.provider?.value} Integration has been successfully ${successSubText}.`,
    );
    this.handleBack();
  }

  onSaveIntegrationFailure(error, originalIssuerUrl?) {
    this.addIntegrationInProgress = false;
    this.oauthIntegrationForm.patchValue({
      issuerUrl: originalIssuerUrl,
    });
    this.errorHandler(error);
  }

  mergeAzureADIssuerUrl() {
    if (this.provider?.value == 'AZURE_AD') {
      this.oauthIntegrationForm.patchValue({
        ...this.oauthIntegrationForm.value,
        issuerUrl:
          this.azureADStaticIssuerUrlPrefix +
          this.oauthIntegrationForm?.value?.issuerUrl,
      });
    }
  }

  handleBack = () => {
    this.router.navigate(['settings', 'oauthClients']);
  };

  providerChange(event) {
    if (['GOOGLE', 'OPEN_ID', 'JUMP_CLOUD']?.includes(event)) {
      this.issuerUrl.clearValidators();
    } else {
      this.issuerUrl.setValidators([
        Validators.required,
        Validators.pattern(whiteSpaceInputPattern),
      ]);
    }
    this.issuerUrl.updateValueAndValidity();

    if (event === 'OPEN_ID') {
      this.authorizationUri.addValidators([Validators.required]);
      this.jwkSetUri.addValidators([Validators.required]);
      this.tokenUri.addValidators([Validators.required]);
      this.userInfoUri.addValidators([Validators.required]);
      this.clientAuthenticationMethod.addValidators([Validators.required]);
      this.scope.setValue('openid, profile, email');
    } else {
      this.authorizationUri.removeValidators([Validators.required]);
      this.jwkSetUri.removeValidators([Validators.required]);
      this.tokenUri.removeValidators([Validators.required]);
      this.userInfoUri.removeValidators([Validators.required]);
      this.clientAuthenticationMethod.removeValidators([Validators.required]);
      this.scope?.setValue('email');
    }
  }

  errorHandler(error: HttpErrorResponse) {
    const { failureSubText } = getToastSubTexts(this.mode);
    this.onAPIErrorService.onAPIError(
      error?.error?.message || `Failed to ${failureSubText} oauth integration`,
      error,
    );
  }
}
