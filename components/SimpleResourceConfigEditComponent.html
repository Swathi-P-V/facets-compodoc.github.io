import { Component, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import {
  MonacoDiffEditorConstructionOptions,
  MonacoEditorConstructionOptions,
} from '@materia-ui/ngx-monaco-editor';
import mixpanel from 'mixpanel-browser';
import { OverrideObject } from 'src/app/cc-api/models';
import {
  UiApplicationControllerService,
  UiCommonClusterControllerService,
} from 'src/app/cc-api/services';
import { SaveChangesFooterComponent } from 'src/app/components/save-changes-footer/save-changes-footer.component';
import { ENVIRONMENT_CONFIGURE } from 'src/app/constants/permissions';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { mergeBlueprintAndOverrideConfig } from 'src/app/utils/app-config-utils';
import { arrayInObject } from 'src/app/utils/object-utils';
import { findDifferences } from './config-diff-utils';
import { ActionEventEmitterService } from 'src/app/services/action-event-emitter.service';
import { actionEvents } from '../../../../../constants/action-events';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
import {
  ENVIRONMENT_SPEC_OVERRIDES_SAVE_FAIL,
  ENVIRONMENT_SPEC_OVERRIDES_SAVE_SUCCESS,
  USER_RESOURCE_GROUP_CREATE_FAIL,
} from 'src/app/constants/analytics-track-events';
const emptyIcon = './assets/empty-override.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-simple-resource-config-edit',
  templateUrl: './simple-resource-config-edit.component.html',
  styleUrls: ['./simple-resource-config-edit.component.scss'],
})
export class SimpleResourceConfigEditComponent extends FacetsPage {
  objectKeys = Object.keys;
  editorOptions: MonacoEditorConstructionOptions = {
    theme: 'vs-light',
    language: 'json',
    minimap: {
      enabled: false,
    },
  };

  disabledEditorOptions: MonacoEditorConstructionOptions = {
    theme: 'vs-light',
    language: 'json',
    readOnly: true,
    minimap: {
      enabled: false,
    },
  };

  diffEditorOptions: MonacoDiffEditorConstructionOptions = {
    theme: 'vs-light',
    readOnly: true,
    minimap: {
      enabled: false,
    },
  };

  modifiedCode: string;
  originalData: OverrideObject;
  originalCode: string;
  updatedData: any = {};
  disabled: boolean;
  clusterId: any;
  appName: any;
  stackName: any;
  resourceType: any;
  fetchingOverrides: boolean;
  icon = emptyIcon;
  modified = false;
  submittingOverrides = false;
  appData: any;
  fetchingAppData: boolean;
  arrayInObject: boolean = false;
  environmentConfigurePermission = ENVIRONMENT_CONFIGURE;
  appJson: any;

  @ViewChild(SaveChangesFooterComponent)
  private footerComponent: SaveChangesFooterComponent;

  constructor(
    _route: ActivatedRoute,
    private clusterService: UiCommonClusterControllerService,
    private router: Router,
    private appService: UiApplicationControllerService,
    private actionEventEmitterService: ActionEventEmitterService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p.clusterId;
    this.appName = p.resourceName;
    this.stackName = p.stackName;
    this.resourceType = p.resourceType;
    this.fetchingOverrides = true;
    this.modifiedCode = undefined;
    this.originalCode = undefined;
    this.originalData = undefined;
    this.appData = undefined;
    this.appJson = undefined;
    this.fetchingAppData = true;
    this.appService
      .getResourceByNameV2UsingGET({
        clusterId: this.clusterId,
        resourceName: this.appName,
        resourceType: this.resourceType,
      })
      .subscribe(
        (data) => {
          this.appJson = data;
          this.arrayInObject = arrayInObject(this.appJson);
          this.appData = JSON.stringify(data, null, '\t');
          this.fetchingAppData = false;
          this.appService
            .getResourceOverrideObjectUsingGET({
              clusterId: this.clusterId,
              resourceName: this.appName,
              resourceType: this.resourceType,
            })
            .subscribe(
              (resourceData) => {
                this.originalData = resourceData;
                const merged = mergeBlueprintAndOverrideConfig(
                  this.appJson || {},
                  this.originalData?.overrides || {},
                );
                this.setEditorCode(merged);
                this.updatedData = resourceData?.overrides || {};
                this.fetchingOverrides = false;
              },
              (err) => {
                this.icon = errorIcon;
                this.fetchingOverrides = false;
                if (err?.status === 404) {
                  const merged = mergeBlueprintAndOverrideConfig(
                    this.appJson || {},
                    this.originalData?.overrides || {},
                  );
                  this.setEditorCode(merged);
                  this.updatedData = {};
                }
              },
            );
        },
        (err) => {
          this.fetchingAppData = false;
          this.fetchingOverrides = false;
        },
      );
  }

  setEditorCode(merged) {
    this.modifiedCode = this.arrayInObject
      ? JSON.stringify(this.originalData?.overrides, null, '\t')
      : JSON.stringify(merged, null, '\t');
    this.originalCode = this.arrayInObject
      ? JSON.stringify(this.originalData?.overrides, null, '\t')
      : JSON.stringify(merged, null, '\t');
  }

  updateOverrides(content) {
    try {
      if (content !== this.modifiedCode) {
        const json = JSON.parse(content);
        this.updatedData = json;
        this.modified = true;
      }
    } catch (error) {}
  }

  submitDisabled() {
    try {
      JSON.parse(this.modifiedCode);
      return false;
    } catch (error) {
      return true;
    }
  }

  submitOverrides = (ref: any) => {
    try {
      if (!this.originalData) {
        this.originalData = {
          resourceName: this.appName,
          resourceType: this.resourceType,
        };
      } else {
        this.originalData.resourceName = this.appName;
        this.originalData.resourceType = this.resourceType;
      }
      this.originalData.overrides = this.arrayInObject
        ? this.updatedData
        : findDifferences(
            JSON.parse(this.appData),
            JSON.parse(this.modifiedCode),
          );
      this.submittingOverrides = true;
      this.appService
        .postResourceOverrideObjectUsingPOST({
          resourceName: this.appName,
          resourceType: this.resourceType,
          overrideRequest: this.originalData,
          clusterId: this.clusterId,
        })
        .subscribe(
          (data) => {
            this.submittingOverrides = false;
            mixpanel.track(ENVIRONMENT_SPEC_OVERRIDES_SAVE_SUCCESS);
            this.modified = false;
            this.customToastService.success(
              'Updated cluster variables, changes will be applied after next release',
            );
            this.sendOverrideCompleteEvent();
            ref.close();
            this.handleBack();
          },
          (err) => {
            mixpanel.track(ENVIRONMENT_SPEC_OVERRIDES_SAVE_FAIL, {
              error: err,
            });
            this.submittingOverrides = false;
            this.onAPIErrorService.onAPIError(
              err?.error?.message || 'Failed to update the changes',
              err,
            );
          },
        );
    } catch (err) {
      this.submittingOverrides = false;
      this.customToastService.error('Update Failed: ' + err?.error?.message);
      ref.close();
    }
  };

  sendOverrideCompleteEvent() {
    this.actionEventEmitterService.emitActionEvent(
      actionEvents?.ON_SAVE_ENV_OVERRIDES,
    );
  }

  handleDelete() {
    this.clusterService
      .deleteOverridesUsingDELETE({
        resourceName: this.appName,
        resourceType: this.resourceType,
        clusterId: this.clusterId,
      })
      .subscribe(
        (data) => {
          console.log(data);
        },
        (err) => {
          console.log(err);
        },
      );
  }

  handleBack = () => {
    this.router.navigate(['../'], {
      relativeTo: this.route,
    });
  };
  onDiscard = (ref: any) => {
    ref.close('discard');
    this.modifiedCode = this.originalCode;
    this.modified = false;
  };
  async canDeactivate(): Promise<boolean> {
    if (!this.modified) {
      return true;
    }

    return this.footerComponent.onDeactivate();
  }
}
