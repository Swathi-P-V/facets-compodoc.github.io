import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { UiChatGptControllerService } from '../../../../cc-api/services/ui-chat-gpt-controller.service';
import cloneDeep from 'lodash/cloneDeep';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { getChatMessagesWithAliveContext } from './k8s-chat-details.utils';

@Component({
  selector: 'app-chat-detail',
  templateUrl: './k8s-chat-detail.component.html',
  styleUrls: ['./k8s-chat-detail.component.scss'],
})
export class K8SChatDetailComponent extends FacetsPage {
  suggestions = [
    {
      label: 'Analyze Cluster',
      filter: false,
      icon: 'kubernetes-blue',
    },
    {
      label: 'Analyze Pods',
      filter: 'Pod',
      icon: 'pod',
    },
    {
      label: 'Analyze ReplicaSets',
      filter: 'ReplicaSet',
      icon: 'replica-set',
    },
    {
      label: 'Analyze Deployments',
      filter: 'Deployment',
      icon: 'deploy',
    },
    {
      label: 'Analyze PVC',
      filter: 'PersistentVolumeClaim',
      icon: 'pvc',
    },
  ];
  chatMessages: any[] = [];
  typing: boolean = false;
  error = '';
  clusterId: string;

  roboImage = './assets/robot.png';
  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private uiChatGptControllerService: UiChatGptControllerService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p.clusterId;
    this.chatMessages = [];
  }

  handleBack = () => {
    this.router.navigate(['../'], {
      relativeTo: this.activatedRoute,
    });
  };

  getAnalyzeK8sGptAnswer(question) {
    this.chatMessages.push({
      content: question.label,
      role: 'user',
      type: 'k8sgpt-cluster-issues',
    });
    this.typing = true;
    this.uiChatGptControllerService
      .analyzeKubernetesClusterUsingGET({
        clusterId: this.clusterId,
        filters: question?.filter ? [question.filter] : [''],
      })
      .subscribe(
        (res) => {
          this.typing = false;
          this.chatMessages.push({
            type: 'k8sgpt-cluster-issues',
            customMessageData: res?.results,
            reply: false,
            date: new Date(),
            role: 'assistant',
          });
        },
        (error) => {
          this.typing = false;
          this.chatMessages.push({
            type: 'error',
            customMessageData: this.getErrorMessage(error),
            reply: false,
            date: new Date(),
            role: 'assistant',
          });
        },
      );
  }

  chatWithK8s($event) {
    this.chatMessages.push({
      content: $event.message,
      role: 'user',
      type: 'k8s-command-results',
    });
    this.typing = true;

    const chatsAfterCompleteStatus = getChatMessagesWithAliveContext(
      this.chatMessages,
    );

    this.uiChatGptControllerService
      .getK8sChatsUsingPOST({
        clusterId: this.clusterId,
        questions: chatsAfterCompleteStatus?.map((message) => ({
          content: message?.content || message?.apiContent || '',
          role: message?.role,
        })),
      })
      .subscribe(
        (res) => {
          this.typing = false;
          this.chatMessages.push({
            type: 'k8s-command-results',
            customMessageData: res,
            reply: false,
            date: new Date(),
            role: 'assistant',
            apiContent: res?.content,
          });
        },
        (error) => {
          this.typing = false;
          this.chatMessages.push({
            type: 'error',
            content: this.getErrorMessage(error),
            reply: false,
            date: new Date(),
            role: 'assistant',
          });
        },
      );
  }

  getErrorMessage(error) {
    return error.status === 403
      ? 'User does not have permission to perform this operation'
      : error?.message ||
          error?.error?.message ||
          'Sorry, something went wrong! Please try again after sometime.';
  }
}
