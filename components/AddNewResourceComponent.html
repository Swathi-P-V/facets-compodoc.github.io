import {
  Component,
  EventEmitter,
  Input,
  Output,
  TemplateRef,
  ViewChild,
} from '@angular/core';
import mixpanel from 'mixpanel-browser';
import { UiBlueprintDesignerControllerService } from 'src/app/cc-api/services';
import {
  intentCategorypMapping,
  intentsCategoryOrder,
} from '../blueprint-designer.constants';
import { KeyValue } from '@angular/common';
import cloneDeep from 'lodash/cloneDeep';
import debounce from 'lodash/debounce';
import groupBy from 'lodash/groupBy';
import { sanitizeString } from 'src/app/utils/string-util';
import { NbDialogService, NbToastrService } from '@nebular/theme';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params, Router } from '@angular/router';
import {
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';

import { BlueprintDesignerGraphService } from './../blueprint-designer-graph/blueprint-designer-graph.service';
import { BlueprintDesignerTableService } from './../blueprint-designer-table/blueprint-designer-table.service';
import { IntentDetailsService } from 'src/app/services/intent-details.service';
import { getFlavorName, processIntentsData } from './add-new-resource.utils';

  BLUEPRINT_ADD_RESOURCE_FAIL,
  BLUEPRINT_ADD_RESOURCE_SUCCESS,
} from 'src/app/constants/analytics-track-events';

@Component({
  selector: 'app-add-new-resource',
  templateUrl: './add-new-resource.component.html',
  styleUrls: ['./add-new-resource.component.scss'],
})
export class AddNewResourceComponent extends FacetsPage {
  @Input() blueprintVscType: string;
  @Input() primaryCloud: 'AWS' | 'AZURE' | 'GCP' | 'KUBERNETES';
  blueprint: string;
  branchName: string;
  searchKey: string;
  allResourceIntents: any[];
  filtered_ResourceIntents: any[];
  allIntentGroups = {};
  fetchingIntents: boolean = false;
  fetchingIntentsErrorMessage: string = null;
  errorStateIcon = errorStateIcon;
  currentRouteMode: 'TABLE' | 'GRAPH' = 'TABLE';

  resourceForm: FormGroup = this.fb.group({
    name: new FormControl('', [
      Validators.required,
      Validators.minLength(3),
      Validators.maxLength(40),
      Validators.pattern('^[a-z0-9-]+$'),
    ]),
    vcsUrl: new FormControl('', [Validators.pattern(whiteSpaceInputPattern)]),
    relativePath: new FormControl('', [
      Validators.pattern(whiteSpaceInputPattern),
    ]),
    flavor: new FormControl('', [Validators.required]),
  });

  get resourceName() {
    return this.resourceForm.get('name');
  }
  get vcsUrl() {
    return this.resourceForm.get('vcsUrl');
  }
  get relativePath() {
    return this.resourceForm.get('relativePath');
  }
  get flavor() {
    return this.resourceForm?.get('flavor');
  }

  @Output() addedResource: EventEmitter<any> = new EventEmitter<any>();

  intentSelectDialog;
  constructor(
    private blueprintDesignerService: UiBlueprintDesignerControllerService,
    private uiBlueprintDesignerControllerService: UiBlueprintDesignerControllerService,
    private dialogService: NbDialogService,
    private router: Router,
    private fb: FormBuilder,
    route: ActivatedRoute,
    private customToast: CustomToastService,
    public blueprintDesignerGraphService: BlueprintDesignerGraphService,
    public blueprintDesignerTableService: BlueprintDesignerTableService,
    private intentDetailsService: IntentDetailsService,
  ) {
    super(route);
  }

  OnPageLoad(p: Params) {
    this.blueprint = p?.stackName;
    this.branchName = p?.branchName;
    this.getAllResourceIntents();
    this.blueprintDesignerTableService.selectedEvent.subscribe((eventType) => {
      this.currentRouteMode = eventType;
    });
    this.blueprintDesignerGraphService.selectedEvent.subscribe((eventType) => {
      this.currentRouteMode = eventType;
    });
  }

  // Fetches all the resource types which Facets provides
  getAllResourceIntents() {
    this.fetchingIntentsErrorMessage = null;
    if (!this.intentDetailsService?.intentLoaded) {
      this.intentDetailsService.updateIntentDataSubject.subscribe(
        (intentStatus) => {
          if (intentStatus) {
            this.loadIntentData();
          }
        },
      );
    } else {
      this.loadIntentData();
    }
  }

  loadIntentData() {
    const intentError = this.intentDetailsService?.getIntentError();
    const intentResponse = this.intentDetailsService?.getAllIntents();
    this.fetchIntentsFromAPI(intentResponse, intentError);
  }

  fetchIntentsFromAPI(intentResponse, intentError) {
    if (!intentError) {
      this.allResourceIntents = processIntentsData(intentResponse);
      this.groupIntentsByCategory(this.allResourceIntents);
    } else {
      this.fetchingIntentsErrorMessage =
        intentError?.error?.message ||
        'There was an error fetching the intents, please try later';
    }
  }

  public onCompare(
    _left: KeyValue<any, any>,
    _right: KeyValue<any, any>,
  ): number {
    return 1;
  }

  groupIntentsByCategory(allIntents) {
    const intentsCopy = cloneDeep(allIntents);
    const groupByCategory = groupBy(intentsCopy, (intent) => intent.category);
    this.allIntentGroups = {};
    intentsCategoryOrder.forEach((category) => {
      if (groupByCategory[category]?.length) {
        this.allIntentGroups[category] = groupByCategory[category];
      }
    });
  }

  // Filter resources based on search input
  filterResources = debounce(() => {
    const searchValue = sanitizeString(this.searchKey);
    const filteredResources = this.allResourceIntents.filter((resourceType) => {
      if (
        sanitizeString(resourceType?.displayName)?.includes(searchValue) ||
        sanitizeString(resourceType?.intent)?.includes(searchValue) ||
        resourceType?.category?.includes(searchValue)
      ) {
        return true;
      }
      return false;
    });
    this.groupIntentsByCategory(filteredResources);
  }, 100);

  onClearSearch() {
    this.searchKey = '';
    this.groupIntentsByCategory(this.allResourceIntents);
  }

  @ViewChild('selectResourceDialog') selectResourceDialog: TemplateRef<any>;

  openSelectResourceDialog() {
    this.intentSelectDialog = this.dialogService.open(
      this.selectResourceDialog,
      {
        closeOnBackdropClick: false,
        hasBackdrop: true,
      },
    );
  }

  resourceDetails: any;
  creatingResource: boolean = false;
  creatingMode: 'CREATE' | 'CREATE_AND_VIEW';
  createResource(
    resourceDetails: any,
    intent: any,
    ref: any,
    redirect?: boolean,
  ) {
    this.creatingResource = true;
    this.creatingMode = redirect ? 'CREATE_AND_VIEW' : 'CREATE';
    let resourceFileRequest = [
      {
        mergeContent: {},
        resourceName: this.resourceName.value,
        resourceType: intent?.intent,
        flavor: this.flavor?.value,
      },
    ];
    if (
      intent?.intent === 'service' &&
      this.vcsUrl?.value &&
      this.relativePath?.value
    ) {
      resourceFileRequest[0].mergeContent = {
        metadata: {
          git: {
            repository_url: this.vcsUrl.value,
            path: this.relativePath?.value || '.',
          },
        },
      };
    }
    this.blueprintDesignerService
      .createResourcesUsingPOST({
        resourceFileRequests: resourceFileRequest,
        branch: this.branchName,
        stackName: this.blueprint,
      })
      .subscribe(
        () => {
          this.creatingResource = false;
          mixpanel.track(BLUEPRINT_ADD_RESOURCE_SUCCESS);
          this.customToast.success(
            'New resource has been successfully added to the blueprint.',
          );
          this.addedResource.emit({
            resourceName: this.resourceName.value,
            resourceType: intent?.intent,
          });
          ref?.close();

          if (redirect) {
            this.redirectToResourceOverview({
              resourceName: this.resourceName.value,
              resourceType: intent?.intent,
            });
          } else {
            this.router.navigate([
              '/',
              'stack',
              this.blueprint,
              'blueprint-designer',
              this.branchName,
              'resource-list',
            ]);
          }
        },
        (error) => {
          this.creatingResource = false;
          mixpanel.track(BLUEPRINT_ADD_RESOURCE_FAIL, { error });
          this.customToast.error(
            error?.error?.message ||
              `Resource ${this.resourceName.value} could not be added`,
          );
        },
      );
  }

  @ViewChild('resourceNameDialog') resourceNameDialog: TemplateRef<any>;
  openResourceNameDialog(selectedIntent, selectedFlavor?) {
    if (!['traefik', 'i3'].includes(selectedIntent?.intent)) {
      this.resourceForm.reset();
      setTimeout(() => {
        this.resourceDetails = {
          name: '',
        };
        const flavorOptions = selectedIntent?.flavors?.map((flavor) => {
          const flavorName = getFlavorName(flavor?.name);
          return {
            label: flavorName,
            value: flavorName,
            clouds: flavor?.clouds,
          };
        });
        this.flavor.setValue(
          getFlavorName(selectedFlavor?.name) ||
            getFlavorName(selectedIntent?.flavors?.[0]?.name),
        );
        this.dialogService.open(this.resourceNameDialog, {
          closeOnBackdropClick: false,
          hasBackdrop: true,
          context: {
            intent: selectedIntent,
            resourceDetails: this.resourceDetails,
            flavorOptions,
          },
        });
      }, 10);
    }
  }

  // Redirect the user to the resource overview
  // page. This might change for the edit button later.
  redirectToResourceOverview(resource: any) {
    this.router.navigate(
      [
        `./resource/type/${resource.resourceType}/name/${resource.resourceName}/spec`,
      ],
      {
        relativeTo: this.route,
        state: {
          mode: this.currentRouteMode || 'TABLE',
        },
      },
    );
  }
  onClickIntent(intent, ref, flavor?) {
    if (flavor) {
      // If multiple flavors present and user selects one of them
      this.intentSelectDialog.close();
      ref?.close();
      this.openResourceNameDialog(intent, flavor);
    }

    // If only one flavour is present, then user is not asked to choose the flavour, so flavor won't be passed here
    if (intent?.flavors?.length < 2 && !flavor) {
      ref?.close();
      this.openResourceNameDialog(intent, flavor);
    }
  }
}
