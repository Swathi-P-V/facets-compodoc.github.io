  constructor(
    private _route: ActivatedRoute,
    private uiDropdownsControllerService: UiDropdownsControllerService,
    private router: Router,
    private stackService: UiStackControllerService,
    private uiResourceGroupController: UiResourceGroupControllerService,
    private uiCommonClusterControllerService: UiCommonClusterControllerService,
    private tableStatefulFiltersService: TableStatefulFiltersService,
    private bulkEnableDisableResourceService: BulkEnableDisableResourcesService,
    private bulkSelectTableEntriesService: BulkSelectTableEntriesService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(_route);

    // Subscribe to data changes in ng2-smart-table's 'LocalDataSource'(this.source).
    // Update 'filteredResources' and recalculate selection state on each change.
    this.ng2SourceSubscription = this.source.onChanged().subscribe((change) => {
      this.filteredResources = change.elements;
      this.setSelectAllCheckboxState();
    });

    // Subscribe to the 'bulkResourceOpsComplete$' observable which emits status of completion of bulk resource operations.
    this.bulkResourceOpsSubscription =
      this.bulkEnableDisableResourceService.bulkResourceOpsComplete.subscribe(
        (status) => {
          this.onBulkEnableDisableResourceComplete(status);
        },
      );
  }

  OnPageLoad(p: Params): void {
    this.resources = [];
    this.showEmptyState = false;
    this.tableData = [];
    this.showSubstack = false;
    if (p?.stackName && p?.clusterId) {
      this.stackName = p?.stackName;
      this.secondaryMsg = `Please add resources for ${this.stackName} blueprint`;
      this.clusterId = p?.clusterId;
      this.resources = {};
      this.fetchResourceAndCluster();
    }
    this.initStatefulTableService();
  }

  fetchResourceAndCluster() {
    this.fetchingResources = true;
    Promise.allSettled([
      this.getAllResources(this.clusterId),
      this.getEnvironmentConfig(),
      this.getAllResourceGroups(),
    ]).then((data) => {
      this.fetchingResources = false;
      this.initResourceMapper();
      if (this.resources?.length > 0) {
        this.prepareTableData(this.resources);
      } else {
        this.showEmptyState = true;
      }
    });
  }

  initStatefulTableService() {
    const initiator: StatefulFiltersInitiator = {
      tableConfig: this.tableSettings,
      tableName: 'resource-center',
      tableSource: this.source,
      updateQueryParams: true,
      sessionPersistence: false,
    };
    this.tableStatefulFiltersService.initializeRecording(initiator);
  }

  async getAllResourceGroups() {
    return new Promise((resolve) => {
      this.uiResourceGroupController
        .findAllUsingGET()
        .subscribe((resourceGrpInfo: any) => {
          this.allResourceGroups = resourceGrpInfo;
          this.prepareResourceGroupNameToIDMap();
        })
        .add(() => {
          resolve(true);
        });
    });
  }

  prepareResourceGroupNameToIDMap() {
    this.resourceGroupNameToIDMap?.clear();
    this.allResourceGroups.forEach((resourceGroup) => {
      this.resourceGroupNameToIDMap.set(resourceGroup.name, resourceGroup.id);
    });
  }

  getAllResources(clusterId) {
    this.showEmptyState = false;
    return new Promise((resolve, reject) => {
      this.uiDropdownsControllerService
        .getAllResourcesByClusterUsingGET({
          clusterId: clusterId,
          includeContent: false,
          includeSubstack: true,
        })
        .subscribe(
          (resources) => {
            this.resources = resources.filter(
              (resource) => resource?.resourceType !== 'UNKNOWN',
            );
            // Add selected field for all the resources
            this.resources.forEach((resource) => {
              resource['selected'] = false;
              resource[
                'id'
              ] = `${resource.resourceType}:${resource.resourceName}`;
            });
            resolve(true);
          },
          (error) => {
            this.resources = [];
            this.showEmptyState = true;
            this.fetchingResources = false;
            this.secondaryMsg =
              error?.error?.message ||
              ' There is some error fetching resources';
            this.primaryMsg = '';
            this.emptyIcon = errorIcon;
            this.status = 'error';
            reject(error);
          },
        );
    });
  }

  getEnvironmentConfig() {
    return new Promise((resolve, reject) => {
      this.uiCommonClusterControllerService
        .getClusterCommonUsingGET(this.clusterId)
        .subscribe(
          async (r) => {
            this.cluster = r;
            resolve(true);
          },
          (error) => {
            this.cluster = null;
            reject(error);
          },
        );
    });
  }

  // Takes in the data received from the getAllResourceGroups API
  // and prepares a map which will include all the resource groups for
  // all the resources
  // key => (resourceType: resourceName), value => (resourceGroups[])
  initResourceMapper() {
    this.resourceMapper = new Map<string, string[]>();
    this.allResourceGroups?.forEach((resourceGroup) => {
      resourceGroup?.resourceInfo?.forEach((stack) => {
        if (stack?.stackName?.toLowerCase() === this.stackName?.toLowerCase()) {
          let resourceGroupName = resourceGroup?.name;
          stack?.resources?.forEach((resourceType) => {
            let resourceTypeVal = resourceType?.resourceType;
            resourceType?.resourceNames?.forEach((resource) => {
              let resourceID = this.getResourceID(resourceTypeVal, resource);
              if (!this.resourceMapper.get(resourceID)) {
                this.resourceMapper.set(resourceID, []);
              }
              this.resourceMapper.get(resourceID).push(resourceGroupName);
            });
          });
        }
      });
    });
  }

  getResourceID(resourceType, resourceName) {
    return resourceType + ':' + resourceName;
  }

  prepareTableData(resources) {
    this.resourceData = [];
    this.resourceData = getResourceTableData(
      resources,
      this.resourceMapper,
      this.cluster,
    );
    this.setTableData();
  }

  setTableData() {
    let tempTableData = this.resourceData.filter(
      (resource) => !resource?.substack,
    );
    this.setTableDataAndFilters(tempTableData);
  }

  setTableDataAndFilters(tempTableData) {
    this.buildResourceFilter(tempTableData, 'resourceType');
    this.buildResourceGroupsFilter(tempTableData);
    this.tableData = [...tempTableData];
    this.source.load([...tempTableData]);
    this.source.setFilter([]);
  }

  /*
   * Build a filter list of resource name/type from available resources name/type
   * Create a set to ensure unique values
   * sort the set values by converting to array
   * map the set array to create filter options
   * set the options to settings
   */
  buildResourceFilter(resourceData, filterType) {
    let resourceFilterData: {
      title;
      value;
    }[] = Array.from(
      new Set(
        resourceData?.map((resource) =>
          filterType === 'resourceType'
            ? resource?.resourceType
            : resource?.resourceName,
        ),
      ),
    )
      .sort()
      .map((name) => {
        return {
          title: name,
          value: name,
        };
      });
    this.tableSettings.columns[filterType].filter.config.list = [
      ...resourceFilterData,
    ];
    this.tableSettings = { ...this.tableSettings };
  }

  buildResourceGroupsFilter(resourceData) {
    let resourceGroupFilterData = [];
    let visitedResourceGroups = new Set<string>();
    resourceData.forEach((resource) => {
      let allResourceGroups = this.resourceMapper.get(
        this.getResourceID(resource.resourceType, resource.resourceName),
      );
      allResourceGroups?.forEach((resGrp) => {
        if (!visitedResourceGroups.has(resGrp)) {
          visitedResourceGroups.add(resGrp);
        }
      });
    });
    resourceGroupFilterData = Array.from(visitedResourceGroups).sort();
    resourceGroupFilterData = resourceGroupFilterData.map((resGrp) => ({
      title: resGrp,
      value: resGrp,
    }));
    this.tableSettings.columns.resourceGroups.filter.config.list = [
      ...resourceGroupFilterData,
    ];
    this.tableSettings = { ...this.tableSettings };
  }

  getLink(resourceData) {
    const url = `type/${resourceData?.resourceType}/resource/${resourceData?.resourceName}/overview`;
    return url;
  }

  listenToActionButtonEvents(event) {
    let clickedButton = this.actionButtons.filter((button) => {
      return button?.id === event;
    })[0];
    clickedButton?.onClick();
  }

  refreshStack() {
    let gitSyncButton = this.actionButtons[0];
    gitSyncButton.loading = true;
    this.stackService.reloadStackUsingGET1(this.stackName).subscribe(
      (stack) => {
        gitSyncButton.loading = false;
        this.customToastService.success(
          'Git Sync Initiated. Refresh page in a few seconds to view updates.',
        );
        this.fetchResourceAndCluster();
      },
      (error) => {
        gitSyncButton.loading = false;
        this.onAPIErrorService.onAPIError(
          error?.error?.message || 'Syncing with git failed',
          error,
        );
      },
    );
  }

  onClickShowSubstackChange(event) {
    this.bulkEnableDisableResourceService.sendFooterData({
      selectedEntitiesForBulkOps: [],
      statusOfSelectedEntities: '',
    });
    this.updateTableColumnsBasedOnSubstackVisibility(event);
    event
      ? this.setTableDataAndFilters(this.resourceData)
      : this.setTableData();
  }

  onSelectAllCheckboxClicked(checked) {
    // Update isSelectAll
    this.isSelectAll = checked;

    //Update the selected property of all resources which are currently loaded in table
    let { updatedSource, filteredEntities } =
      this.bulkSelectTableEntriesService.updateSelectedPropOnSelectAllClick(
        this.source,
        this.filteredResources,
        checked,
        'id',
      );
    this.source.load([...updatedSource]);
    this.filteredResources = cloneDeep(filteredEntities);

    // Update the page footer based on currentSelectedEntities
    this.updateFooterState();
  }

  onCheckboxClicked(data) {
    // Update the selected property of resource whose checkbox is clicked
    let updatedSourceData =
      this.bulkSelectTableEntriesService.updateSelectedPropOnCheckboxClick(
        data,
        this.source,
        this.bulkEnableDisableResourceService.findResource(data),
      );
    this.source.load([...updatedSourceData]);

    // Determine the state of 'selectAll' checkbox
    this.setSelectAllCheckboxState();

    // Update the page footer based on currentSelectedEntities
    this.updateFooterState();
  }

  updateFooterState() {
    let selectedEntitiesState =
      this.bulkSelectTableEntriesService.determineSelectedEntityState(
        this.source,
        'status',
      );
    this.bulkEnableDisableResourceService.sendFooterData(selectedEntitiesState);
  }

  setSelectAllCheckboxState() {
    const { isIndeterminate, isSelectAll } =
      this.bulkSelectTableEntriesService.getSelectAllCheckboxState(
        this.source,
        'id',
        this.filteredResources,
      );
    this.isIndeterminate = isIndeterminate;
    this.isSelectAll = isSelectAll;
  }

  handleCancel() {
    this.isIndeterminate = false;
    this.isSelectAll = false;
    if (this.showSubstack) {
      this.tableData = [...this.resourceData]; //Not sure why are we setting this
      this.source.load([...this.resourceData]);
    } else {
      let tempTableData = this.resourceData.filter(
        (resource) => !resource?.substack,
      );
      this.tableData = [...tempTableData]; //Not sure why are we setting this
      this.source.load([...tempTableData]);
    }
  }

  onBulkEnableDisableResourceComplete(complete) {
    // If resources are enabled/disabled successfully then, reload the page, set selected resource to empty
    this.isIndeterminate = false;
    this.isSelectAll = false;
    complete ? this.fetchResourceAndCluster() : this.handleCancel();
  }

  updateTableColumnsBasedOnSubstackVisibility(event) {
    const columns = this.tableSettings.columns || {};

    // If showSubstack is true, add substack as the first column
    if (event) {
      this.tableSettings.columns = {
        substack: {
          title: '',
          width: '3%',
          filter: false,
          type: 'custom',
          valuePrepareFunction: (cell, row) => {
            return JSON.stringify({
              tooltipText: cell ? 'Substack Name: ' + row?.stackName : '',
              imgLink: cell ? this.resourceIcon : '',
            });
          },
          renderComponent: SmartTableIconTooltipComponent,
        },
        ...columns,
      };
    } else {
      // If showSubstack is false, remove the substack column
      delete this.tableSettings.columns['substack'];
    }
  }
  ngOnDestroy() {
    this.bulkResourceOpsSubscription.unsubscribe();
    this.ng2SourceSubscription.unsubscribe();
  }
}
