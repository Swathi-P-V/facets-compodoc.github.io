import { Component } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { map } from 'rxjs';
import mixpanel from 'mixpanel-browser';
import {
  UiCustomRoleControllerService,
  UiDropdownsControllerService,
} from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { Permission } from '../user-roles.component';
import { whiteSpaceInputPattern } from './../../../../constants/inputPatterns';
import { CUSTOM_ROLE_WRITE } from 'src/app/constants/permissions';
import {
  ENVIRONMENT_LAUNCH,
  ENVIRONMENT_DESTROY,
  RELEASE_FULL,
  RELEASE_FULL_ALLOW_DESTROY,
  RELEASE_SELECTIVE,
  RELEASE_SELECTIVE_ALLOW_DESTROY,
  RELEASE_CUSTOM,
  RELEASE_CUSTOM_ALLOW_DESTROY,
  RELEASE_SCALE_UP,
  RELEASE_SCALE_DOWN,
} from 'src/app/constants/permissions';
import { sortPermissionsInPermissionGroups } from 'src/app/utils/permissions/permissions.utils';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
import {
  USER_ROLE_CREATE_FAIL,
  USER_ROLE_CREATE_SUCCESS,
  USER_ROLE_EDIT_FAIL,
  USER_ROLE_EDIT_SUCCESS,
} from 'src/app/constants/analytics-track-events';
const errorStateIcon = './assets/error.svg';

@Component({
  selector: 'app-create-edit-custom-role',
  templateUrl: './create-edit-custom-role.component.html',
  styleUrls: ['./create-edit-custom-role.component.scss'],
})
export class CreateEditCustomRoleComponent extends FacetsPage {
  mode: 'Edit' | 'Create' = 'Create';

  // Role(s)
  origRole: any; // Original copy of the role
  role: any; // The role for which the component is invoked
  fetchingRoles: boolean = false;
  fetchingRolesError: boolean = false;
  secondaryMessage: string;
  noDataIcon: any;
  allRoles: any[]; // Comes from either the custom roles table component or is fetched from the BE

  // Basic information form
  basicInfoForm: FormGroup;
  basicInfoFormKeys: string[];

  // Permissions form
  allPermissions: Permission[];
  permissionGroups: Map<string, Permission[]> = new Map();
  permissionGroupsArray: Array<{ categoryLabel: any; categoryName: any }> = [];
  permissionsFormError: boolean = false;
  permissionsFormErrorMessage: boolean = false;
  fetchingPermissions: boolean = false;
  permissionsForm: FormGroup;
  permissionsFormKeys: string[];

  // Clone permissions
  fetchingClonePermissions: boolean = false;
  copyRole: any;
  cloneRequired: boolean = false;
  clonePermissions: Permission[];
  populateBasicInfo: boolean = false;
  customRoleWrite = CUSTOM_ROLE_WRITE;

  releasePermissionsOrder = [
    RELEASE_FULL,
    RELEASE_SELECTIVE,
    RELEASE_CUSTOM,
    RELEASE_FULL_ALLOW_DESTROY,
    RELEASE_SELECTIVE_ALLOW_DESTROY,
    RELEASE_CUSTOM_ALLOW_DESTROY,
    RELEASE_SCALE_UP,
    RELEASE_SCALE_DOWN,
    ENVIRONMENT_LAUNCH,
    ENVIRONMENT_DESTROY,
  ];

  // Create
  savingRole: boolean = false;

  constructor(
    private uiCustomRoleControllerService: UiCustomRoleControllerService,
    private uiDropDownController: UiDropdownsControllerService,
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private customRoleControllerService: UiCustomRoleControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    if (p.roleName) {
      this.role = {
        roleName: p.roleName,
      };
      this.origRole = { ...this.role };
      this.mode = 'Edit';
      this.populateBasicInfo = true;
    }
    this.setRouterStateListener();
  }

  setRouterStateListener() {
    this.activatedRoute.paramMap
      .pipe(map(() => window.history.state))
      .subscribe((state) => {
        // console.log('state => ', state);
        if (!state.allRoles || !state.allRoles.length) {
          this.fetchAllRoles();
        } else {
          this.allRoles = state.allRoles;
        }
        if (state.currentRole) {
          this.role = state.currentRole;
          this.origRole = { ...this.role };
          this.populateBasicInfo = true;
          this.mode = 'Edit';
        }
        this.initForm();
      });
  }

  fetchAllRoles() {
    function setErrorData(error: any, _this: any) {
      _this.fetchingRolesError = true;
      (_this.primaryMessage = 'Failed to fetch roles'),
        (_this.secondaryMessage =
          error?.error?.message ||
          'Something went wrong, please try again later or contact your admin');
      _this.noDataIcon = errorStateIcon;
    }

    this.fetchingRoles = true;
    this.fetchingRolesError = false;

    this.customRoleControllerService
      .getAllRolesUsingGET()
      .subscribe(
        (allRoles: any[]) => {
          this.allRoles = allRoles.filter((role) => role?.baseRole);
        },
        (error: any) => {
          setErrorData(error, this);
        },
      )
      .add(() => {
        this.fetchingRoles = false;
      });
  }

  async initForm() {
    let fetch = await this.fetchAllPermissions();
    if (!fetch) {
      // Handle error
    } else {
      // Create form
      this.initBasicInfoForm();
      this.initPermissionsForm();
      if (this.role) {
        this.copyRole = this.role;
        this.basicInfoForm.controls['name'].setValue(this.role.roleName);
        this.copyRolePermissions();
      }
    }
  }

  async fetchAllPermissions() {
    return new Promise((resolve) => {
      this.uiDropDownController
        .getAllPermissionsUsingGET()
        .subscribe(
          (permissions: any) => {
            const filteredPermissions = permissions.filter(
              (permission) => permission?.key !== 'K8S_PERMISSION',
            );
            this.allPermissions = filteredPermissions;
            filteredPermissions.forEach((p) => {
              let permissionGroup = p?.category?.categoryName;
              if (!this.permissionGroups.has(permissionGroup)) {
                this.permissionGroups.set(permissionGroup, []);
                this.permissionGroupsArray.push(p?.category);
              }
              this.permissionGroups.get(permissionGroup).push(p);
            });
            /*
             * If release permissions are present in the permissions list,
             * then sort them according to the predefined sorted permission keys list
             */
            this.permissionGroups = sortPermissionsInPermissionGroups(
              this.permissionGroups,
            );
            if (this.permissionGroups.get('RELEASE')?.length > 0) {
              this.permissionGroups.set(
                'RELEASE',
                this.permissionGroups
                  .get('RELEASE')
                  ?.sort(
                    (a, b) =>
                      this.releasePermissionsOrder.indexOf(a?.key) -
                      this.releasePermissionsOrder.indexOf(b?.key),
                  ),
              );
            }
            this.permissionGroupsArray.sort((a, b) =>
              a?.categoryLabel?.localeCompare(b?.categoryLabel),
            );
            resolve(true);
          },
          (error: any) => {
            this.permissionsFormError = true;
            this.permissionsFormErrorMessage =
              error?.error?.message ||
              'Something went wrong, please try again later or contact your admin';
            resolve(false);
          },
        )
        .add(() => {
          this.fetchingPermissions = false;
        });
    });
  }

  initBasicInfoForm() {
    this.basicInfoForm = new FormGroup({
      name: new FormControl(null, [
        Validators.required,
        Validators.minLength(1),
        Validators.pattern(whiteSpaceInputPattern),
      ]),
      description: new FormControl(null),
    });
    this.basicInfoFormKeys = Object.keys(this.basicInfoForm.controls);
  }

  initPermissionsForm() {
    // Create the form controls object looping through the received permissions;

    let formControls = {};
    this.permissionGroupsArray.forEach((permissionGroup) => {
      let permissionTypes = this.permissionGroups.get(
        permissionGroup?.categoryName,
      );
      permissionTypes.forEach((permission: Permission) => {
        formControls[permission.key] = new FormControl('');
      });
    });
    this.permissionsForm = new FormGroup(formControls);

    this.permissionsFormKeys = Object.keys(this.permissionsForm.controls);
  }

  togglePermission(permission: Permission) {
    let currentVal = this.permissionsForm.controls[permission.key].value;
    this.permissionsForm.controls[permission.key].setValue(!currentVal);
  }

  copyRolePermissions() {
    this.fetchingClonePermissions = true;
    this.clonePermissions = [];
    this.uiCustomRoleControllerService
      .getCustomRoleUsingGET(this.copyRole.roleName)
      .subscribe(
        (roleData: any) => {
          if (this.populateBasicInfo) {
            this.basicInfoForm.controls['description'].setValue(
              roleData.description,
            );
          }
          this.clonePermissions = roleData.entities;
        },
        (error: any) => {
          this.onAPIErrorService.onAPIError(
            'Could not fetch permissions - ' + error?.error?.message,
            error,
          );
        },
      )
      .add(() => {
        this.syncPermissionsWithClone();
        this.fetchingClonePermissions = false;
      });
  }

  syncPermissionsWithClone() {
    this.permissionsFormKeys.forEach((permissionKey) => {
      this.permissionsForm.controls[permissionKey].setValue(false);
    });

    this.clonePermissions.forEach((permission: Permission) => {
      this.permissionsForm.controls[permission.key].setValue(true);
    });
  }

  saveRole() {
    this.savingRole = true;

    let allPermissions = [];
    this.permissionsFormKeys.forEach((permissionKey) => {
      if (this.permissionsForm.controls[permissionKey].value) {
        allPermissions.push(permissionKey);
      }
    });
    let role = {
      roleName: this.basicInfoForm.controls.name.value,
      description: this.basicInfoForm.controls.description.value,
      permissions: allPermissions,
    };

    this.mode === 'Create' ? this.createRole(role) : this.updateRole(role);
  }

  createRole(newRole: any) {
    this.uiCustomRoleControllerService
      .createCustomRoleUsingPOST(newRole)
      .subscribe(
        (rsp) => {
          mixpanel.track(USER_ROLE_CREATE_SUCCESS);
          this.customToastService.success(
            `Successfully created custom role ${newRole.roleName}`,
          );
          this.backToUserManagement();
        },
        (error) => {
          mixpanel.track(USER_ROLE_CREATE_FAIL, { error });
          this.onAPIErrorService.onAPIError(
            `Failed to create custom role - ${error?.error?.message}`,
            error,
          );
        },
      )
      .add(() => {
        this.savingRole = false;
      });
  }

  updateRole(existingRole: any) {
    this.uiCustomRoleControllerService
      .updateCustomRoleUsingPUT({
        request: existingRole,
        roleName: this.origRole.roleName,
      })
      .subscribe(
        (rsp) => {
          mixpanel.track(USER_ROLE_EDIT_SUCCESS);
          this.customToastService.success(
            `Successfully updated role ${existingRole.roleName}`,
          );
          this.backToUserManagement();
        },
        (error) => {
          mixpanel.track(USER_ROLE_EDIT_FAIL, { error });
          this.onAPIErrorService.onAPIError(
            error?.error?.message || `Failed to update role`,
            error,
          );
        },
      )
      .add(() => {
        this.savingRole = false;
      });
  }

  backToUserManagement = () => {
    this.router.navigate(['/settings/users/roles']);
  };
}
