import { Component } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { MonacoDiffEditorConstructionOptions } from '@materia-ui/ngx-monaco-editor';
import { UiApplicationControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { mergeBlueprintAndOverrideConfig } from 'src/app/utils/app-config-utils';
const emptyIcon = './assets/empty-override.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-simple-resource-config',
  templateUrl: './simple-resource-config.component.html',
  styleUrls: ['./simple-resource-config.component.scss'],
})
export class SimpleResourceConfigComponent extends FacetsPage {
  objectKeys = Object.keys;
  editorOptions: MonacoDiffEditorConstructionOptions = {
    theme: 'vs-light',
    originalEditable: false,
    readOnly: true,
    minimap: {
      enabled: false,
    },
  };
  code: string;
  disabled: boolean;
  clusterId: any;
  resourceName: any;
  stackName: any;
  resourceType: any;
  fetchingOverrides: boolean;
  icon = emptyIcon;
  overrideJson: any;
  fetchingConfig: boolean;
  appJson: any;
  originalCode: string;
  primaryMsg: string = 'No overrides applied!';
  secondaryMsg: string =
    'You can add overrides to override any variables etc. for this resource';
  showBtn: boolean = true;
  status: 'info' | 'error' = 'info';

  constructor(
    _route: ActivatedRoute,
    private appController: UiApplicationControllerService,
    private router: Router,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p.clusterId;
    this.resourceName = p.resourceName;
    this.stackName = p.stackName;
    this.resourceType = p.resourceType;
    this.fetchingConfig = true;
    this.code = undefined;
    this.appJson = undefined;
    this.originalCode = undefined;
    this.appController
      .getResourceByNameV2UsingGET({
        resourceName: this.resourceName,
        resourceType: this.resourceType,
        clusterId: this.clusterId,
      })
      .subscribe(
        (data: any) => {
          this.appJson = data;
          this.originalCode = JSON.stringify(data, null, '\t');
          this.fetchingConfig = false;
          this.code = JSON.stringify(data, null, '\t');
          this.fetchingOverrides = true;
          this.appController
            .getResourceOverrideObjectUsingGET({
              resourceName: this.resourceName,
              resourceType: this.resourceType,
              clusterId: this.clusterId,
            })
            .subscribe(
              (json) => {
                this.overrideJson = json;
                const merged = mergeBlueprintAndOverrideConfig(
                  this.appJson || {},
                  this.overrideJson?.overrides || {},
                );
                this.code = JSON.stringify(merged, null, '\t');
                this.fetchingOverrides = false;
              },
              (err) => (this.fetchingOverrides = false),
            );
        },
        (error) => {
          this.icon = errorIcon;
          this.secondaryMsg =
            error?.error?.message || ' There is some error fetching resources';
          this.fetchingConfig = false;
          this.fetchingOverrides = false;
          this.primaryMsg = '';
          this.showBtn = false;
          this.status = 'error';
        },
      );
  }

  createOverride() {
    this.router.navigate(['edit'], {
      relativeTo: this.route,
    });
  }
}
