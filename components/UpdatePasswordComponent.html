import { Component, OnInit } from '@angular/core';
import {
  AbstractControl,
  FormControl,
  FormGroup,
  ValidationErrors,
  ValidatorFn,
  Validators,
} from '@angular/forms';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { UiUserControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { UserService } from 'src/app/services/user.service';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-update-password',
  templateUrl: './update-password.component.html',
  styleUrls: ['./update-password.component.scss'],
})
export class UpdatePasswordComponent extends FacetsPage {
  changePasswordModel: FormGroup = new FormGroup(
    {
      oldPassword: new FormControl('', [
        Validators.required,
        Validators.minLength(8),
        Validators.maxLength(20),
      ]),
      newPassword: new FormControl('', [
        Validators.required,
        Validators.pattern(
          '^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.*[!@#$%^&*])([a-zA-Z0-9!@#$%^&*]+)',
        ),
        Validators.minLength(8),
        Validators.maxLength(20),
      ]),
      confirmPassword: new FormControl('', [Validators.required]),
    },
    [this.passwordMatchValidator('newPassword', 'confirmPassword')],
  );
  showPassword = {
    oldPassword: 'password',
    newPassword: 'password',
    confirmPassword: 'password',
  };
  changePassInProgress: boolean = false;
  hasPassword: boolean = false;
  loadingPasswordForm: boolean = true;
  userSubscription: any;

  constructor(
    private activatedroute: ActivatedRoute,
    private router: Router,
    private userControllerService: UiUserControllerService,
    private userService: UserService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedroute);
  }

  OnPageLoad(p: Params): void {
    this.loadingPasswordForm = true;
    if (this.userService.userLoaded) {
      this.getUserInfo(this.userService.user?.name);
    } else {
      this.userSubscription = this.userService.userObserver.subscribe(
        (userLoaded) => {
          if (userLoaded) {
            this.getUserInfo(this.userService?.user?.name);
          }
        },
      );
    }
  }

  get oldPassword() {
    return this.changePasswordModel.get('oldPassword');
  }

  get newPassword() {
    return this.changePasswordModel.get('newPassword');
  }

  get confirmPassword() {
    return this.changePasswordModel.get('confirmPassword');
  }

  get passwordMatchError() {
    return this.changePasswordModel.getError('passwordMatchError');
  }

  handleBack = () => {
    this.router.navigate(['accountSettings', 'profile']);
  };

  // Toggles show/hide for password input box
  toggleShowPassword(passwordType) {
    this.showPassword[passwordType] === 'password'
      ? (this.showPassword[passwordType] = 'text')
      : (this.showPassword[passwordType] = 'password');
  }

  changePassword() {
    this.changePassInProgress = true;
    const payLoad = {
      newPassword: this.newPassword?.value,
      oldPassword: this.oldPassword?.value,
    };
    this.userControllerService.updatePasswordUsingPOST(payLoad).subscribe(
      (data) => {
        this.changePassInProgress = false;
        this.customToastService.success(
          data?.message || 'Password changed successfully',
        );
        this.handleBack();
      },
      (error) => {
        this.changePassInProgress = false;
        this.onAPIErrorService.onAPIError(
          error?.error?.message || 'Provided passwords are incorrect',
          error,
        );
      },
    );
  }

  passwordMatchValidator(source: string, target): ValidatorFn {
    return (control: AbstractControl): ValidationErrors | null => {
      const newPassword = control.get(source).value;
      const confirmPassword = control.get(target).value;

      if (confirmPassword.length) {
        return newPassword && confirmPassword && newPassword === confirmPassword
          ? null
          : { passwordMatchError: true };
      }
      return null;
    };
  }

  getUserInfo(userName: string) {
    this.userControllerService.getAllUsersUsingGET().subscribe(
      async (users) => {
        let user = users.find((user) => user?.userName === userName);
        this.hasPassword = user.password !== null;

        if (!this.hasPassword) {
          this.oldPassword.clearValidators();
          this.oldPassword.updateValueAndValidity();
        }
        this.loadingPasswordForm = false;
      },
      (error) => {
        this.loadingPasswordForm = false;
      },
    );
  }
}
