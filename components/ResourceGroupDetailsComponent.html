import { Component, TemplateRef, ViewChild } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { NbDialogService, NbTagComponent } from '@nebular/theme';
import mixpanel from 'mixpanel-browser';
import {
  UiDropdownsControllerService,
  UiResourceGroupControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { SmartTableRouterlinkComponent } from 'src/app/components/smart-table-routerlink/smart-table-routerlink.component';
import {
  USER_RESOURCE_GROUP_EDIT_FAIL,
  USER_RESOURCE_GROUP_EDIT_SUCCESS,
} from 'src/app/constants/analytics-track-events';
import { RESOURCE_GROUP_WRITE } from 'src/app/constants/permissions';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { ContextUpdatesService } from 'src/app/services/context-updates.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const noAccessIcon = './assets/alert-empty.svg';
const errorStateIcon = './assets/error.svg';
const emptyStateIcon = './assets/empty_resources.svg';

@Component({
  selector: 'app-resource-group-details',
  templateUrl: './resource-group-details.component.html',
  styleUrls: ['./resource-group-details.component.scss'],
})
export class ResourceGroupDetailsComponent extends FacetsPage {
  noAccessIcon = noAccessIcon;
  noDataIcon = emptyStateIcon;
  errorIcon = errorStateIcon;
  RESOURCE_GROUP_WRITE = RESOURCE_GROUP_WRITE;

  resourceGroupName: string;
  resourceGroupID: string;
  stackName: string;

  selectedBlueprint: any;
  allBlueprints: any[];
  allResourcesForBlueprint: any[];
  resourceMapForBlueprint: Map<string, string[]>;

  allResources: any[];
  fetchingResources: boolean = false;
  fetchingError: boolean = false;

  resourceForm: FormGroup;

  fetchingResource: boolean = false;
  fetchingBlueprints: boolean = false;
  tempResourceList: any = [];
  addingResource: boolean = false;

  changeRegistered: boolean = false;

  @ViewChild('createResourceDialog', { static: true })
  private createResourceDialog: TemplateRef<any>;

  resourceTableConfig: any = {
    columns: {
      stackName: {
        title: 'Blueprint',
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [],
          },
        },
        valuePrepareFunction: (value) => {
          const displayVal = {
            url: ['/', 'stack', value],
            display: value,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: SmartTableRouterlinkComponent,
        type: 'custom',
      },
      resourceType: {
        title: 'Resource Type',
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [],
          },
        },
        type: 'text',
        valuePrepareFunction(type, row) {
          return type?.length > 0 ? type : 'N/A';
        },
      },
      resourceName: {
        title: 'Resource name',
        filter: true,
      },
    },
    pager: {
      display: false,
      perPage: 10,
    },
    mode: 'external',
    actions: false,
  };

  constructor(
    private _route: ActivatedRoute,
    private router: Router,
    private dialogService: NbDialogService,
    private uiResourceGroupController: UiResourceGroupControllerService,
    private contextUpdateService: ContextUpdatesService,
    private uiStackControllerService: UiStackControllerService,
    private uiDropdownsControllerService: UiDropdownsControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.resourceGroupName = this.route.snapshot.paramMap.get('resourceGroup');
    this.initResourceForm();
    this.stackName = p.stackName;
    this.getResources();
  }

  initResourceForm() {
    this.resourceForm = new FormGroup({
      blueprint: new FormControl('', [Validators.required]),
      resources: new FormControl([]),
    });
  }

  backToUserManagement = () => {
    this.router.navigate(['/settings/users/resource-groups']);
  };

  getResources() {
    this.fetchingResources = true;
    this.allResources = [];

    this.uiResourceGroupController
      .findAllUsingGET()
      .subscribe(
        (resources) => {
          // THIS IS ASSUMING that 2 resource groups cannot have the same name
          let resourceGroup: any = resources.find(
            (resourceGrp) => resourceGrp.name === this.resourceGroupName,
          );
          this.resourceGroupID = resourceGroup.id;
          resourceGroup?.resourceInfo?.forEach((stack: any) => {
            let resources = [];
            stack.resources?.forEach((resourceType: any) => {
              resourceType?.resourceNames?.forEach((resource) => {
                resources.push({
                  resourceName: resource,
                  resourceType: resourceType.resourceType,
                  stackName: stack.stackName,
                });
              });
            });
            this.allResources.push(...resources);
            this.buildResourceFilter(this.allResources, 'resourceType');
            this.buildResourceFilter(this.allResources, 'stackName');
          });
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Could not fetch the resources',
            error,
          );
        },
      )
      .add(() => {
        this.fetchingResources = false;
      });
  }

  async promptCreateResourceDialog() {
    this.allBlueprints = this.contextUpdateService.getAllBlueprints();
    this.addingResource = false;
    this.tempResourceList = [];
    this.fetchingResource = false;
    this.resourceMapForBlueprint?.clear();

    this.initResourceForm();

    if (!this.allBlueprints?.length) {
      await this.getAllBlueprints();
    }

    this.dialogService.open(this.createResourceDialog, {
      closeOnBackdropClick: false,
    });
  }

  removeResource(item: any) {
    const currentResources = this.resourceForm.get('resources').value;
    const newItems = currentResources.filter((i: any) => i !== item);
    this.resourceForm.get('resources').setValue(newItems);
  }

  getAllBlueprints() {
    return new Promise((resolve) => {
      this.fetchingBlueprints = true;
      this.uiStackControllerService.getStacksUsingGET1().subscribe((stacks) => {
        this.allBlueprints = stacks.map((stack: any) => stack.name);
        this.fetchingBlueprints = false;
        resolve(true);
      });
    });
  }

  async getAllResourcesForBlueprint() {
    this.resourceForm.patchValue({
      resources: [],
    });
    this.changeRegistered = false;
    await this.fetchAllResourcesForBlueprint();
    this.preFillResources();
  }

  preFillResources() {
    this.tempResourceList = [];
    this.allResources?.forEach((resource) => {
      if (resource.stackName === this.resourceForm.get('blueprint')?.value) {
        this.tempResourceList.push({
          title: resource.resourceType,
          value: resource.resourceName,
        });
      }
    });
    if (this.tempResourceList?.length) {
      this.tempResourceList = [...this.tempResourceList];
    }
  }

  fetchAllResourcesForBlueprint() {
    this.tempResourceList = [];
    return new Promise((resolve) => {
      this.fetchingResource = true;
      this.uiDropdownsControllerService
        .getAllResourcesByStackUsingGET({
          stackName: this.resourceForm.get('blueprint')?.value,
          includeContent: false,
        })
        .subscribe((data) => {
          this.resourceMapForBlueprint = new Map<string, string[]>();
          data.forEach((resource: any) => {
            if (resource.resourceType.toLowerCase() != 'unknown') {
              if (!this.resourceMapForBlueprint.has(resource.resourceType)) {
                this.resourceMapForBlueprint.set(resource.resourceType, []);
              }
              this.resourceMapForBlueprint
                .get(resource.resourceType)
                .push(resource.resourceName);
            }
          });
        })
        .add(() => {
          this.fetchingResource = false;
          resolve(true);
        });
    });
  }

  // get selectedTags from reactive form
  removeTag(tag: NbTagComponent) {
    const tags = this.resourceForm.get('resources')?.value;
    const index = tags.indexOf(tag.text);
    if (index !== -1) {
      tags.splice(index, 1);
    }
    this.resourceForm.patchValue({
      resources: tags,
    });
  }

  getEntriesOfMap() {
    if (this.resourceMapForBlueprint) {
      return Array.from(this.resourceMapForBlueprint?.keys())?.sort();
    }
  }

  createResource(ref) {
    let stack = this.resourceForm.get('blueprint').value;
    const payLoad: any = {
      resources: new Map<string, string[]>(),
      stackName: stack,
    };
    this.tempResourceList.forEach((resource: any) => {
      if (!payLoad.resources.has(resource.title)) {
        payLoad.resources.set(resource.title, []);
      }
      payLoad.resources.get(resource.title).push(resource.value);
    });
    const resourcesList = [];
    Array.from(payLoad.resources.keys()).forEach((resourceType) => {
      resourcesList.push({
        resourceType: resourceType,
        resourceNames: payLoad.resources.get(resourceType),
      });
    });
    payLoad.resources = resourcesList;
    this.addingResource = true;
    this.uiResourceGroupController
      .updateResourceUsingPATCH({
        resourceGroupId: this.resourceGroupID,
        resourceUpdateRequest: payLoad,
      })
      .subscribe(
        (data) => {
          mixpanel.track(USER_RESOURCE_GROUP_EDIT_SUCCESS);
          this.customToastService.success(
            `Updated resources for ${this.resourceGroupName}`,
          );
          ref.close();
          this.getResources();
        },
        (error) => {
          mixpanel.track(USER_RESOURCE_GROUP_EDIT_FAIL, { error });
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Failed to update the resources',
            error,
          );
        },
      )
      .add(() => {
        this.addingResource = true;
        this.tempResourceList = [];
      });
  }

  selectedOption(event) {
    this.changeRegistered = true;
    this.tempResourceList = [...event?.visibleList];
  }

  removedItem(event) {
    this.changeRegistered = true;
    this.tempResourceList = [...event?.visibleList];
  }

  buildResourceFilter(resourceData, filterType) {
    let resourceFilterData: {
      title;
      value;
    }[] = Array.from(
      new Set(resourceData?.map((resource) => resource[filterType])),
    )
      .sort()
      .map((name) => {
        return {
          title: name,
          value: name,
        };
      });
    this.resourceTableConfig.columns[filterType].filter.config.list = [
      ...resourceFilterData,
    ];
    this.resourceTableConfig = { ...this.resourceTableConfig };
  }
}
