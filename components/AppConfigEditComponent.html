import { Component, EventEmitter, Output, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { LocalDataSource } from 'ng2-smart-table';
import { UiApplicationControllerService } from '../../../../../cc-api/services/ui-application-controller.service';
import { OverrideStatusBadgeComponent } from '../../../../../components/override-status-badge/override-status-badge.component';
import { UiCommonClusterControllerService } from '../../../../../cc-api/services/ui-common-cluster-controller.service';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  getEffectiveValue,
  getOverridesObjectFromModel,
  getUserVars,
} from 'src/app/utils/app-config-utils';
import { SaveChangesFooterComponent } from 'src/app/components/save-changes-footer/save-changes-footer.component';
import { ENVIRONMENT_CONFIGURE } from 'src/app/constants/permissions';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const emptyIcon = './assets/empty-override.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-app-config-edit',
  templateUrl: './app-config-edit.component.html',
  styleUrls: ['./app-config-edit.component.scss'],
})
export class AppConfigEditComponent extends FacetsPage {
  clusterId: any;
  appName: any;
  stackName: any;
  appJson: any;
  appJsonVersion2 = true;
  overrideJson: any;
  userVarsCount: number = 0;
  userDefinedVariables: LocalDataSource = new LocalDataSource();
  userVarsChanged: any = {};
  configChanged: any = {};
  addOverrideSpinner: boolean = false;
  objectKeys = Object.keys;
  containerConfigurationModel: any = {};
  originalData: any = {};
  icon = emptyIcon;
  updatedData: any = {};
  primaryMsg: string = 'No variables found!';
  errMsg: string = '';
  status: 'info' | 'error' = 'info';
  @Output() byClose: EventEmitter<any> = new EventEmitter<any>();

  fetchingApplicationData = false;
  fetchingOverrides = false;
  reviewComments: string = '';

  settings = {
    columns: {
      name: {
        title: 'Variable Name',
        filter: true,
        width: '40%',
        editable: false,
        sort: true,
      },
      value: {
        title: 'Value',
        filter: true,
        width: '40%',
        editable: true,
        editor: { type: 'password' },
      },
      overridden: {
        type: 'custom',
        renderComponent: OverrideStatusBadgeComponent,
        title: 'Override Status',
        filter: {
          type: 'list',
          config: {
            selectText: 'Status',
            list: [
              { value: 'NO CHANGE', title: 'No change' },
              { value: 'OVERRIDDEN', title: 'Overridden' },
            ],
          },
        },
        width: '20%',
        editable: false,
      },
    },
    noDataMessage: 'No variables found',
    pager: {
      display: false,
      perPage: 10,
    },
    edit: {
      inputClass: '',
      editButtonContent: '<i class="eva-edit-outline eva"></i>',
      saveButtonContent: '<i class="eva-checkmark-outline eva"></i>',
      cancelButtonContent: '<i class="eva-close-outline eva"></i>',
      confirmSave: true,
    },
    actions: {
      add: false,
      edit: true,
      delete: false,
      position: 'right',
    },
  };
  resourceType: string;
  configKeyPaths = {
    disabled: { paths: ['disabled'], defaultValue: false },
    size: { paths: ['size', 'spec.runtime.size.value'], defaultValue: 'NA' },
    resourceAllocationStrategy: {
      paths: ['resourceAllocationStrategy', 'spec.runtime.size.namespace'],
      defaultValue: 'NA',
    },
    hpaMinReplicas: {
      paths: ['scaling.hpaMinReplicas', 'spec.runtime.autoscaling.min'],
      defaultValue: '',
    },
    hpaMaxReplicas: {
      paths: ['scaling.hpaMaxReplicas', 'spec.runtime.autoscaling.max'],
      defaultValue: '',
    },
    hpaMetricThreshold: {
      paths: [
        'scaling.hpaMetricThreshold',
        'spec.runtime.autoscaling.cpuThreshold',
      ],
      defaultValue: '',
    },
  };
  environmentConfigurePermission = ENVIRONMENT_CONFIGURE;

  @ViewChild(SaveChangesFooterComponent)
  private footerComponent: SaveChangesFooterComponent;
  constructor(
    _route: ActivatedRoute,
    private router: Router,
    private uiApplicationControllerService: UiApplicationControllerService,
    private uiCommonClusterControllerService: UiCommonClusterControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.fetchingApplicationData = true;
    this.fetchingOverrides = true;
    this.clusterId = p.clusterId;
    this.appName = p.resourceName;
    this.resourceType = p.resourceType;
    this.stackName = p.stackName;
    this.appJson = undefined;
    this.overrideJson = undefined;
    this.userVarsCount = 0;
    this.userDefinedVariables = new LocalDataSource();
    this.containerConfigurationModel = {};
    this.originalData = {};
    this.updatedData = {};
    this.uiApplicationControllerService
      .getResourceByNameV2UsingGET({
        resourceName: this.appName,
        clusterId: this.clusterId,
        resourceType: this.resourceType,
      })
      .subscribe(
        (json) => {
          this.appJson = json;
          if ('v2' !== this.appJson?.apiVersion) {
            this.appJsonVersion2 = false;
          }
          this.fetchingApplicationData = false;

          this.fetchingOverrides = true;
          this.fetchingApplicationData = false;
          this.uiApplicationControllerService
            .getResourceOverrideObjectUsingGET({
              resourceName: this.appName,
              resourceType: this.resourceType,
              clusterId: this.clusterId,
            })
            .subscribe(
              (json) => {
                this.overrideJson = json?.overrides;
                this.fetchingOverrides = false;
              },
              (err) => (this.fetchingOverrides = false),
            )
            .add(() => {
              this.populateUserVars(this.appJson, this.overrideJson);
              this.loadDataForConfiguration(this.appJson, this.overrideJson);
            });
        },
        (error) => {
          this.fetchingApplicationData = false;
          this.fetchingOverrides = false;
          this.icon = errorIcon;
          this.primaryMsg = '';
          this.errMsg =
            error?.error?.message || ' There is some error fetching resource';
          this.status = 'error';
        },
      );
  }

  loadDataForConfiguration(json: any, overrides: any) {
    const effectiveConfigValues = {};
    Object.keys(this.configKeyPaths).forEach((key) => {
      effectiveConfigValues[key] = getEffectiveValue(
        this.configKeyPaths[key].paths,
        json,
        overrides,
        this.configKeyPaths[key].defaultValue,
      );
    });
    let model: any = {};
    Object.keys(this.configKeyPaths).forEach((key) => {
      model[key] = effectiveConfigValues[key].effectiveValue;
      this.originalData[key] = effectiveConfigValues[key].effectiveValue;
    });

    this.containerConfigurationModel = { ...model };
  }

  populateUserVars(json: any, overrides: any) {
    const { varsCount, dataSource = [] } = getUserVars(
      json,
      overrides,
      this.appJsonVersion2,
    );
    this.userVarsCount = varsCount;
    this.userDefinedVariables.load(dataSource);
    dataSource.forEach((data: any) => {
      this.originalData[data.name] = data.value;
    });
  }

  validateVars(event: any) {
    if (event.newData.value != this.originalData[event.newData.name]) {
      this.userVarsChanged[event.newData.name] = event.newData.value;
    } else {
      delete this.userVarsChanged[event.newData.name];
    }
    event.confirm.resolve(event.newData);
  }

  submitClusterOverrides = async (ref: any) => {
    this.addOverrideSpinner = true;
    try {
      const overrides = getOverridesObjectFromModel(
        this.overrideJson,
        this.configChanged,
        this.userVarsChanged,
        this.appJsonVersion2,
      );
      const body = {
        resourceName: this.appName,
        resourceType: this.resourceType,
        changeLog: this.reviewComments,
        overrides: overrides,
      };
      this.uiCommonClusterControllerService
        .overrideSizingUsingPOST1({
          request: [body],
          clusterId: this.clusterId,
        })
        .subscribe(
          (r) => {
            this.customToastService.success(
              'Updated cluster variables, changes will be applied after next release',
            );
            ref.close();
            this.userVarsChanged = {};
            this.configChanged = {};
            this.addOverrideSpinner = false;
            // this.handleBack();
            this.byClose.emit();
          },
          (err) => {
            this.addOverrideSpinner = false;
            this.onAPIErrorService.onAPIError(
              err?.error?.message || 'Failed to update the changes',
              err,
            );
          },
        );
    } catch (err) {
      this.addOverrideSpinner = false;
      this.customToastService.error('Update failed: ' + err?.error?.message);
      ref.close();
    }
  };

  // custom onChange subscriber to track container configuration changes
  onChange(event, fieldName: string, displayName: string) {
    if (event?.target?.value) {
      if (event.target.value != this.originalData[fieldName]) {
        this.configChanged[fieldName] = {
          value: event.target.value,
          name: displayName,
        };
      } else {
        delete this.configChanged[fieldName];
      }
    }
  }

  onCheckboxChange(event, fieldName: string, displayName: string) {
    if (event?.target?.checked != this.originalData[fieldName]) {
      this.configChanged[fieldName] = {
        value: event?.target?.checked,
        name: displayName,
      };
    } else {
      delete this.configChanged[fieldName];
    }
  }

  handleBack = () => {
    this.router.navigate(['../'], {
      relativeTo: this.route,
    });
  };

  discardChanges = (ref) => {
    ref.close('discard');
    this.userVarsChanged = {};
    this.configChanged = {};
    this.reviewComments = '';
    this.loadDataForConfiguration(this.appJson, this.overrideJson);
  };

  async canDeactivate(): Promise<boolean> {
    if (
      this.objectKeys(this.userVarsChanged)?.length == 0 &&
      this.objectKeys(this.configChanged)?.length == 0
    ) {
      return true;
    }

    return this.footerComponent.onDeactivate();
  }
}
