import { Component } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { UiStackControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
const emptyStateLogo = './assets/empty-env.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-single-diff-select',
  templateUrl: './single-diff-select.component.html',
  styleUrls: ['./single-diff-select.component.scss'],
})
export class SingleDiffSelectComponent extends FacetsPage {
  clusterId: any;
  appName: any;
  stackName: any;
  resourceType: any;
  environments: any;
  currentEnvironment: any;
  environmentOptions = [];
  environmentsToCompare = [];
  icon = emptyStateLogo;
  fetchingEnvs = false;
  primaryMsg: string = 'No environment to compare!';
  secondaryMsg: string = 'Select at least one environment to compare diff';
  showBtn: boolean = true;
  status: 'info' | 'error' = 'info';

  constructor(
    _route: ActivatedRoute,
    private router: Router,
    private uiStackControllerService: UiStackControllerService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p.clusterId;
    this.appName = p.resourceName;
    this.stackName = p.stackName;
    this.resourceType = p.resourceType;
    this.fetchingEnvs = true;
    this.uiStackControllerService
      .getClustersUsingGET1(this.stackName)
      .subscribe(
        (json) => {
          this.environments = json;
          this.currentEnvironment = this.environments.find(
            (env: any) => env.id === this.clusterId,
          );
          this.prepareOptions(this.environments);
          this.fetchingEnvs = false;
        },
        (error) => {
          this.fetchingEnvs = false;
          this.secondaryMsg =
            error?.error?.message ||
            ' There is some error fetching environment information';
          this.primaryMsg = '';
          this.showBtn = false;
          this.icon = errorIcon;
          this.status = 'error';
        },
      );
  }

  prepareOptions(allEnvironments) {
    this.environmentOptions = allEnvironments
      .filter((o) => o.id !== this.clusterId)
      .map((env) => ({
        label: env.name,
        value: env.id,
      }));
  }

  handleBack = () => {
    this.router.navigate([
      '/',
      this.stackName,
      'cluster',
      this.clusterId,
      'resources',
    ]);
  };

  handleCompare() {
    this.router.navigate([
      '/',
      this.stackName,
      'cluster',
      this.clusterId,
      'resources',
      'type',
      this.resourceType,
      'resource',
      this.appName,
      'diff',
      'compare',
      'envs',
      this.environmentsToCompare.join('__'),
    ]);
  }

  createCluster(): void {
    this.router.navigate(['/', this.stackName, 'chooseClusterCreate']);
  }
}
