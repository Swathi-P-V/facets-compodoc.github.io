import {
  Component,
  ViewChild,
  EventEmitter,
  Output,
  Input,
} from '@angular/core';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { take, filter } from 'rxjs';
import {
  MonacoEditorConstructionOptions,
  MonacoEditorLoaderService,
  MonacoDiffEditorConstructionOptions,
} from '@materia-ui/ngx-monaco-editor';
import {
  PublicApIsService,
  UiThemeFileControllerService,
} from 'src/app/cc-api/services';
import customThemeSchema from 'src/custom-theme-schema';
import { ROLE_CC_ADMIN } from 'src/app/constants/permissions';
import isEmpty from 'lodash/isEmpty';
import customTheme from 'src/custom-theme';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { CanDeactivateComponent } from 'src/app/can-deactivate.guard';
import { SaveChangesFooterComponent } from 'src/app/components/save-changes-footer/save-changes-footer.component';

@Component({
  selector: 'app-custom-theme-editor',
  templateUrl: './custom-theme-editor.component.html',
  styleUrls: ['./custom-theme-editor.component.scss'],
})
export class CustomThemeEditorComponent
  extends FacetsPage
  implements CanDeactivateComponent
{
  @Input() currentThemeCode: any = null;
  @ViewChild(SaveChangesFooterComponent)
  private footerComponent: SaveChangesFooterComponent;
  editorOptions: MonacoEditorConstructionOptions = {
    theme: 'vs-light',
    automaticLayout: true,
    language: 'json',
    readOnly: false,
    formatOnType: true,
    formatOnPaste: true,
    inlineSuggest: {
      enabled: true,
    },
    wordWrap: 'on',
    minimap: {
      enabled: true,
    },
    quickSuggestions: true,
    suggestOnTriggerCharacters: true,
    suggest: {
      snippetsPreventQuickSuggestions: false,
    },
  };
  popupEditorOptions: MonacoDiffEditorConstructionOptions = {
    theme: 'vs-light',
    originalEditable: false,
    readOnly: true,
    minimap: {
      enabled: false,
    },
  };
  modelUri: any;
  @Output() codeModifiedEvent: EventEmitter<any> = new EventEmitter();
  themePermission = ROLE_CC_ADMIN;
  themeCodeModified: boolean = false;
  themeCodeValid: boolean = false;
  modifiedCode: any = null;
  codeToCompare: any = null;
  changeThemeInProgress: boolean = false;
  fetchingTheme: boolean = false;

  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private monacoLoaderService: MonacoEditorLoaderService,
    private publicApiService: PublicApIsService,
    private toastService: CustomToastService,
    private themeControllerService: UiThemeFileControllerService,
  ) {
    super(activatedRoute);
    /*
      loading default schema from static schema theme variables 
    */
    this.monacoLoaderService.isMonacoLoaded$
      .pipe(
        filter((isLoaded) => isLoaded),
        take(1),
      )
      .subscribe(() => {
        monaco.languages.json.jsonDefaults.setDiagnosticsOptions({
          enableSchemaRequest: true,
          validate: true,
          schemas: [
            {
              uri: 'http://json-schema.org/draft-07/schema#',
              fileMatch: ['*'],
              schema: customThemeSchema,
            },
          ],
        });
      });
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.currentThemeCode = JSON.stringify({});
    this.getCurrentTheme();
  }

  getCurrentTheme() {
    this.fetchingTheme = true;
    this.publicApiService.retrieveThemeFileUsingGET().subscribe(
      (data) => {
        if (data?.content) {
          try {
            this.currentThemeCode = JSON.stringify(
              JSON.parse(data?.content),
              null,
              '\t',
            );
            if (isEmpty(JSON.parse(data?.content))) {
              this.currentThemeCode = JSON.stringify(customTheme, null, '\t');
            }
            this.setCustomThemeVars();
          } catch (error) {
            this.showErrMsg(error);
          }
        }
        this.fetchingTheme = false;
      },
      (error) => {
        if (error?.error?.code === '404') {
          this.currentThemeCode = JSON.stringify(customTheme, null, '\t');
          this.setCustomThemeVars();
        }
        this.showErrMsg(error);
        this.fetchingTheme = false;
      },
    );
  }

  setCustomThemeVars() {
    this.submitDisabled();
    this.codeToCompare = this.currentThemeCode;
    this.modifiedCode = this.currentThemeCode;
  }

  showErrMsg(error) {
    this.toastService?.error(
      error?.error?.message || `Unable to load theme contents`,
    );
  }

  submitDisabled() {
    try {
      JSON.parse(this.modifiedCode);
      this.themeCodeValid = true;
    } catch (error) {
      this.themeCodeValid = false;
    }
  }

  codeUpdate(modifiedCode) {
    this.modifiedCode = modifiedCode;
    this.themeCodeModified = modifiedCode !== this.codeToCompare;
    this.submitDisabled();
  }

  submitThemeSettings(ref) {
    this.changeThemeInProgress = true;
    if (this.themeCodeValid && this.themeCodeModified) {
      this.themeControllerService
        .saveThemeFileUsingPOST(this.modifiedCode)
        .subscribe(
          (saveStatus) => {
            this.changeThemeInProgress = false;
            this.toastService?.success(`Theme changes saved successfully`);
            ref?.close();
            window.location.reload();
          },
          (error) => {
            if (
              (error.status === 500 || error.status === 403) &&
              error?.error.message === 'Access is denied'
            ) {
              this.toastService?.error(
                'Please contact your administrator for more details',
              );
            } else {
              this.toastService?.error(
                error?.error?.message || `Theme changes failed`,
              );
            }
            this.changeThemeInProgress = false;
          },
        );
    }
  }

  onDiscard = (ref: any) => {
    this.currentThemeCode = JSON.stringify({});
    this.modifiedCode = null;
    /*
      input property for monaco editor code does not detect change in properties of object after discard.
      For this, we need to set the stringified object to empty value first and then to value after discarded changes
      This way, the changes are loaded properly after discard.
    */
    setTimeout(() => {
      this.currentThemeCode = this.codeToCompare;
    }, 10);
    ref.close('discard');
  };

  async canDeactivate(): Promise<boolean> {
    if (this.modifiedCode === this.codeToCompare) {
      return true;
    }
    return this.footerComponent.onDeactivate();
  }
}

  [show]="true"
  saveBtnPermission="{{ themePermission }}"
  [disableSaveBtn]="!this.themeCodeModified || !this.themeCodeValid"
  (saveChanges)="submitThemeSettings($event)"
  [showLoader]="changeThemeInProgress"
  (discardChanges)="onDiscard($event)"
  modalSize="large"
  modalWidth="70rem"
>

    <ngx-monaco-diff-editor
      [options]="editorOptions"
      [original]="codeToCompare"

    ></ngx-monaco-diff-editor>
  </div>
