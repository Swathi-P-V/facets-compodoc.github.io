import { Component, Input, OnInit } from '@angular/core';
import { ActivatedRoute, Params } from '@angular/router';
import { UiStackControllerService } from 'src/app/cc-api/services';
import { SmartTableRouterlinkComponent } from 'src/app/components/smart-table-routerlink/smart-table-routerlink.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { UiDropdownsControllerService } from 'src/app/cc-api/services/ui-dropdowns-controller.service';
import { SmartTableWithBadgeComponent } from 'src/app/components/smart-table-with-badge/smart-table-with-badge.component';
import { cloneDeep } from 'lodash';
import { TextCircleTooltipComponent } from '../../resource-center/resource-list/text-circle-tooltip.component';
import { OverrideStatusBadgeComponent } from 'src/app/components/override-status-badge/override-status-badge.component';
import {
  overriddenStatusMap,
  providedNormalMap,
  resourceStatusMap,
} from 'src/app/constants/resource-details';
const emptyStateLogo = './assets/empty-env.svg';
const errorLogo = './assets/error.svg';

@Component({
  selector: 'app-resource-environments',
  templateUrl: './resource-environments.component.html',
  styleUrls: ['./resource-environments.component.scss'],
})
export class ResourceEnvironmentsComponent extends FacetsPage {
  @Input() blueprint: string;

  resourceName: string;
  resourceType: string;
  blueprintName: string;

  errorStateIcon = errorLogo;
  emptyStateIcon = emptyStateLogo;

  environmentsTableData: any[];
  environmentsOverviewList: any;

  STATUS_CONFIG = {
    enabled: {
      statusText: 'ENABLED',
      background: '#EBFFF3',
      color: '#139E4A',
    },
    disabled: {
      statusText: 'IN PROGRESS',
      background: '#FFEBEB',
      color: '#BF0D0D',
    },
  };

  resourceEnvStatusColumn = {
    title: 'Status',
    width: '20%',
    type: 'custom',
    valuePrepareFunction: (cell) => {
      return JSON.stringify({
        severity: cell?.toLowerCase(),
        badgeClass: 'p-2 text-capitalize mb-0',
      });
    },
    renderComponent: SmartTableWithBadgeComponent,
    filter: {
      type: 'list',
      config: {
        selectText: 'All',
        list: [
          { value: 'Disabled', title: 'Disabled' },
          { value: 'Enabled', title: 'Enabled' },
        ],
      },
    },
  };
  providedStatus = {
    type: 'custom',
    title: 'Type',
    width: '10%',
    filter: {
      type: 'list',
      config: {
        selectText: 'All',
        list: [
          { value: 'P', title: 'Provided' },
          { value: 'N', title: 'Normal' },
        ],
      },
    },
    valuePrepareFunction: (cell, row) => {
      return JSON.stringify({
        value: cell?.[0]?.toUpperCase(),
        setClass: cell?.length > 0 ? 'tablet' : '',
        showTooltip: cell?.length > 0,
        toolTip: cell === 'Provided' ? 'Provided resource' : 'Normal resource',
      });
    },
    renderComponent: TextCircleTooltipComponent,
  };
  overridden = {
    type: 'custom',
    renderComponent: OverrideStatusBadgeComponent,
    title: 'Override Status',
    filter: {
      type: 'list',
      config: {
        selectText: 'All',
        list: [
          { value: 'NO CHANGE', title: 'No change' },
          { value: 'OVERRIDDEN', title: 'Overridden' },
        ],
      },
    },
    width: '15%',
    editable: false,
  };
  environmentsTableConfig: any = {
    columns: {
      name: {
        title: 'Environment Name',
        filter: true,
        valuePrepareFunction: (value, row, cell) => {
          const displayVal = {
            url: [
              '/',
              this.blueprint,
              'cluster',
              row?.id,
              'resources',
              'type',
              this.resourceType,
              'resource',
              this.resourceName,
              'configuration',
            ],
            display: value,
            extraParams: { previousComponent: 'blueprint-designer' },
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: SmartTableRouterlinkComponent,
        type: 'custom',
      },
      cloud: {
        title: 'Cloud Provider',
        filter: true,
      },
      releaseStream: {
        title: 'Release Stream',
        filter: true,
      },
    },
    pager: {
      display: false,
      perPage: 10,
    },
    actions: {
      edit: false,
      delete: false,
      add: false,
      position: 'right',
      custom: [],
    },
    hideSubHeader: false,
  };

  constructor(
    private uiStackControllerService: UiStackControllerService,
    private uiDropdownsControllerService: UiDropdownsControllerService,
    activatedRoute: ActivatedRoute,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    const resourceID = p?.resourceID;
    this.resourceType = p?.resourceType;
    this.resourceName = p?.resourceName;
    this.blueprintName = p?.stackName;
    this.getEnvDetails();
    this.getResourceInfoForAllClusters();
  }

  fetchingEnvironments: boolean = true;
  errorInFetchingEnvironments: boolean = false;
  errorMessage: string;
  getEnvDetails() {
    this.fetchingEnvironments = true;
    this.errorInFetchingEnvironments = false;
    this.uiStackControllerService
      .getClustersOverviewUsingGET(this.blueprint)
      .subscribe(
        (res) => {
          this.environmentsOverviewList = res;
          this.setEnvironmentsTableData();
          this.getResourceInfoForAllClusters();
        },
        (error) => {
          this.errorInFetchingEnvironments = true;
          this.errorMessage =
            error?.error?.message ||
            ' There was some error fetching environments';
        },
      )
      .add(() => {
        this.fetchingEnvironments = false;
      });
  }

  setEnvironmentsTableData() {
    let tableData = [];
    this.environmentsOverviewList.forEach((env) => {
      let environment = {};
      environment['id'] = env?.['cluster']['id'];
      environment['name'] = env?.['cluster']['name']
        ? env?.['cluster']['name']
        : '--';
      environment['cloud'] = env['cluster']['cloud'] || '--';
      environment['releaseStream'] = env['cluster']['releaseStream'] || '--';
      environment['resourceEnvStatus'] = '-';
      environment['provided'] = '-';
      environment['overridden'] = '-';
      tableData.push(environment);
    });
    this.environmentsTableData = tableData;
  }

  getResourceInfoForAllClusters() {
    this.uiDropdownsControllerService
      .getAllClusterResourcesByStackUsingGET({
        resourceName: this.resourceName,
        resourceType: this.resourceType,
        stackName: this.blueprintName,
      })
      .subscribe(
        (response) => {
          if (!response?.length) {
            delete this.environmentsTableConfig?.resourceEnvStatus;
            return;
          }
          let tableSettings = cloneDeep(this.environmentsTableConfig);
          tableSettings = {
            ...tableSettings,
            columns: {
              ...tableSettings?.columns,
              resourceEnvStatus: this.resourceEnvStatusColumn,
              overridden: this.overridden,
              provided: this.providedStatus,
            },
          };
          this.environmentsTableConfig = Object.assign({}, tableSettings);

          const envResourceMap = new Map(
            response.map((env) => {
              return [env.clusterId, env];
            }),
          );
          this.environmentsTableData = this.environmentsTableData?.map(
            (envData) => ({
              ...envData,
              resourceEnvStatus:
                resourceStatusMap[
                  `${envResourceMap.get(envData?.id)?.info?.disabled}`
                ] || '',
              provided:
                providedNormalMap[
                  `${envResourceMap.get(envData?.id)?.info?.provided}`
                ] || '',
              overridden:
                overriddenStatusMap[
                  `${envResourceMap.get(envData?.id)?.overridden}`
                ] || '',
            }),
          );
        },
        () => {
          delete this.environmentsTableConfig?.resourceEnvStatus;
        },
      );
  }
}

<ng-container *ngIf="!fetchingEnvironments; else showLoader">
  <nb-card>
    <nb-card-header> Environments </nb-card-header>
    <ng-container>
      <nb-card-body
        [ngClass]="{
          'p-0': !fetchingEnvironments && environmentsTableData?.length
        }"
      >
        <ng2-smart-table
          *ngIf="environmentsTableData?.length"
          [settings]="environmentsTableConfig"
          [source]="environmentsTableData"
        ></ng2-smart-table>
        <app-message
          *ngIf="!environmentsTableData?.length"
          [imgUrl]="
            errorInFetchingEnvironments ? errorStateIcon : emptyStateIcon
          "
          [primaryMsg]="
            errorInFetchingEnvironments ? 'Error occurred' : 'Nothing found'
          "
          [secondaryMsg]="
            errorInFetchingEnvironments
              ? errorMessage
              : 'Resource is not linked to any environment'
          "
          [showBtn]="false"
        ></app-message>
      </nb-card-body>
    </ng-container>
  </nb-card>
</ng-container>
<ng-template #showLoader>
  <nb-card
    *ngIf="!errorInFetchingEnvironments && fetchingEnvironments"
    [nbSpinner]="true"
    class="background-grey-light py-5"
  ></nb-card>
</ng-template>
