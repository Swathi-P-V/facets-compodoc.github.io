import {
  Component,
  Input,
  OnInit,
  Output,
  EventEmitter,
  SimpleChanges,
  OnChanges,
} from '@angular/core';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params } from '@angular/router';
import {
  FormControl,
  FormGroup,
  Validators,
  FormGroupDirective,
} from '@angular/forms';

@Component({
  selector: 'app-multi-select-search',
  templateUrl: './multi-select-search.component.html',
  styleUrls: ['./multi-select-search.component.scss'],
  providers: [FormGroupDirective],
})
export class MultiSelectSearchComponent
  extends FacetsPage
  implements OnChanges
{
  @Input() control: FormControl;
  @Input() value: string | number;
  @Input() dropdownContent: any = {};
  dropdownContentOriginal: any = {};
  @Input() dropDownList: any = [];
  dropDownListOriginal: any = [];
  @Input() name: string = '';
  @Input() placeholder: string = '';
  @Input() size: string = 'small';
  @Input() title: string = '';
  @Input() multi: boolean = false;
  @Input() fieldSize: string = 'small';
  visibleOptions: string[] = [];
  selectedOptionVal: string = '';
  @Output() selectedOption = new EventEmitter<any>();
  @Output() removedItem = new EventEmitter<string>();
  selectedList = [];
  visibleList: { title: string; value: string }[] = [];
  searchString: FormControl = new FormControl('');
  autoComplete: FormControl = new FormControl('');
  @Input() preselectVal: string = '';
  @Input() preselectList: any = [];
  @Input() showClear: boolean = true;
  @Input() formcontrolName: string;
  @Input() groupName: FormGroup;
  @Input() showSpinner: boolean = false; // Set to true to show a spinner on the input field
  @Input() itemName: string = '';
  keysPresent: boolean;

  constructor(
    private activatedRoute: ActivatedRoute,
    private rootFormGroup: FormGroupDirective,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    if (this.dropDownList.length > 0) {
      this.dropDownListOriginal = [...this.dropDownList];
    }
  }

  ngOnChanges(changes: SimpleChanges) {
    if (changes?.preselectVal?.currentValue) {
      this.selectedOptionVal = changes?.preselectVal?.currentValue.toString();
    }
    if (changes?.multi || this.multi) {
      if (changes?.preselectList?.currentValue?.length > 0) {
        this.visibleList = [...changes?.preselectList?.currentValue];
      } else {
        this.visibleList = [];
      }
    }
    if (changes?.dropDownList?.currentValue?.length > 0) {
      this.dropDownListOriginal = changes.dropDownList.currentValue;
    }
    if (changes.showSpinner?.currentValue != null) {
      this.showSpinner = changes.showSpinner.currentValue;
    }
    if (changes.dropdownContent?.currentValue) {
      if (changes.dropdownContent?.currentValue instanceof Map) {
        this.dropdownContentOriginal = Object.fromEntries(
          changes.dropdownContent?.currentValue,
        );
      } else {
        this.dropdownContentOriginal = { ...this.dropdownContent };
      }
      this.searchString.setValue('');
    }
    if (changes.itemName?.currentValue?.length) {
      this.itemName = changes.itemName.currentValue;
    }
    this.keysPresent = Object.keys(this.dropdownContentOriginal)?.length > 0;
  }

  changeResourceVal(event, option, groupOption) {
    if (event) {
      event.stopPropagation();
    }
    let optionPresent = false;
    this.visibleList.forEach((item) => {
      if (
        item.title.toLowerCase() === groupOption.toLowerCase() &&
        item.value.toLowerCase() === option.toLowerCase()
      ) {
        optionPresent = true;
      }
    });
    if (option && optionPresent) {
      let tempList = [];
      this.visibleList.forEach((item) => {
        if (
          item.title.toLowerCase() === groupOption.toLowerCase() &&
          item.value.toLowerCase() === option.toLowerCase()
        ) {
          console.log('item', item);
        } else {
          tempList.push(item);
        }
      });
      this.visibleList = [...tempList];
    } else {
      this.visibleList.push({
        title: groupOption.toLowerCase(),
        value: option.toLowerCase(),
      });
    }
    this.selectedOption.emit({ visibleList: this.visibleList });
  }

  checkResourcePresent(optionCheck, key) {
    let tempChecklist = this.visibleList.filter((option) => {
      if (option.title === key.toLowerCase() && option.value === optionCheck) {
        return true;
      }
    });
    if (tempChecklist.length > 0) {
      return true;
    }
    return false;
  }

  stopEventPropagation(event) {
    event.stopPropagation();
  }

  checkSingleOptionPresent(option) {
    if (option.value.toString() === this.selectedOptionVal.toString()) {
      return true;
    }
    return false;
  }

  onResourceInputChange() {
    this.getFilteredOptions(this.searchString.value);
  }

  getFilteredOptions(values: string) {
    if (this.multi) {
      let filteredOps = {};
      if (values?.length > 0) {
        Object.keys(this.dropdownContentOriginal).forEach((parentOption) => {
          let filteredList = [];
          this.dropdownContentOriginal[parentOption].forEach((item: string) => {
            if (item.toLowerCase().includes(values.toLowerCase())) {
              filteredList.push(item);
            }
          });
          if (filteredList.length > 0) {
            filteredOps[parentOption] = filteredList;
          }
        });
        this.dropdownContent = { ...filteredOps };
      } else {
        this.dropdownContent = { ...this.dropdownContentOriginal };
      }
    } else {
      let filteredList = [];
      if (values?.length > 0) {
        this.dropDownListOriginal.forEach((option) => {
          console.log(
            'option',
            option,
            option.value.toLowerCase().includes(values.toLowerCase()),
          );
          if (option.value.toLowerCase().includes(values.toLowerCase())) {
            filteredList.push(option);
          }
        });
        this.dropDownList = [...filteredList];
      } else {
        this.dropDownList = [...this.dropDownListOriginal];
      }
    }
  }

  removeItem(option) {
    this.visibleList = this.visibleList.filter(
      (item) => item.value.toLowerCase() !== option.value.toLowerCase(),
    );
    //this.changeBtnText();
    let tempList = [];
    this.visibleList.forEach((item) => {
      if (
        item.title.toLowerCase() === option.title.toLowerCase() &&
        item.value.toLowerCase() === option.value.toLowerCase()
      ) {
        console.log('item', item);
      } else {
        tempList.push(item);
      }
    });
    this.visibleList = [...tempList];
    this.selectedOption.emit({ visibleList: this.visibleList });
  }

  selectVal(event) {
    this.selectedOptionVal = event;
    if (this.groupName) {
      this.groupName.get(this.formcontrolName).setValue(event);
    }
    this.selectedOption.emit(event);
  }
}

    <input

      (input)="onResourceInputChange()"
