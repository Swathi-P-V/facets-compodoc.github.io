import { Component, OnDestroy, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import { cloneDeep } from 'lodash';
import { ClipboardService } from 'ngx-clipboard';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { UserService } from 'src/app/services/user.service';
import { VIEW_RESOURCE_SECRETS } from 'src/app/constants/permissions';
import { UiApplicationControllerService } from 'src/app/cc-api/services';
import { SmartTableMultiTypeComponent } from 'src/app/components/smart-table-multi-type/smart-table-multi-type.component';
import { MonacoEditorConstructionOptions } from '@materia-ui/ngx-monaco-editor';
import {
  areObjectKeysSameType,
  areArrayKeysOfType,
} from './interfaces-attributes-utils';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

const errorStateIcon = './assets/error.svg';

@Component({
  selector: 'app-interfaces-attributes',
  templateUrl: './interfaces-attributes.component.html',
  styleUrls: ['./interfaces-attributes.component.scss'],
})
export class InterfacesAttributesComponent
  extends FacetsPage
  implements OnDestroy
{
  clusterId: any;
  resourceType: any;
  resourceName: any;
  //Booleans
  loading: boolean = false;
  hasPasswordViewPermission: boolean = false;
  showPassword: boolean = false;
  showInterfacePassword: boolean = false;
  showAttributePassword: boolean = false;
  //Interfaces and attributes raw and table data
  interfacesRawData = {};
  attributesRawData = {};
  interfaces = [];
  attributes = [];
  cummulativeInterfaceTableData: {} = {};
  //Interfaces dropdown
  interfaceInstances = [];
  selectedInterfaceInstance = 'writer';
  //error state props
  errorMessage: string = '';
  errorStateIcon = errorStateIcon;
  //subscriptions
  userSubscription: any;
  smartTableCustomActionButtons: any = [
    {
      id: 'reference-value',
      icon: 'dollar-outline',
      isFciPack: true,
      toolTip: 'Get reference value',
      status: 'primary',
      size: 'small',
      iconStatus: 'primary',
      appearance: 'ghost',
      fciIconFontSize: '1rem',
      onClick: (data) => {
        this.showReferenceVariable(data);
      },
    },
    {
      id: 'copy-to-clipboard',
      icon: 'copy-outline',
      isFciPack: true,
      status: 'primary',
      size: 'small',
      appearance: 'ghost',
      iconStatus: 'primary',
      fciIconFontSize: '1rem',
      onClick: (rowData) => {
        this.copyToClipboard(rowData);
      },
    },
  ];
  tableSettings: any = {
    columns: {
      key: {
        title: 'Key',
        filter: false,
        type: 'text',
        width: '20%',
      },
      value: {
        title: 'Value',
        filter: false,
        type: 'custom',
        width: '70%',
        valuePrepareFunction: (cell, row) => {
          let component = this.getComponentAndData(cell, row?.key);
          return JSON.stringify(component);
        },
        renderComponent: SmartTableMultiTypeComponent,
        onComponentInitFunction: (instance) => {
          instance?.byClickCellWithEllipsisLink?.subscribe((data) => {
            this.openAttributesModal(data);
          });

          instance?.byClickCellWithRouterLink?.subscribe((data: any) => {
            this.openAttributesModal(data);
          });
        },
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        filter: false,
        width: '10%',
        valuePrepareFunction: (value, row, cell) => {
          let actionButtons = cloneDeep(this.smartTableCustomActionButtons);
          //Based on few conditions we need to show or hide the action buttons
          if (!this.showCopyToClipboardAction(row)) {
            actionButtons = actionButtons.filter(
              (button) => button.id !== 'copy-to-clipboard',
            );
          }
          if (!this.showDollarReferenceAction(row)) {
            actionButtons = actionButtons.filter(
              (button) => button.id !== 'reference-value',
            );
          }
          const displayVal = {
            smartTableCustomActionButtons: actionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
      perPage: 15,
    },
    mode: 'external',
    actions: false,
  };

  @ViewChild('referenceVarTemplate', { static: true })
  private referenceVarTemplate: TemplateRef<any>;
  @ViewChild('attributeExpandModal', { static: true })
  private attributeExpandModal: TemplateRef<any>;

  constructor(
    private _route: ActivatedRoute,
    private clipboardService: ClipboardService,
    private dialogService: NbDialogService,
    private userService: UserService,
    private uiApplicationController: UiApplicationControllerService,
    private customToastService: CustomToastService,
  ) {
    super(_route);
  }

  OnPageLoad(params: Params): void {
    this.loading = true;
    this.clusterId = params?.clusterId;
    this.resourceType = params?.resourceType;
    this.resourceName = params?.resourceName;

    this.resetComponentPropsOnPageLoad();

    if (this.userService.userLoaded) {
      let auth = this.userService.user.authorities.find(
        (ele) => ele.authority === VIEW_RESOURCE_SECRETS,
      );
      this.hasPasswordViewPermission = auth !== undefined;
    } else {
      this.userSubscription = this.userService.userObserver.subscribe(
        (userLoaded) => {
          if (userLoaded) {
            let auth = this.userService.user.authorities.find(
              (ele) => ele.authority === VIEW_RESOURCE_SECRETS,
            );
            this.hasPasswordViewPermission = auth !== undefined;
          }
        },
      );
    }

    this.getInterfacesAndAttributes(
      this.clusterId,
      this.resourceName,
      this.resourceType,
    );
  }

  resetComponentPropsOnPageLoad() {
    this.interfacesRawData = {};
    this.attributesRawData = {};
    this.cummulativeInterfaceTableData = {};
    this.interfaces = [];
    this.attributes = [];
    this.showPassword = false;
    this.showAttributePassword = false;
    this.showInterfacePassword = false;
    this.interfaceInstances = [];
    this.selectedInterfaceInstance = 'writer';
    this.errorMessage = '';
  }

  onInterfaceShowPasswordToggled(event) {
    this.showInterfacePassword = event;

    //Construct table data for all interfaces
    Object.keys(this.interfacesRawData).forEach((key) => {
      this.cummulativeInterfaceTableData[key] = this.constructTableData(
        this.interfacesRawData[key],
        'interface',
      );
    });

    //Set the interfaces value to selected option from dropdown
    this.interfaces =
      this.cummulativeInterfaceTableData[this.selectedInterfaceInstance];
  }

  onAttributeShowPasswordToggled(event) {
    this.showAttributePassword = event;

    if (Object.keys(this.attributesRawData)?.length > 0) {
      this.attributes = this.constructTableData(
        this.attributesRawData,
        'attribute',
      );
    }
  }

  copyToClipboard({ key, value }) {
    this.clipboardService.copy(value);
    this.customToastService.show({
      title: 'Copied!',
      content: `${key} copied`,
      duration: 1000,
      type: 'success',
    });
  }

  showReferenceVariable(rowData) {
    let referenceVar = '';
    if (rowData?.type === 'interface') {
      referenceVar = `\${${this.resourceType}.${
        this.resourceName
      }.out.interfaces.${
        this.selectedInterfaceInstance
      }.${rowData.key.toLowerCase()}}`;
    } else {
      referenceVar = `\${${this.resourceType}.${
        this.resourceName
      }.out.attributes.${this.getKeyPath(
        this.attributesRawData,
        rowData?.key,
      )}}`;
    }

    this.dialogService.open(this.referenceVarTemplate, {
      context: { reference: referenceVar, rowData: rowData },
    });
  }

  getKeyPath(obj, targetKey, currentPath = '') {
    for (const key in obj) {
      if (key === targetKey) {
        return currentPath ? `${currentPath}.${key}` : key;
      }

      const value = obj[key];
      if (typeof value === 'object' && value !== null) {
        const nestedPath = this.getKeyPath(
          value,
          targetKey,
          currentPath ? `${currentPath}.${key}` : key,
        );
        if (nestedPath) {
          return nestedPath;
        }
      }
    }

    return null;
  }

  getInterfacesAndAttributes(clusterId, resourceName, resourceType) {
    this.loading = true;
    this.uiApplicationController
      .getResourceOutPropertiesUsingGET({
        clusterId: clusterId,
        resourceName: resourceName,
        resourceType: resourceType,
      })
      .subscribe(
        (response) => {
          this.errorMessage = '';
          this.processApiData(response);
        },
        (error) => {
          this.loading = false;
          this.errorMessage =
            error?.error?.message ||
            `Failed to fetch interfaces and attributes for ${resourceType}:${resourceName}`;
        },
      );
  }

  processApiData(res) {
    this.interfacesRawData = res?.interfaces;
    this.attributesRawData = res?.attributes;

    if (Object.keys(this.interfacesRawData)?.length > 0) {
      //Form dropdown based on keys in interfaces object and set first dropdown value as default selected.
      this.interfaceInstances = this.constructDropdown(this.interfacesRawData);
      this.selectedInterfaceInstance = this.interfaceInstances[0].value;

      //Construct table data for all interfaces
      Object.keys(this.interfacesRawData).forEach((key) => {
        this.cummulativeInterfaceTableData[key] = this.constructTableData(
          this.interfacesRawData[key],
          'interface',
        );
      });

      //Set the interfaces value to selected option from dropdown
      this.interfaces =
        this.cummulativeInterfaceTableData[this.selectedInterfaceInstance];
    }

    if (Object.keys(this.attributesRawData)?.length > 0) {
      this.attributes = this.constructTableData(
        this.attributesRawData,
        'attribute',
      );
    }
    this.loading = false;
  }

  constructDropdown(interfaces) {
    let options = [];
    options = Object.keys(interfaces).map((key) => {
      return {
        label: key,
        value: key,
      };
    });
    return options;
  }

  //This function takes an object and converts it into array of objects.
  //Each object has properties - key and value which holds data of each key-value pair of obj.
  constructTableData(instance, type: 'interface' | 'attribute') {
    let tableData = [];
    let secrets = instance?.secrets;
    Object.entries(instance).forEach(([key, value]) => {
      if (key !== 'secrets') {
        tableData.push({
          key: key,
          value: this.getValue(
            key,
            value,
            secrets,
            type === 'interface'
              ? this.showInterfacePassword
              : this.showAttributePassword,
          ),
          isSecret: secrets?.includes(key),
          type: type,
        });
      }
    });
    return tableData;
  }

  //This function is used to get actual value or asterisk based on whether user has permission to view secrets variables and whether show password is turned on.
  getValue(key, value, secrets, showPassword) {
    if (secrets?.includes(key)) {
      if (this.hasPasswordViewPermission) {
        return showPassword ? value : '*****';
      } else {
        return '*****';
      }
    }
    return value;
  }

  onInterfaceChange() {
    this.interfaces =
      this.cummulativeInterfaceTableData[this.selectedInterfaceInstance];
  }

  openAttributesModal(data) {
    let modalProps: any = {};
    //For router-link component we get entire data, so use extraData to get popupType and values
    //For link-ellipsis component we get directly the extraData. So directly access popupType and values
    if (data?.extraData?.popupType === 'table' || data?.popupType === 'table') {
      let formattedTableData = [];
      let rawTableData = data?.extraData?.populateData || data?.populateData;
      formattedTableData = Object.entries(rawTableData).map(([key, value]) => {
        return {
          key: key,
          value: value,
        };
      });
      let clonedTableSettings = cloneDeep(this.tableSettings);
      if (formattedTableData?.length === 0) {
        clonedTableSettings.noDataMessage = `No ${
          data?.extraData?.name || data?.name
        } found.`;
      }
      modalProps = {
        name: data?.extraData?.name || data?.name,
        type: 'table',
        tableData: formattedTableData,
        tableSettings: clonedTableSettings,
      };
    } else if (data?.extraData?.popupType === 'json-editor') {
      const editorOptions: MonacoEditorConstructionOptions = {
        theme: 'vs-light',
        language: 'json',
        readOnly: true,
        automaticLayout: true,
        lineNumbers: 'on',
        minimap: {
          enabled: false,
        },
      };
      modalProps = {
        name: data?.extraData?.name,
        type: 'json-editor',
        editorData: JSON.stringify(data?.extraData?.populateData, null, 2),
        editorOptions,
      };
    }
    this.dialogService.open(this.attributeExpandModal, {
      context: {
        ...modalProps,
      },
    });
  }

  getComponentAndData(value, key): { type: string; props: {} } {
    let type = 'default';
    let props = {};
    //If value is string or null or undefined then render it as a plain text. For null and undefined render '-'
    if (typeof value === 'string' || value === null || value === undefined) {
      props = {
        value: value || '-',
      };
    } else if (Array.isArray(value)) {
      /* If value is array, then handle following conditions
         -empty array
         -array of strings
         -array of objects
     */
      if (value.length === 0) {
        props = {
          value: JSON.stringify(value),
        };
      } else {
        if (areArrayKeysOfType(value, 'string')) {
          let cellCustomComponentData = value.map((element) => ({
            name: element,
          }));
          type = 'link-ellipsis';
          props = {
            data: cellCustomComponentData,
            decorate: true,
            status: 'success',
            defaultDisplay: 3,
            showAsLink: false,
            dataObject: true,
          };
        } else if (areArrayKeysOfType(value, 'object')) {
          const mergedObject = value.reduce(
            (acc, obj) => ({ ...acc, ...obj }),
            {},
          );
          type = 'router-link';
          props = this.constructRouterLinkProps('View List', key, mergedObject);
        }
      }
    } else {
      /* If value is JSON object, then handle following conditions
         -empty object
         -Object with different data types
         -Object with same data type : strings or objects
     */
      let keys = Object.keys(value);
      if (keys?.length === 0) {
        props = {
          value: JSON.stringify(value),
        };
      } else {
        if (!areObjectKeysSameType(value)) {
          type = 'router-link';
          props = this.constructRouterLinkProps(
            'View ' + `${key}`,
            key,
            value,
            'json-editor',
          );
        } else {
          //As all values are of same type we can get the typeof of value from first element
          let firstElement = value[keys[0]];
          if (typeof firstElement === 'string' || Array.isArray(firstElement)) {
            type = 'router-link';
            props = this.constructRouterLinkProps(
              'View ' + `${key}`,
              key,
              value,
            );
          } else if (typeof firstElement === 'object') {
            let cellCustomComponentData = Object.entries(value)?.map(
              ([key, val]) => ({
                name: key,
                populateData: val,
                popupType: 'table',
              }),
            );
            type = 'link-ellipsis';
            props = {
              data: cellCustomComponentData,
              extraData: cellCustomComponentData,
              decorate: false,
              status: 'success',
              defaultDisplay: 3,
              showAsLink: true,
              dataObject: true,
              linkUnderline: true,
            };
          }
        }
      }
    }
    return { type, props };
  }

  /* Do not show action buttons for following conditions
     - if value is null, undefined, empty string/array/object
     - if value is marked secret and user do not have permission to view secret values
     - if value is marked secret, user has permission to view secret but show password is turned-off
     - if value is array of objects
     - if value is JSON object
  */
  showCopyToClipboardAction(row) {
    const hasValue = row?.value && row.value.length;
    const isSecret = row?.isSecret;
    const showPasswordEnabled =
      row.type === 'interface'
        ? this.showInterfacePassword
        : this.showAttributePassword;
    if (
      !hasValue ||
      (isSecret && (!this.hasPasswordViewPermission || !showPasswordEnabled))
    ) {
      return false;
    } else if (
      Array.isArray(row.value) &&
      areArrayKeysOfType(row.value, 'object')
    ) {
      return false;
    } else if (typeof row?.value === 'object' && !Array.isArray(row?.value)) {
      return false;
    }
    return true;
  }

  showDollarReferenceAction(row) {
    if (Array.isArray(row.value) && areArrayKeysOfType(row.value, 'object')) {
      return false;
    } else if (
      typeof row?.value === 'object' &&
      row.value !== null &&
      !Array.isArray(row?.value)
    ) {
      return false;
    }
    return true;
  }

  constructRouterLinkProps(
    display: string,
    name: string,
    populateData,
    popupType = 'table',
  ) {
    return {
      url: null,
      display: display,
      extraData: {
        name: name,
        populateData: populateData,
        popupType: popupType,
      },
    };
  }

  ngOnDestroy() {
    this.userSubscription?.unsubscribe();
  }
}
