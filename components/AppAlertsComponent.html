import { Component, OnInit, ViewChild } from '@angular/core';
import { NbPopoverDirective } from '@nebular/theme';
import { interval } from 'rxjs';
import { switchMap } from 'rxjs/operators';
import { UiAlertsControllerService } from 'src/app/cc-api/services';
import { Router } from '@angular/router';
import { AlertOverview } from './../../cc-api/models/alert-overview';

@Component({
  selector: 'app-alerts',
  templateUrl: './alerts.component.html',
  styleUrls: ['./alerts.component.scss'],
})
export class AppAlertsComponent implements OnInit {
  @ViewChild(NbPopoverDirective) popover: NbPopoverDirective;
  warningAlertsData: any = [];
  criticalAlertsData: any = [];
  alertsData: AlertOverview[] = [];
  loadingAlerts: boolean = false;
  alertsEmptyIcon: string = './assets/empty-env.svg';
  errorIcon: string = './assets/error-red.svg';
  showErrorState: boolean = false;
  errorMessage: string = '';
  loadingTotalAlerts: boolean = false;
  totalAlertsCount: number = 0;
  pollInterval: number = 60000;
  tabsData = [
    {
      title: 'Critical State',
      data: this.criticalAlertsData,
      badgeStatus: 'danger',
      count: '0',
    },
    {
      title: 'Risk State',
      data: this.warningAlertsData,
      badgeStatus: 'warning',
      count: '0',
    },
  ];

  constructor(
    private alertsControllerService: UiAlertsControllerService,
    private router: Router,
  ) {}

  ngOnInit(): void {
    this.getTotalAlerts();
    this.getAlertsCount();
  }

  getAlertsData(): void {
    this.alertsData = [];
    this.criticalAlertsData = [];
    this.warningAlertsData = [];
    this.loadingAlerts = true;
    this.showErrorState = false;
    this.alertsControllerService.getAlertsOverviewUsingGET().subscribe(
      (data) => {
        this.alertsData = data;
        this.alertsData.forEach((alert) => {
          if (alert?.stackName?.length) {
            if (alert?.severityCounts?.critical > 0) {
              this.criticalAlertsData.push(alert);
            }
            if (alert?.severityCounts?.warning > 0) {
              this.warningAlertsData.push(alert);
            }
          }
        });
        this.tabsData[0].data = this.criticalAlertsData;
        this.tabsData[0].count = this.criticalAlertsData.reduce(
          (sum, alert) => sum + alert?.severityCounts?.critical,
          0,
        );
        this.tabsData[1].data = this.warningAlertsData;
        this.tabsData[1].count = this.warningAlertsData.reduce(
          (sum, alert) => sum + alert?.severityCounts?.warning,
          0,
        );
        this.loadingAlerts = false;
      },
      (err) => {
        this.loadingAlerts = false;
        this.showErrorState = true;
        this.errorMessage =
          err?.error?.message || 'There was some error fetching alerts data';
      },
    );
  }

  /*
   * Interval to call api every 5 seconds.
   * Switchmap to run only 1 request at any point in time.
   * get response and call getTotalAlerts again
   */
  getTotalAlerts() {
    interval(this.pollInterval)
      .pipe(
        switchMap(() => this.alertsControllerService.getAlertsCountUsingGET()),
      )
      .subscribe(
        (alertsCountResponse: any) => {
          this.totalAlertsCount = alertsCountResponse?.total;
        },
        () => {
          this.getTotalAlerts();
        },
      );
  }

  navigateToAlerts(alert) {
    this.closeAlertPopover();
    this.router.navigate([
      '/',
      alert?.stackName,
      'cluster',
      alert?.clusterId,
      'alerts',
    ]);
  }

  openAlertsPopover() {
    if (!this.popover.isShown) {
      this.getAlertsData();
      this.getAlertsCount();
    }
  }

  getAlertsCount() {
    this.alertsControllerService
      .getAlertsCountUsingGET()
      .subscribe((alertsCountResponse) => {
        this.totalAlertsCount = alertsCountResponse?.total;
      });
  }

  closeAlertPopover() {
    this.popover.hide();
  }
}
