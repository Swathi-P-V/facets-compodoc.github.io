import { Component, ViewChild, TemplateRef } from '@angular/core';
import { UiCommonClusterControllerService } from '../../../cc-api/services/ui-common-cluster-controller.service';
import { Router, ActivatedRoute, Params } from '@angular/router';
import { StackTemplateInput } from '../../../cc-api/models/stack-template-input';
import { NbDialogService } from '@nebular/theme';
import cloneDeep from 'lodash/cloneDeep';
import { UiStackControllerService } from '../../../cc-api/services/ui-stack-controller.service';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { TEMPLATE_INPUTS_DOCS_LINK } from 'src/app/constants/urls';
import { TEMPLATE_WRITE, TEMPLATE_DELETE } from 'src/app/constants/permissions';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

const emptyStateLogo = './assets/error.svg';
const DEFAULT_COLUMN_WIDTH = '240px';

@Component({
  selector: 'app-cluster-template-inputs',
  templateUrl: './cluster-template-inputs.component.html',
  styleUrls: ['./cluster-template-inputs.component.scss'],
})
export class ClusterTemplateInputsComponent extends FacetsPage {
  private clusterId: any;

  templates: { [p: string]: Array<StackTemplateInput> } = {};
  isTemplatesDefined: boolean;
  fetchingTemplate: boolean = false;
  deletionInProgress: boolean = false;
  emptyStateLogo = emptyStateLogo;
  creationInProgress: boolean = false;
  tableSourceData: {} = {};
  errorFetchingTemplateInputs: string = '';
  errorFetchingTemplateInputTypes: string = '';
  fetchingTemplateInputsMeta: boolean = true;
  metaData: {} = {};
  selectedTab: any;
  stackName: any;
  templateWritePermission: string = TEMPLATE_WRITE;
  templateDeletePermission: string = TEMPLATE_DELETE;

  smartTableCustomActionButtons = [
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete Template',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: this.templateDeletePermission,
      onClick: (data) => {
        this.deleteTemplateConfirmPopup(data);
      },
    },
  ];

  tableCustomColumn = {
    type: 'custom',
    title: 'Actions',
    valuePrepareFunction: (value, row, cell) => {
      //Considering uid as uniquie column here
      let smartTableCustomActionButtons = cloneDeep(
        this.smartTableCustomActionButtons,
      );
      smartTableCustomActionButtons.forEach((button) => {
        button['uniqueId'] = button.id + '_' + row.uid;
      });
      // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
      const displayVal = {
        smartTableCustomActionButtons,
      };
      return JSON.stringify(displayVal);
    },
    renderComponent: CustomActionsComponent,
    onComponentInitFunction: (instance) => {
      instance.byCustomActionButtonClick.subscribe((data) => {
        let customActionButton = this.smartTableCustomActionButtons.filter(
          (actionButton) => {
            return actionButton.id === data.actionButton.id;
          },
        )[0];
        customActionButton.onClick(data.rowData);
      });
    },
    width: '140px',
    filter: false,
  };

  tableColumns = {
    uid: {
      title: 'Unique Id',
      filter: true,
      editable: false,
      width: '240px',
    },
  };

  settings = {
    noDataMessage: 'No Templates Found',
    columns: this.tableColumns,
    hideSubHeader: false,
    actions: false,
    pager: {
      display: false,
    },
  };

  templateInputActionButtons = [
    {
      id: 'create-template-input-button',
      title: 'Create Template Input Type',
      buttonType: 'primary',
      nbTooltip: 'Create new Template input type. Coming Soon',
      disabled: true,
      action: undefined,
      hidden: false,
    },
  ];

  @ViewChild('deleteTemplateUI', { static: true })
  private deleteTemplateUI: TemplateRef<any>;

  constructor(
    private commonClusterService: UiCommonClusterControllerService,
    private activatedRoute: ActivatedRoute,
    private dialogService: NbDialogService,
    private stackControllerService: UiStackControllerService,
    private router: Router,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {
    /** This query param is needed to show the previously selected tab back when they navigate from creation flow */
    this.activatedRoute.queryParams.subscribe((params) => {
      if (params && params['templateInputType']) {
        this.selectedTab = params['templateInputType'];
      }
    });
  }

  OnPageLoad(p: Params): void {
    this.templates = {};
    this.metaData = {};
    this.tableSourceData = {};
    this.clusterId = p?.clusterId;
    this.stackName = p?.stackName;
    this.getAllTemplateInputsAndMeta();
  }

  getAllTemplateInputsAndMeta() {
    this.fetchingTemplateInputsMeta = true;
    this.fetchingTemplate = true;
    this.getTemplatesData();
    this.getTemplatesMeta();
  }

  getTemplatesData() {
    this.commonClusterService
      .getAllTemplateInputsUsingGET(this.clusterId)
      .subscribe(
        (response) => {
          this.templates = response;
          this.tableSourceData = {};
          this.prepareInputsTableSourceData();
          this.fetchingTemplate = false;
          this.errorFetchingTemplateInputs = '';
        },
        (error) => {
          this.templates = {};
          this.tableSourceData = {};
          this.fetchingTemplate = false;
          this.errorFetchingTemplateInputs =
            error?.error?.message ||
            error?.message ||
            'There is some error fetching template inputs';
        },
      );
  }

  getTemplatesMeta() {
    this.stackControllerService
      .getAllTemplateInputsMetaUsingGET(this.stackName)
      .subscribe(
        (response) => {
          this.metaData = response;
          this.fetchingTemplateInputsMeta = false;
          this.errorFetchingTemplateInputTypes = '';

          this.isTemplatesDefined = Object.keys(this.metaData)?.length !== 0;

          if (this.isTemplatesDefined) {
            this.prepareInputsTableSettings(Object.keys(this.metaData)?.[0]);
          }
        },
        (error) => {
          this.metaData = null;
          this.isTemplatesDefined = false;
          this.fetchingTemplateInputsMeta = false;
          this.errorFetchingTemplateInputTypes =
            error?.error?.message ||
            error?.message ||
            'There is some error fetching template input types';
        },
      );
  }

  prepareInputsTableSourceData() {
    /* Pushing the dynamic fields to table colum settings,
    to show the dynamic fields in the flatten structure in table */
    Object.entries(this.templates).forEach(([key, value]) => {
      this.tableSourceData[key] = [];
      this.templates[key].forEach(({ data, ...input }) => {
        this.tableSourceData[key].push({
          ...input,
          ...data,
        });
      });
    });
  }
  prepareInputsTableSettings(templateInput) {
    /* Number of columns for each input type might be diffrent,
    hence prepare the table column settings dynamically based on the active tab */
    const newSettings = {
      ...this.settings,
      columns: {
        ...this.tableColumns,
      },
    };

    this.metaData[templateInput]?.spec?.fields?.forEach((field) => {
      newSettings.columns[field] = {
        title: field,
        filter: true,
        editable: false,
        width: DEFAULT_COLUMN_WIDTH,
      };
    });
    newSettings.columns['custom'] = this.tableCustomColumn;
    this.settings = Object.assign({}, newSettings);
  }

  tabSelected($event: any) {
    this.selectedTab = $event.tabTitle;
    this.prepareInputsTableSettings($event.tabTitle);
  }

  deleteTemplateConfirmPopup(rowData) {
    this.dialogService.open(this.deleteTemplateUI, { context: rowData });
  }

  deleteTemplate(data, ref) {
    this.deletionInProgress = true;
    let templateDeleteParams = {
      uid: data.uid,
      inputType: data.inputType,
      clusterId: this.clusterId,
    };
    this.commonClusterService
      .deleteTemplateInputsUsingDELETE(templateDeleteParams)
      .subscribe(
        (t) => {
          this.customToastService.success(
            'Template input successfully deleted ',
          );
          this.deletionInProgress = false;
          ref.close();
          this.getAllTemplateInputsAndMeta();
        },
        (error) => {
          this.deletionInProgress = false;
          ref.close();
          const errorMessage = error?.error?.message || error?.message;
          this.onAPIErrorService.onAPIError(
            'Template input delete failed. ' + errorMessage,
            error,
          );
        },
      );
  }
  openTemplateInputsDocs() {
    window.open(TEMPLATE_INPUTS_DOCS_LINK, '_blank');
  }
  navigateToCreateInput(templateInputType) {
    this.router.navigate([`../template-inputs/${templateInputType}/create`], {
      relativeTo: this.route,
    });
  }
}
