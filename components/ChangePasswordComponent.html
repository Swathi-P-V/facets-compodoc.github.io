import { Component, OnInit } from '@angular/core';
import {
  FormControl,
  FormGroup,
  Validators,
  FormBuilder,
} from '@angular/forms';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { ApplicationControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-change-password',
  templateUrl: './change-password.component.html',
  styleUrls: ['./change-password.component.scss'],
})
export class ChangePasswordComponent extends FacetsPage {
  changePasswordModel: FormGroup;
  showPassword: boolean = false; // Toggle for showing and hiding password for 'create' and 'changePassword' modes
  changePassInProgress: boolean = false;
  userId: string;
  showConfirmPassword: boolean = false;

  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private applicationController: ApplicationControllerService,
    private fb: FormBuilder,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.initChangePasswordModel();
    this.userId = p?.userId;
  }

  // Initializes the newPasswordModel form
  initChangePasswordModel() {
    this.changePasswordModel = this.fb.group(
      {
        password: new FormControl('', [
          Validators.required,
          Validators.minLength(8),
          Validators.maxLength(20),
          Validators.pattern(
            '^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.*[!@#$%^&*])([a-zA-Z0-9!@#$%^&*]+)',
          ),
        ]),
        confirmedPassword: new FormControl('', [Validators.required]),
      },
      {
        validator: this.confirmedValidator('password', 'confirmedPassword'),
      },
    );
  }

  confirmedValidator(controlName: string, matchingControlName: string) {
    return (formGroup: FormGroup) => {
      const control = formGroup?.controls[controlName];
      const matchingControl = formGroup?.controls[matchingControlName];
      if (
        matchingControl?.errors &&
        !matchingControl?.errors?.confirmedValidator
      ) {
        return;
      }
      if (control?.value !== matchingControl?.value) {
        matchingControl.setErrors({ confirmedValidator: true });
      } else {
        matchingControl.setErrors(null);
      }
    };
  }

  get password() {
    return this.changePasswordModel.get('password');
  }

  get confirmedPassword() {
    return this.changePasswordModel.get('confirmedPassword');
  }

  // Maps password input box's type
  getInputType() {
    if (this.showPassword) {
      return 'text';
    }
    return 'password';
  }

  // Toggles show/hide for password input box
  toggleShowPassword() {
    this.showPassword = !this.showPassword;
  }

  // Redirects back to the parent component [user management]
  backToUserManagement = () => {
    this.router.navigate(['/settings/users']);
  };

  // Calls backend to update exisiting user's password
  updatePassword() {
    this.changePassInProgress = true;
    const payLoad = {
      userId: this.userId,
      pwdChange: {
        newPassword: this.changePasswordModel.get('password').value,
      },
    };

    this.applicationController.changePasswordUsingPUT(payLoad).subscribe(
      (rsp) => {
        this.changePassInProgress = false;
        this.customToastService.success(`Updated password successfully`);
        this.backToUserManagement();
      },
      (err) => {
        this.changePassInProgress = false;
        this.onAPIErrorService.onAPIError(
          err?.error?.message || `Could not update the password`,
          err,
        );
      },
    );
  }

  toggleConfirmPassword() {
    this.showConfirmPassword = !this.showConfirmPassword;
  }
}
