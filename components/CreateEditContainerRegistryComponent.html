import { Component } from '@angular/core';
import { UiArtifactoryControllerService } from '../../../cc-api/services/ui-artifactory-controller.service';
import { ECRArtifactory } from '../../../cc-api/models/ecrartifactory';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { AccessDeniedService } from 'src/app/services/access-denied.service';
import { ARTIFACTORY_WRITE } from './../../../constants/permissions';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { extractAwsInfoFromUri } from 'src/app/utils/aws.utils';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { getToastSubTexts } from 'src/app/utils/toast.utils';
import { BasicDockerArtifactory, BasicDockerDTO } from 'src/app/cc-api/models';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-create-edit-container-registry',
  templateUrl: './create-edit-container-registry.component.html',
  styleUrls: ['./create-edit-container-registry.component.scss'],
})
export class CreateEditContainerRegistryComponent extends FacetsPage {
  updateInProgress: boolean = false;
  fetchingContainerRegistry: boolean = false;
  readonly ARTIFACTORY_WRITE = ARTIFACTORY_WRITE;

  mode: 'create' | 'edit' = 'create';
  containerRegistryForm = new FormGroup({
    registryName: new FormControl('', [

      Validators.minLength(3),
      Validators.pattern('[a-zA-Z0-9-]+'),
      Validators.pattern(whiteSpaceInputPattern),
    ]),
    registryType: new FormControl('ECR', [
      Validators.required,
      Validators.pattern(whiteSpaceInputPattern),
    ]),
  });

  registryTypeDropdown = [
    {
      label: 'AWS ECR',
      value: 'ECR',
    },
    {
      label: 'Other',
      value: 'OTHERS',
      tooltip:
        'Choose Others for DockerHub, Nexus, GAR (Google Artifact Registry), ACR (Azure Container Registry).',
    },
  ];

  registryAPIInProgress: boolean = false;

  readonly defaultControlFieldNames = ['registryName', 'registryType'];

  readonly fieldsBasedOnRegitryType = {
    ECR: [
      {
        fieldName: 'Registry URL',
        inputType: 'string',
        required: true,
        tooltip: 'Provide the URL of your ECR container registry.',
        placeholder: 'Registry URL',
        id: 'uri',
        formControl: 'uri',
        invalidPatternMsg: '',
        invalidLengthMsg: '',
        defaultValue: '',
        validators: [
          Validators.required,
          Validators.pattern(whiteSpaceInputPattern),
        ],
      },
      {
        fieldName: 'AWS Key',
        inputType: 'string',
        required: true,
        tooltip: 'AWS Key which has access to this ECR Repository',
        placeholder: 'AWS key',
        id: 'awsKey',
        formControl: 'awsKey',
        invalidPatternMsg: '',
        invalidLengthMsg: '',
        defaultValue: '',
        validators: [
          Validators.required,
          Validators.pattern(whiteSpaceInputPattern),
        ],
      },
      {
        fieldName: 'AWS Secret',
        inputType: 'string',
        required: true,
        tooltip: 'AWS Secret for the above key',
        placeholder: 'AWS Secret',
        id: 'awsSecret',
        formControl: 'awsSecret',
        invalidPatternMsg: '',
        invalidLengthMsg: '',
        defaultValue: '',
        validators: [
          Validators.required,
          Validators.pattern(whiteSpaceInputPattern),
        ],
      },
    ],
    OTHERS: [
      {
        fieldName: 'Registry URL',
        inputType: 'string',
        required: true,
        tooltip:
          'Provide the URL of your container registry. This is the address where your registry is hosted and accessible.',
        placeholder: 'Registry URL',
        id: 'uri',
        formControl: 'uri',
        invalidPatternMsg: '',
        invalidLengthMsg: '',

        defaultValue: '',

          Validators.required,
          Validators.pattern(whiteSpaceInputPattern),
        ],
      },

  containerRegistry: ECRArtifactory | BasicDockerArtifactory;
  registryId: string;
  constructor(
    route: ActivatedRoute,
    private router: Router,
    private artifactoryControllerService: UiArtifactoryControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(route);
    this.setFormControlsBasedOnRegistryType();
  }

  get registryName() {
    return this.containerRegistryForm.get('registryName');
  }

  get registryType() {
    return this.containerRegistryForm.get('registryType');
  }

  get uri() {
    return this.containerRegistryForm.get('uri');
  }

  get awsKey() {
    return this.containerRegistryForm.get('awsKey');
  }

  get awsSecret() {
    return this.containerRegistryForm.get('awsSecret');
  }

  get username() {
    return this.containerRegistryForm.get('username');
  }

  get password() {
    return this.containerRegistryForm.get('password');
  }

  OnPageLoad(params: Params): void {
    this.registryId = params?.registryId;
    if (params?.registryId) {
      this.mode = 'edit';
      this.getContainerRegistry(params?.registryId);
    }
  }

  setFormControlsBasedOnRegistryType(apiResponse?) {
    Object.keys(this.containerRegistryForm?.controls)?.forEach(
      (controlFieldName) => {
        if (!this.defaultControlFieldNames?.includes(controlFieldName)) {
          this.containerRegistryForm?.removeControl(controlFieldName);
        }
      },
    );

    this.fieldsBasedOnRegitryType?.[this.registryType.value]?.forEach(
      (field) => {
        const defaultValue =
          this.mode === 'create'
            ? field?.defaultValue
            : apiResponse?.[field.formControl];
        this.containerRegistryForm?.addControl(
          field?.id,
          new FormControl(defaultValue, field?.validators),
        );
      },
    );
  }

  getContainerRegistry(registryId) {
    this.fetchingContainerRegistry = true;
    this.artifactoryControllerService
      .getArtifactoryByIdUsingGET(registryId)
      .subscribe(
        (response) => {
          const registry = response;
          this.containerRegistryForm.patchValue({
            registryName: registry.name,
            registryType: response?.artifactoryType,
          });
          this.setFormControlsBasedOnRegistryType(response);
          this.fetchingContainerRegistry = false;
        },
        (error) => {
          this.fetchingContainerRegistry = false;
        },
      );
  }

  getDockerHubRequestPayload() {
    const requestPayload: BasicDockerDTO = {
      artifactoryName: this.registryName?.value?.trim(),
      uri: this.uri?.value?.trim(),
      username: this.username?.value?.trim(),
      password: this.password?.value?.trim(),
    };
    return requestPayload;
  }

  getECRRequestPayload() {
    const { awsRegion, awsAccountId } =
      extractAwsInfoFromUri(this.uri?.value) || {};
    const requestPayload: ECRArtifactory = {
      awsKey: this.awsKey?.value?.trim(),
      awsSecret: this.awsSecret?.value?.trim(),
      name: this.registryName?.value?.trim(),
      awsAccountId: awsAccountId?.trim(),
      awsRegion: awsRegion?.trim(),
      uri: this.uri?.value?.trim(),
      artifactoryType: 'ECR',
    };
    return requestPayload;
  }
  getAPISignattureForECR() {
    switch (this.mode) {
      case 'create': {
        return this.artifactoryControllerService.createECRArtifactoryUsingPOST1(
          this.getECRRequestPayload(),
        );
      }
      case 'edit': {
        return this.artifactoryControllerService.updateECRArtifactoryUsingPOST({
          artifactoryId: this.registryId,
          ecrArtifactory: this.getECRRequestPayload(),
        });
      }
    }
  }
  getAPISignatureForDockerHub() {
    switch (this.mode) {
      case 'create': {
        return this.artifactoryControllerService.createBasicDockerArtifactoryUsingPOST(
          this.getDockerHubRequestPayload(),
        );
      }
      case 'edit': {
        return this.artifactoryControllerService.updateBasicDockerArtifactoryUsingPOST(
          {
            artifactoryId: this.registryId,
            basicDockerDTO: this.getDockerHubRequestPayload(),
          },
        );
      }
    }
  }

  createOrEditRegistry() {
    this.registryAPIInProgress = true;

    let httpReq;
    switch (this.registryType?.value) {
      case 'ECR':
        {
          httpReq = this.getAPISignattureForECR();
        }
        break;
      case 'OTHERS':
        {
          httpReq = this.getAPISignatureForDockerHub();
        }
        break;
    }

    let { successSubText, failureSubText } = getToastSubTexts(this.mode);

    httpReq?.subscribe(
      () => {
        this.customToastService.success(
          `Container Registry ${successSubText} Successfully`,
        );
        this.registryAPIInProgress = false;
        this.handleBack();
      },
      (error) => {
        this.onAPIErrorService.onAPIError(
          error?.error?.message ||
            `Failed to ${failureSubText} Container Registry`,
          error,
        );
        this.registryAPIInProgress = false;
      },
    );
  }
  handleBack = () => {
    this.router.navigate(['settings/artifactories']);
  };
}
