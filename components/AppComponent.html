import { Component, OnInit, OnDestroy } from '@angular/core';
import { Title } from '@angular/platform-browser';
import { NavigationEnd, Router } from '@angular/router';
import { filter } from 'rxjs';
import mixpanel from 'mixpanel-browser';
import { SimpleOauth2User } from './cc-api/models';
import { ApplicationControllerService } from './cc-api/services/application-controller.service';
import { UiStackControllerService } from './cc-api/services/ui-stack-controller.service';
import { PendoService } from './services/pendo.service';
import { RegisterSvgIconsService } from './services/register-svg-icons.service';
import { CookieUtilWrapperService } from './services/cookie-util-wrapper.service';
import { IntentDetailsService } from './services/intent-details.service';
import { FeatureFlagsService } from './services/feature-flags.service';
import { FaviconService } from './services/favicon.service';
import { CPCloudService } from './services/cp-cloud.service';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { PublicApIsService } from 'src/app/cc-api/services/public-ap-is.service';
import isEmpty from 'lodash/isEmpty';

@Component({
  selector: 'app-root',
  template: '<router-outlet *ngIf="featureFlagsLoaded"></router-outlet>',
  styleUrls: ['./app.component.scss'],
})
export class AppComponent implements OnInit, OnDestroy {
  currentBlueprintEnvironmentURL: any;
  featureFlagsLoaded: boolean = false;
  customTheme = null;
  constructor(
    private router: Router,
    private titleService: Title,
    private uiStackControllerService: UiStackControllerService,
    private cookieService: CookieUtilWrapperService,
    private applicationControllerService: ApplicationControllerService,
    private pendo: PendoService,
    private iconPack: RegisterSvgIconsService,
    private featureFlagsService: FeatureFlagsService,
    private faviconService: FaviconService,
    private intentDetailsService: IntentDetailsService,
    private customToast: CustomToastService,
    private publicApiService: PublicApIsService,
  ) {
    mixpanel.init('c5b49b241a26c86813f3feb683481ccb', {
      debug: true,
      track_pageview: true,
      persistence: 'localStorage',
    });
    this.iconPack.register();
    this.attachDynamicTitlesListener();
    this.currentBlueprintEnvironmentURL = this.cookieService.get('selection');
    this.makeFeatureFlagsCall();
    this.intentDetailsService?.setAllIntentData();
  }

  /**
   * Subscribe to route change events and filter out NaviagtionEnd event.
   * Navigate to the deepest child route and access the title property from data object
   * Set title using Angular's Title service
   * @private
   */
  private attachDynamicTitlesListener() {
    this.router.events
      .pipe(filter((event) => event instanceof NavigationEnd))
      .subscribe(() => {
        let root = this.router.routerState.snapshot.root;
        while (root) {
          if (root?.children && root?.children?.length) {
            root = root?.children[0];
          } else if (root?.data && root?.data?.title && root?.params) {
            // If running locally set clientName to 'Dev' else extract client name from hostname.
            const clientName =
              window.location.hostname === 'localhost'
                ? 'Dev'
                : window.location.hostname.split('.')[0];
            this.titleService.setTitle(`${root?.data?.title} - ${clientName}`);

            this.setCurrentBlueprintEnvironmentURL(root);
            return;
          } else {
            return;
          }
        }
      });
  }

  ngOnDestroy(): void {
    this.intentDetailsService?.resetData();
    this.featureFlagsService?.resetData();
  }

  async ngOnInit() {
    this.applicationControllerService.meUsingGET().subscribe(
      (user: SimpleOauth2User) => {
        this.pendo.initialize(user.name);
        mixpanel.identify(user.name);
      },
      (error) => {},
    );
  }

  private setCurrentBlueprintEnvironmentURL(root) {
    const p = root.params;
    if (
      p.stackName &&
      p.clusterId &&
      this.currentBlueprintEnvironmentURL.indexOf(p.clusterId) === -1
    ) {
      this.currentBlueprintEnvironmentURL =
        '/' + p.stackName + '/cluster/' + p.clusterId;
      // this.cookieService.set('selection', this.currentBlueprintEnvironmentURL, {
      //   domain: window.location.hostname,
      //   path: '/',
      // });
      this.cookieService.set(
        'selection',
        this.currentBlueprintEnvironmentURL,
        undefined,
        '/',
      );
    } else if (
      p.stackName &&
      !p.clusterId &&
      this.currentBlueprintEnvironmentURL.indexOf(p.stackName) === -1
    ) {
      //get all cluster of selected stack and set first cluster as default cluster in cookie
      this.uiStackControllerService.getClustersUsingGET1(p.stackName).subscribe(
        (c) => {
          if (c?.length) {
            let clusterId = c[0].id;
            this.currentBlueprintEnvironmentURL =
              '/' + p.stackName + '/cluster/' + clusterId;
            this.cookieService.set(
              'selection',
              this.currentBlueprintEnvironmentURL,
              undefined,
              '/',
            );
          }
        },
        (error) => {
          if (error) {
            this.currentBlueprintEnvironmentURL =
              '/' + p.stackName + '/cluster';
            this.cookieService.set(
              'selection',
              this.currentBlueprintEnvironmentURL,
              undefined,
              '/',
            );
          }
        },
      );
    }
    return;
  }
  async makeFeatureFlagsCall() {
    this.featureFlagsLoaded = false;
    try {
      await this.featureFlagsService?.getAllFeatureFlags();
      this.featureFlagsLoaded = true;
    } catch (error) {
      this.featureFlagsLoaded = true;
    }
    this.setFavIcon();
    this.publicApiService.retrieveThemeFileUsingGET().subscribe((data) => {
      try {
        this.customTheme = data?.content ? JSON.parse(data?.content) : null;
        if (!isEmpty(this.customTheme)) {
          this.setCustomTheme();
        }
      } catch (error) {
        this.customToast.error(
          error?.error?.message || `Unable to load theme contents`,
        );
      }
    });
  }

  setCustomTheme() {
    const bodyElement = document.body.style;
    Object.keys(this.customTheme).forEach((key) => {
      bodyElement.setProperty(`--${key}`, this.customTheme[key]);
    });
  }

  setFavIcon() {
    const favIcon = this.featureFlagsService.getFavIcon();
    if (favIcon) {
      this.faviconService.setFavicon(favIcon);
    }
  }
}
