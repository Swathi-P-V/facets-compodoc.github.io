import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
const emptyStateLogo = './assets/secret-empty.svg';
const errorStatelogo = './assets/error.svg';
import cloneDeep from 'lodash/cloneDeep';
import { UiArtifactRoutingRuleControllerService } from 'src/app/cc-api/services';
import { ArtifactRoutingRuleResponse } from 'src/app/cc-api/models';
import { NbDialogService } from '@nebular/theme';
import { SmartTableRouterlinkComponent } from 'src/app/components/smart-table-routerlink/smart-table-routerlink.component';
import { SmartTableTimeagoTooltipComponent } from 'src/app/components/smart-table-timeago-tooltip/smart-table-timeago-tooltip.component';
import {
  ARTIFACT_ROUTING_RULE_WRITE,
  ARTIFACT_ROUTING_RULE_DELETE,
} from '../../../../constants/permissions';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
@Component({
  selector: 'app-list-ci-rules',
  templateUrl: './list-ci-rules.component.html',
  styleUrls: ['./list-ci-rules.component.scss'],
})
export class ListCiRulesComponent extends FacetsPage {
  stackName: any;
  ciRules: ArtifactRoutingRuleResponse[] = [];
  errorMessage: string = '';
  //Icons
  emptyStateLogo = emptyStateLogo;
  errorStatelogo = errorStatelogo;
  //Loaders
  loading: boolean = true;
  deleteRuleInProgress: boolean = false;
  //Element/Template references
  @ViewChild('confirmDeleteCIRulePopup', { static: true })
  private confirmDeleteCIRulePopup: TemplateRef<any>;
  //Buttons
  createCiPermission = ARTIFACT_ROUTING_RULE_WRITE;
  ciRulesActionButtons = [
    {
      id: 'create-ci-rule',
      title: 'Create CI Rule',
      buttonType: 'primary',
      nbTooltip: 'Create CI Rule',
      action: this.createCiPermission,
      onClick: () => {
        this.navigateToCreateCIRule();
      },
    },
  ];
  smartTableCustomActionButtons = [
    {
      id: 'edit',
      icon: 'edit-outline',
      isFciPack: true,
      toolTip: 'Edit CI Rule',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: ARTIFACT_ROUTING_RULE_WRITE,
      onClick: (data) => {
        this.navigateToEditCIRule(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-outline',
      isFciPack: true,
      toolTip: 'Delete CI Rule',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: ARTIFACT_ROUTING_RULE_DELETE,
      onClick: (data) => {
        this.deleteCIRuleConfirmPopup(data);
      },
    },
  ];
  //Smart table settings
  ciRulesTableSettings: any = {
    columns: {
      ruleName: {
        title: 'Rule Name',
        type: 'custom',
        filter: true,
        valuePrepareFunction: (value, row, cell) => {
          const displayVal = {
            url: [
              '/',
              'stack',
              this.stackName,
              'artifacts',
              'ci-rules',
              'edit',
              row?.id,
            ],
            display: value,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: SmartTableRouterlinkComponent,
      },
      createdBy: {
        title: 'Created By',
        type: 'custom',
        filter: true,
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            user: row.createdBy,
            label: row.creationDate,
          });
        },
        renderComponent: SmartTableTimeagoTooltipComponent,
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          //Considering ruleName as uniquie column here
          let formattedSmartTableCustomActionButtons = cloneDeep(
            this.smartTableCustomActionButtons,
          );
          formattedSmartTableCustomActionButtons.forEach((button) => {
            button['uniqueId'] = button?.id + '_' + row?.ruleName;
          });
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {
            smartTableCustomActionButtons:
              formattedSmartTableCustomActionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance: any) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        filter: false,
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
      perPage: 10,
    },
    mode: 'external',
    actions: false,
  };

  constructor(
    _route: ActivatedRoute,
    private router: Router,
    private uiArtifactRoutingRuleController: UiArtifactRoutingRuleControllerService,
    private dialogService: NbDialogService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.stackName = p?.stackName;
    this.getAllCIRules();
  }

  getAllCIRules() {
    this.loading = true;
    this.uiArtifactRoutingRuleController
      .getAllArtifactRoutingRulesUsingGET({ stackName: this.stackName })
      .subscribe(
        (data) => {
          this.ciRules = data;
          this.errorMessage = '';
          this.loading = false;
        },
        (error) => {
          this.errorMessage =
            error?.error?.message ||
            'There is some error while fetching the CI Rules';
          this.loading = false;
        },
      );
  }

  handleCIRuleActions(event) {
    let clickedButton = this.ciRulesActionButtons.find(
      (button) => button?.id === event,
    );
    clickedButton?.onClick();
  }

  deleteCIRuleConfirmPopup(rule) {
    this.dialogService.open(this.confirmDeleteCIRulePopup, { context: rule });
  }

  deleteCIRule(rule, ref) {
    this.deleteRuleInProgress = true;
    this.uiArtifactRoutingRuleController
      .deleteArtifactRoutingRuleUsingDELETE(rule?.id)
      .subscribe(
        (data) => {
          this.deleteRuleInProgress = false;
          ref.close();
          this.customToastService.success(
            `Rule '${rule?.ruleName}' has been successfully deleted.`,
          );
          this.getAllCIRules();
        },
        (error) => {
          this.deleteRuleInProgress = false;
          ref.close();
          this.onAPIErrorService.onAPIError(
            error?.error?.message ||
              `Rule '${rule?.ruleName}' couldn't be deleted.`,
            error,
          );
        },
      );
  }

  navigateToCreateCIRule() {
    this.router.navigate([
      '/',
      'stack',
      this.stackName,
      'artifacts',
      'ci-rules',
      'create',
    ]);
  }

  navigateToEditCIRule(rule) {
    this.router.navigate([
      '/',
      'stack',
      this.stackName,
      'artifacts',
      'ci-rules',
      'edit',
      rule?.id,
    ]);
  }
}

  <nb-card style="width: 45vw">

        (click)="ref.close()"

  </nb-card>
</ng-template>
