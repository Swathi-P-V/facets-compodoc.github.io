import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { UiArtifactsControllerService } from '../../cc-api/services/ui-artifacts-controller.service';
import { NbDialogService } from '@nebular/theme';
import { Artifact } from '../../cc-api/models/artifact';
import { UiDropdownsControllerService } from 'src/app/cc-api/services';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CustomToastService } from '../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const emptyStateLogo = './assets/artifact-empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-artifacts',
  templateUrl: './artifacts.component.html',
  styleUrls: ['./artifacts.component.scss'],
})
export class ArtifactsComponent extends FacetsPage {
  artifacts: any = [];
  registry: string;
  releaseStreams: any = [];
  fetchingArtifacts = false;
  fetchingOptions = false;
  deleteArtifactInProgress: boolean = false;
  showEmptyState: boolean = false;
  emptyStateLogo = emptyStateLogo;
  primaryMsg: string = 'There are no registered artifacts';
  secondaryMsg: string = 'To start registering, update your blueprint';
  status: 'info' | 'error' = 'info';
  promotionInProgress: boolean = false;

  smartTableCustomActionButtons = [
    {
      id: 'promote',
      icon: 'star-outline',
      toolTip: 'Promote Artifact',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: 'ARTIFACTS_WRITE',
      onClick: (data) => {
        this.promoteArtifactConfirmPopup(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete Artifact',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: 'ARTIFACTS_DELETE',
      onClick: (data) => {
        this.deleteArtifactConfirmPopup(data);
      },
    },
  ];

  @ViewChild('confirmPromote', { static: true })
  private confirmPromote: TemplateRef<any>;

  @ViewChild('confirmDelete', { static: true })
  private confirmDelete: TemplateRef<any>;

  tableSettings: any = {
    columns: {
      applicationName: {
        title: 'Artifact Name',
        type: 'text',
        filter: true,
        width: '20%',
      },
      buildId: {
        title: 'Build Id',
        filter: false,
        type: 'text',
        width: '10%',
      },
      registrationType: {
        title: 'Registration Type',
        width: '10%',
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [
              { value: 'Environment ID', title: 'Environment ID' },
              { value: 'Release Stream', title: 'Release Stream' },
            ],
          },
        },
        type: 'text',
      },
      registrationValue: {
        title: 'Registration Value',
        filter: true,
        type: 'text',
        width: '10%',
      },
      artifactUri: {
        title: 'Artifact URI',
        filter: false,
        type: 'text',
        width: '30%',
      },
      createdBy: {
        title: 'Registered By',
        filter: false,
        type: 'text',
        width: '10%',
        valuePrepareFunction(value, row) {
          if (!value) {
            return 'API';
          }
          return value;
        },
      },
      buildDescription: {
        title: 'Build Description',
        filter: false,
        type: 'text',
        width: '20%',
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {};
          //If release stream is PROD or registrationtype is clusterID, hide promote action button
          if (row.releaseStream === 'PROD' || row.clusterId) {
            displayVal['smartTableCustomActionButtons'] =
              this.smartTableCustomActionButtons.filter(
                (actionButton) => actionButton.id === 'delete',
              );
          } else {
            displayVal['smartTableCustomActionButtons'] =
              this.smartTableCustomActionButtons;
          }
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance: any) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '10%',
        filter: false,
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
      perPage: 10,
    },
    mode: 'external',
    actions: false,
  };

  //Non used properties
  private data: any = [];
  selectedItem: any;
  name: string;

  constructor(
    private activatedRoute: ActivatedRoute,
    private uiArtifactsControllerService: UiArtifactsControllerService,
    private dialogService: NbDialogService,
    private dropdownService: UiDropdownsControllerService,
    private router: Router,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.registry = p.registry;
    this.name = p.name;
    this.fetchingArtifacts = true;
    this.fetchingOptions = true;
    this.uiArtifactsControllerService.getAllUsingGET({}).subscribe(
      (r) => {
        let artifacts = r;
        this.artifacts = this.setRegistrationTypeAndValue(artifacts);
        if (this.artifacts?.length > 0) {
          this.showEmptyState = false;
        } else {
          this.showEmptyState = true;
        }
        this.fetchingArtifacts = false;
      },
      (error) => {
        this.fetchingArtifacts = false;
        this.showEmptyState = true;
        this.emptyStateLogo = errorIcon;
        this.secondaryMsg =
          error?.error?.message || 'There is some error fetching artifacts';
        this.primaryMsg = '';
        this.status = 'error';
      },
    );
    this.dropdownService.getReleaseStreamsUsingGET().subscribe(
      (json) => {
        this.releaseStreams = json;
        this.fetchingOptions = false;
      },
      (err) => (this.fetchingOptions = false),
    );
  }

  getReleaseStreamFilters() {
    const opts = this.releaseStreams;
    const res = [];
    opts.forEach((opt) => {
      res.push({ value: opt, title: opt });
    });
    return res;
  }

  promoteArtifactConfirmPopup(rowData) {
    this.dialogService.open(this.confirmPromote, { context: rowData });
  }

  promote(data: Artifact, ref) {
    data.promoted = true;
    data.releaseStream = data['releaseStreamNew'];
    this.promotionInProgress = true;
    this.uiArtifactsControllerService
      .registerArtifactUsingPOST1(data)
      .subscribe(
        (r) => {
          let artifacts = r;
          this.artifacts = this.setRegistrationTypeAndValue(artifacts);
          this.promotionInProgress = false;
          this.customToastService.success('Artifact promoted successfully');
          ref.close();
        },
        (err) => {
          this.onAPIErrorService.onAPIError(
            err?.error?.message || 'Couldnt promote the artifact',
            err,
          );
          this.promotionInProgress = false;
          ref.close();
        },
      );
  }

  deleteArtifactConfirmPopup(rowData) {
    this.dialogService.open(this.confirmDelete, { context: rowData });
  }

  deleteArtifact(data: Artifact, ref) {
    this.deleteArtifactInProgress = true;
    this.uiArtifactsControllerService
      .deleteArtifactUsingDELETE(data.id)
      .subscribe(
        (data) => {
          this.deleteArtifactInProgress = false;
          this.customToastService.success('Artifact deleted successfully');
          ref.close();

          this.fetchingArtifacts = true;
          this.uiArtifactsControllerService.getAllUsingGET({}).subscribe(
            (r) => {
              let artifacts = r;
              this.artifacts = this.setRegistrationTypeAndValue(artifacts);
              if (this.artifacts?.length <= 0) {
                this.showEmptyState = true;
              }
              this.fetchingArtifacts = false;
            },
            (err) => {
              this.fetchingArtifacts = false;
              this.showEmptyState = true;
            },
          );
        },
        (error) => {
          console.log(error);
          this.deleteArtifactInProgress = false;
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Artifact delete failed',
            error,
          );
          ref.close();
        },
      );
  }

  setRegistrationTypeAndValue(artifacts) {
    artifacts.forEach((artifact) => {
      if (artifact.clusterId && !artifact.releaseStream) {
        artifact.registrationType = 'Environment ID';
        artifact.registrationValue = artifact.clusterId;
      } else if (!artifact.clusterId && artifact.releaseStream) {
        artifact.registrationType = 'Release Stream';
        artifact.registrationValue = artifact.releaseStream;
      }
    });
    return artifacts;
  }
}
