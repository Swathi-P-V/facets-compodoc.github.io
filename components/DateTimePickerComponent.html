import { Component, EventEmitter, Input, Output } from '@angular/core';
import { NbCalendarRange, NbDateService } from '@nebular/theme';

@Component({
  selector: 'app-date-time-picker',
  templateUrl: './date-time-picker.component.html',
  styleUrls: ['./date-time-picker.component.scss'],
})
export class DateTimePickerComponent {
  @Input() calendarMode: 'single' | 'range' = 'single';
  @Input() unsetLabel: string = 'No filters applied';
  @Output() filterChanged = new EventEmitter<any>();

  showPicker: boolean = false;
  range: NbCalendarRange<Date>;
  singleDate: Date = new Date();
  isRangeMode: boolean = false;
  timeIntervals: { value: any; label: string }[];
  quickRanges: { value: any; label: string }[];
  selectedOption: any;
  selectedValue: any;
  selectedLabel: string;
  selectedLabelTemp: string;
  today: Date;

  constructor(protected dateService: NbDateService<Date>) {
    this.today = this.dateService.today();
    this.toggleMode();
    if (this.calendarMode === 'range') {
      this.range = {

        // ),

        //   this.dateService.getMonthEnd(new Date()),
        //   -3,
        // ),
        start: new Date(new Date().getTime() - 60 * 60 * 1000),
        end: new Date(),
      };
    } else {
    }
    this.initDefaultTimeIntervals();
  }

  getDateString(date: Date) {
    if (!date) {
      return 'not selected';
    }
    let values = date.toString()?.split(' ');
    return values[1] + '/' + values[2] + '/' + values[3];
  }

  getLastWeek(today: Date) {
    let day = today.getDay() - 1;
    let monday = new Date(
      today.getFullYear(),
      today.getMonth(),
      today.getDate() - day - 7,
    ).setHours(0, 0, 0, 0);
    let sunday = new Date(
      today.getFullYear(),
      today.getMonth(),
      today.getDate() - day - 1,
    ).setHours(23, 59, 59, 999);
    return { monday: new Date(monday), sunday: new Date(sunday) };
  }

  getLastMonth(today: Date) {
    let month = today.getMonth();
    let lastStart = new Date(today.getFullYear(), month - 1, 1).setHours(
      0,
      0,
      0,
      0,
    );
    let lastEnd = new Date(today.getFullYear(), month, 0).setHours(
      23,
      59,
      59,
      999,
    );
    return { lastStart: new Date(lastStart), lastEnd: new Date(lastEnd) };
  }

  getLastYear(today: Date) {
    let year = today.getFullYear();
    let lastStart = new Date(year - 1, 0, 1).setHours(0, 0, 0, 0);
    let lastEnd = new Date(year, 0, 0).setHours(23, 59, 59, 999);
    return { lastStart: new Date(lastStart), lastEnd: new Date(lastEnd) };
  }

  toggleMode() {
    this.calendarMode = this.calendarMode === 'range' ? 'single' : 'range';
    this.isRangeMode = this.calendarMode === 'range';
  }

  onOptionChange() {
    setTimeout(() => {
      // console.log('SELECTED OPTION => ', this.selectedOption);
      if (typeof this.selectedOption != 'string') {
        this.selectedValue = { ...this.selectedOption };
        Object.keys(this.selectedValue).forEach((dateAttribute) => {
          this.selectedValue[dateAttribute] = new Date(
            this.selectedValue[dateAttribute],
          );
        });
      } else {
        this.selectedValue = this.selectedOption.split('-');
        let today = Math.floor(Date.now() / 1000);
        let magnitude = +this.selectedValue[0];
        let unit = this.selectedValue[1];
        let from: Date;
        let till = today;
        switch (unit) {
          case 'm':
            from = new Date((till - 60 * magnitude) * 1000);
            break;
          case 'h':
            from = new Date((till - 60 * 60 * magnitude) * 1000);
            break;
          case 'd':
            from = new Date((till - 60 * 60 * 24 * magnitude) * 1000);
            break;
          case 'mt':
            from = new Date((till - 60 * 60 * 24 * 30 * magnitude) * 1000);
            break;
          case 'y':
            from = new Date((till - 60 * 60 * 24 * 365 * magnitude) * 1000);
            break;
          default:
            from = new Date();
        }
        this.selectedValue = { from: new Date(from), till: new Date() };
        console.log('range => ', this.range);
      }

      // console.log('selectedOption => ', this.selectedOption);

      if (this.selectedValue.date) {
        this.updateSingleDate(this.selectedValue.date);
      }

      if (this.calendarMode === 'single') {
        this.toggleMode();
      }

      if (this.isSameDate(this.selectedValue.from, this.selectedValue.till)) {
        if (this.calendarMode === 'range') {
          this.toggleMode();
          this.singleDate = this.selectedValue.from;
        }
      }

      this.range = {
        start: this.selectedValue.from,
        end: this.selectedValue.till,
      };

      // if(this.selectedLabelTemp?.substring(0, 4)?.toLocaleLowerCase() !== 'last'){
      //   this.range = {
      //     start: this.dateService.addDay(this.selectedValue.from, 0),
      //     end: this.dateService.addDay(this.selectedValue.till, 0),
      //   };
      // }
    }, 100);
  }

  getPreviousMonday() {
    var date = new Date();
    var day = date.getDay();
    var prevMonday = new Date();
    if (date.getDay() == 0) {
      prevMonday.setDate(date.getDate() - 7);
    } else {
      prevMonday.setDate(date.getDate() - (day - 1));
    }
    return prevMonday;
  }

  updateSingleDate(date: Date) {
    this.selectedValue.from = new Date(date.setHours(0, 0, 0, 0));
    this.selectedValue.till = new Date(date.setHours(23, 59, 59, 999));
    delete this.selectedValue.date;
  }

  isSameDate(dateA: Date, dateB: Date) {
    return (
      dateA.getFullYear() == dateB.getFullYear() &&
      dateA.getMonth() == dateB.getMonth() &&
      dateA.getDate() == dateB.getDate()
    );
  }

  selectFromCalendar() {
    setTimeout(() => {
      this.selectedOption = {};
      // if (this.calendarMode == 'single') {
      //   this.updateSingleDate(this.singleDate);
      //   this.selectedLabelTemp = this.getDateString(this.singleDate);
      // } else {
      this.selectedValue = {
        from: this.range.start,
        till: this.range?.end
          ? this.range.end
          : new Date(this.range.end?.setHours(23, 59, 59, 999)),
      };
      this.selectedLabelTemp =
        this.getDateString(this.range.start) +
        ' - ' +
        this.getDateString(this.range.end);
      // }
    }, 100);
  }

  apply() {
    let value: { from: Date; till: Date };
    value = {
      from: this.selectedValue?.from || this.range.start,
      till: this.selectedValue?.till || this.range.start,
    };
    this.filterChanged.emit(value);
    this.selectedLabel = this.selectedLabelTemp;
    this.showPicker = false;
  }
}
