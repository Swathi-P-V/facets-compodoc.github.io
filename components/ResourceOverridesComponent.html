  constructor(
    activatedRoute: ActivatedRoute,
    private uiDropdownsControllerService: UiDropdownsControllerService,
    private dialogService: NbDialogService,

  OnPageLoad(p: Params): void {
    this.stackName = p?.stackName;
    this.branchName = p?.branchName;
    this.resourceName = p?.resourceName;
    this.resourceType = p?.resourceType;
    this.fetchResourceConfigs();
  }

  /*
    1. Get Stack configuration JSON 
    2. Get Environment overrides configuration
    3. Flatten the stack config to match the field to the environment config
  */
  fetchResourceConfigs(): void {
    this.setInitialValues();
    this.blueprintConfig = {};
    this.flattenedBlueprintConfig = {};
    Promise.all([this.getResourceStackInfo(), this.getResourceEnvInfo()]).then(
      (promiseRes) => {
        try {
          this.blueprintConfig = JSON.parse(this.resourceData?.content);
        } catch (error) {
          this.blueprintConfig = null;
        }

        if (this.blueprintConfig) {
          this.flattenedBlueprintConfig = flattenObject(this.blueprintConfig);
          this.prepareConfigData();
        } else {
          this.setEmptyStateValues();
        }
        this.loadingTableData = false;
      },
      (error) => {
        this.loadingTableData = false;
        this.tableData = [];
        this.setEmptyStateValues(error);
      },
    );
  }

  setEmptyStateValues(error = null): void {
    this.emptyStateLogo = error ? errorIcon : this.emptyStateLogo;
    this.secondaryMsg = error
      ? error?.error?.message ||
        'There is some error fetching override configuration'
      : 'No overrides in resource configuration yet';
    this.status = error ? 'error' : 'info';
  }

  // Reset all the values
  setInitialValues() {
    this.overridesData = [];
    this.tableData = [];
    this.loadingTableData = true;
    this.currentConfig.reset();
    this.envConfigCode.reset();
    this.updatedConfig.reset();
  }

  /*
   * Prepare configuration fields data to show in the form of table
   * First, it flattens entire config object for each environment
   * Then it finds unique key for each environment and that key becomes a key in resultant new object
   * For each key, it updates the count of how many environments it occurs and adds environment information in env_metadata
   */
  prepareConfigData() {
    const configData = processConfigData(
      this.overridesData,
      this.flattenedBlueprintConfig,
    );
    this.flattenEnvironments(configData);
    this.prepareTableData(configData);
  }

  newOverridesTable: any[];
  overridesDataSource: LocalDataSource = new LocalDataSource();
  flattenEnvironments(configData) {
    const newOverridesTable = [];
    Object.keys(configData)?.forEach((path) => {
      configData[path]?.env_metaData?.forEach((env) => {
        let row = {
          path: path,
          blueprintValue: configData[path]?.blueprintVal,
          ...env,
        };
        delete row?.key;
        newOverridesTable.push(row);
      });
    });
    this.newOverridesTable = newOverridesTable;
    this.originalDataBackup.clear();
    newOverridesTable.forEach((row) => {
      this.originalDataBackup.set(
        `${row.path}$-$${row.clusterId}$-$${row.clusterName}`,
        cloneDeep(row.value),
      );
    });
    this.overridesDataSource.load(newOverridesTable);
  }

  /*
    Prepares final table array from the resultant object.
  */
  prepareTableData(configData) {
    this.tableData = Object.keys(configData).map((configKey) => {
      return {
        key: configKey,
        ...configData[configKey],
      };
    });
    this.loadingTableData = false;
  }

  //Get resource stack configuration
  getResourceStackInfo() {
    this.loadingStackConfig = true;
    return new Promise((resolve, reject) => {
      this.uiDropdownsControllerService
        .getResourceByStackUsingGET({
          stackName: this.stackName,
          resourceName: this.resourceName,
          resourceType: this.resourceType,
          branch: this.branchName,
        })
        .subscribe(
          (resource) => {
            this.resourceData = resource;
            this.loadingStackConfig = false;
            resolve(true);
          },
          (error) => {
            this.errorInFetchingBlueprintConfig = true;
            this.loadingStackConfig = false;
            reject(error);
          },
        );
    });
  }

  //Get environment override configuration for each resource
  getResourceEnvInfo() {
    this.loadingEnvConfig = true;
    return new Promise((resolve, reject) => {
      this.stackControllerService
        .getOverridesUsingGET2({
          stackName: this.stackName,
          resourceName: this.resourceName,
          resourceType: this.resourceType,
        })
        .subscribe(
          (resource) => {
            this.overridesData = resource;
            this.loadingEnvConfig = false;
            resolve(true);
          },
          (error) => {
            this.errorInFetchingBlueprintConfig = true;
            this.loadingEnvConfig = false;
            reject(error);
          },
        );
    });
  }

  resetConfigInputFields() {
    this.envConfigCode.setValue(null);
    this.currentConfig.setValue(null);
    this.updatedConfig.setValue(null);
    this.updatedConfig.reset();
  }

  //open configuration popup to view
  openConfigPopup(data) {
    this.resetConfigInputFields();
    this.envConfigCode.setValue(
      JSON.stringify(
        data?.context === 'BLUEPRINT' ? data?.blueprintValue : data?.value,
        null,
        '\t',
      ),
    );
    this.dialogService.open(this.envConfigEditor, { context: data });
  }

  submitOverrides = (updatedObject, configData, overrideKey, ref) => {
    return new Promise(async (resolve, reject) => {
      try {
        this.submittingOverrides = true;
        this.appService
          .postResourceOverrideObjectUsingPOST({
            resourceName: this.resourceName,
            resourceType: this.resourceType,
            overrideRequest: updatedObject,
            clusterId: configData?.clusterId,
          })
          .subscribe(
            (data) => {
              this.submittingOverrides = false;
              resolve(data);
            },
            (err) => {
              this.submittingOverrides = false;
              this.onAPIErrorService.onAPIError(
                `Update Failed for ${overrideKey.split('$-$')[0]} for ${
                  overrideKey.split('$-$')[2]
                }: ` + err?.error?.message,
                err,
              );
              reject(err);
            },
          );
      } catch (err) {
        this.submittingOverrides = false;
        this.customToasterService.error(
          `Update Failed for ${overrideKey.split('$-$')[0]} for ${
            overrideKey.split('$-$')[2]
          }: ` + err?.error?.message,
        );
        reject(err);
      }
    });
  };

  discardChanges = (ref?) => {
    ref?.close('discard');
    this.newOverrideChanges = {};
    this.fetchResourceConfigs();
  };

  async setResourceOverride(overriddenEnvConfig, configData, overrideKey, ref) {
    return new Promise(async (resolve, reject) => {
      const dataType = this.dataType.value;
      let value = this.newOverrideChanges[overrideKey];
      let dataTypeValueMismatch = false;
      let errorMessage = 'Invalid value provided for selected data type';
      switch (dataType) {
        case 'BOOLEAN':
          value = value === 'true';
          break;
        case 'NUMBER':
          value = parseFloat(value);
          break;
        case 'OBJECT':
          try {
            value = JSON.parse(value);
          } catch (error) {
            errorMessage = error;
            dataTypeValueMismatch = true;
            reject(errorMessage);
            return;
          }
          break;
        default:
          break;
      }
      overriddenEnvConfig.flattenedConfig[overrideKey?.split('$-$')[0]] = value;
      overriddenEnvConfig.overrideObject.overrides = unflattenObject(
        overriddenEnvConfig?.flattenedConfig,
      );
      const updatedObject = omit(overriddenEnvConfig, 'flattenedConfig');
      try {
        const rsp = await this.submitOverrides(
          updatedObject?.overrideObject,
          configData,
          overrideKey,
          ref,
        );
        resolve(rsp);
      } catch (err) {
        reject(err);
      }
    });
  }

  async confirmOverride(configData, overrideKey, ref) {
    return new Promise(async (resolve, reject) => {
      let overriddenEnvConfig = cloneDeep(
        this.overridesData.find(
          (data) => configData.clusterId === data?.clusterId,
        ),
      );

      try {
        const rsp = await this.setResourceOverride(
          overriddenEnvConfig,
          configData,
          overrideKey,
          ref,
        );
        resolve(rsp);
      } catch (err) {
        reject(err);
      }
    });
  }

  validateChanges(event) {
    try {
      event.newData.value = JSON.parse(event?.newData?.value);
    } catch {}

    const key = `${event?.newData?.path}$-$${event?.newData?.clusterId}$-$${event?.newData?.clusterName}`;
    if (
      JSON.stringify(event?.newData?.value) !==
      JSON.stringify(this.originalDataBackup.get(key))
    ) {
      this.newOverrideChanges[key] =
        JSON.stringify(event?.newData?.value) === '[]'
          ? '[]'
          : event?.newData?.value;
    } else {
      delete this.newOverrideChanges[key];
    }

    event?.confirm?.resolve(event?.newData);
  }

  isEditorRequired(override) {
    return isParsableObject(override.value);
  }

  async saveAllOverrideChanges(ref) {
    const overrideKeys = Object.keys(this.newOverrideChanges);
    await this.processOverrideKeysSequentially(overrideKeys, 0, 0, ref);
  }

  async processOverrideKeysSequentially(
    overrideKeys,
    currentIndex,
    successfulHits,
    ref,
  ) {
    if (currentIndex < overrideKeys.length) {
      const overrideKey = overrideKeys[currentIndex];
      const clusterId = overrideKey.split('$-$')[1];
      const configData = { clusterId };

      try {
        const rsp: any = await this.confirmOverride(
          configData,
          overrideKey,
          ref,
        );
        successfulHits += 1;
        for (let i = 0; i < this.overridesData.length; i++) {
          if (this.overridesData[i]?.clusterId === configData?.clusterId) {
            this.overridesData[i].overrideObject = rsp;
            this.overridesData[i].flattenedConfig[
              overrideKey?.split('$-$')[0]
            ] = this.newOverrideChanges[overrideKey];
            break;
          }
        }
        delete this.newOverrideChanges[overrideKey];
      } catch (err) {
        console.log('FAILED, SKIPPING');
      }
      await this.processOverrideKeysSequentially(
        overrideKeys,
        currentIndex + 1,
        successfulHits,
        ref,
      );
    } else {
      this.customToasterService.success(
        `Updated overrides for ${successfulHits} out of ${overrideKeys.length} changes successfully`,
      );
      ref?.close();
      this.fetchResourceConfigs();
    }
  }
}
