import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { Route } from '@playwright/test';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { ToastButton } from '../custom-toast/toast-interfaces';

@Component({
  selector: 'app-toasters',
  templateUrl: './toasters.component.html',
  styleUrls: ['./toasters.component.scss'],
})
export class ToastersComponent extends FacetsPage {
  subscription: any;
  toasterLinks = [
    {
      id: 'env-overview',
      label: 'Overview',
      click: () => {
        console.log(
          'Navigated to overview page. This console is inline statement of click method',
        );
      },
    },
    {
      id: 'launch-env',
      label: 'Launch Environment',
      icon: 'arrow-ios-forward-outline',
      action: 'ENVIRONMENT_CONFIGURE',
      click: () => {
        this.routeToEnvLaunch();
      },
    },

    {
      id: 'release',
      label: 'Release',
      icon: 'arrow-ios-forward-outline',
      action: 'ENVIRONMENT_CONFIGURE',
      click: () => {
        this.routeToEnvLaunch();
      },
    },
  ];

  constructor(
    private customToast: CustomToastService,
    private router: Router,
    private _route: ActivatedRoute,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {}

  OnPageInit(): void {
    // Subscribe to the link click event from the toaster service
    this.subscription = this.customToast.toastActionClick$.subscribe((item) => {
      console.log('Item clicked is ', item);
      item.click();
    });
  }

  button: ToastButton[] = [
    {
      id: 'button-id',
      label: 'Retry',
      status: 'primary',
      appearance: 'outline',
      tooltip:
        'If you want to retry release, click on this button. Iam a tooltip to assist you.',
      click: () => {
        console.log(
          'An inline function definition can be executed when a button is clicked',
        );
      },
    },

    {
      id: 'button-id-2',
      label: 'Go Back',
      appearance: 'ghost',
      click: () => {
        this.callSomeMethod();
      },
    },
  ];

  singleBtn: ToastButton[] = [
    {
      id: 'button-id-3',
      label: 'Get Info',
      status: 'danger',
      appearance: 'outline',
      closeToast: false,
      click: () => {
        console.log(
          'The button is clicked but the modal is n ot closed. This is using closeToast key',
        );
      },
    },
  ];

  showCustomToaster(title, content, type, button) {
    this.customToast.show({
      title: title,
      content: content,
      type: type,
      buttons: button,
    });
  }

  showCustomToasterWithMultipleLinks(title, content, type, links) {
    this.customToast.show({
      title: title,
      content: content,
      type: type,
      duration: 0,
      links: links,
    });
  }

  showCustomToasterWithMultipleButtons(title, content, type, buttons) {
    this.customToast.show({
      title: title,
      content: content,
      type: type,
      duration: 0,
      buttons: buttons,
    });
  }

  showSuccessWithDefaults() {
    this.customToast.success(
      'This is a success toast with just content passed to it as a string. This toast should close in 5 seconds',
    );
  }

  showErrorWithDefaults() {
    this.customToast.error(
      "This is a error toast with just content passed to it as a string. This toast shouldn't close until user clicks on close button",
    );
  }

  showSuccessWithOverrides() {
    this.customToast.success({
      content:
        'This is a success method usage demo with overrides. Title and duration is overridden. This is success toast and will not close in 5seconds',
      title: 'Custom title for Success toast',
      duration: 0,
    });
  }

  showErrorWithOverrides() {
    this.customToast.error({
      content:
        'This is a error method usage demo with overrides. Title and duration is overridden. This is error toast and will close in 8 seconds',
      title: 'Custom title for Error toast',
      duration: 8000,
    });
  }

  callSomeMethod() {
    console.log('Button 2 click calls a method of this class');
  }

  routeToEnvCreation() {
    this.router.navigate(['/infra-dev/chooseClusterCreate/clusterCreate/aws']);
  }

  routeToEnvLaunch() {
    this.router.navigate([
      '/infra-dev/cluster/62970be8920da80001f3f495/releases',
    ]);
  }
}
