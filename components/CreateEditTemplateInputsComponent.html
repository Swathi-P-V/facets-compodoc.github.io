import { Component } from '@angular/core';
import { Router, ActivatedRoute, Params } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';

import { UiStackControllerService } from '../../../../cc-api/services/ui-stack-controller.service';
import { StackTemplateInput } from '../../../../cc-api/models/stack-template-input';
import { UiCommonClusterControllerService } from '../../../../cc-api/services/ui-common-cluster-controller.service';
import { TEMPLATE_WRITE } from 'src/app/constants/permissions';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const errorIcon = '../assets/error.svg';

@Component({
  selector: 'app-create-edit-template-inputs',
  templateUrl: './create-edit-template-inputs.component.html',
  styleUrls: ['./create-edit-template-inputs.component.scss'],
})
export class CreateEditTemplateInputsComponent extends FacetsPage {
  constructor(
    private commonClusterService: UiCommonClusterControllerService,
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private stackControllerService: UiStackControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  clusterId: string;
  stackName: string;
  metaData: {} = {};
  templateInputType: string;
  loadingTemplateInputsMeta: boolean = false;
  templateRequestPayload: StackTemplateInput = {
    data: {},
  };
  templateInputForm: FormGroup;
  creationInProgress: boolean = false;
  mode: string = 'create';
  templateInputUniqueId: string;
  errorFetchingTemplatesMeta: string = '';
  errorImage = errorIcon;
  templateWritePermission: string = TEMPLATE_WRITE;

  OnPageInit(): void {}
  OnPageLoad(p: Params): void {
    /** If the stack changes from the breadcrumb, and it is not same stack as the current selected on,
     * then redirect back template inputs listing page */
    if (this.stackName && this.stackName !== p.stackName) {
      this.router.navigate([`../../../template-inputs`], {
        relativeTo: this.route,
      });
    }

    this.metaData = {};
    this.loadingTemplateInputsMeta = true;
    this.clusterId = p?.clusterId;
    this.stackName = p?.stackName;
    this.templateInputType = p?.templateInputType;
    this.templateInputUniqueId = p?.inputId;
    this.mode = this.templateInputUniqueId ? 'edit' : 'create';

    this.getAllTemplateInputsMeta();
  }

  initTemplateInputFormModel() {
    const dynamicFormField = {};
    this.metaData[this.templateInputType]?.spec?.fields?.forEach((field) => {
      dynamicFormField[field] = new FormControl('', [
        Validators.required,
        Validators.pattern(/[\S]/),
      ]);
    });
    this.templateInputForm = new FormGroup({
      uid: new FormControl('', [
        Validators.required,
        Validators.pattern(/[\S]/),
      ]),
      ...dynamicFormField,
    });
  }
  get uid() {
    return this.templateInputForm.get('uid');
  }
  getAllTemplateInputsMeta() {
    this.stackControllerService
      .getAllTemplateInputsMetaUsingGET(this.stackName)
      .subscribe(
        (response) => {
          this.loadingTemplateInputsMeta = false;
          this.metaData = response;
          this.errorFetchingTemplatesMeta = '';
          this.initTemplateInputFormModel();
        },
        (error) => {
          this.loadingTemplateInputsMeta = false;
          this.errorFetchingTemplatesMeta =
            error?.error?.message || error?.message;
        },
      );
  }
  onCancel = () => {
    this.router.navigate([`../../../template-inputs`], {
      relativeTo: this.route,
      queryParams: { templateInputType: this.templateInputType },
    });
  };
  onClickCreateInput() {
    this.commonClusterService
      .createTemplateInputsUsingPOST({
        clusterId: this.clusterId,
        inputType: this.templateInputType,
        templateInputs: [this.templateRequestPayload],
      })
      .subscribe(
        (res) => {
          this.creationInProgress = false;
          this.customToastService.success(
            'Template input created successfully',
          );

          this.router.navigate([`../../../template-inputs`], {
            relativeTo: this.route,
            queryParams: { templateInputType: this.templateInputType },
          });
        },
        (error) => {
          const errorMessage = error?.error?.message || error?.message;
          this.onAPIErrorService.onAPIError(
            errorMessage || 'There is some error creating template input ',
            error,
          );
          this.creationInProgress = false;
        },
      );
  }
  submitTemplateInputForm() {
    this.creationInProgress = true;
    this.templateRequestPayload = {
      uid: this.templateInputForm.get('uid').value,
      data: {},
    };
    this.metaData[this.templateInputType]?.spec?.fields?.forEach((field) => {
      this.templateRequestPayload.data[field] =
        this.templateInputForm.get(field).value;
    });
    this.onClickCreateInput();
  }
}
