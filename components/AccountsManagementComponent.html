import { SmartTableCellWithEllipsisComponent } from './../../components/smart-table-cell-with-ellipsis/smart-table-cell-with-ellipsis.component';
import { Component, TemplateRef, ViewChild } from '@angular/core';
import {
  UiAccountsControllerService,
  UiMetaControllerService,
} from 'src/app/cc-api/services';
import { NbDialogService } from '@nebular/theme';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { Account } from 'src/app/cc-api/models';
import { SmartTableWithIconComponent } from 'src/app/components/smart-table-with-icon/smart-table-with-icon.component';
import { SmartTableTimeagoTooltipComponent } from 'src/app/components/smart-table-timeago-tooltip/smart-table-timeago-tooltip.component';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ACCOUNTS_WRITE, ACCOUNTS_DELETE } from 'src/app/constants/permissions';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { convertToSentenceCase } from 'src/app/utils/string-util';
import { CustomToastService } from '../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

const emptyStateLogo = './assets/secret-empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-accounts-management',
  templateUrl: './accounts-management.component.html',
  styleUrls: ['./accounts-management.component.scss'],
})
export class AccountsManagementComponent extends FacetsPage {
  @ViewChild('versionControl', { static: true })
  private versionControl: TemplateRef<any>;
  @ViewChild('awsAccount', { static: true })
  private awsAccount: TemplateRef<any>;
  @ViewChild('azureAccount', { static: true })
  private azureAccount: TemplateRef<any>;
  @ViewChild('gcpAccount', { static: true })
  private gcpAccount: TemplateRef<any>;
  @ViewChild('kubernetesModal', { static: true })
  private kubernetesModal: TemplateRef<any>;
  @ViewChild('confirmDelete', { static: true })
  private confirmDelete: TemplateRef<any>;
  @ViewChild('coderModal', { static: true })
  private coderModal: TemplateRef<any>;
  accountId: string = '';
  fetchingAccounts: boolean = false;
  fetchingAccountId: boolean = false;
  accounts: Account[];
  deletionInProgress: boolean = false;
  versionControlForm: FormGroup = this.fb.group({
    name: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    username: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    hostName: [null, [Validators.pattern(whiteSpaceInputPattern)]],
    token: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
  });
  smartTableCustomActionButtons = [
    {
      id: 'edit',
      icon: 'edit-2-outline',
      toolTip: 'Edit Account',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: ACCOUNTS_WRITE,
      onClick: (data) => {
        this.editAccount(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete Account',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: ACCOUNTS_DELETE,
      onClick: (data) => {
        this.deleteAccount(data);
      },
    },
  ];
  tableSettings: any = {
    columns: {
      name: {
        title: 'Name',
        filter: true,
        type: 'custom',
        valuePrepareFunction: (cell, row) => {
          let imageUrl = '';
          if (row?.provider) {
            imageUrl = this.imgToProviderMapping[row?.provider];
          }
          return JSON.stringify({
            // url: `./${row.id}`,
            labelLink: true,
            textContent: cell,
            nbIconLink: imageUrl,
            is_reverse: true,
            noTooltip: true,
            badgeClass: 'ml-2 mb-0',
            imgClass: 'wide-icon-class',
            showIcon: true,
            showNbIcon: true,
            nbIconPack: 'fci',
            isTextLink: true,
            url: '/settings/organization-accounts/' + row?.id,
            iconFontSize: '2.1rem',
          });
        },
        renderComponent: SmartTableWithIconComponent,
      },
      provider: {
        title: 'Type',
        type: 'text',
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [
              { value: 'BITBUCKET', title: 'BITBUCKET' },
              { value: 'GITHUB', title: 'GITHUB' },
              { value: 'GITLAB', title: 'GITLAB' },
              { value: 'AWS', title: 'AWS' },
              { value: 'AZURE', title: 'AZURE' },
              { value: 'GCP', title: 'GCP' },
              { value: 'KUBERNETES', title: 'KUBERNETES' },
              { value: 'CODER', title: 'CODER' },
            ],
          },
        },
      },
      inUse: {
        title: 'Status',
        type: 'text',
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [
              { value: true, title: 'in-use' },
              { value: false, title: 'not-used' },
            ],
          },
        },
        valuePrepareFunction: (value) => {
          return value ? 'in-use' : 'not-used';
        },
      },
      envs: {
        title: 'Environments',
        type: 'custom',
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [],
          },
        },
        valuePrepareFunction: (cell, row) => {
          let dataToSend = [];
          if (row?.associatedTo?.length) {
            row?.associatedTo.forEach((stack) => {
              if (stack?.clusters?.length > 0) {
                stack?.clusters.forEach((env) => {
                  dataToSend.push({
                    name: env?.clusterName,
                    url:
                      '/' +
                      stack?.stackName +
                      '/cluster/' +
                      env?.clusterId +
                      '/overview',
                  });
                });
              }
            });
          }
          return JSON.stringify({
            data: dataToSend,
            decorate: true,
            showAsLink: true,
            dataObject: true,
            defaultDisplay: 1,
            linkUnderline: true,
          });
        },
        renderComponent: SmartTableCellWithEllipsisComponent,
      },
      lastModifiedBy: {
        title: 'Modified by',
        filter: false,
        type: 'custom',
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            user: row.lastModifiedBy || row.createdBy,
            label: row.lastModifiedDate || row.creationDate,
          });
        },
        renderComponent: SmartTableTimeagoTooltipComponent,
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {
            smartTableCustomActionButtons: this.smartTableCustomActionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '12%',
        filter: false,
      },
    },
    noDataMessage: 'No accounts found',
    pager: {
      display: false,
    },
    mode: 'external',
    actions: false,
  };

  error: any;
  primaryMsg: string = '';
  secondaryMsg: string = '';
  status: 'error' | 'info' = 'info';
  showEmptyState: boolean = false;
  loading: boolean;
  emptyStateLogo: any = emptyStateLogo;
  errorIcon: any = errorIcon;
  githubIcon = 'github-account';
  gitlabIcon = 'gitlab-account';
  azureIcon = 'azure';
  awsIcon = 'aws';
  gcpIcon = 'gcp';
  coderIcon = 'coder-account';
  bitbucketIcon = 'bitbucket-account';
  kubernetes = 'kubernetes';
  botIcon = './assets/bot.svg';

  OnPageLoad(p: Params): void {
    this.getAccountId();
    this.getAccountDetails();

  getAccountId() {
    this.fetchingAccountId = true;
    this.uiMetaControllerService.getCPAccountUsingGET().subscribe({
      next: (res) => {
        this.accountId = res?.accountId;
        this.fetchingAccountId = false;
      },
      error: (res) => {
        this.accountId = '';
        this.fetchingAccountId = false;
      },
    });
  }

  isLoading() {
    return this.fetchingAccountId || this.fetchingAccounts;
  }

  getAccountDetails() {
    this.fetchingAccounts = true;
    this.accountControllerService.getAllAccountsUsingGET().subscribe(
      (response) => {
        this.accounts = response;
        this.buildEnvFilter();
        if (!this.accounts?.length) {
          this.emptyStateLogo = this.errorIcon;
          this.showEmptyState = true;
          this.primaryMsg = 'No accounts added!';
          this.secondaryMsg =
            'Add accounts to create blueprint and environments effortlessly';
        }
        this.fetchingAccounts = false;
      },
      (error) => {
        this.fetchingAccounts = false;
        this.showEmptyState = true;
        this.primaryMsg = '';
        this.secondaryMsg =
          error?.error?.message ||
          'There is some error fetching accounts information';
        this.emptyStateLogo = this.errorIcon;
        this.status = 'error';
      },
    );
  }

  buildEnvFilter() {
    let clusterFilterData = [];
    const clustersToAdd = new Set<string>();
    this.accounts.forEach((account) => {
      const clusters = [];
      account?.associatedTo?.forEach((stack) => {
        stack?.clusters?.forEach((cluster) => {
          clusters.push(cluster?.clusterName);
          if (!clustersToAdd.has(cluster?.clusterName)) {
            clustersToAdd.add(cluster?.clusterName);
          }
        });
      });
      account['envs'] = clusters;
    });
    clusterFilterData = Array.from(clustersToAdd).sort();
    clusterFilterData = clusterFilterData.map((resGrp) => ({
      title: resGrp,
      value: resGrp,
    }));
    this.tableSettings.columns.envs.filter.config.list = [...clusterFilterData];
    this.tableSettings = { ...this.tableSettings };
  }

  editAccount(data) {
    switch (data.provider) {
      case 'GITHUB':
      case 'GITLAB':
      case 'BITBUCKET':
        this.versionControlForm.patchValue({
          name: data?.name || '',
          username: data?.userName,
          token: data?.accessToken,
          host: data?.enterpriseHostName,
        });
        this.dialogService.open(this.versionControl, {
          context: {
            type: 'edit',
            formType: convertToSentenceCase(data.provider),
            data: data,
          },
        });
        break;
      case 'AWS':
        this.dialogService.open(this.awsAccount, {
          context: { type: 'edit', formType: 'AWS', awsAccountData: data },
        });
        break;
      case 'AZURE':
        this.dialogService.open(this.azureAccount, {
          context: { type: 'edit', formType: 'Azure', azureAccountData: data },
        });
        break;
      case 'GCP':
        this.dialogService.open(this.gcpAccount, {
          context: { type: 'edit', formType: 'GCP', gcpAccountData: data },
        });
        break;
      case 'KUBERNETES':
        this.dialogService.open(this.kubernetesModal, {
          context: {
            type: 'edit',
            formType: 'KUBERNETES',
            kubernetesAccountData: data,
          },
        });
        break;
      case 'CODER':
        this.dialogService.open(this.coderModal, {
          context: {
            type: 'edit',
            formType: 'KUBERNETES',
            coderAccountData: data,
          },
        });
        break;
    }
  }

          errorMessage = err?.error

          errorMessage = 'Account deletion failed';

        ref.close();
      },
    );
  }

  editVersionControlAccount(data, type, ref) {
    let accountName =
      this.versionControlForm?.controls?.name?.value?.trim() || '';
    let username =
      this.versionControlForm?.controls?.username?.value?.trim() || '';
    let token = this.versionControlForm?.controls?.token?.value?.trim() || '';
    let payloadKey = '';
    switch (type) {
      case 'Github':
        payloadKey = 'githubAccountRequest';
        break;
      case 'Gitlab':
        payloadKey = 'gitlabAccountRequest';
        break;
      case 'Bitbucket':
        payloadKey = 'bitbucketAccountRequest';
        break;
    }
    const requestPayload = {
      id: data?.id,
      [payloadKey]: {
        accessToken: token,
        userName: username,
        name: accountName,
        accountType: data?.accountType,
        provider: data?.provider,
        id: data?.id,
        vcsUrl: data?.vcsUrl,
        createdBy: data?.createdBy,
        creationDate: data?.creationDate,
        enterpriseHostName: data?.enterpriseHostName,
      },
    };
    if (type === 'Github') {
      this.editGithubAccount(requestPayload, ref);
    } else if (type === 'Gitlab') {
      this.editGitlabAccount(requestPayload, ref);
    } else if (type === 'Bitbucket') {
      this.editBitbucketAccount(requestPayload, ref);
    }
  }

  goToEnv(env, ref) {
    ref.close();

    this.router.navigate([url]);
  }

  optionClicked(option, data, ref) {
    ref.close();
    if (data?.rowData?.accountType === 'VERSION_CONTROL') {
      let selectedStack;
      data?.stacks.forEach((stack) => {
        if (stack.name === option) {
          selectedStack = stack;
        }
      });
      const url = '/stack/' + selectedStack.name;
      this.router.navigate([url]);
    }
    if (data?.rowData?.accountType === 'CLOUD') {
      let selectedEnv;
      data?.envs.forEach((env) => {
        if (env.name === option) {
          selectedEnv = env;
        }
      });
      const url =
        '/' +
        selectedEnv.stackName +
        '/cluster/' +
        selectedEnv.id +
        '/overview';
      this.router.navigate([url]);
    }
  }

  accountVerification(formDetails: any) {
    this.accVerificationInProgress = true;
    let payload = this.constructAccountValidationPayload(formDetails?.formType);
    let httpReq;

    switch (formDetails?.formType.toUpperCase()) {
      case 'BITBUCKET':
        httpReq =
          this.accountControllerService.validateBitbucketAccountUsingPOST(
            payload,
          );
        break;

      case 'GITHUB':
        httpReq =
          this.accountControllerService.validateGithubAccountUsingPOST(payload);
        break;

      case 'GITLAB':
        httpReq =
          this.accountControllerService.validateGitlabAccountUsingPOST(payload);
        break;
    }

    httpReq.subscribe(
      (data) => {
        this.accVerificationInProgress = false;
        this.customToastService.success('This is a verified account');
      },
      (error) => {
        this.accVerificationInProgress = false;
        this.onAPIErrorService.onAPIError(
          error?.error?.message || 'Something went wrong, try again later.',
          error,
        );
      },
    );
  }

  constructAccountValidationPayload(accType: string) {
    let payload;
    switch (accType?.toUpperCase()) {
      case 'GITHUB':
      case 'BITBUCKET':
      case 'GITLAB': {
        const { name, token, username, hostName } =
          this.versionControlForm?.controls || {};
        let vcsAccDetails: any = {
          accessToken: token?.value?.trim(),
          userName: username?.value?.trim(),
          name: name?.value?.trim(),
          accountType: 'VERSION_CONTROL',
          provider: accType.toUpperCase(),
        };
        if (accType.toUpperCase() === 'GITHUB') {
          vcsAccDetails.enterpriseHostName = hostName?.value?.trim() || null;
        }
        payload = vcsAccDetails;
        break;
      }
    }
    return payload;
  }

  onSaveAccount() {
    this.getAccountDetails();
  }
  onSaveCoderAccount(event) {
    this.getAccountDetails();
  }
