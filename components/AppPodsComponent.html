import {
  Component,
  ElementRef,
  OnDestroy,
  TemplateRef,
  ViewChild,
} from '@angular/core';
import { NbDialogService } from '@nebular/theme';
import { ClipboardService } from 'ngx-clipboard';
import { Pod } from '../../../../cc-api/models/pod';
import { ActivatedRoute, Params } from '@angular/router';
import { UiApplicationControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { FormatDateService } from 'src/app/services/format-date.service';
import { TailLogsComponent } from '../tail-logs/tail-logs.component';
import { HttpClient, HttpParams } from '@angular/common/http';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const emptyIcon = './assets/pods-empty.svg';
const errorIcon = './assets/error.svg';
@Component({
  selector: 'app-app-pods',
  templateUrl: './app-pods.component.html',
  styleUrls: ['./app-pods.component.scss'],
})
export class AppPodsComponent extends FacetsPage implements OnDestroy {
  clusterId: any = '';
  errorMessage: string = '';
  fetchingPods = false;
  appName: string = '';
  podInfo: Array<Pod> = [];
  pods: Array<any> = [];
  icon = emptyIcon;
  message = 'No pods found!';
  rollingRestartInProgress: boolean = false;
  status: 'info' | 'error' = 'info';
  errMsg: string = '';
  clipboardServiceSubscriber: any;
  private rootUrl: string = window.location.origin;
  tailLogsParams = {
    tail: 100,
    podName: '',
    clusterId: '',
  };
  resourceType: string;
  podsUrl: string;

  @ViewChild('tailLogsComponent') tailLogsComponent: TailLogsComponent;

  constructor(
    private dialogService: NbDialogService,
    private clipboardService: ClipboardService,
    activatedRoute: ActivatedRoute,
    private uiApplicationControllerService: UiApplicationControllerService,
    public formatDateService: FormatDateService,
    private httpClient: HttpClient,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
    this.handleClipboardResponse();
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.podInfo = [];
    this.pods = [];
    this.fetchingPods = true;
    this.errorMessage = '';
    this.clusterId = p?.clusterId;
    this.appName = p?.resourceName;
    this.resourceType = p?.resourceType;
    this.uiApplicationControllerService
      .listPodsUsingGET({
        applicationName: this.appName,
        clusterId: this.clusterId,
      })
      .subscribe(
        (data) => {
          this.podInfo = data;
          this.loadPodData();
          this.fetchingPods = false;
        },
        (error) => {
          this.podInfo = null;
          this.pods = null;
          this.fetchingPods = false;
          this.message = '';
          this.errMsg =
            error?.error?.message || ' There is some error fetching pods';
          this.icon = errorIcon;
          this.status = 'error';
        },
      );
  }

  loadPodData() {
    this.podInfo.forEach((pod) => {
      let data = {
        name: pod?.metadata?.name || 'N/A',
        status: pod?.status?.phase || 'N/A',
        startTime:
          this.formatDateService.getFormattedDate(pod?.status?.startTime) ||
          'N/A',
        hostIp: pod?.status?.hostIP || 'N/A',
        podIp: pod?.status?.podIP || 'N/A',
        containers: [],
      };
      if (pod?.status?.containerStatuses) {
        pod.status.containerStatuses.forEach((container) => {
          let cData = {
            name: container.name,
            image: container.image,
            state: this.getContainerState(container.state),
          };
          data.containers.push(cData);
        });
      }
      this.pods.push(data);
    });
  }

  getContainerState(obj) {
    if (obj.running) {
      return { status: 'Running', meta: { startedAt: obj.running.startedAt } };
    } else if (obj.waiting) {
      return {
        status: 'Waiting',
        meta: {
          message: obj.waiting.message,
          reason: obj.waiting.reason,
        },
      };
    } else if (obj.terminated) {
      return {
        status: 'Terminated',
        meta: {
          startedAt: obj.terminated.startedAt,
          finishedAt: obj.terminated.finishedAt,
          containerID: obj.terminated.containerID,
          exitCode: obj.terminated.exitCode,
          message: obj.terminated.message,
          reason: obj.terminated.reason,
          signal: obj.terminated.signal,
        },
      };
    } else {
      return { status: '--', meta: {} };
    }
  }

  // Whenever this dialog box is opened, the auto-refresh mode for fetching logs is set as default with the interval of 5 seconds
  openTailLogsDialog(tailLogsUI, pod: Pod) {
    this.podsUrl = `${this.rootUrl}/cc-ui/v1/clusters/${this.clusterId}/pods/${pod?.['name']}/logs?tail=${this.tailLogsParams?.tail}&resourceType=${this.resourceType}&resourceName=${this.appName}`;
    this.dialogService.open(tailLogsUI, {
      context: pod,
      closeOnBackdropClick: false,
    });
  }

  copyLogsToClipboard() {
    let logLines = '';
    const logs = this.tailLogsComponent.getLogsData();
    logs.forEach((logLine) => {
      logLines += `${logLine.message}\n`;
    });
    this.clipboardService.copy(logLines);
  }

  handleClipboardResponse() {
    this.clipboardServiceSubscriber =
      this.clipboardService.copyResponse$.subscribe((res: any) => {
        if (res.isSuccess) {
          this.customToastService.show({
            title: 'Copied',
            content: 'Logs copied to clipboard',
            duration: 1000,
            type: 'success',
          });
        }
      });
  }
  showDetails(dialog: TemplateRef<any>) {
    this.dialogService.open(dialog);
  }

  confirmRollingRestart(confirmRollingRestartUI) {
    this.dialogService.open(confirmRollingRestartUI);
  }

  rollingRestartAll(ref) {
    this.rollingRestartInProgress = true;
    // uiApplicationControllerService.rollingRestartUsingPOST
    const params = new HttpParams()
      .set('resourceName', this.appName)
      .set('resourceType', this.resourceType);
    this.httpClient
      .post(
        `${this.rootUrl}/cc-ui/v1/clusters/${this.clusterId}/restart/${this.appName}`,
        null,
        { params: params },
      )
      .subscribe(
        (data) => {
          this.rollingRestartInProgress = false;
          ref.close();
          this.customToastService.success('Rolling Restart Started');
        },
        (error) => {
          this.rollingRestartInProgress = false;
          ref.close();
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Failed to trigger Rolling Restart',
            error,
          );
        },
      );
  }

  ngOnDestroy() {
    this.clipboardServiceSubscriber?.unsubscribe();
  }
}
