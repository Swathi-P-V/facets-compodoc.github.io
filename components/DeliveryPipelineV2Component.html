import { Component, Injector, OnDestroy } from '@angular/core';
import { ActivatedRoute, Params } from '@angular/router';
import {
  UiDeliveryPipelineControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  DeliveryPipelineUtilService,
  EnvironmentConfigPipeline,
  PipelineTreeNode,
} from 'src/app/services/delivery-pipeline.util.service';
import { Graph, Cell, Path } from '@antv/x6';
import Hierarchy from '@antv/hierarchy';
import '@antv/x6-angular-shape';
import { DlvPplEnvironmentNodeComponent } from './dlv-ppl-environment-node/dlv-ppl-environment-node.component';
import { ContextUpdatesService } from 'src/app/services/context-updates.service';
import { PipelineNode } from 'src/app/cc-api/models';
import { CustomToastService } from '../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

class MindMapData {
  id: string;
  width: number = 192;
  height: number = 160;
  children?: MindMapData[];
  environment?: PipelineTreeNode;
}

interface HierarchyResult {
  id: string;
  x: number;
  y: number;
  data: MindMapData;
  children?: HierarchyResult[];
}

@Component({
  selector: 'app-delivery-pipeline-v2',
  templateUrl: './delivery-pipeline-v2.component.html',
  styleUrls: ['./delivery-pipeline-v2.component.scss'],
})
export class DeliveryPipelineV2Component
  extends FacetsPage
  implements OnDestroy
{
  graph: Graph;
  blueprint: string; // Current blueprint
  nodeMap: Map<string, PipelineTreeNode> = new Map<string, PipelineTreeNode>(); // Stores the node against an ID
  pipelineTree: PipelineTreeNode; // Points to the first level of nodes directly connected to the blueprint
  allEnvironments: any[];
  editModeOn: boolean;
  fetchingPipelineData: boolean;
  errorPipelineData: boolean;
  emptyPipeline: boolean;
  pipelineSerialized: EnvironmentConfigPipeline[];
  environmentMap: Map<string, any> = new Map<string, any>();
  componentLoaded = false;
  updated: boolean = false;
  rootCell: any;

  OnPageLoad(p: Params): void {
    this.registerGraphModules();
    this.blueprint = p.stackName;
    this.getCurrentPipeline();
  }

  constructor(
    private contextUpdatesService: ContextUpdatesService,
    public deliveryPipelineUtilService: DeliveryPipelineUtilService,
    private deliveryPipelineService: UiDeliveryPipelineControllerService,
    private uiStackControllerService: UiStackControllerService,
    private injector: Injector,
    activatedRoute: ActivatedRoute,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  ngOnDestroy(): void {
    this.graph?.dispose();
    this.unregisterGraphModules();
  }

  setAllEnvironments() {
    this.contextUpdatesService.successfulContextUpdates.subscribe((context) => {
      let newBlueprint = context.currentContext.blueprint;
      if (newBlueprint != this.blueprint) {
        this.allEnvironments =
          this.contextUpdatesService.getCurrentEnvironments();
        this.createEnvironmentMap();
        this.blueprint = newBlueprint;
        this.getCurrentPipeline();
      }
    });
  }

  unregisterGraphModules() {
    Graph.unregisterConnector('mindmap');
    Graph.unregisterConnector('algo-connector');
    Graph.unregisterEdge('mindmap-edge');
    Graph.unregisterAngularContent('dlv-ppl-env-node');
  }

  registerGraphModules() {
    if (this.componentLoaded) {
      return;
    }
    this.componentLoaded = true;

    this.deliveryPipelineUtilService.graphConfigUpdates.subscribe(
      (newPipeline: PipelineTreeNode) => {
        this.pipelineTree = newPipeline;
        this.updated = true;
        this.render();
      },
    );

    Graph.registerConnector(
      'mindmap',
      (sourcePoint, targetPoint, routerPoints, options) => {
        const midX = sourcePoint.x + 10;
        const midY = sourcePoint.y;
        const ctrX = (targetPoint.x - midX) / 5 + midX;
        const ctrY = targetPoint.y;
        const pathData = `
         M ${sourcePoint.x} ${sourcePoint.y}
         L ${midX} ${midY}
         Q ${ctrX} ${ctrY} ${targetPoint.x} ${targetPoint.y}
        `;
        return options.raw ? Path.parse(pathData) : pathData;
      },
      true,
    );
    Graph.registerConnector(
      'algo-connector',
      (s, e) => {
        const offset = 4;
        const deltaY = Math.abs(e.y - s.y);
        const control = Math.floor((deltaY / 3) * 2);

        const v1 = { x: s.x, y: s.y + offset + control };
        const v2 = { x: e.x, y: e.y - offset - control };

        return Path.normalize(
          `M ${s.x} ${s.y}
           L ${s.x} ${s.y + offset}
           C ${v1.x} ${v1.y} ${v2.x} ${v2.y} ${e.x} ${e.y - offset}
           L ${e.x} ${e.y}
          `,
        );
      },
      true,
    );
    Graph.registerEdge(
      'mindmap-edge',
      {
        inherit: 'edge',
        connector: {
          name: 'mindmap',
        },
        attrs: {
          line: {
            targetMarker: '',
            stroke: '#04084936',
            strokeWidth: 2,
          },
        },
        zIndex: 0,
      },
      true,
    );
    Graph.registerAngularContent('dlv-ppl-env-node', {
      injector: this.injector,
      content: DlvPplEnvironmentNodeComponent,
    });

    this.renderGraph();
  }

  // Fetched from the breadcrumbs component through a service; stores all the envs for a BP
  // Fetches the pipeline array from the backend and builds the tree to be rendered
  async getCurrentPipeline() {
    await this.fetchAllEnvironments();
    this.editModeOn = false;
    this.deliveryPipelineUtilService.setEditModeTo('off');
    this.updated = false;
    this.fetchingPipelineData = true;
    this.errorPipelineData = false;
    this.emptyPipeline = !this.allEnvironments?.length;
    this.deliveryPipelineService
      .getDeliveryPipelineUsingGET(this.blueprint)
      .subscribe(
        (data: any) => {
          this.fillInaccessibleEnvironmentsToMap(data);
          this.pipelineSerialized = this.topologicalSort(data);
          this.fetchingPipelineData = false;
          this.buildTree();
        },
        (error: any) => {
          this.fetchingPipelineData = false;
          this.errorPipelineData = true;
        },
      );
  }

  // Constructs an N-ary tree out of the array with pipelineTree as the root
  deserializePipeline() {
    this.pipelineTree = new PipelineTreeNode(
      null,
      null,
      null,
      null,
      null,
      200,
      50,
    ); // Root
    let root = this.pipelineTree;
    this.nodeMap.clear();
    this.nodeMap.set('root', root);
    for (let i = 0; i < this.pipelineSerialized.length; i++) {
      const env = this.pipelineSerialized[i];
      let currentNode = new PipelineTreeNode(
        env.id,
        env.clusterName,
        env.clusterId,
        env.parentClusterId,
        env.config,
        210,
        env.config.autoPromoteArtifact && env.config.requiresSignOff
          ? 175
          : 135,
      );
      this.nodeMap.set(env.clusterId, currentNode);
      const parentNode = this.findParent(env.parentClusterId, root);
      parentNode.children.push(currentNode);
      parentNode.isLeafNode = false;
    }
    this.deliveryPipelineUtilService.setNodeMap(this.nodeMap);
    // console.log('Pipeline tree: ', this.pipelineTree);
  }

  // Recursive function for finding the parent node
  findParent(id: string, root: PipelineTreeNode) {
    if (id == root.clusterId) {
      return root;
    } else {
      for (let i = 0; i < root.children.length; i++) {
        const childNode = root.children[i];
        let parent = this.findParent(id, childNode);
        if (parent) {
          return parent;
        }
      }
    }
  }

  // Takes in the entire graph structure and renders it using the X6 library
  renderGraph() {
    this.graph = new Graph({
      container: document.getElementById('container'),
      resizing: false,
      grid: true,
      selecting: {
        enabled: true,
      },
      keyboard: {
        enabled: true,
      },
      panning: true,
      connecting: {
        connectionPoint: 'anchor',
      },
      mousewheel: {
        enabled: true,
        zoomAtMousePosition: true,
        modifiers: 'ctrl',
        minScale: 0.5,
        maxScale: 3,
      },
    });
  }

  zoomContainer(mode: 'in' | 'out') {
    const zoom = this.graph.zoom();
    if (zoom < 3 && mode == 'in') {
      this.graph.zoom(0.1);
    }
    if (zoom > 0.5 && mode == 'out') {
      this.graph.zoom(-0.1);
    }
  }

  render() {
    this.pipelineTree.id = 'root';
    // console.log('DATA to be rendered => ', this.pipelineTree);
    const result: HierarchyResult = Hierarchy.mindmap(this.pipelineTree, {
      direction: 'H',
      getHeight(d: MindMapData) {
        return d.height || '180';
      },
      getWidth(d: MindMapData) {
        return d.width || '200';
      },
      getHGap() {
        return 40;
      },
      getVGap() {
        return 40;
      },
      getSide: () => {
        return 'right';
      },
    });
    const cells: Cell[] = [];
    const traverse = (hierarchyItem: HierarchyResult) => {
      if (hierarchyItem) {
        const { data, children } = hierarchyItem;
        cells.push(
          this.graph.createNode({
            id: data.id,
            data: {
              ngArguments: {
                environment: data,
                blueprint: this.blueprint,
              },
            },
            x: hierarchyItem.x,
            y: hierarchyItem.y,
            width: data.width,
            height: data.height,
            shape: 'angular-shape',
            componentName: 'dlv-ppl-env-node',
          }),
        );
        if (children) {
          children.forEach((item: HierarchyResult) => {
            const { id } = item;
            cells.push(
              this.graph.createEdge({
                shape: 'mindmap-edge',
                source: {
                  cell: hierarchyItem.id,
                  anchor: {
                    name: 'right',
                    args: {
                      dy: data?.id == 'root' ? 0 : -40,
                    },
                  },
                },
                target: {
                  cell: id,
                  anchor: {
                    name: 'left',
                    args: {
                      dx: 10,
                      dy: -40,
                    },
                  },
                },
              }),
            );
            traverse(item);
          });
        }
      }
    };

    traverse(result);
    this.graph.resetCells(cells);
    this.deliveryPipelineUtilService.setGraph(this.graph);
    this.graph.scaleContentToFit({
      padding: 10,
      maxScale: 1,
    });
    // this.graph.centerContent();
    this.centreGraph();
  }

  // Calls multiple functions to render the pipeline tree
  buildTree() {
    this.deserializePipeline();
    this.deliveryPipelineUtilService.setPipelineTree(this.pipelineTree);
    this.render();
  }

  // Fetches all the environments if the page is opened directly routed to the delivery pipeline
  // and the environments data isn't available through the breadcrumb's component / contextUpdates service
  async fetchAllEnvironments() {
    return new Promise((resolve) => {
      this.uiStackControllerService
        .getClustersUsingGET1(this.blueprint)
        .subscribe((envs) => {
          this.allEnvironments = envs;
          this.createEnvironmentMap();
          resolve(true);
        });
    });
  }

  centreGraph() {
    // Get the div container and the graph elements
    const container = document.getElementById('container');

    // Calculate the dimensions of the div container and the graph
    const containerWidth = container.clientWidth;
    const containerHeight = container.clientHeight;
    const graphBBox = this.graph.getCellsBBox(this.graph.getNodes());

    // Determine the scaling factor needed to fit the graph inside the div container
    let scaleFactor = Math.min(
      containerWidth / graphBBox.width,
      containerHeight / graphBBox.height,
    );

    // Set a maximum scaling factor
    const maxScaleFactor = 1.25;

    // Set a min scaling factor
    const minScaleFactor = 0.8;

    // Limit the scaling factor to the maximum/minimum if necessary
    if (scaleFactor > maxScaleFactor) {
      scaleFactor = maxScaleFactor;
    }

    if (scaleFactor < minScaleFactor) {
      scaleFactor = minScaleFactor;
    }

    // Apply the scaling factor to the graph to resize it accordingly
    this.graph.scale(scaleFactor);

    // Center the graph inside the div container if necessary
    const xOffset =
      (containerWidth - graphBBox.width * scaleFactor) / 2 -
      graphBBox.x * scaleFactor;
    const yOffset =
      (containerHeight - graphBBox.height * scaleFactor) / 2 -
      graphBBox.y * scaleFactor;
    this.graph.translate(xOffset, yOffset);
  }

  // Maps environment metadata with the clusterId as the key
  createEnvironmentMap() {
    this.environmentMap.clear();
    this.allEnvironments?.forEach((env) => {
      env.name = env.name;
      this.environmentMap.set(env.id, env);
    });
    this.deliveryPipelineUtilService.setAllEnvironments(
      this.allEnvironments,
      this.environmentMap,
    );
  }

  // Some users might not have the access to all the environments
  // and thus they might not be present inside the response
  // which we receive from the getAllClusters API response because
  // that endpoint is RBAC enabled and would be only returning the accessible
  // environments to the user. Although, the delivery pipeline nodes should still be
  // visible when they are inaccessible. We need to loop through the nodes we receive from
  // the pipeline API and just fill in the remaining nodes inside the map too.
  fillInaccessibleEnvironmentsToMap(environments: any[]) {
    environments.forEach((environment: any) => {
      if (
        !this.deliveryPipelineUtilService
          .getEnvironmentMap()
          .has(environment.id)
      ) {
        this.deliveryPipelineUtilService.addToEnvironmentMap(
          environment.clusterId,
          { ...environment, ...{ name: environment.clusterName } },
        );
      }
    });
  }

  // We need all the parents to appear before their children to parse the array properly
  // Backend might send a shuffled list so we need to topologically sort the array first and then proceed
  discovered: Set<string> = new Set();
  topoSortedPipeline: EnvironmentConfigPipeline[] = [];
  currentDfsStack: EnvironmentConfigPipeline[];

  // Recursive DFS function for traversing the array component by component
  topoSortDFS(
    env: EnvironmentConfigPipeline,
    pipeline: EnvironmentConfigPipeline[],
  ) {
    if (!env || this.discovered.has(env.clusterId)) {
      return;
    }
    this.discovered.add(env.clusterId);
    this.currentDfsStack.push(env);
    let parent = pipeline.find((e) => e.clusterId === env.parentClusterId);
    this.topoSortDFS(parent, pipeline);
  }

  topologicalSort(pipeline: EnvironmentConfigPipeline[]) {
    this.discovered = new Set();
    this.topoSortedPipeline = [];
    pipeline.forEach((env) => {
      if (!this.discovered.has(env.clusterId)) {
        this.currentDfsStack = [];
        this.topoSortDFS(env, pipeline);
        while (this.currentDfsStack.length) {
          this.topoSortedPipeline.push(this.currentDfsStack.pop());
        }
      }
    });
    // console.log('topoSortedPipeline => ', this.topoSortedPipeline);
    return this.topoSortedPipeline;
  }

  // Once the user clicks on save changes, we need to serialize this tree back into an array
  // and update the interface as the API requires
  saveNewPipeline() {
    this.serializeTree();
  }

  // Traverses the tree in a depth first manner and adds all the nodes of the tree to an array
  // This array is by default topologically sorted
  // Once we have saved a tree from the UI, the backend should receive a topologically sorted list of nodes
  // which we don't really have to sort ourselves after receiving the pipeline from the GET call in the first place
  // BUT -- if the UI is receiving a pipeline for the first time, it might not be topologically sorted and might end up
  // messing the logic; and thus, we need to run the topo sort function every time
  serializeTree() {
    let pipeline: PipelineNode[] = [];
    let root = this.pipelineTree;
    let stack: PipelineTreeNode[] = [root];
    while (stack.length) {
      let top = stack.pop();
      let children = top.children;
      if (top.clusterId && top.clusterId.substring(0, 4) != 'D__-') {
        pipeline.push({
          id: top.id,
          clusterId: top.clusterId,
          parentClusterId: top.parentClusterId,
          config: top.config,
          stackName: this.blueprint,
        });
      }
      children.forEach((child) => {
        stack.push(child);
      });
    }
    this.savePipeline(pipeline);
  }

  savingPipeline: boolean = false;
  // Update the pipeline at the backend
  savePipeline(pipeline: PipelineNode[]) {
    this.savingPipeline = true;
    this.deliveryPipelineService
      .updateDeliveryPipelineUsingPUT({
        stackName: this.blueprint,
        pipelineNodes: pipeline,
      })
      .subscribe(
        (pipeline) => {
          this.customToastService.success(
            `Saved new pipeline for ${this.blueprint} successfully`,
          );
          this.editModeOn = false;
          this.getCurrentPipeline();
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            `Failed to update the pipeline for ${this.blueprint}`,
            error,
          );
        },
      )
      .add(() => {
        this.savingPipeline = false;
      });
  }
}
