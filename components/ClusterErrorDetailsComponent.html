import { Component, Input, EventEmitter, Output } from '@angular/core';
import { ActivatedRoute, Params } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { FormControl } from '@angular/forms';
// @ts-ignore
import * as Convert from 'ansi-to-html';

@Component({
  selector: 'app-cluster-error-details',
  templateUrl: './cluster-error-details.component.html',
  styleUrls: ['./cluster-error-details.component.scss'],
})
export class ClusterErrorDetailsComponent extends FacetsPage {
  @Input() data: any;
  @Input() ref: any;
  @Input() showLogsBtn: boolean = false;
  @Input() showCloseBtn: boolean = true;
  searchKey = new FormControl('');
  errorIcon = './assets/error-1.svg';
  errLogo: string = './assets/hazard.svg';
  convert: any;
  @Output() logsClick = new EventEmitter<any>();
  changes = './assets/changes.svg';

  constructor(route: ActivatedRoute) {
    super(route);
    this.convert = new Convert();
  }

  OnPageLoad(p: Params): void {}

  replaceElem(key, item, logItem, className) {
    const text = new DOMParser().parseFromString(
      item?.innerHTML?.toLowerCase(),
      'text/html',
    ).documentElement?.textContent;
    let HTMLLowerCase: string = text?.toLowerCase();
    // @ts-ignore
    let newHTML = HTMLLowerCase.replaceAll(
      new RegExp(key, 'ig'),
      '<span class="highlight__">' + key + '</span>',
    );
    let newSpan = document.createElement('span');
    newSpan.classList.add(className);
    newSpan.innerHTML = newHTML;
    logItem.replaceChild(newSpan, item);
  }

  search(event) {
    const key = event?.target?.value?.toLowerCase();

    let allTitleParent = Array.from(document.querySelectorAll('.logItem'));
    let allTitleItems = Array.from(document.querySelectorAll('.errorTitle'));
    allTitleParent.forEach((logItem, index) => {
      this.replaceElem(key, allTitleItems?.[index], logItem, 'errorTitle');
    });

    let allStackParent = Array.from(document.querySelectorAll('.errorStack'));
    let allStackItems = Array.from(document.querySelectorAll('.stack'));
    allStackParent.forEach((logItem, index) => {
      this.replaceElem(key, allStackItems?.[index], logItem, 'stack');
    });
  }

  getHTML(title) {
    if (title) {
      let stackString = this.convert.toHtml(title);
      const convertSpan = document.createElement('span');
      convertSpan.innerHTML = stackString;
      return (
        convertSpan?.textContent?.toLowerCase() ||
        convertSpan?.innerText?.toLowerCase()
      );
    }
    return '';
  }

  changeStackStyle(item) {
    item.style.height = item?.style?.height === '2.6rem' ? 'auto' : '2.6rem';
    item.style.overflow = item?.style?.overflow === 'auto' ? 'hidden' : 'auto';
  }

  showMoreLess(item) {
    item.style.display = item?.style?.display === 'block' ? 'none' : 'block';
  }

  showLogDetails(data) {
    this.logsClick.emit({
      data: data,
    });
  }
}
