import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { AbstractCluster, Account, Stack } from 'src/app/cc-api/models';
import {
  UiAccountsControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { organizationAccountIconMap } from 'src/app/constants/org-account-icon-map';
const errorStateIcon = './assets/error.svg';

@Component({
  selector: 'app-account-details',
  templateUrl: './account-details.component.html',
  styleUrls: ['./account-details.component.scss'],
})
export class AccountDetailsComponent extends FacetsPage {
  accountId: string = '';
  accountDetails: Account & { [key: string]: any };
  allEnvironments: AbstractCluster[] = [];
  allBlueprints: Stack[] = [];
  fetchingAccountDetails: boolean = true;
  errorMessage: string = '';
  errorStateIcon = errorStateIcon;

  providerToLogoMap = organizationAccountIconMap;

  providerToAttributeMap = {
    AWS: [
      { label: 'External Id', id: 'externalId', type: 'text' },
      { label: 'IAM Role', id: 'iamRole', type: 'text' },
      { label: 'Environments', id: 'environments', type: 'ellipsis' },
    ],
    AZURE: [
      { label: 'Subscription ID', id: 'subscriptionId', type: 'text' },
      { label: 'Tenant ID', id: 'tenantId', type: 'text' },
      { label: 'Client ID', id: 'clientId', type: 'text' },
      { label: 'Client Secret', id: 'clientSecret', type: 'text' },
      { label: 'Environments', id: 'environments', type: 'ellipsis' },
    ],
    GCP: [
      {
        label: 'Service account key',
        id: 'serviceAccountKey',
        type: 'longText',
      },
      { label: 'Project ID', id: 'project', type: 'text' },
      { label: 'Environments', id: 'environments', type: 'ellipsis' },
    ],
    GITHUB: [
      { label: 'Username', id: 'userName', type: 'text' },
      { label: 'Personal Access Token', id: 'accessToken', type: 'text' },
      { label: 'Blueprints', id: 'blueprints', type: 'ellipsis' },
    ],
    GITLAB: [
      { label: 'Username', id: 'userName', type: 'text' },
      { label: 'Personal Access Token', id: 'token', type: 'text' },
      { label: 'Blueprints', id: 'blueprints', type: 'ellipsis' },
    ],
    BITBUCKET: [
      { label: 'Username', id: 'userName', type: 'text' },
      { label: 'Password', id: 'password', type: 'text' },
      { label: 'Blueprints', id: 'blueprints', type: 'ellipsis' },
    ],
    KUBERNETES: [
      { label: 'Host', id: 'host', type: 'text' },
      { label: 'Environments', id: 'environments', type: 'ellipsis' },
    ],
    CODER: [
      { label: 'template ID', id: 'templateId', type: 'text' },
      { label: 'Coder URL', id: 'url', type: 'text' },
      { label: 'Organization ID', id: 'organizationId', type: 'text' },
    ],
  };
  constructor(
    activatedRoute: ActivatedRoute,
    private router: Router,
    private accountControllerService: UiAccountsControllerService,
    private stackControllerService: UiStackControllerService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.accountId = p?.accountId;
    this.getAccountDetails(this.accountId);
  }

  getAccountDetails(accountId) {
    this.fetchingAccountDetails = true;
    this.accountControllerService.getAccountUsingGET(accountId).subscribe(
      async (data) => {
        this.accountDetails = data;
        this.accountDetails['providerImg'] =
          this.providerToLogoMap[this.accountDetails?.provider];
        this.accountDetails['details'] = [];
        await this.addDetails(this.accountDetails?.provider);
        this.fetchingAccountDetails = false;
      },
      (error) => {
        this.errorMessage =
          error?.error?.message || 'Error while fetching account details';
        this.fetchingAccountDetails = false;
      },
    );
  }

  addDetails(provider) {
    return new Promise((resolve, reject) => {
      let attributes = this.providerToAttributeMap[provider];
      attributes?.forEach(
        (attr: { label: string; id: string; type: string }) => {
          if (attr?.id === 'environments') {
            attr['value'] = this.accountDetails?.associatedTo;
            this.accountDetails?.['details']?.push(attr);
          } else if (attr?.id === 'blueprints') {
            attr['value'] = this.accountDetails?.associatedTo?.map(
              (stack) => stack?.stackName,
            );
            this.accountDetails?.['details']?.push(attr);
          } else {
            attr['value'] = this.accountDetails?.[attr?.id] || '--';
            this.accountDetails?.['details']?.push(attr);
          }
        },
      );
      resolve(true);
    });
  }

  handleBack = () => {
    this.router.navigate(['settings', 'organization-accounts']);
  };
}
