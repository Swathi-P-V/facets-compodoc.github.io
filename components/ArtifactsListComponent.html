import { SmartTableRouterlinkComponent } from './../../../components/smart-table-routerlink/smart-table-routerlink.component';
import { Component, OnDestroy, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import {
  UiArtifactCiControllerService,
  UiArtifactRoutingRuleControllerService,
  UiPromotionWorkflowControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { NbDialogService } from '@nebular/theme';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { SmartTableWithIconComponent } from 'src/app/components/smart-table-with-icon/smart-table-with-icon.component';
import { PromotionWorkflow } from 'src/app/cc-api/models';
import { FormatDateService } from 'src/app/services/format-date.service';
import { SmartTableTimeagoTooltipComponent } from '../../../components/smart-table-timeago-tooltip/smart-table-timeago-tooltip.component';

import { UserService } from 'src/app/services/user.service';
import { UserAuthoritiesService } from 'src/app/services/user-authorities.service';
import { ARTIFACT_CI_WRITE } from '../../../constants/permissions';
import { RbacButton } from 'src/app/models/rbac-button.interface';
import { CustomToastService } from '../../custom-toast/custom-toast.service';

const emptyStateLogo = './assets/artifact-empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-artifacts-list',
  templateUrl: './artifacts-list.component.html',
  styleUrls: ['./artifacts-list.component.scss'],
})
export class ArtifactsListComponent extends FacetsPage implements OnDestroy {
  @ViewChild('workflowDetails', { static: true }) //promotional workflow details popover
  private workflowDetails: TemplateRef<any>;
  artifacts: LocalDataSource; //artifacts CI list
  fetchingArtifacts = false; //artifacts CI loader
  fetchingAllEnvs: boolean = false;
  allEnvs: any;
  envIdNameMapping: any = {};
  deleteArtifactCIInProgress: boolean = false; //delete CI integration loader
  showEmptyState: boolean = false; //show empty state condition variable
  emptyStateLogo = emptyStateLogo; //empty state logo
  primaryMsg: string = 'No CI integration'; //empty state primary message
  secondaryMsg: string = 'Create new CI integration to promote your artifacts'; //empty state secondary message
  showBtn: boolean = false; //empty state show button conditional variable
  buttonText: string = 'Create CI Integration'; //empty state button text
  stackName: string = ''; //stack name for CI integration api
  fetchingPromotion: boolean = false; //fetch promotion workflow
  promotionWorkflows: PromotionWorkflow[] = []; //promotional workflows
  workflowIdNameMapping: any = {};
  status: 'info' | 'error' = 'info';
  CIRuleIdNameMap: {} = {};
  createPermission = ARTIFACT_CI_WRITE;

  //header action buttons
  ciActionButtons = [
    {
      id: 'create-artifact',
      title: 'Create CI Integration',
      buttonType: 'primary',
      nbTooltip: 'Create new CI Integration',
      icon: '',
      action: this.createPermission,
    },
  ];

  //table edit and delete custom action buttons
  smartTableCustomActionButtons = [
    {
      id: 'edit',
      icon: 'edit-2-outline',
      toolTip: 'Edit CI',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: 'ARTIFACT_CI_WRITE',
      onClick: (data) => {
        this.editCi(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete CI',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: 'ARTIFACT_CI_DELETE',
      onClick: (data) => {
        this.deleteArtifactCIConfirmPopup(data);
      },
    },
  ];

  @ViewChild('confirmDelete', { static: true })
  private confirmDelete: TemplateRef<any>;

  @ViewChild('attachWorkflowModal', { static: true })
  private attachWorkflowModal: TemplateRef<any>;

  tableSettings: any = {
    // selectMode: 'multi',
    columns: {
      selected: {
        type: 'custom',
        renderComponent: SoftDeleteCheckboxComponent,
        filter: false,
        onComponentInitFunction: (instance: any) => {
          instance.change.subscribe((data) => {
            this.onClickSelectCheckbox(data);
          });
        },
      },
      ciName: {
        title: 'CI Integration',
        filter: true,
        width: '15%',
        type: 'custom',
        valuePrepareFunction: (value, row, cell) => {
          // DATA FROM HERE GOES TO renderComponent
          const displayVal = {
            url: ['/', 'artifact-ci-details', row?.id],
            display: value,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: SmartTableRouterlinkComponent,
      },
      registrationTypeName: {
        title: 'Registration Type',
        width: '10%',
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [
              { value: 'Environment', title: 'Environment' },
              { value: 'Release Stream', title: 'Release Stream' },
              { value: 'Hybrid', title: 'Hybrid' },
            ],
          },
        },
        type: 'text',
      },
      stackName: {
        title: 'Blueprint Name',
        filter: false,
        type: 'text',
        width: '10%',
        valuePrepareFunction(stackName, row) {
          if (stackName?.length > 0) {
            return stackName;
          }
          return 'All';
        },
      },
      promotionWorkflowId: {
        title: 'Promotion Workflow',
        filter: false,
        type: 'custom',
        width: '15%',
        valuePrepareFunction: (cell, row) => {
          let tooltip = '';
          return JSON.stringify({
            isTextLink: this.workflowIdNameMapping[cell],
            url: null,
            textContent:
              cell?.length > 0 ? this.workflowIdNameMapping[cell] : 'N/A',
            showIcon: false,
            noTooltip: true,
            badgeClass:
              'mb-0 ' +
              (cell?.length > 0 ? 'selected-text cursor-pointer' : ''),
          });
        },
        renderComponent: SmartTableWithIconComponent,
        onComponentInitFunction: (instance) => {
          instance.byTitleClick.subscribe((data) => {
            if (data?.rowData?.promotionWorkflowId?.length > 0) {
              this.openWorkflowDetails(data?.rowData);
            }
          });
        },
      },
      ruleName: {
        title: 'CI Rule',
        filter: true,
        type: 'text',
      },
      createdBy: {
        title: 'Created By',
        width: '10%',
        filter: true,
        type: 'custom',
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            user: row.createdBy,
            label: row.creationDate,
          });
        },
        renderComponent: SmartTableTimeagoTooltipComponent,
      },

            this.smartTableCustomActionButtons;

        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance: any) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton?.id === data?.actionButton?.id;

            customActionButton.onClick(data?.rowData);
          });

        filter: false,
      },
    },
    noDataMessage: 'No data found',
    pager: {
      display: false,
      perPage: 10,
    },
    mode: 'external',
    actions: false,
  };

  //Non used properties
  selectedItem: any;
  name: string;

  footerButtons: RbacButton[] = [
    {
      id: 'attach-workflow',
      label: 'Attach Promotion Workflow',
      permissions: 'ARTIFACT_CI_WRITE',
      tooltip: 'Select artifacts in bulk to attach a promotion workflow',
      status: 'primary',
      disabled: true,
      appearance: 'outline',
    },
  ];
  selectedCIIntegrations: any = [];
  selectedPromotionWorkflow: string;
  filteredArtifacts: any = [];
  isSelectAll = false;
  differentRegTypeError = '';
  workflowsByRegistrationType: {};
  registrationTypeOfSelectedCI = '';
  attachWorkflowInProgress = false;
  showAllIndeterminate = false;
  hasAccessToArtifactCIWrite: boolean = false;
  userSubscription: any;
  defaultPromotionWorkflow: string = '';

  constructor(
    private activatedRoute: ActivatedRoute,
    private dialogService: NbDialogService,
    private router: Router,
    private uiArtifactCiController: UiArtifactCiControllerService,
    private promotionWorkflowService: UiPromotionWorkflowControllerService,
    private formatDateService: FormatDateService,
    private uiStackController: UiStackControllerService,
    private userService: UserService,
    private userAuthoritiesService: UserAuthoritiesService,
    private uiAritifactRoutingRuleController: UiArtifactRoutingRuleControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
    this.artifacts = new LocalDataSource();

    this.artifacts.onChanged().subscribe((change) => {
      this.filteredArtifacts = change.elements;
      this.setIndeterminateState();
    });
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.fetchingArtifacts = true;
    this.stackName = p?.stackName;
    this.name = p?.name;
    this.checkAndSubscribeUserService();
    this.resetPromotionWorkflowVars();
    Promise.all([
      this.getAllEnvironments(),
      this.getAllWorkflows(),
      this.getAllCIRules(this.stackName),
    ]).then(
      (promiseRes) => {
        this.fetchArtifacts();
      },
      (error) => {
        this.fetchingArtifacts = false;
        this.setErrorState(error);
      },
    );
  }
  checkAndSubscribeUserService() {
    if (this.userService.userLoaded) {
      this.hasAccessToArtifactCIWrite = this.checkUserAccessArtifactWrite();
      if (!this.hasAccessToArtifactCIWrite) {
        this.updateTableColumns();
        return;
      }
    } else {
      this.userSubscription = this.userService.userObserver.subscribe(
        this.onChangeUser,
      );
    }
  }

  updateTableColumns() {
    const newTableSettings = {
      ...this.tableSettings,
      columns: {
        ...this.tableSettings.columns,
      },
    };
    delete newTableSettings.columns.selected;
    this.tableSettings = Object.assign({}, newTableSettings);
  }
  onChangeUser = (userLoaded) => {
    if (userLoaded) {
      this.hasAccessToArtifactCIWrite = this.checkUserAccessArtifactWrite();
      if (!this.hasAccessToArtifactCIWrite) {
        this.updateTableColumns();
        return;
      }
    }
  };

  fetchArtifacts() {
    this.fetchingArtifacts = true;
    this.uiArtifactCiController
      .getArtifactCisByStackUsingGETResponse(this.stackName)
      .subscribe(
        (response) => {
          if (response?.body?.length > 0) {
            this.artifacts?.load([
              ...this.setRegistrationTypeAndValue(response?.body),
            ]);
          }
          this.showEmptyState = response?.body?.length <= 0;
          this.showBtn = response?.body?.length <= 0;
          this.fetchingArtifacts = false;
        },
        (error) => {
          this.fetchingArtifacts = false;
          this.setErrorState(error);
        },
      );
  }

  setErrorState(error?) {
    this.showEmptyState = true;
    this.emptyStateLogo = errorIcon;
    this.secondaryMsg =
      error?.error?.message || 'There is some error fetching CI integration';
    this.primaryMsg = '';
    this.status = 'error';
    this.showBtn = false;
  }

  //setting registration type and value in artifact CI list based on registration type values from api.
  setRegistrationTypeAndValue(ciIntegrations) {
    let clonedCiIntegrations = cloneDeep(ciIntegrations);
    clonedCiIntegrations = clonedCiIntegrations.map((ciIntegration) => ({
      ...ciIntegration,
      registrationTypeName: convertToSentenceCase(
        ciIntegration?.registrationType,
      ),
      selected: false,
      ruleName: ciIntegration?.ruleId
        ? this.CIRuleIdNameMap[ciIntegration?.ruleId]
        : 'N/A',
    }));
    return clonedCiIntegrations;
  }

  getAllEnvironments = (populateVals = false) => {
    return new Promise((resolve, reject) => {
      this.uiStackController
        .getAllClustersUsingGET({
          page: 0,
          perPage: 10000,
          sortBy: 'ID',
        })
        .subscribe(
          (envs) => {
            this.allEnvs = envs?.content;
            this.allEnvs.forEach((env) => {
              this.envIdNameMapping[env?.id] = env?.name;
            });
            resolve(true);
          },
          (error) => {
            reject(error);
          },
        );
    });
  };

  getAllWorkflows = () => {
    return new Promise((resolve, reject) => {
      this.promotionWorkflowService.getAllWorkflowsUsingGET().subscribe(
        (response) => {
          this.promotionWorkflows = response;
          this.promotionWorkflows?.forEach((workflow) => {
            this.workflowIdNameMapping[workflow.id] = workflow?.workflowName;
          });
          resolve(true);
        },
        (error) => {
          reject(error);
        },
      );
    });
  };

  getAllCIRules = (stackName) => {
    return new Promise((resolve, reject) => {
      this.CIRuleIdNameMap = {};
      this.uiAritifactRoutingRuleController
        .getAllArtifactRoutingRulesUsingGET({ stackName: stackName })
        .subscribe(
          (res) => {
            if (res?.length > 0) {
              res.forEach((rule) => {
                this.CIRuleIdNameMap[rule?.id] = rule?.ruleName;
              });
            }
            resolve(true);
          },
          (error) => {
            reject(error);
          },
        );
    });
  };

  deleteArtifactCIConfirmPopup(rowData) {
    this.uiArtifactCiController
      .getArtifactsForCIUsingGET(rowData?.ciName)
      .subscribe(
        (res) => {
          if (res?.length > 0) {
            res.forEach((artifact) => {
              if (artifact.artifactId) {
                rowData.showMessage = true;
              }
            });
          } else {
            rowData.showMessage = false;
          }
          this.dialogService.open(this.confirmDelete, { context: rowData });
        },
        (err) => {
          rowData.showMessage = false;
          this.dialogService.open(this.confirmDelete, { context: rowData });
        },
      );
  }

  deleteArtifactCI(data, ref) {
    this.deleteArtifactCIInProgress = true;
    this.uiArtifactCiController.deleteArtifactCIUsingDELETE(data?.id).subscribe(
      (res) => {
        this.deleteArtifactCIInProgress = false;
        this.customToastService.success('CI Integration deleted successfully');
        ref.close();
        this.fetchArtifacts();
      },
      (error) => {
        this.deleteArtifactCIInProgress = false;
        this.onAPIErrorService.onAPIError(
          error?.error?.message || 'CI Integration delete failed',
          error,
        );
        ref.close();
      },
    );
  }

  createCI() {
    this.router.navigate([
      '/',
      'stack',
      this.stackName,
      'artifacts',
      'create-ci',
    ]);
  }

  editCi(data) {
    this.router.navigate(
      ['/', 'stack', this.stackName, 'artifacts', 'edit-ci', data?.id],
      {
        state: {
          previousComponent: 'artifactsList',
          stackName: this.stackName,
        },
      },
    );
  }

  listenToActionButtonEvents(event) {
    if (event === 'create-artifact') {
      this.createCI();
    }
  }

  openWorkflowDetails(rowData) {
    let workflow = this.promotionWorkflows.find(
      (workflow) => workflow?.id === rowData?.promotionWorkflowId,
    );
    this.dialogService.open(this.workflowDetails, {
      context: { rowData: workflow },
    });
  }

  getFormattedDate(date) {
    return this.formatDateService.getFormattedDate(date);
  }

  onArtifactSelect() {
    this.differentRegTypeError = '';
    this.registrationTypeOfSelectedCI = '';

    const totalSelectedArtifacts = this.artifacts?.['data'].filter(
      (artifact) => artifact.selected,
    );
    const groupByRegistration = groupBy(
      totalSelectedArtifacts,
      (ciIntegration) => ciIntegration.registrationType,
    );
    if (Object.keys(groupByRegistration).length > 1) {
      this.differentRegTypeError =
        'Please select CI integration of the same registration type';
      this.footerButtons[0].disabled = true;
      return;
    }

    this.registrationTypeOfSelectedCI = Object.keys(groupByRegistration)[0];
    this.selectedCIIntegrations = totalSelectedArtifacts;
    if (this.selectedCIIntegrations?.length) {
      this.footerButtons[0].disabled = false;
    } else {
      this.footerButtons[0].disabled = true;
    }
  }

  setDefaultPromotionWorkflow() {
    this.selectedPromotionWorkflow =
      this.workflowsByRegistrationType[this.registrationTypeOfSelectedCI].find(
        (workflow) => workflow.defaultWorkflow,
      )?.id || '';
  }
  addHierarchyText() {
    this.promotionWorkflows = this.promotionWorkflows?.map((workflow) => ({
      ...workflow,
      hierarchyText: this.getHierarchyText(workflow),
    }));
  }
  handleFooterButtons(buttonId) {
    if (buttonId === 'attach-workflow') {
      this.addHierarchyText();
      this.groupWorkflowsByRegType();
      this.setDefaultPromotionWorkflow();
      this.dialogService.open(this.attachWorkflowModal, { context: '' });
    }
  }

  attachPromotionWorkflow(ref) {
    this.attachWorkflowInProgress = true;
    this.uiArtifactCiController
      .bulkEditWorkflowUsingPOST({
        artifactCiIds: this.selectedCIIntegrations.map(
          (ciIntegration) => ciIntegration.id,
        ),
        workflowId: this.selectedPromotionWorkflow,
      })
      .subscribe(
        () => {
          this.attachWorkflowInProgress = false;
          ref.close();

          const selectedWorkflowDetails =
            this.promotionWorkflows.find(
              (workflow) => workflow.id === this.selectedPromotionWorkflow,
            ) || {};
          this.customToastService.success(
            `"${selectedWorkflowDetails?.workflowName}" has been successfully attached to ${this.selectedCIIntegrations.length} Artifact CIs`,
          );
          this.resetPromotionWorkflowVars();
          this.fetchArtifacts();
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            error?.error?.message ||
              error?.message ||
              'Something went wrong while attaching promotion workflow.',
            error,
          );
          this.attachWorkflowInProgress = false;
        },
      );
  }

  onChangeSelectAll(checked) {
    this.isSelectAll = checked;
    let filteredArtifactsWithSelectedProp = this.filteredArtifacts?.map(
      (artifact) => ({
        ...artifact,
        selected: checked,
      }),
    );

    const artifactsMapById = new Map(
      this.artifacts?.['data'].map((item) => [item.id, item]),
    );

    filteredArtifactsWithSelectedProp.forEach((artifact) => {
      const item = artifactsMapById.get(artifact.id);
      if (item) {
        item.selected = artifact.selected;
      }
    });
    this.artifacts.load(Array.from(artifactsMapById.values()));

    this.onArtifactSelect();
  }

  setIndeterminateState() {
    const artifactMapById = new Map(
      this.artifacts?.['data'].map((item) => [item.id, item]),
    );
    const currentSelectedArtifacts = this.filteredArtifacts.filter(
      (artifact) => artifactMapById.get(artifact.id).selected,
    );
    this.showAllIndeterminate = currentSelectedArtifacts.length
      ? currentSelectedArtifacts?.length !== this.filteredArtifacts.length
      : false;
    this.isSelectAll = currentSelectedArtifacts.length > 0;
  }

  onClickSelectCheckbox(data) {
    if (typeof data.val !== 'boolean') {
      return;
    }
    const clonedArtifacts = cloneDeep(this.artifacts?.['data']);
    const selectedArtifact = clonedArtifacts.find(
      (artifact) => artifact.id === data.rowData.id,
    );
    selectedArtifact.selected = data.val;

    this.artifacts.load([...clonedArtifacts]);

    this.setIndeterminateState();
    this.onArtifactSelect();
  }

  getHierarchyText(workflow) {
    let hierarchyText = [];
    hierarchyText = workflow?.hierarchies?.map((hierarchy) => {
      return workflow.registrationType === 'ENVIRONMENT'
        ? this.envIdNameMapping[hierarchy.registrationValue]
        : hierarchy.registrationValue;
    });
    return `(${hierarchyText?.join(' -> ')})`;
  }

  groupWorkflowsByRegType() {
    this.workflowsByRegistrationType = groupBy(
      this.promotionWorkflows,
      (workflow) => workflow.registrationType,
    );
    this.workflowsByRegistrationType['ENVIRONMENT'] =
      this.workflowsByRegistrationType['ENVIRONMENT']?.filter((workflow) => {
        return workflow.stackName === this.stackName;
      });
    this.workflowsByRegistrationType['RELEASE_STREAM'] =
      this.workflowsByRegistrationType['RELEASE_STREAM']?.filter((workflow) => {
        return (
          workflow.stackName === this.stackName || !workflow.stackName?.length
        );
      });
    this.workflowsByRegistrationType['HYBRID'] = [
      ...(this.workflowsByRegistrationType['ENVIRONMENT'] || []),
      ...(this.workflowsByRegistrationType['RELEASE_STREAM'] || []),
    ];
  }
  checkUserAccessArtifactWrite() {
    return this.userService.hasSomeroles(['ARTIFACT_CI_WRITE']);
  }
  disablePromotionWorkflowOption(): boolean {
    return this.userAuthoritiesService.isActionDisabled('ARTIFACT_CI_WRITE');
  }
  openCreateWorkflow(ref, event) {
    ref.close();
    event.stopPropagation();
    this.router.navigate(['../create-workflow'], {
      relativeTo: this.route,
    });
  }
  resetPromotionWorkflowVars() {
    this.selectedCIIntegrations = [];
    this.selectedPromotionWorkflow = '';
    this.isSelectAll = false;
    this.differentRegTypeError = '';
    this.workflowsByRegistrationType = {};
    this.registrationTypeOfSelectedCI = '';
    this.attachWorkflowInProgress = false;
    this.showAllIndeterminate = false;
    this.defaultPromotionWorkflow = '';
    this.footerButtons[0].disabled = true;
  }
  ngOnDestroy(): void {
    if (this.userSubscription) {
      this.userSubscription.unsubscribe();
    }
  }
}
