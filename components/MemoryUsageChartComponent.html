import { Component, Input, OnChanges, SimpleChanges } from '@angular/core';
import { AgChartOptions } from 'ag-charts-community';
import * as agCharts from 'ag-charts-community';
import {
  GrafanaServiceService,
  MemoryGrafanaStats,
} from 'src/app/services/grafana-service.service';
import { Subscription } from 'rxjs';
import { AppNativeChartsService } from '../app-native-charts.service';
const errorStateIcon = './assets/error.svg';

const emptyStateIcon = './assets/alert-empty.svg';

@Component({
  selector: 'app-memory-usage-chart',
  templateUrl: './memory-usage-chart.component.html',
  styleUrls: ['./memory-usage-chart.component.scss'],
})
export class MemoryUsageChartComponent implements OnChanges {
  @Input() clusterID: any;
  @Input() resource: any;
  @Input() filter: any;

  series: any[];
  contextSubscription: Subscription;

  chartLoading: boolean = true;
  options: AgChartOptions;

  usedStats: any;
  limitStats: any;

  chartDataUsedStats: any[] = [];
  chartDataLimitStats: any[] = [];
  chartDataFinal: any[] = [];

  error: boolean = false;
  errorStateIcon = errorStateIcon;
  empty: boolean = false;
  emptyStateIcon = emptyStateIcon;

  filterStart: number;
  filterEnd: number;

  graphColors: string[] = [
    '#7EB16D',
    '#EAB839',
    '#040849',
    '#AC9FBB',
    '#F7717D',
    '#5299D3',
    '#ADF7B6',
    '#6B7FD7',
    '#D55672',
    '#F5E663',
  ];
  lastColorUsed: number = -1;

  constructor(
    private grafanaService: GrafanaServiceService,
    private appNativeChartsService: AppNativeChartsService,
  ) {}

  ngOnChanges(changes: SimpleChanges): void {
    if (changes) {
      if (changes.clusterID?.currentValue) {
        this.clusterID = changes.clusterID?.currentValue;
      }
      if (changes.resource?.currentValue) {
        this.resource = changes.resource?.currentValue;
      }
      if (changes.filter?.currentValue) {
        this.applyFilter(changes.filter?.currentValue);
      } else {
        this.setGraphData();
      }
    }
  }

  applyFilter(value) {
    this.chartLoading = true;
    this.filterStart = value?.from.getTime();
    this.filterEnd = value?.till.getTime();
    this.setGraphData();
  }

  async setGraphData() {
    this.empty = false;
    this.error = false;
    await this.getMemoryUsedStats();
    await this.getMemoryLimitStats();
    this.createSeries();
    this.setOptions();
    this.chartLoading = false;
  }

  setOptions() {
    const dayDiff = Math.floor(
      Math.floor((this.filterEnd - this.filterStart) / 3600000) / 24,
    );
    this.options = this.appNativeChartsService.getOptions(
      this.chartDataFinal,
      this.series,
      'Container GBytes of Memory Used',
      dayDiff,
    );
    this.options = {
      autoSize: true,
      data: this.chartDataFinal,
      theme: {
        overrides: {
          line: {
            series: {
              highlightStyle: {
                series: {
                  dimOpacity: 0.2,
                  strokeWidth: 1,
                },
              },
            },
          },
        },
      },
      title: {
        text: 'Container GBytes of Memory Used',
        fontSize: 12,
      },
      series: this.series,
      axes: [
        {
          type: 'number',
          position: 'left',
        },
        {
          position: 'bottom',
          type: 'time',
          nice: true,
          tick: {
            count: agCharts.time.minute.every(10),
          },
          title: {
            text: 'Time',
          },
          label: {
            format:
              isNaN(dayDiff) || dayDiff < 4
                ? '%H' + ':' + '%M'
                : '%-m' + '/' + '%-d',
          },
        },
      ],
      legend: {
        item: {
          marker: {
            shape: 'square', // 'square', 'diamond', 'cross', 'plus', 'triangle'
          },
        },
        position: 'bottom',
      },
    };
    // console.log('OPTIONS', this.options);
    if (!this.chartDataFinal || !this.chartDataFinal.length) {
      this.empty = true;
    }
  }

  getRequest(
    quotaType:
      | 'kube_pod_container_resource_limits'
      | 'container_memory_usage_bytes',
  ) {
    const request: MemoryGrafanaStats = {
      clusterID: this.clusterID,
      resource: this.resource,
      quotaType: quotaType,
      steps: 30,
    };

    if (this.filterStart && this.filterEnd) {
      request.start = Math.floor(this.filterStart / 1000);
      request.end = Math.floor(this.filterEnd / 1000);
      request.steps = Math.floor((request.end - request.start) / 120);
    }

    return request;
  }

  async getMemoryUsedStats() {
    return new Promise((resolve) => {
      this.grafanaService
        .getMemoryStats(this.getRequest('container_memory_usage_bytes'))
        .subscribe(
          (rsp) => {
            this.usedStats = rsp;
          },
          (err) => {
            this.error = true;
          },
        )
        .add(() => {
          resolve(true);
        });
    });
  }

  getMemoryLimitStats() {
    return new Promise((resolve) => {
      this.grafanaService
        .getMemoryStats(this.getRequest('kube_pod_container_resource_limits'))
        .subscribe(
          (rsp) => {
            this.limitStats = rsp;
          },
          (err) => {
            this.error = true;
          },
        )
        .add(() => {
          resolve(true);
        });
    });
  }

  getData(): any[] {
    return this.chartDataFinal;
  }

  createSeries() {
    this.series = [];
    this.chartDataUsedStats = [];
    this.chartDataLimitStats = [];
    this.setSeriesFor(
      'used',
      this.usedStats?.data?.result,
      this.chartDataUsedStats,
    );
    this.setSeriesFor(
      'limit',
      this.limitStats?.data?.result,
      this.chartDataLimitStats,
    );
    this.createFinalSeries(this.chartDataUsedStats, this.chartDataLimitStats);
  }

  createFinalSeries(...args: any) {
    let maxLen = -1;
    for (let i = 0; i < args.length; i++) {
      maxLen = Math.max(maxLen, args[i].length);
    }
    // console.log('maxLen', maxLen);
    this.chartDataFinal = [];
    for (let i = 0; i < maxLen; i++) {
      this.chartDataFinal.push({
        ...(this.chartDataLimitStats[i] || {}),
        ...(this.chartDataUsedStats[i] || {}),
      });
    }
  }

  setSeriesFor(name: string, results: any[], chartSeries: any[]) {
    let allResults = results;
    this.lastColorUsed = -1;
    for (let i = 0; i < results?.length; i++) {
      const metric = allResults[i]['metric']['pod'];
      const values = allResults[i]['values'];
      let randomColor =
        this.lastColorUsed < this.graphColors.length - 1
          ? this.graphColors[++this.lastColorUsed]
          : this.grafanaService.getRandomColorForGraph();
      this.series.push({
        type: 'line',
        xKey: 'time',
        yKey: `${name}-${metric}`,
        stroke: randomColor,
        marker: {
          stroke: randomColor,
          fill: randomColor,
          size: 2,
        },
      });
      for (let j = 0; j < values.length; j++) {
        const dataPoint = values[j];
        const time = new Date(dataPoint[0] * 1000);
        const value = dataPoint[1];
        const point = {};
        point[`${name}-${metric}`] = +value;
        point['time'] = time;
        chartSeries.push(point);
      }
    }
    chartSeries.sort(this.appNativeChartsService.compareByAttributeName);
  }
}
