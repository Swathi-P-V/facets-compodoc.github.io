import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { Router } from '@angular/router';
import mixpanel from 'mixpanel-browser';
import { UiResourceGroupControllerService } from 'src/app/cc-api/services';
import { RESOURCE_GROUP_WRITE } from 'src/app/constants/permissions';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
import {
  USER_RESOURCE_GROUP_CREATE_FAIL,
  USER_RESOURCE_GROUP_CREATE_SUCCESS,
} from 'src/app/constants/analytics-track-events';

@Component({
  selector: 'app-create-resource-group',
  templateUrl: './create-resource-group.component.html',
  styleUrls: ['./create-resource-group.component.scss'],
})
export class CreateResourceGroupComponent implements OnInit {
  createResGrpForm: FormGroup;
  resourceGroupWritePermission = RESOURCE_GROUP_WRITE;

  constructor(
    private router: Router,
    private uiResourceGroupController: UiResourceGroupControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {}

  ngOnInit(): void {
    this.createResGrpForm = new FormGroup({
      name: new FormControl('', [
        Validators.required,
        Validators.minLength(3),
        Validators.maxLength(15),
        Validators.pattern(whiteSpaceInputPattern),
      ]),
    });
  }

  get resGroupName() {
    return this.createResGrpForm.get('name');
  }

  backToUserManagement = () => {
    this.router.navigate(['/settings/users/resource-groups']);
  };

  creating: boolean = false;
  createResGrp() {
    this.creating = true;
    const name = this.createResGrpForm.get('name')?.value;
    this.uiResourceGroupController
      .createUsingPOST({
        name: name,
      })
      .subscribe(
        (rsp: any) => {
          this.creating = false;
          mixpanel.track(USER_RESOURCE_GROUP_CREATE_SUCCESS);
          this.customToastService.success(
            `Successfully created resource group: ${name}`,
          );
          this.backToUserManagement();
        },
        (error: any) => {
          mixpanel.track(USER_RESOURCE_GROUP_CREATE_FAIL, { error });
          this.onAPIErrorService.onAPIError(
            error?.error?.message || `Could not create resource group: ${name}`,
            error,
          );
          this.creating = false;
        },
      );
  }
}
