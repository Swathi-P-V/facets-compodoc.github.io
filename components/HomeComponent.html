import { SmartTableTimeagoTooltipComponent } from 'src/app/components/smart-table-timeago-tooltip/smart-table-timeago-tooltip.component';
import { Component, OnInit, ViewChild, TemplateRef } from '@angular/core';
import { UiStackControllerService } from '../../cc-api/services/ui-stack-controller.service';
import { Stack } from '../../cc-api/models/stack';
import { Router } from '@angular/router';
import { SmartTableRouterlinkComponent } from '../../components/smart-table-routerlink/smart-table-routerlink.component';
import { NbDialogService } from '@nebular/theme';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { ContextUpdatesService } from 'src/app/services/context-updates.service';
import { UiAccountsControllerService } from 'src/app/cc-api/services';
import cloneDeep from 'lodash/cloneDeep';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import {
  BLUEPRINT_TEMPLATE_DELETE,
  BLUEPRINT_TEMPLATE_WRITE,
  STACK_DELETE,
  STACK_WRITE,
} from 'src/app/constants/permissions';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const documentLogo = './assets/document.svg';
const videoLogo = './assets/video.svg';

@Component({
  selector: 'app-home',
  templateUrl: './home.component.html',
  styleUrls: ['./home.component.scss'],
})
export class HomeComponent implements OnInit {
  @ViewChild('deleteStackUI', { static: true })
  private deleteStackUI: TemplateRef<any>;
  @ViewChild('demoVideo', { static: true })
  private demoVideo: TemplateRef<any>;
  selectedTab: 'Blueprints' | 'Templates' = 'Blueprints';
  stacks: Stack[] = [];
  loading: boolean = false;
  deletionInProgress: boolean = false;
  documentLogo: any = documentLogo;
  videoLogo: any = videoLogo;
  callInProgress: boolean = false;
  accounts: any = [];
  Object = Object;
  homeData = {
    Blueprints: [],
    Templates: [],
  };
  smartTableCustomActionButtons = [
    {
      id: 'edit',
      icon: 'edit-2-outline',
      toolTip: 'Edit Blueprint',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: STACK_WRITE,
      onClick: (data) => {
        this.editBluePrint(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete Blueprint',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: STACK_DELETE,
      onClick: (data) => {
        this.deleteBlueprintConfirmPopup(data);
      },
    },
  ];

  tableSettings = {
    columns: {
      name: {
        title: 'Name',
        filter: true,
        valuePrepareFunction: (value, row, cell) => {
          // DATA FROM HERE GOES TO renderComponent
          const displayVal = {
            url: [
              '/',
              'stack',
              value,
              'blueprint-designer',
              this.contextUpdatesService.getLastBranchForBlueprint(value) ||
                row?.branch,
              'resource-list',
            ],
            display: value,
          };
          return JSON.stringify(displayVal);
        },
        width: '10%',
        renderComponent: SmartTableRouterlinkComponent,
        type: 'custom',
      },
      vcsUrl: {
        title: 'Repository',
        filter: true,
        valuePrepareFunction: (cell) => {
          return "<a href='" + cell + "' target='_blank'>" + cell + '</a>';
        },
        type: 'html',
        width: '20%',
      },
      relativePath: {
        title: 'Path',
        filter: false,
        width: '10%',
      },
      versionControlAccountId: {
        title: 'Account',
        filter: false,
        valuePrepareFunction: (cell) => {
          let accName = this.accounts
            .filter((acc) => acc?.id === cell)
            .map((acc) => acc?.name);
          const displayVal = {
            url: ['/', 'settings', 'organization-accounts', cell],
            display: accName,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: SmartTableRouterlinkComponent,
        type: 'custom',
        width: '10%',
      },
      createdBy: {
        title: 'Modified By',
        width: '20%',
        filter: true,
        type: 'custom',
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            user: row.lastModifiedBy,
            label: row.lastModifiedDate,
          });
        },
        renderComponent: SmartTableTimeagoTooltipComponent,
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          //Add another field "uniqueId" formed by <smartActionButtonId>_<unique_column_value> to each row entry
          //Considering blueprint name as uniquie column here
          let formattedSmartTableCustomActionButtons = cloneDeep(
            this.smartTableCustomActionButtons,
          );

          formattedSmartTableCustomActionButtons.forEach((button) => {
            button['uniqueId'] = button.id + '_' + row.name;
            if (button?.id === 'edit') {
              button.toolTip = row?.template
                ? 'Edit Template'
                : 'Edit Blueprint';
              button.action = row?.template
                ? BLUEPRINT_TEMPLATE_WRITE
                : STACK_WRITE;
            } else {
              button.toolTip = row?.template
                ? 'Delete Template'
                : 'Delete Blueprint';
              button.action = row?.template
                ? BLUEPRINT_TEMPLATE_DELETE
                : STACK_DELETE;
            }
          });
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {
            smartTableCustomActionButtons:
              formattedSmartTableCustomActionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '12%',
        filter: false,
      },
    },
    pager: {
      display: false,
    },
    actions: false,
  };

  blueprintActionButtons = [
    {
      id: 'new-blueprint-button',
      title: 'New Blueprint',
      buttonType: 'primary',
      nbTooltip: 'Create New Blueprint',
      action: 'STACK_WRITE',
      hidden: false,
      click: () => {
        this.createStack();
      },
    },
  ];

  constructor(
    private uiStackControllerService: UiStackControllerService,
    private router: Router,
    private dialogService: NbDialogService,
    private contextUpdatesService: ContextUpdatesService,
    private accountControllerService: UiAccountsControllerService,
    private customToast: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {}

  ngOnInit(): void {
    this.getAccountDetails();
  }

  gotoPage(x): void {
    if (x.action === 'View') {
      const stackName = x.data.name;
      this.router.navigate(['/', 'stack', stackName]);
    }
  }

  openVideo() {
    this.dialogService.open(this.demoVideo, { context: '' });
  }

  createStack() {
    this.router.navigate(['/', 'createStack']);
  }

  editBluePrint(rowData) {
    const stackName = rowData.name;
    this.router.navigate(['/', 'stack', stackName, 'edit']);
  }

  deleteBlueprintConfirmPopup(rowData) {
    this.dialogService.open(this.deleteStackUI, { context: rowData });
  }

  getAccountDetails() {
    this.loading = true;
    this.accountControllerService
      .getAccountsByTypeUsingGET('VERSION_CONTROL')
      .subscribe(
        (response) => {
          this.accounts = response;
          this.getStacks();
        },
        (error) => {
          this.getStacks();
        },
      );
  }

  getStacks() {
    this.homeData = {
      Templates: [],
      Blueprints: [],
    };
    this.uiStackControllerService.getStacksUsingGET1().subscribe(
      (s) => {
        this.loading = false;
        let stacks: Stack[] = s;
        if (this.accounts?.length > 0) {
          stacks.forEach((stack) => {
            this.accounts?.forEach((account) => {
              if (stack?.versionControlAccountId === account?.id) {
                stack.user = account?.userName ? account?.userName : '--';
              }
            });
          });
        }
        stacks.sort((s1, s2) => s1?.name.localeCompare(s2?.name));
        stacks.forEach((stack) => {
          if (stack?.template) {
            this.homeData.Templates.push(stack);
          } else {
            this.homeData.Blueprints.push(stack);
          }
        });
        return (this.stacks = [...stacks]);
      },
      (error) => {
        this.loading = false;
      },
    );
  }

  deleteBlueprint(data, ref) {
    this.deletionInProgress = true;
    this.uiStackControllerService.deleteStackUsingDELETE(data.name).subscribe(
      (t) => {
        this.customToast.success(
          `${data?.template ? 'Template' : 'Blueprint'} ${
            data?.name
          } has been deleted successfully.`,
        );
        this.deletionInProgress = false;
        ref.close();
        this.contextUpdatesService.requestContextSwitcherRefresh();
        //Get the stacks to refresh the list
        this.loading = true;
        this.uiStackControllerService.getStacksUsingGET1().subscribe((s) => {
          this.stacks = s;
          this.getStacks();
        });
      },
      (error) => {
        this.deletionInProgress = false;
        ref.close();
        this.onAPIErrorService.onAPIError(
          error?.error?.message ||
            `${data?.template ? 'Template' : 'Blueprint'} ${
              data?.name
            } could not be deleted`,
          error,
        );
      },
    );
  }

  handleActions(event) {
    this.blueprintActionButtons?.find((button) => button.id === event)?.click();
  }

  changedTab(tab) {
    this.selectedTab = tab?.tabTitle;
  }
}
