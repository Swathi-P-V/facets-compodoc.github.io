import { Component, TemplateRef, ViewChild, OnDestroy } from '@angular/core';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { UiSettingsControllerService } from 'src/app/cc-api/services';
import { UiCommonClusterControllerService } from 'src/app/cc-api/services';
import { AwsCluster } from 'src/app/cc-api/models';
import { NbDialogService } from '@nebular/theme';
import { FormControl, FormGroup } from '@angular/forms';
import { SaveChangesFooterComponent } from 'src/app/components/save-changes-footer/save-changes-footer.component';
import { CanDeactivateComponent } from 'src/app/can-deactivate.guard';
import { ENVIRONMENT_CONFIGURE } from 'src/app/constants/permissions';
import { cloneDeep } from 'lodash';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { VPN_ENABLE_DISABLE } from 'src/app/constants/cluster';

const errLogo = './assets/error.svg';

@Component({
  selector: 'app-cluster-settings',
  templateUrl: './cluster-settings.component.html',
  styleUrls: ['./cluster-settings.component.scss'],
})
export class ClusterSettingsComponent
  extends FacetsPage
  implements CanDeactivateComponent
{
  clusterSettingsForm: FormGroup;
  cluster: AwsCluster;
  clusterId: string;
  stackName: any;
  fetchingSetting: Boolean = false;
  fetchingCluster: Boolean = false;
  entityType:
    | 'CLUSTER'
    | 'BLUE_PRINT'
    | 'TEMPLATE_INPUT'
    | 'CONTROL_PLANE'
    | 'IAC'
    | 'ARTIFACT_CI'
    | 'USER_GROUP'
    | 'ACCOUNT'
    | 'ARTIFACTORY';
  clusterDefaultSettings: any;
  clusterSettings: {};
  clusterSettingOriginal: {} = {};
  clusterSettingChanged: {} = {};
  selectOptions: {} = {};
  reviewComments: string = '';
  changeSettings: boolean = false;
  settingsError: boolean = false;
  clusterError: boolean = false;
  errIcon: any = errLogo;
  status: 'info' | 'error' = 'info';
  primaryMsg: string = 'No settings available';
  secondaryMsg: string = '';
  showBtn: boolean = true;
  buttonText: string = 'Reload';
  tooltip: string = 'Load Settings';
  showTooltip: boolean = true;
  environmentConfigPermission = ENVIRONMENT_CONFIGURE;
  VPN_ENABLE_DISABLE = VPN_ENABLE_DISABLE;

  objectKeys = Object.keys;

  @ViewChild(SaveChangesFooterComponent)
  private footerComponent: SaveChangesFooterComponent;

  @ViewChild('vpnInstructionsModal')
  private vpnInstructionsModal: TemplateRef<any>;

  releaseNowButton = {
    id: 'release-now',
    label: 'Release Now',
    icon: 'arrow-ios-forward-outline',
    action: ENVIRONMENT_CONFIGURE,
    click: () => {
      this.navigateToReleasesPage();
    },
  };
  toastSubscription: any;

  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private uiSettingService: UiSettingsControllerService,
    private uiCommonClusterControllerService: UiCommonClusterControllerService,
    private toastService: CustomToastService,
    private dialogService: NbDialogService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {
    this.toastSubscription = this.toastService.buttonClick$.subscribe(
      (data) => {
        if (this.releaseNowButton?.id === data?.link?.id) {
          this.releaseNowButton?.click();
        }
      },
    );
  }

  OnPageLoad(p: Params): void {
    this.fetchingCluster = true;
    this.stackName = p.stackName;
    this.clusterId = p.clusterId;
    this.uiCommonClusterControllerService
      .getClusterCommonUsingGET(p.clusterId)
      .subscribe(
        (r) => {
          this.cluster = r;
          this.entityType = this.cluster.entityType;
          this.fetchingCluster = false;
          this.getClusterSettings();
        },
        (error) => {
          this.fetchingCluster = false;
          this.clusterError = true;
          this.primaryMsg =
            error?.error?.message ||
            ' There is some error fetching environment information';
          this.secondaryMsg = '';
          this.showBtn = false;
          this.status = 'error';
        },
      );
  }

  handleBack = () => {
    const url =
      '/' + this.stackName + '/cluster/' + this.clusterId + '/overview';
    this.router.navigate([url]);
  };

  getClusterSettings() {
    this.clusterDefaultSettings = [];
    this.selectOptions = {};
    this.clusterSettingChanged = {};
    this.clusterSettingOriginal = {};
    this.clusterSettings = {};
    this.fetchingSetting = true;
    let settingParams = {
      entityType: this.entityType,
      entityId: this.clusterId,
    };
    this.settingsError = false;
    this.uiSettingService.getSettingValueUsingGET(settingParams).subscribe(
      (r) => {
        this.clusterSettings = r;
        this.uiSettingService
          .getAllSettingsForEntityUsingGET(this.entityType)
          .subscribe(
            (settingsResponse) => {
              const updatedResponse = cloneDeep(settingsResponse);
              this.clusterDefaultSettings = cloneDeep(updatedResponse);
              this.clusterDefaultSettings.forEach((setting) => {
                this.prepareOptions(setting);
                if (
                  this.clusterSettings &&
                  Object.keys(this.clusterSettings).indexOf(setting.id) > -1
                ) {
                  this.clusterSettingOriginal[setting.id] =
                    this.clusterSettings[setting.id];
                } else {
                  this.clusterSettings[setting.id] = setting.defaultValue;
                  this.clusterSettingOriginal[setting.id] =
                    setting.defaultValue;
                }
              });
              this.setClusterSettingForm();
              this.fetchingSetting = false;
            },
            (error) => {
              this.fetchingSetting = false;
              this.settingsError = true;
            },
          );
      },
      (error) => {
        this.fetchingSetting = false;
        this.settingsError = true;
      },
    );
  }

  setClusterSettingForm = () => {
    const group: any = {};
    Object.keys(this.clusterSettings).forEach((setting) => {
      group[setting] = new FormControl(this.clusterSettings[setting]);
    });
    this.clusterSettingsForm = new FormGroup(group);
  };

  prepareOptions(setting) {
    let selectOption = [];
    if (setting.settingType === 'ENUM') {
      setting.allowedValues.forEach((value) => {
        selectOption.push({
          label: value,
          value: value,
        });
      });
      this.selectOptions[setting.id] = selectOption;
    } else if (setting.settingType === 'BOOLEAN') {
      selectOption.push({
        label: true,
        value: true,
      });
      selectOption.push({
        label: false,
        value: false,
      });
      this.selectOptions[setting.id] = selectOption;
    }
  }

  determineNewValue(value: any, type: string) {
    switch (type) {
      case 'num':
        return parseInt(value);
      case 'float':
        return parseFloat(value);
      default:
        return value;
    }
  }

  onChange(event, setting, type) {
    if (event?.target?.value) {
      if (event.target.value != this.clusterSettingOriginal[setting.id]) {
        this.clusterSettingChanged[setting.id] = {
          value: this.determineNewValue(event.target.value, type),
          name: setting.name,
        };
      } else {
        delete this.clusterSettingChanged[setting.id];
      }
    }
  }

  onChangeSelect(value, setting) {
    if (value != this.clusterSettingOriginal[setting.id]) {
      this.clusterSettingChanged[setting.id] = {
        value: value,
        name: setting.name,
      };
    } else {
      delete this.clusterSettingChanged[setting.id];
    }
  }
  onChangeToggle(value, setting) {}

  submitClusterSettings = (ref) => {
    this.changeSettings = true;
    let insertSettingParams: {
      settingValueRequest: {};
      entityType:
        | 'CLUSTER'
        | 'BLUE_PRINT'
        | 'TEMPLATE_INPUT'
        | 'CONTROL_PLANE'
        | 'IAC'
        | 'ARTIFACT_CI'
        | 'USER_GROUP'
        | 'ACCOUNT'
        | 'ARTIFACTORY';
      entityId: string;
    } = {
      entityId: this.clusterId,
      entityType: this.entityType,
      settingValueRequest: {},
    };
    const settingValues = {};
    for (const setting in this.clusterSettingChanged) {
      settingValues[setting] = this.clusterSettingChanged[setting].value;
    }
    insertSettingParams['settingValueRequest'] = settingValues;
    this.uiSettingService
      .addSettingValueUsingPUT(insertSettingParams)
      .subscribe(
        (response) => {
          const isVPNEnabled = response[VPN_ENABLE_DISABLE];
          let toasterTitle = isVPNEnabled
            ? 'VPN enabled, Release pending!'
            : 'Settings saved!';
          let toasterDescription = isVPNEnabled
            ? 'Perform a release for VPN settings to take effect.'
            : 'Release updated settings to enact changes.';

          this.toastService.show({
            title: toasterTitle,
            content: toasterDescription,
            type: 'success',
            link: this.releaseNowButton,
          });
          ref.close();
          this.getClusterSettings();
          this.changeSettings = false;
        },
        (error) => {
          ref.close();
          this.changeSettings = false;
          if (
            (error.status === 500 || error.status === 403) &&
            error?.error.message === 'Access is denied'
          ) {
            this.toastService.error(
              'Please contact your administrator for more details',
            );
          } else {
            this.toastService.error(
              'Settings change failed. ' + error?.error?.message,
            );
          }
        },
      );
  };

  onDiscard = (ref: any) => {
    this.clusterSettingChanged = {};
    this.clusterSettings = this.clusterSettingOriginal;

    this.setClusterSettingForm();
    ref.close('discard');
  };

  async canDeactivate(): Promise<boolean> {
    if (this.objectKeys(this.clusterSettingChanged)?.length == 0) {
      return true;
    }

    return this.footerComponent.onDeactivate();
  }

  showVPNInstructions() {
    this.dialogService.open(this.vpnInstructionsModal, {
      context: {},
    });
  }

  navigateToReleasesPage() {
    this.router.navigate([
      `/${this.stackName}/cluster/${this.clusterId}/releases`,
    ]);
  }

  navigateToEnvOverview(event, ref) {
    ref?.close();
    event?.preventDefault();
    this.router.navigate([
      `/${this.stackName}/cluster/${this.clusterId}/overview`,
    ]);
  }

  OnDestroy() {
    this.toastSubscription?.unsubscribe();
  }
}
