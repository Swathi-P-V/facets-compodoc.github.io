import {
  Component,
  Input,
  OnInit,
  TemplateRef,
  ViewChild,
} from '@angular/core';
import { ViewCell } from 'ng2-smart-table';
import { getConfigKeyToShow } from 'src/app/utils/app-config-utils';
import { FormControl } from '@angular/forms';
import { MonacoEditorConstructionOptions } from '@materia-ui/ngx-monaco-editor';
import { NbDialogService } from '@nebular/theme';
import { SmartTableRouterlinkComponent } from 'src/app/components/smart-table-routerlink/smart-table-routerlink.component';
import { NbPopoverDirective } from '@nebular/theme';
import { Resource } from 'src/app/cc-api/models';
import { UiDropdownsControllerService } from 'src/app/cc-api/services';
import * as iconMapping from 'src/app/constants/facetsIconMap.json';
import { filterUnequalValues } from 'src/app/utils/array-utils';
const emptyStateLogo = './assets/empty-override.svg';
import { flattenObject } from 'src/app/utils/object-utils';
import { LocalDataSource } from 'ng2-smart-table';
import cloneDeep from 'lodash/cloneDeep';

@Component({
  selector: 'app-override-details',
  templateUrl: './override-details.component.html',
  styleUrls: ['./override-details.component.scss'],
})
export class OverrideDetailsComponent implements ViewCell, OnInit {
  @ViewChild('envConfigEditor', { static: true })
  private envConfigEditor: TemplateRef<any>;
  @ViewChild(NbPopoverDirective) overridePopover: NbPopoverDirective;
  overridden: boolean;
  @Input() value: string | number;
  @Input() rowData: any;
  @Input() overridesData: any[] = [];
  @Input() resource: Resource;
  showOverridePopup: boolean = false;
  overrideDataSource: LocalDataSource = new LocalDataSource();
  emptyStateLogo = emptyStateLogo;
  secondaryMsg =
    'There are no overridden values which are different from blueprint values';
  envConfigCode: FormControl = new FormControl();
  tableSettings: any = {
    columns: {
      key: {
        title: 'Key',
        type: 'html',
        filter: true,
        valuePrepareFunction: (cell, row) => {
          return `<div class='width-limit'>${cell}</div>`;
        },
        width: '20%',
      },
      blueprintVal: {
        title: 'Blueprint Value',
        filter: false,
        type: 'custom',
        valuePrepareFunction: (value, row, cell) => {
          return getConfigKeyToShow(value);
        },
        hide: false,
        renderComponent: SmartTableRouterlinkComponent,
        onComponentInitFunction: (instance) => {
          instance.byLinkClick.subscribe((data) => {
            data.context = 'BLUEPRINT';
            this.openConfigPopup(data);
          });
        },
        width: '20%',
      },
      overrideVal: {
        title: 'Overridden Value',
        type: 'custom',
        filter: false,
        valuePrepareFunction: (value, row, cell) => {
          return getConfigKeyToShow(value);
        },
        renderComponent: SmartTableRouterlinkComponent,
        onComponentInitFunction: (instance) => {
          instance.byLinkClick.subscribe((data) => {
            data.context = 'CLUSTER';
            this.openConfigPopup(data);
          });
        },
        width: '20%',
      },
    },
    noDataMessage:
      'There are no overridden values which are different from blueprint',
    pager: {
      display: false,
      perPage: 10,
    },
    mode: 'external',
    actions: false,
  };
  editorOptionsReadOnly: MonacoEditorConstructionOptions = {
    theme: 'vs-light',
    language: 'json',
    readOnly: true,
    minimap: {
      enabled: false,
    },
  };
  resourceTypeIcon = '';
  resourceContentData: any;
  loadingBlueprintData: boolean = false;

  constructor(
    private dialogService: NbDialogService,
    private uiDropdownsControllerService: UiDropdownsControllerService,
  ) {}

  ngOnInit(): void {
    this.overridden = this.value === 'OVERRIDDEN';
    const overridesData = this.rowData?.configList?.length
      ? this.rowData?.configList
      : [];
    this.overridesData = filterUnequalValues(
      overridesData,
      'blueprintVal',
      'overrideVal',
    );
    this.showOverridePopup = this.rowData?.showOverridePopup;
  }

  openConfigPopup(data) {
    this.envConfigCode.setValue(
      JSON.stringify(
        data?.context === 'BLUEPRINT' ? data?.blueprintVal : data?.overrideVal,
        null,
        '\t',
      ),
    );
    this.dialogService
      .open(this.envConfigEditor, { context: data })
      .onClose.subscribe(() => {
        setTimeout(() => {
          this.overridePopover.show();
        }, 0);
      });
  }

  closeOverrides() {
    this.overridePopover.hide();
  }

  openPopup() {
    this.overridePopover.show();
  }

  setBlueprintConfig() {
    let flattenedBlueprintConfig = {};
    try {
      flattenedBlueprintConfig = this.resourceContentData
        ? flattenObject(JSON.parse(this.resourceContentData))
        : {};
      this.overridesData.forEach((override) => {
        override.blueprintVal = flattenedBlueprintConfig[override?.key];
      });
    } catch (err) {
      flattenedBlueprintConfig = {};
    }
    this.overridesData = filterUnequalValues(
      this.overridesData,
      'blueprintVal',
      'overrideVal',
    );
    this.resourceTypeIcon = iconMapping[this.rowData?.resourceType]
      ? this.rowData?.resourceType
      : 'default';
    this.overrideDataSource.load(this.overridesData);
  }

  popoverStateChange(event) {
    if (event?.isShown) {
      if (!this.resourceContentData) {
        this.loadingBlueprintData = true;
        this.uiDropdownsControllerService
          .getResourceByStackUsingGET({
            stackName: this.rowData?.stackName,
            resourceName: this.rowData?.resourceName,
            resourceType: this.rowData?.resourceType,
          })
          .subscribe(
            (resource) => {
              this.resourceContentData = resource?.content;
              this.tableSettings.columns.blueprintVal.hide = false;
              this.setBlueprintConfig();
              this.loadingBlueprintData = false;
            },
            (error) => {
              this.loadingBlueprintData = false;
              this.tableSettings.columns.blueprintVal.hide = true;
            },
          );
      } else {
        this.setBlueprintConfig();
      }
    }
  }
}
