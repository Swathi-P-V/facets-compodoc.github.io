import { Component, ViewChild, TemplateRef } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { NbPopoverDirective, NbDialogService } from '@nebular/theme';
import mixpanel from 'mixpanel-browser';
import { Role, Account } from 'src/app/cc-api/models';
import {
  UiCustomRoleControllerService,
  UiStackControllerService,
  UiUserGroupControllerService,
  UiResourceGroupControllerService,
  UiAccountsControllerService,
} from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import cloneDeep from 'lodash/cloneDeep';
import { UserGroup } from 'src/app/cc-api/models/user-group';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { USER_GROUP_WRITE } from 'src/app/constants/permissions';
import { ResourceGroup } from './../../../cc-api/models/resource-group';
import { SmartTableRouterlinkComponent } from './../../../components/smart-table-routerlink/smart-table-routerlink.component';
import { setAccountEnvs } from './create-user-group-utils';
import { organizationAccountIconMap } from 'src/app/constants/org-account-icon-map';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
import {
  USER_GROUP_CREATE_FAIL,
  USER_GROUP_CREATE_SUCCESS,
  USER_GROUP_EDIT_FAIL,
  USER_GROUP_EDIT_SUCCESS,
} from 'src/app/constants/analytics-track-events';

@Component({
  selector: 'app-create-user-group',
  templateUrl: './create-user-group.component.html',
  styleUrls: ['./create-user-group.component.scss'],
})
export class CreateUserGroupComponent extends FacetsPage {
  @ViewChild('organizationAccountPopover', { static: true })
  private organizationAccountPopover: TemplateRef<any>;
  pageHeading = {
    create: 'Create User Group',
    edit: 'Edit User Group',
  };
  mode: string = 'create';

  allRoles: Array<Role> = []; //AllRoles
  baseRoles: Array<Role> = []; // All base roles for user form [passed in from parent through state]

  fetchingResourceGroups: boolean = false;
  allResourceGroups: Array<ResourceGroup> = [];
  fetchingError: boolean = false;
  showResourceGroups: boolean = false;
  resourceGroupSelectOptions: {
    label: string;
    value: string;
    checked: boolean;
  }[] = [];

  allEnvs = []; //All environments available in control plane
  envStackClusterMap: any = {}; // All [Stack -> Clusters] mapping,

  envSpecificRoleEnvDropdown: any = {};
  envAccessibleEnvDropdown: any = {};
  envAccessibleEnvDropdownOriginal: any = {};
  envAccessibleEnvSelect: any = {};
  envSpecificRoleEnvFilteredDropdown: any = {};

  accessPriviligesList;
  baseRoleMutationObserver: MutationObserver;
  cdkOverlayContainerNode: any;
  selectedRole: any;
  userGroupWrite = USER_GROUP_WRITE;

  // Hint string for additionalRole desc
  additionalRoleHints = {
    K8S_READER: 'Can download K8s credentials and do read operations on K8s',
    K8S_DEBUGGER: 'Can download K8s credentials and do write operations on K8s',
    CLI_ARTIFACT_PUSH: 'Can push artifacts using CLI',
  };

  userGroupModel: FormGroup;
  clusterRoleBindings: {} = {};
  envIdNameMap: {} = {}; //All [environmentId -> environmentName] map
  envIdStackMap: {} = {};
  envNameIdMap: {} = {}; //All [environmentName -> environmentId] map
  accessibleEnvs: { id: string; name: string }[] = []; //Used to render badge pills view for selected accessible envs
  roleIdNameMap: {} = {}; //All [roleId -> roleName] map
  groupId: string; //Stores Id of the group that is to be edited.
  existingGroup: UserGroup; //Stores the user group that is to be edited.
  loadingUserGroupForm: boolean = true;
  loading: boolean;

  tableSettings: any = {
    columns: {
      name: {
        title: 'Environment Name',
        type: 'custom',
        filter: true,
        valuePrepareFunction: (value, row, cell) => {
          // DATA FROM HERE GOES TO renderComponent
          const displayVal = {
            url: ['/', row?.stackName, 'cluster', row?.id, 'overview'],
            display: value,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: SmartTableRouterlinkComponent,
      },
      role: {
        title: 'Role',
        type: 'text',
        filter: true,
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
      perPage: 5,
    },
    mode: 'external',
    actions: {
      position: 'right',
      add: false,
      edit: false,
      delete: false,
      custom: [
        {
          name: 'Delete',
          title:
            '<i class="eva-close-circle-outline eva ml-3 text-danger"></i>',
          type: 'html',
        },
      ],
    },
  };
  clusterRoleBindingTableData: {
    id: string;
    stackName: string;
    name: string;
    role: string;
  }[] = []; //Stores data source to render env specific permission table
  stackOptions: { label: string; value: string }[] = [];
  accessibleBlueprints: string[] = [];
  accountsList: Account[];
  fetchingAccounts: boolean = false;
  accountSelectOptions: {
    label: string;
    value: string;
    icon: string;
    checked?: boolean;
  }[] = [];
  accountIconMap = organizationAccountIconMap;

  @ViewChild(NbPopoverDirective) popover: NbPopoverDirective;

  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private uiStackControllerService: UiStackControllerService,
    private customRoleService: UiCustomRoleControllerService,
    private userGroupControllerService: UiUserGroupControllerService,
    private dialogService: NbDialogService,
    private uiResourceGroupController: UiResourceGroupControllerService,
    private accountsControllerService: UiAccountsControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.loadingUserGroupForm = true;
    this.initUserGroupModel();
    this.asyncPageLoadFunc(p);
  }

  async asyncPageLoadFunc(params) {
    this.groupId = params?.groupId;
    await this.getAllResourceGroups();
    await this.getBaseAndAdditionalRoles();
    await this.getAllEnvironments();
    await this.getAllAccounts();
    if (this.groupId) {
      this.mode = 'edit';
      this.getGroupDetailsForEdit().then(
        async (data) => {
          await this.removeDeletedEnvs();
          this.filterEnvOptionsForStacks();
          setTimeout(() => {
            this.patchUserGroupModel();
            if (this.existingGroup.clusterIds.length > 0) {
              this.getAccessibleEnvs();
            }
            this.accessibleBlueprints = this.existingGroup.stackNames;
          }, 500);
          this.setSelectMenuUserGroups();
          this.setSelectedAccounts();
          this.clusterRoleBindings = this.existingGroup.clusterRoleBindings;

  //Adds environment specific permissions to clusterRoleBindings object and smart table data source
  addEnvironmentSpecificPermissions() {
    //Add clusterId:role keyvalue pair to clusterRoleBindings Object
    this.clusterRoleBindings[this.envNameIdMap[this.envName.value]?.envId] =
      this.envSpecificRole.value;

    //Add the keypair to table source data
    this.prepareEnvironmentSpecificTableData(
      this.envName.value,
      this.envSpecificRole.value,
    );

    //Remove this environment from envSpecificRoleEnvDropdown.
    //To remove environment set isAdd to false
    this.filterEnvSpecificRoleEnvDropdown(
      this.envNameIdMap[this.envName.value].envId,
      false,
    );

    //Reset the form values
    this.userGroupModel
      .get('environmentSpecificRoleModel')
      .patchValue({ envName: '', envSpecificRole: '' });
  }

  //This function adds or deletes entry to/from clusterRoleBindingTableData.
  //To add pass form values and to delete pass form value and set isDelete to true.
  prepareEnvironmentSpecificTableData(name, role, isDelete?: boolean) {
    //Get a copy of table data
    let tempData = cloneDeep(this.clusterRoleBindingTableData);

    //Add or delete data from copiedTable data
    if (!isDelete) {
      tempData.push({
        id: this.envNameIdMap[name]?.envId,
        stackName: this.envNameIdMap[name]?.stackName,
        name: name,
        role: this.roleIdNameMap[role],
      });
    } else {
      tempData = tempData.filter((ele) => ele.name !== name);
    }

    //Assign the updated data to actual table source
    this.clusterRoleBindingTableData = tempData;
  }

  //Delete the entry from table's source data and from clusterBindings object
  deleteEnvSpecificPermission(event) {
    //delete clusterId:role keyvalue pair from clusterRoleBindings object
    delete this.clusterRoleBindings[event.data.id];

    //Remove keypair from table source
    this.prepareEnvironmentSpecificTableData(
      event.data.name,
      event.data.role,
      true,
    );

    //Add this environment to envSpecificRoleEnvDropdown.
    //To add environment set isAdd(second param) to true

    // Extract env ids of the current dropdown values -- Implementation can be improved
    let envNames = [];
    Object.values(this.envSpecificRoleEnvDropdown).map((envArrays: []) => {
      envArrays.map((env) => {
        envNames.push(env['id']);
      });
    });
    //Send existing drodownvalues and the one deleted
    this.filterEnvSpecificRoleEnvDropdown([...envNames, event.data.id], true);
  }

  //When accessible environments selection changes this function is called
  //This will update accessible environments array that renders badge pills
  getAccessibleEnvs() {
    this.accessibleEnvs = this.userGroupModel
      .get('clusterIds')
      .value.map((envId) => {
        return { id: envId, name: this.envIdNameMap[envId] };
      });
    //As user can set env specific permission for only accessible environments,
    //Remove all the environments other than accessible environments from envSpecificRoleEnvDropdown
    if (this.accessibleEnvs.length > 0) {
      if (!this.clusterRoleBindings) {
        this.filterEnvSpecificRoleEnvDropdown(
          this.userGroupModel.get('clusterIds').value,
          true,
        );
      } else {
        let deletedRoleBindedClusters = Object.keys(
          this.clusterRoleBindings,
        ).filter((key) => this.clusterIds.value.indexOf(key) === -1);

        deletedRoleBindedClusters.forEach((cluster) => {
          delete this.clusterRoleBindings[cluster];

          this.prepareEnvironmentSpecificTableData(
            this.envIdNameMap[cluster],
            this.clusterRoleBindings[cluster],
            true,
          );
        });

        let bindings = Object.keys(this.clusterRoleBindings);
        let ids = this.clusterIds.value.filter(
          (ids) => bindings.indexOf(ids) === -1,
        );
        this.filterEnvSpecificRoleEnvDropdown(ids, true);
      }
    } else {
      //If accessibleEnvs length is 0 then user is able to access all envs.
      this.filterEnvSpecificRoleEnvDropdown(
        Object.keys(this.envIdNameMap),
        true,
      );
    }
  }

  getAccessibleStack() {
    this.accessibleBlueprints = this.userGroupModel.get('stackNames').value;
    let accessibleEnvs = this.userGroupModel?.get('clusterIds')?.value;
    let filteredEnvs: any = [];
    if (this.accessibleBlueprints?.length) {
      const filteredEnvOptions = {};
      Object.keys(this.envAccessibleEnvDropdownOriginal).forEach((stack) => {
        if (
          this.accessibleBlueprints.includes(stack) &&
          this.envAccessibleEnvDropdownOriginal[stack]?.length
        ) {
          filteredEnvOptions[stack] =
            this.envAccessibleEnvDropdownOriginal[stack];
        }
      });
      this.envAccessibleEnvDropdown = cloneDeep(filteredEnvOptions);
      filteredEnvs = accessibleEnvs.filter((env) =>
        this.accessibleBlueprints.includes(this.envIdStackMap[env]),
      );
    } else {
      this.envAccessibleEnvDropdown = {
        ...this.envAccessibleEnvDropdownOriginal,
      };
      filteredEnvs = [];
    }
    setTimeout(() => {
      this.userGroupModel
        .get('clusterIds')
        .setValue(this.accessibleBlueprints?.length ? filteredEnvs : []);
      this.getAccessibleEnvs();
    }, 0);
  }

  //Deselects the environment from dropdown, removes badge
  onTagRemove(envId): void {
    this.accessibleEnvs = this.accessibleEnvs.filter((ele) => ele.id !== envId);

    let clusters = this.clusterIds.value.filter((id) => id !== envId);
    this.userGroupModel.patchValue({
      clusterIds: clusters,
    });

    //Once you remove the tag, if there are accessible envs then remove any env specific permission related to this env
    if (this.accessibleEnvs.length > 0) {
      if (this.clusterRoleBindings.hasOwnProperty(envId)) {
        //Delete from cluster bindings
        delete this.clusterRoleBindings[envId];

        //Delete from table
        this.prepareEnvironmentSpecificTableData(
          this.envIdNameMap[envId],
          this.clusterRoleBindings[envId],
          true,
        );
      }
      //Delete the environment from envSpecificRoleEnvDropdown
      this.filterEnvSpecificRoleEnvDropdown(envId, false);
    } else {
      //If accessibleEnvs length is 0 then user is able to access all envs.
      this.filterEnvSpecificRoleEnvDropdown(
        Object.keys(this.envIdNameMap),
        true,
      );
    }
  }

  filterEnvSpecificRoleEnvDropdown(clusterId, isAdd: boolean) {
    if (isAdd) {
      this.envSpecificRoleEnvDropdown = {};
      this.allEnvs
        .filter((cluster) => {
          if (typeof clusterId !== 'string') {
            return clusterId.indexOf(cluster.id) !== -1;
          } else {
            return cluster.id === clusterId;
          }
        })
        .map((cluster) => {
          let stackName = cluster.stackName;
          if (this.envSpecificRoleEnvDropdown.hasOwnProperty(stackName)) {
            this.envSpecificRoleEnvDropdown[stackName].push(cluster);
          } else {
            this.envSpecificRoleEnvDropdown[stackName] = [cluster];
          }
        });
    } else {
      //Deletes environment from dropdown
      Object.keys(this.envSpecificRoleEnvDropdown).forEach((key) => {
        this.envSpecificRoleEnvDropdown[key] = this.envSpecificRoleEnvDropdown[
          key
        ].filter((cluster) => {
          if (typeof clusterId !== 'string') {
            return clusterId.indexOf(cluster.id) === -1;
          } else {
            return cluster.id !== clusterId;
          }
        });
      });
    }
    this.envSpecificRoleEnvFilteredDropdown = this.envSpecificRoleEnvDropdown;
  }

  submitForm() {
    this.loading = true;

    let userGroup: any = new (class implements UserGroup {
      baseRole: string;
      clusterIds: Array<string>;
      clusterRoleBindings: { [key: string]: string };
      groupName: string;
      id: string;
      associatedResourceGroups?: Array<string>;
    })();

    if (this.mode === 'edit') {
      userGroup.id = this.userGroupModel.get('id').value;
    }

    userGroup.groupName = this.groupName.value;
    userGroup.baseRole = this.baseRole.value;
    userGroup.accounts = this.accounts?.value;
    userGroup.clusterIds = this.clusterIds.value;
    userGroup.stackNames = this.stackNames.value;
    userGroup.clusterRoleBindings = this.clusterRoleBindings;
    userGroup.associatedResourceGroups = this.associatedResourceGroups.value;

    if (this.mode === 'create') {
      this.createUserGroup(userGroup);
    } else if (this.mode === 'edit') {
      this.editUserGroup(userGroup);
    }
  }

  getFilteredOptions(values: string) {
    let filteredOps = {};

    Object.values(this.envSpecificRoleEnvDropdown).forEach((value: []) => {
      value
        .filter((cluster: any) => cluster['name'].includes(values))
        .map((cluster: any) => {
          let stackName = cluster.stackName;
          filteredOps.hasOwnProperty(stackName)
            ? filteredOps[stackName].push(cluster)
            : (filteredOps[stackName] = [cluster]);
        });

      //TODO - If stack has no environments, remove the stack name
    });

    this.envSpecificRoleEnvFilteredDropdown = filteredOps;
  }

  //When any environment is deleted, the environment specific permission or accessible environments list in user group still contains those environments.

  removeDeletedEnvs() {
    return new Promise((resolve, reject) => {
      //Existing Environments list
      let envsIds = Object.keys(this.envIdNameMap);

      //Remove from clusterRolebindings
      let deletedRoleBindedClusters = Object.keys(
        this.existingGroup.clusterRoleBindings,
      ).filter((key) => envsIds.indexOf(key) === -1);

      deletedRoleBindedClusters.forEach((cluster) => {
        delete this.existingGroup.clusterRoleBindings[cluster];
      });

      //Remove from accessible clusters
      let existingAccessibleClusters = this.existingGroup.clusterIds.filter(
        (cluster) => envsIds.indexOf(cluster) !== -1,
      );

      this.existingGroup.clusterIds = existingAccessibleClusters;
      resolve(true);
    });
  }
