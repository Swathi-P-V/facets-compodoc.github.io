import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import {
  UiBlueprintDesignerControllerService,
  UiCoderControllerService,
  UiDropdownsControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ResourceDetailsService } from './resource-details.service';
import { UiAccountsControllerService } from 'src/app/cc-api/services';
import * as iconMapping from 'src/app/constants/facetsIconMap.json';
import { getResourceHeaderIcons } from 'src/app/utils/resource-details/header-utils';
import {
  disableResourceButton,
  enableResourceButton,
  workspaceButton,
} from 'src/app/constants/resource-details';
import { STACK_CONFIGURE, STACK_WRITE } from 'src/app/constants/permissions';
import { ContextUpdatesService } from 'src/app/services/context-updates.service';
import { BreadCrumbConfig } from 'src/app/models/breadcrumb.model';
import { CoderAccount, Stack } from 'src/app/cc-api/models';
import cloneDeep from 'lodash/cloneDeep';
import { map } from 'rxjs';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

const errorStateIcon = './assets/error.svg';

@Component({
  selector: 'app-designer-resource-details',
  templateUrl: './resource-details.component.html',
  styleUrls: ['./resource-details.component.scss'],
})
export class ResourceDetailsComponent extends FacetsPage {
  readonly STACK_WRITE = STACK_WRITE;
  @ViewChild('coderDialog') coderDialog: TemplateRef<any>;
  resource: any;
  blueprint: string;
  branchName: string;
  errorStateIcon = errorStateIcon;
  resourceJson: any = {};
  defaultHeaderDescription: string;
  resourceType: string;
  resourceName: string;
  resourceStatusProps: string;
  previousComponent: string;
  loadingStackDetails: boolean = false;
  stack: Stack;
  coderSupportedResourceTypes = ['service'];
  previousComponentMode: 'GRAPH' | 'TABLE' = 'TABLE';

  editResourceButton = {
    id: 'edit-resource-name',
    title: 'Edit Name',
    buttonType: 'primary',
    icon: 'edit-2-outline',
    nbTooltip: 'Edit Resource Name',
    disabled: false,
    loading: false,
    action: STACK_WRITE,
    status: 'primary',
  };
  deleteResourceButton = {
    id: 'delete-resource',
    title: 'Delete',
    icon: 'trash-2-outline',
    buttonType: 'tertiary',
    nbTooltip: 'Delete resource',
    disabled: false,
    loading: false,
    action: STACK_WRITE,
    status: 'danger',
  };
  actionButtons = [];

  headingIcons = [];
  primaryIcon = {
    id: 'resource-type',
    icon: '',
    pack: 'fci',
    status: 'primary',
    iconTooltip: '',
    iconSize: '36px',
  };

  @ViewChild('enableDisableConfirmation')
  enableDisableConfirmation: TemplateRef<any>;
  showCoderButton: boolean = false;

  private coderAccounts: CoderAccount[] = [];
  private coderButtonTooltip: string;

  constructor(
    private uiDropdownsControllerService: UiDropdownsControllerService,
    private resourceDetailsService: ResourceDetailsService,
    private uiBlueprintDesignerControllerService: UiBlueprintDesignerControllerService,
    private dialogService: NbDialogService,
    private router: Router,
    private contextService: ContextUpdatesService,
    activatedRoute: ActivatedRoute,
    private uiStackControllerService: UiStackControllerService,
    private uiAccountsControllerService: UiAccountsControllerService,
    private uiCoderControllerService: UiCoderControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.blueprint = p?.stackName;
    this.resourceType = p?.resourceType;
    this.resourceName = p?.resourceName;
    this.branchName = p?.branchName;
    this.resourceStatusProps = '';
    this.makeGETAPICalls();
    this.route.queryParams.subscribe((params) => {
      this.previousComponent = params['from'];
    });
    this.route.paramMap
      .pipe(map(() => window.history.state))
      .subscribe((data) => {
        this.previousComponentMode = data?.mode || 'TABLE';
      });
  }

  makeGETAPICalls() {
    Promise.allSettled([
      this.getResourceDetails(),
      this.getStackDetails(),
      this.getCoderAccount(),
    ]).then((data) => {
      this.setTabs();
      this.setButtons();
    });
  }

  setButtons() {
    const enableDisableButton = this.resource?.info?.disabled
      ? enableResourceButton
      : disableResourceButton;

    enableDisableButton.action = STACK_CONFIGURE;

    const showCoderButton =
      this.coderAccounts?.length && this.stack?.vcs === 'GITHUB';

    let buttons = [];
    if (showCoderButton) {
      buttons = [
        {
          ...workspaceButton,
          disabled: !!this.coderButtonTooltip,
          nbTooltip: this.coderButtonTooltip,
        },
        enableDisableButton,
        {
          ...this.editResourceButton,
          buttonType: 'tertiary',
          type: 'secondary',
        },
        this.deleteResourceButton,
      ];
    } else {
      buttons = [
        this.editResourceButton,
        enableDisableButton,
        this.deleteResourceButton,
      ];
    }
    this.actionButtons = buttons;
  }
  setTabs() {
    let originalTab = cloneDeep(this.tabs);
    if (
      this.stack?.vcs === 'GITHUB' &&
      this.coderSupportedResourceTypes.includes(this.resourceType) &&
      this.resource?.info?.githubRepositoryUrl?.length
    ) {
      this.tabs = [...originalTab, ...this.additionalTabs];
    } else {
      this.tabs = [...originalTab];
      if (
        this.router.url.includes('/pull-requests') ||
        this.router.url.includes('/github-actions')
      ) {
        this.router.navigate([
          `/stack/${this.blueprint}/blueprint-designer/${this.branchName}/resource/type/${this.resourceType}/name/${this.resourceName}/overview`,
        ]);
      }
    }
  }

  getCoderAccount() {
    this.coderAccounts = [];
    return new Promise((resolve, reject) => {
      this.uiAccountsControllerService
        .getAccountsByTypeUsingGET('CODER')
        .subscribe(
          async (response) => {
            this.coderAccounts = response;
            if (response?.length) {
              await this.setCoderDisabledStatus();
            }

            resolve(true);
          },
          (err) => {
            reject(err);
          },
        );
    });
  }

  setCoderDisabledStatus = () => {
    this.coderButtonTooltip = '';
    return new Promise((resolve, reject) => {
      this.uiCoderControllerService
        .identifyCoderLaunchEligibleUsingGET(this.blueprint)
        .subscribe(
          ({ coderAccountPresent, userVCSTokenPresent }) => {
            if (!coderAccountPresent) {
              this.coderButtonTooltip =
                'Please link your coder.com account in Account Management under Settings.';
            } else if (!userVCSTokenPresent) {
              this.coderButtonTooltip =
                'Kindly map your GitHub Personal Access Token in the profile settings.';
            }
            resolve(true);
          },
          () => {
            this.coderButtonTooltip = '';
            reject(false);
          },
        );
    });
  };

  fetchingResource: boolean = false;
  errorInFetchingResource: boolean = false;
  getResourceDetails() {
    this.resource = null;
    this.primaryIcon.icon = iconMapping[this.resourceType]
      ? this.resourceType
      : 'default';
    this.fetchingResource = true;
    this.errorInFetchingResource = false;
    this.resourceDetailsService.clearResource();
    return new Promise((resolve, reject) => {
      this.uiDropdownsControllerService
        .getResourceByStackUsingGET({
          resourceName: this.resourceName,
          resourceType: this.resourceType,
          stackName: this.blueprint,
          branch: this.branchName,
        })
        .subscribe(
          (rsp) => {
            this.resource = rsp;
            this.setOverridesPageVisibility();
            this.resourceDetailsService.setResource(this.resource);
            try {
              this.resourceJson = JSON.parse(rsp.content || '{}');
            } catch (error) {
              this.resourceJson = {};
            }
            this.setHeaderDescription();
            this.setHeaderIcons();
            resolve(true);
          },
          (err) => {
            this.errorInFetchingResource = true;
            if (err?.status != 404) {
              this.onAPIErrorService.onAPIError(
                err?.error?.message || 'Failed to fetch the resource',
                err,
              );
            }
            
            reject(err);
          },
        )
        .add(() => {
          this.fetchingResource = false;
        });
    });
  }
  setHeaderDescription() {
    this.resource?.info?.flavor || this.resource?.info?.flavour
      ? `Blueprint Resource Details: <b>${this.resource?.resourceName}</b> ${
          this.resource.resourceType
        } with <b>${
          this.resource?.info?.flavor || this.resource?.info?.flavour
        }</b> flavor implementation`
      : 'See overview and manage configuration of the resource in the blueprint';
  }
  setOverridesPageVisibility() {
    const ext = this.resource?.filename?.split('.')?.[1];
    if (ext !== 'json') {
      this.tabs = this.tabs.filter((tab) => tab?.id !== 'overrides');
    }
  }

  handlePageActions(event: string) {
    switch (event) {
      case 'edit-resource-name':
        this.openEditPopup();
        break;
      case 'delete-resource':
        this.openDeleteDialog(this.resource);
        break;
      case 'enable-resource':
      case 'disable-resource':
        this.openEnableDisableConfirmDialog(event);
        break;
      case 'launch-workspace':
        this.openWorkspacePopup(event);
        break;
      default:
        break;
    }
  }

  newResourceName: string;
  @ViewChild('editResourceNameDialog') editResourceNameDialog: TemplateRef<any>;
  openEditPopup() {
    this.newResourceName = this.resource.resourceName;
    this.dialogService.open(this.editResourceNameDialog, {
      hasBackdrop: true,
      closeOnBackdropClick: true,
    });
  }

  updatingResource: boolean;
  saveResource(ref) {
    this.updatingResource = true;
    this.uiBlueprintDesignerControllerService
      .renameResourceUsingPUT({
        branch: this.branchName,
        stackName: this.blueprint,
        resourceRenameRequest: {
          newResourceName: this.newResourceName,
          oldResourceName: this.resource.resourceName,
          resourceType: this.resource.resourceType,
          oldFileName: this.resource.filename,
          directory: this.resource.directory,
        },
      })
      .subscribe(
        (rsp) => {
          this.customToastService.success(
            `Resource name updated to ${this.newResourceName}`,
          );
          ref.close();
          this.resource.resourceName = this.newResourceName;
          this.resourceDetailsService.setResource(this.resource);
          const url = this.router.url.split('/');
          const componentName = url[url.length - 1];
          setTimeout(() => {
            // We need to destroy this component for the children component to reload properly
            // Although since we are redirecting to the same URL with different parameters
            // The reloading strategy isn't working in the forChild method so we need to redirect
            // the application to another URL and then send it back to this URL by using the timeout API
            // so that it can execute after this component is no longer up
            this.router.navigate([
              'stack',
              this.blueprint,
              'blueprint-designer',
              this.branchName,
              'resource',
              'type',
              `${this.resource.resourceType}`,
              'name',
              `${this.newResourceName}`,
              componentName,
            ]);
          }, 10);
          this.router.navigate([
            `stack/${this.blueprint}/blueprint-designer/${this.branchName}/${this.resourceType}/${this.newResourceName}/loading`,
          ]);
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            error?.error?.message || `Failed to update resource name`,
            error,
          );
        },
      )
      .add(() => {
        this.updatingResource = false;
      });
  }

  @ViewChild('deleteConfirmation') deleteConfirmation: TemplateRef<any>;
  openDeleteDialog(resource) {
    this.dialogService.open(this.deleteConfirmation, {
      context: {
        id: resource.resourceName,
        name: resource.resourceName,
        ...resource,
      },
    });
  }

  // Sends the user back to the resource details table component
  handleBack = () => {
    if (this.previousComponent === 'resource-centre') {
      const previousContext: BreadCrumbConfig =
        this.contextService.getContextFromCookies('previousContext');
      this.router.navigate([
        `${previousContext.blueprint}/cluster/${previousContext.environment}/resources/type/${previousContext.resourceType}/resource/${previousContext.resource}/overview`,
      ]);
    } else {
      const queryParams = { mode: this.previousComponentMode || 'TABLE' };
      if (this.resourceDetailsService.isLastContextPresent()) {
        const prevURL = this.resourceDetailsService.getLastContext();
        this.router.navigate([
          `stack/${this.blueprint}/blueprint-designer/${this.branchName}/${this.resourceType}/${this.resourceName}/loading`,
        ]);
        setTimeout(() => {
          this.router.navigate([prevURL]);
        }, 10);
        return;
      }

      this.router.navigate(
        [
          '/',
          'stack',
          this.blueprint,
          'blueprint-designer',
          this.branchName,
          'resource-list',
        ],
        {
          state: {
            mode: this.previousComponentMode || 'TABLE',
          },
        },

  deleteLoader: boolean = false;
  onDeleteREsource({ id, name, ...resource }, ref) {
    this.deleteLoader = true;
    this.uiBlueprintDesignerControllerService
      .deleteResourcesUsingDELETE({
        stackName: this.blueprint,
        branch: this.branchName,
        resourceFileRequests: [
          {
            content: {},
            resourceName: resource.resourceName,
            resourceType: resource.resourceType,
            filename: this.resource.filename,
            directory: this.resource.directory,
          },
        ],
      })
      .subscribe(
        (response) => {
          this.customToastService.success(
            `Resource ${resource.resourceName} deleted successfully`,
          );
          this.router.navigate([
            'stack',
            this.blueprint,
            'blueprint-designer',
            this.branchName,
            'resource-list',
          ]);
          ref?.close();
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            error?.error?.message ||
              `Failed to delete Resource ${resource.resourceName}`,
            error,
          );
        },
      )
      .add(() => {
        this.deleteLoader = false;
      });
  }
  setHeaderIcons() {
    this.headingIcons = getResourceHeaderIcons({
      resource: this.resource,
      clusterState: undefined,
      type: 'blueprint',
    });

    this.resourceStatusProps = JSON.stringify({
      severity: this.resource?.info?.disabled ? 'disabled' : 'enabled',
      badgeClass: 'p-2 text-capitalize mb-0',
    });
  }

  openEnableDisableConfirmDialog(buttonId) {
    this.dialogService.open(this.enableDisableConfirmation, {
      context: {
        actionText: buttonId === 'enable-resource' ? 'enable' : 'disable',
        resourceName: this.resource?.resourceName,
      },
    });
  }
  enableDisableResource(ref, actionText) {
    this.updatingResource = true;
    this.uiBlueprintDesignerControllerService
      .bulkEditDisabledForResourcesUsingPUT({
        stackName: this.blueprint,
        branch: this.branchName,
        resourceEnableDisableRequestList: [
          {
            resourceName: this.resourceName,
            resourceType: this.resourceType,
            disabled: !this.resource?.info?.disabled,
            filename: this.resource.filename,
            directory: this.resource.directory,
          },
        ],
      })
      .subscribe(
        async () => {
          this.updatingResource = false;
          ref?.close();
          this.customToastService.success(
            `Resource has been ${actionText}d successfully`,
          );
          await this.getResourceDetails();
          this.setButtons();
        },
        (error) => {
          this.updatingResource = false;
          this.onAPIErrorService.onAPIError(
            error?.error?.message || `Failed to ${actionText} resource`,
            error,
          );
        },
      );
  }

  getStackDetails() {
    this.loadingStackDetails = true;
    return new Promise((resolve, reject) => {
      this.uiStackControllerService.getStackUsingGET(this.blueprint).subscribe(
        (s) => {
          this.stack = s;
          this.loadingStackDetails = false;
          resolve(true);
        },
        (err) => {
          this.loadingStackDetails = false;
          let originalTab = cloneDeep(this.tabs);
          this.tabs = [...originalTab];
          reject(err);
        },
      );
    });
  }

  openWorkspacePopup(event) {
    this.dialogService.open(this.coderDialog);
  }
}
