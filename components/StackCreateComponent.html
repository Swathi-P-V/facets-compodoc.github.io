import { Component, ViewChild, TemplateRef } from '@angular/core';
import {
  UiStackControllerService,
  UiAccountsControllerService,
} from 'src/app/cc-api/services';
import { StackCreateService } from 'src/app/services/stack-create.service';
import { ActivatedRoute, Router, Params } from '@angular/router';
import { Stack } from './../../../cc-api/models/stack';
import { NbDialogService, NbPopoverDirective } from '@nebular/theme';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ContextUpdatesService } from 'src/app/services/context-updates.service';
import { Subscription } from 'rxjs';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Account } from 'src/app/cc-api/models';
import {
  whiteSpaceInputPattern,
  whiteSpaceAllPattern,
  urlDetectionPattern,
} from 'src/app/constants/inputPatterns';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-stack-create',
  templateUrl: './stack-create.component.html',
  styleUrls: ['./stack-create.component.scss'],
})
export class StackCreateComponent extends FacetsPage {
  @ViewChild('userAccountPopover', { static: true })
  private userAccountPopover: TemplateRef<any>;
  @ViewChild('versionControl', { static: true })
  private versionControl: TemplateRef<any>;
  @ViewChild(NbPopoverDirective) popover: NbPopoverDirective;
  loadingStackDetails: boolean = false; //This loader is used for displaying page level loader
  loadingAccounts: boolean = false;
  stackName: string = '';
  stackForm: FormGroup = this.fb.group({
    name: ['', [Validators.required, Validators.pattern(whiteSpaceAllPattern)]],
    creationType: ['NEW', [Validators.required]],
    templateId: ['', Validators.required],
    vcs: [''],
    vcsUrl: [
      '',
      [
        Validators.required,
        Validators.pattern(whiteSpaceInputPattern),
        Validators.pattern(urlDetectionPattern),
      ],
    ],
    accountId: ['', Validators.required],
    branch: ['master', Validators.pattern(whiteSpaceInputPattern)],
    relativePath: ['.', Validators.pattern(whiteSpaceInputPattern)],
    primaryCloud: [null],
  });
  versionControlForm: FormGroup = this.fb.group({
    provider: ['GITHUB', [Validators.required]],
    accountName: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    userName: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    token: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
  });
  stack: Stack = {};
  creationTypeOptions = [
    { value: 'NEW', label: 'New Blueprint' },
    { value: 'EXISTING', label: 'Import Existing Blueprint' },
  ];
  versionControlOptions = [
    { value: 'GITHUB', label: 'Github' },
    { value: 'BITBUCKET', label: 'Bitbucket' },
    { value: 'GITLAB', label: 'Gitlab' },
  ];
  userAccounts: any = [
    { label: 'Github', url: 'github-light' },
    { label: 'Gitlab', url: 'gitlab-light' },
    { label: 'Bitbucket', url: 'bitbucket-light' },
  ];
  accounts: Account[];
  accountsOptions: { label: string; value: string; provider: string }[] = [];
  loadingTemplate: boolean = false; //This loader is used to display loader for template widget
  popoverAccount: any = {};
  popoverId: string = '';
  cloudProviderList = [
    { label: 'AWS', value: 'AWS' },
    { label: 'GCP', value: 'GCP' },
    { label: 'AZURE', value: 'AZURE' },
  ];
  isCreate = true;
  callInProgress: boolean = false; //This loader is used for create and update button.
  contextUpdates$: Subscription;
  versionControlInProgress: boolean = false;
  stacks: Stack[] = [];
  templates: {
    label: string;
    value: string;
    description: string;
  }[] = [];
  selectedTemplate: {
    label: string;
    value: string;
    description: string;
  } = { label: '', value: '', description: '' };
  versionControlAccountIconMap: any = {
    GITHUB: 'github-light',
    GITLAB: 'gitlab-light',
    BITBUCKET: 'bitbucket-light',
  };
  providerDocLinkMap = {
    GITHUB:
      'https://docs.github.com/en/authentication/keeping-your-account-and-data-secure/creating-a-personal-access-token',
    GITLAB:
      'https://docs.gitlab.com/ee/user/profile/personal_access_tokens.html',
    BITBUCKET:
      'https://support.atlassian.com/bitbucket-cloud/docs/app-passwords/',
  };

  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private uiStackControllerService: UiStackControllerService,
    private stackCreateService: StackCreateService,
    private contextUpdatesService: ContextUpdatesService,
    private accountControllerService: UiAccountsControllerService,
    private fb: FormBuilder,
    private dialogService: NbDialogService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  get name() {
    return this.stackForm.get('name');
  }

  get creationType() {
    return this.stackForm.get('creationType');
  }

  get templateId() {
    return this.stackForm.get('templateId');
  }

  get vcsUrl() {
    return this.stackForm.get('vcsUrl');
  }

  get vcs() {
    return this.stackForm.get('vcs');
  }

  get accountId() {
    return this.stackForm.get('accountId');
  }

  get branch() {
    return this.stackForm.get('branch');
  }

  get relativePath() {
    return this.stackForm.get('relativePath');
  }

  get primaryCloud() {
    return this.stackForm.get('primaryCloud');
  }

  get provider() {
    return this.versionControlForm.get('provider');
  }

  get accountName() {
    return this.versionControlForm.get('accountName');

  get userName() {
    return this.versionControlForm.get('userName');
  }

  get token() {
    return this.versionControlForm.get('token');
  }

  OnPageInit(): void {
    this.getAllStacks();
    this.getTemplates();
  }

  getTemplates() {
    this.loadingTemplate = true;
    this.uiStackControllerService.getStackTemplatesUsingGET().subscribe(
      (t) => {
        this.templates = t.map((template) => {
          return {
            label: template?.name,
            description: template?.description,
            value: template?.id,
          };
        });
        this.loadingTemplate = false;
      },
      (err) => {
        this.loadingTemplate = false;
      },
    );
  }

  getAllStacks() {
    this.uiStackControllerService.getStacksUsingGET1().subscribe(
      (s) => {
        this.stacks = s;
      },
      (err) => {
        console.log(err);
      },
    );
  }

  getAccountDetails(rememberVcs, accountId = '') {
    this.loadingAccounts = true;
    this.accountControllerService
      .getAccountsByTypeUsingGET('VERSION_CONTROL')
      .subscribe(
        (response) => {
          this.accounts = response;
          response.forEach((account) => {
            if (account?.id?.length) {
              this.accountsOptions.push({
                label: account.name,
                value: account.id,
                provider: account.provider,
              });
            }
          });
          this.accountsOptions.push({
            label: 'Create new account',
            value: 'createAccount',
            provider: '',
          });
          if (rememberVcs) {
            this.onVcsChange(this.vcs?.value, accountId);
          }
          if (this.stackName) {
            this.getStackDetails();
          }
          this.loadingAccounts = false;
        },
        (error) => {
          this.loadingAccounts = false;
          if (this.stackName) {
            this.getStackDetails();
          }
        },
      );
  }

  onVcsChange(event, accountId = '') {
    this.versionControlForm.reset();
    setTimeout(() => {
      this.accountId.markAsPristine();
      this.accountId.markAsUntouched();
      this.accountId.setValue(accountId);
    }, 250);
  }

  OnPageLoad(p: Params): void {
    this.stackName = p.stackName;
    /*
     * Remove the validations on template selection for edit mode
     */
    if (this.stackName) {
      this.isCreate = false;
      this.name.disable();
      this.creationType.clearValidators();
      this.creationType.updateValueAndValidity();
      this.templateId.clearValidators();
      this.templateId.updateValueAndValidity();
    }
    this.getAccountDetails(false);
  }

  getStackDetails() {
    this.loadingStackDetails = true;
    this.uiStackControllerService.getStackUsingGET(this.stackName).subscribe(
      (s) => {
        this.stack = s;
        if (this.stack) {
          this.onVcsChange(this.stack.vcs, this.stack.versionControlAccountId);

          this.stackForm.patchValue({
            name: this.stack?.name || '',
            vcsUrl: this.stack?.vcsUrl || '',
            vcs: this.stack?.vcs || '',
            accountId: this.stack?.versionControlAccountId || '',
            branch: this.stack?.branch || '',
            relativePath: this.stack?.relativePath || '',
            primaryCloud: this.stack?.primaryCloud,
          });
        }
        this.loadingStackDetails = false;
      },
      (err) => {
        this.loadingStackDetails = false;
      },
    );
  }

  createStack() {
    try {
      this.callInProgress = true;
      let params: Stack = {
        name: this.name?.value,
        vcsUrl: this.vcsUrl?.value,
        vcs: this.vcs?.value,
        versionControlAccountId: this.accountId?.value,
        branch: this.branch?.value,
        relativePath: this.relativePath?.value,
        templateId:
          this.templateId?.value === 'NOTEMPLATE' ? '' : this.templateId?.value,
        primaryCloud: this.primaryCloud?.value,
      };
      this.uiStackControllerService.createStackUsingPOST1(params).subscribe(
        (cluster) => {
          this.customToastService.success('Blueprint creation success!! ');
          this.stackForm.reset();
          this.callInProgress = false;
          this.sendStackCreateEvent();
          this.contextUpdatesService.requestContextSwitcherRefresh();
          this.contextUpdates$ =
            this.contextUpdatesService.contextSwitcherUpdated.subscribe(
              (signal) => {
                if (signal.updated) {
                  this.router.navigate(['/', 'home']);
                  this.contextUpdates$.unsubscribe();
                }
              },
            );
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            'Failed: ' + error?.error?.message || error?.message,
            error,
          );
          this.callInProgress = false;
        },
      );
    } catch (err) {
      console.log(err);
      this.callInProgress = false;
    }
  }

  sendStackCreateEvent() {
    this.stackCreateService.sendStackEvent(true);
  }

  updateStack() {
    try {
      this.callInProgress = true;
      let params: Stack = {
        name: this.name?.value,
        vcsUrl: this.vcsUrl?.value,
        vcs: this.vcs?.value,
        versionControlAccountId: this.accountId?.value,
        branch: this.branch?.value,
        relativePath: this.relativePath?.value,
        primaryCloud: this.primaryCloud?.value,
      };
      this.uiStackControllerService
        .updateStackUsingPUT1({ stackName: this.stack?.name, stack: params })
        .subscribe(
          (cluster) => {
            this.customToastService.success('Blueprint change success!! ');
            this.stackForm.reset();
            this.callInProgress = false;
            this.router.navigate(['/home']);
          },
          (error) => {
            this.callInProgress = false;
            this.onAPIErrorService.onAPIError(
              'Failed ' + error?.error?.message || error?.message,
              error,
            );
          },
        );
    } catch (err) {
      console.log(err);
      this.callInProgress = false;
    }
  }

  toHome = () => {
    this.router.navigate(['/', 'home']);
  };

  goToStack(stack, ref) {
    ref.close();
    const url = '/stack/' + stack.name;
    this.router.navigate([url]);
  }

  openPopover(event, value) {
    event.stopPropagation();
    let stacks = [];
    this.stacks.forEach((stack) => {
      if (stack.versionControlAccountId === value) {
        stacks.push(stack);
      }
    });
    this.popoverAccount = this.accounts.filter(
      (account) => account.id === value,
    )[0];
    this.popoverId = value;
    this.dialogService.open(this.userAccountPopover, {
      context: { rowData: this.popoverAccount, stacks: stacks },
    });
  }

  openCreate(event) {
    event?.stopPropagation();
    this.accountId?.setValue('');
    this.provider.setValue('GITHUB');
    this.dialogService.open(this.versionControl);
  }

  submitAccount(ref) {
    let accountParams: any = {
      accessToken: this.token?.value || '',
      userName: this.userName?.value || '',
      name: this.accountName?.value || '',
      accountType: 'VERSION_CONTROL',
      provider: this.provider?.value || '',
    };
    if (this.provider?.value === 'GITHUB') {
      this.createGithubAccount(accountParams, ref);
    } else if (this.provider?.value === 'GITLAB') {
      this.createGitlabAccount(accountParams, ref);
    } else if (this.provider?.value === 'BITBUCKET') {
      this.createBitbucketAccount(accountParams, ref);
    }
  }

  createGithubAccount(accountParams, ref) {
    this.versionControlInProgress = true;
    this.accountControllerService
      .createGitHubAccountUsingPOST(accountParams)
      .subscribe(
        (data) => {
          this.customToastService.success(
            'Github account created successfully',
          );
          ref.close();
          this.versionControlInProgress = false;
          this.resetVersionControlForm();
          this.getAccountDetails(true, data?.id);
        },
        (err) => {
          this.versionControlInProgress = false;
          this.resetVersionControlForm();
          ref.close();
          this.onAPIErrorService.onAPIError(
            err?.error?.message || 'Github account creation failed',
            err,
          );
        },
      );
  }

  createGitlabAccount(accountParams, ref) {
    this.versionControlInProgress = true;
    this.accountControllerService
      .createGitLabAccountUsingPOST(accountParams)
      .subscribe(
        (data) => {
          this.customToastService.success(
            'Gitlab account created successfully',
          );
          ref.close();
          this.versionControlInProgress = false;
          this.resetVersionControlForm();
          this.getAccountDetails(true, data?.id);
        },
        (err) => {
          this.versionControlInProgress = false;
          this.resetVersionControlForm();
          ref.close();
          this.onAPIErrorService.onAPIError(
            err?.error?.message || 'Gitlab account creation failed',
            err,
          );
        },
      );
  }

  createBitbucketAccount(accountParams, ref) {
    this.versionControlInProgress = true;
    this.accountControllerService
      .createBitBucketAccountUsingPOST(accountParams)
      .subscribe(
        (data) => {
          this.customToastService.success(
            'Bitbucket account created successfully',
          );
          ref.close();
          this.versionControlInProgress = false;
          this.resetVersionControlForm();
          this.getAccountDetails(true, data?.id);
        },
        (err) => {
          this.versionControlInProgress = false;
          this.resetVersionControlForm();
          ref.close();
          this.onAPIErrorService.onAPIError(
            err?.error?.message || 'Bitbucket account creation failed',
            err,
          );
        },
      );
  }

  resetVersionControlForm() {
    this.versionControlForm.reset();
    this.provider.setValue('GITHUB');
  }

  providerChange(event) {
    this.versionControlForm.reset();
  }

  changeCreationType(event) {
    event === 'EXISTING'
      ? this.templateId.setValue('NOTEMPLATE')
      : this.templateId?.reset();
  }

  onAccountSelectionChange(event, provider) {
    this.vcs.setValue(provider);
  }
}
