import { Component, TemplateRef, ViewChild } from '@angular/core';
import { UiArtifactoryControllerService } from '../../cc-api/services/ui-artifactory-controller.service';
import { Artifactory } from '../../cc-api/models/artifactory';
import { NbDialogService } from '@nebular/theme';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params, Router } from '@angular/router';
import {
  ARTIFACTORY_WRITE,
  ARTIFACTORY_DELETE,
} from './../../constants/permissions';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { SmartTableTextTooltipComponent } from 'src/app/components/smart-table-text-tooltip/smart-table-text-tooltip.component';
import { SmartTableTimeagoTooltipComponent } from 'src/app/components/smart-table-timeago-tooltip/smart-table-timeago-tooltip.component';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const emptyStateLogo = './assets/pods-empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-artifactory-management',
  templateUrl: './artifactory-management.component.html',
  styleUrls: ['./artifactory-management.component.scss'],
})
export class ArtifactoryManagementComponent extends FacetsPage {
  @ViewChild('deleteArtifactoryUI', { static: true })
  private deleteArtifactoryUI: TemplateRef<any>;
  deletionInProgress: boolean = false;
  artifactories: Array<Artifactory>;
  smartTableCustomActionButtons = [
    {
      id: 'edit',
      icon: 'edit-outline',
      isFciPack: true,
      toolTip: 'Edit Container Registry',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: ARTIFACTORY_WRITE,
      onClick: (data) => {
        this.navigateToEditRegistry(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-outline',
      isFciPack: true,
      toolTip: 'Delete Container Registry',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: ARTIFACTORY_DELETE,
      onClick: (data) => {
        this.deleteArtifactoryConfirmPopup(data);
      },
    },
  ];
  tableSettings: any = {
    columns: {
      name: {
        title: 'Name',
        filter: true,
        type: 'text',
      },
      uri: {
        title: 'URL',
        filter: true,
        type: 'custom',
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            truncateText: true,
            label: cell,
            tooltipText: cell,
            allowCopy: true,
          });
        },
        renderComponent: SmartTableTextTooltipComponent,
      },
      artifactoryType: {
        title: 'Type',
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [
              { value: 'ECR', title: 'ECR' },
              { value: 'OTHERS', title: 'Other' },
            ],
          },
        },
        valuePrepareFunction: (value) => {
          const registryTypeLabelMap = {
            ECR: 'ECR',
            OTHERS: 'Other',
          };
          return registryTypeLabelMap[value];
        },
      },
      lastModifiedBy: {
        title: 'Modified By',
        filter: true,
        type: 'custom',
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            user: row?.lastModifiedBy || row?.createdBy,
            label: row?.lastModifiedDate || row?.creationDate,
          });
        },
        renderComponent: SmartTableTimeagoTooltipComponent,
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {
            smartTableCustomActionButtons: this.smartTableCustomActionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '12%',
        filter: false,
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
      perPage: 15,
    },
    mode: 'external',
    actions: false,
  };

  error: any;
  primaryMsg: string = 'No artifactories created!';
  secondaryMsg: string =
    'Create artifactories to add docker image repositories';
  buttonText: string = 'Add Container Registry';
  toolTip: string = 'Add Container Registry';
  showBtn: boolean = true;
  status: 'error' | 'info' = 'info';
  showEmptyState: boolean = false;
  loading: boolean;
  emptyStateLogo: any = emptyStateLogo;
  errorIcon: any = errorIcon;
  readonly ARTIFACTORY_DELETE = ARTIFACTORY_DELETE;

  actionButtons = [
    {
      id: 'create-artifactory',
      title: 'Add Container Registry',
      buttonType: 'primary',
      disabled: false,
      loading: false,
      action: ARTIFACTORY_WRITE,
    },
  ];

  constructor(
    route: ActivatedRoute,
    private router: Router,
    private artifactoryControllerService: UiArtifactoryControllerService,
    private dialogService: NbDialogService,
    private customToasterService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(route);
  }

  OnPageLoad(p: Params): void {
    this.getArtifactories();
  }

  handlePageActions(event: string) {
    switch (event) {
      case 'create-artifactory':
        this.navigateToCreateRegistry();
        break;
      default:
        break;
    }
  }

  navigateToCreateRegistry() {
    this.router.navigate(['settings/artifactories/create']);
  }

  navigateToEditRegistry(rowData) {
    this.router.navigate([`settings/artifactories/edit/${rowData?.id}`]);
  }

  getArtifactories() {
    this.loading = true;
    this.artifactoryControllerService.getAllArtifactoriesUsingGET1().subscribe(
      (response) => {
        this.artifactories = response;
        this.showEmptyState = this.artifactories?.length <= 0;
        this.loading = false;
      },
      (error) => {
        this.loading = false;
        this.showEmptyState = true;
        this.primaryMsg =
          error?.error?.message || 'There is some error fetching artifacts';
        this.emptyStateLogo = this.errorIcon;
        this.showBtn = false;
        this.status = 'error';
      },
    );
  }

  deleteArtifactory(data, ref) {
    this.deletionInProgress = true;
    this.artifactoryControllerService
      .deleteArtifactoryUsingDELETE(data?.id)
      .subscribe(
        (response) => {
          this.customToasterService.success(
            `Container Registry ${data?.name} deleted successfully `,
          );
          this.deletionInProgress = false;
          ref.close();
          this.getArtifactories();
        },
        (error) => {
          this.error = error?.error?.message || error?.message;
          this.onAPIErrorService.onAPIError(
            error?.error?.message ||
              error?.message ||
              `Failed to delete Container Registry ${data?.name}`,
            error,
          );
          this.deletionInProgress = false;
          ref.close();
        },
      );
  }

  deleteArtifactoryConfirmPopup(rowData) {
    this.dialogService.open(this.deleteArtifactoryUI, { context: rowData });
  }
}

<div class="row">
  <div class="col-12">
    <app-page-header
      heading="Container Registries"
      subheading="Add container registries to securely store your images"
      [actionButtons]="actionButtons"
      (click)="handlePageActions($event)"
    >
    </app-page-header>
  </div>
</div>
<div class="d-flex justify-content-between mb-3 mt-3">
  <nb-card [ngClass]="{ 'py-5': loading, 'w-100': true }" [nbSpinner]="loading">
    <nb-card-body *ngIf="!loading && !showEmptyState">
      <ng2-smart-table
        class="cc-no-standard"
        [settings]="tableSettings"
        [source]="artifactories"
      ></ng2-smart-table>
    </nb-card-body>
    <nb-card-body *ngIf="!loading && showEmptyState">
      <app-message
        [imgUrl]="emptyStateLogo"
        [primaryMsg]="primaryMsg"
        [secondaryMsg]="secondaryMsg"
        (buttonClick)="navigateToCreateRegistry()"
        [buttonText]="buttonText"
        [showBtn]="showBtn"
        [toolTip]="toolTip"
        [status]="status"
      ></app-message>
    </nb-card-body>
  </nb-card>

<ng-template #deleteArtifactoryUI let-data let-ref="dialogRef">
  <app-delete-confirmation
    [dialogRef]="ref"
    [deleteItem]="data"
    [showSpinner]="deletionInProgress"
    headerText="Container Registry"
    [permanentDelete]="false"
    [deleteBtnPermission]="ARTIFACTORY_DELETE"
    (deleteConfirmEvent)="deleteArtifactory(data, ref)"
  >
  </app-delete-confirmation>
</ng-template>
