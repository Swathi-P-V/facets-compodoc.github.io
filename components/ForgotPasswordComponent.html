import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { ActivatedRoute, Router } from '@angular/router';
import { UiNoAuthUserControllerService } from 'src/app/cc-api/services';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ApplicationControllerService } from 'src/app/cc-api/services';
import { SimpleOauth2User } from '../../../cc-api/models/simple-oauth-2user';
import { ResetPasswordRequest } from 'src/app/cc-api/models';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { CustomToastService } from '../../custom-toast/custom-toast.service';

@Component({
  selector: 'app-forgot-password',
  templateUrl: './forgot-password.component.html',
  styleUrls: ['./forgot-password.component.scss'],
})
export class ForgotPasswordComponent implements OnInit {
  errorMsg: string = '';
  logo: any;
  isCustomIntegrationPresent: boolean = false;
  redirectTo: string = null;
  validationInProgress: boolean = false;
  show: boolean = false;
  forgotPasswordForm: FormGroup;
  showSuccessSection: boolean = false;

  constructor(
    private http: HttpClient,
    private route: ActivatedRoute,
    private router: Router,
    private readonly fb: FormBuilder,
    private uiNoAuthService: UiNoAuthUserControllerService,
    private applicationControllerService: ApplicationControllerService,
    private customToastService: CustomToastService,
  ) {}

  ngOnInit(): void {
    this.forgotPasswordForm = this.fb.group({
      username: [
        '',
        [
          Validators.required,
          Validators.email,
          Validators.pattern(whiteSpaceInputPattern),
        ],
      ],
    });

    //If the user is already logged-in then redirect to home page
    this.applicationControllerService.meUsingGET().subscribe(
      (x: SimpleOauth2User) => {
        this.router.navigate(['/home']);
      },
      (error) => {},
    );
    this.http
      .get('/public/v1/logo', { responseType: 'text' })
      .subscribe((x) => {
        this.logo = x;
      });
  }

  get username() {
    return this.forgotPasswordForm.get('username');
  }

  sendEmail() {
    this.errorMsg = '';
    this.validationInProgress = true;
    let params: ResetPasswordRequest = {
      email: this.username.value,
    };
    this.uiNoAuthService.resetPasswordRequestUsingPOST(params).subscribe(
      (response) => {
        this.validationInProgress = false;
        this.showSuccessSection = true;
      },
      (error) => {
        this.validationInProgress = false;
        if (error?.error?.code === '404') {
          this.errorMsg =
            'The email address you entered doesn’t belong to an account.';
        } else {
          this.customToastService.error(
            error?.error?.message || 'Failed to send email',
          );
        }
      },
    );
  }

  returnToLogin() {
    this.router.navigate(['/', 'login']);
  }
}
