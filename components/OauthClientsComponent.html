import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { UiOAuthControllerService } from '../../cc-api/services/ui-oauth-controller.service';
import { CustomOAuth2ClientRegistration } from '../../cc-api/models/custom-oauth-2client-registration';
import { NbDialogService } from '@nebular/theme';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  OAUTH_INTEGRATION_WRITE,
  OAUTH_INTEGRATION_DELETE,
} from 'src/app/constants/permissions';
import cloneDeep from 'lodash/cloneDeep';
import { CustomToastService } from '../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const emptyStateLogo = './assets/oauth-empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-oauth-clients',
  templateUrl: './oauth-clients.component.html',
  styleUrls: ['./oauth-clients.component.scss'],
})
export class OauthClientsComponent extends FacetsPage {
  integrations: Array<CustomOAuth2ClientRegistration>;
  loading: boolean = false;
  emptyStateLogo = emptyStateLogo;
  errorLogo = errorIcon;
  primaryMsg: string = 'No integrations added!';
  secondaryMsg: string =
    'Add OAuth integration to login to Facets Control Plane';
  showEmptyState: boolean = false;
  status: 'info' | 'error' = 'info';
  showBtn: boolean = true;
  buttonText: string = 'Add OAuth Integration';
  tooltip: string = 'Add OAuth Integration';
  registrationIdForDeletion: string = '';
  oauthIntegrationWrite = OAUTH_INTEGRATION_WRITE;
  smartTableCustomActionButtons = [
    {
      id: 'edit',
      icon: 'edit-2-outline',
      toolTip: 'Edit',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: this.oauthIntegrationWrite,
      onClick: (data) => {
        this.navigateToAddOrEditIntegrations(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: OAUTH_INTEGRATION_DELETE,
      onClick: (data) => {
        this.deleteIntegrationPopup(data);
      },
    },
  ];
  tableSettings: any = {
    columns: {
      registrationId: {
        title: 'Registration Id',
      },
      provider: {
        title: 'Provider',
      },
      loginButtonText: {
        title: 'Login Button',
      },
      clientId: {
        title: 'Client Id',
        filter: false,
      },
      scope: {
        title: 'Scope',
        filter: false,
      },
      systemConfigured: {
        title: 'Type',
        filter: false,
        valuePrepareFunction: (cell) => {
          return cell ? 'System Defined' : 'Custom';
        },
      },
      issuerUrl: {
        title: 'Issuer Url',
        filter: false,
        valuePrepareFunction: (cell) => {
          return cell || 'N/A';
        },
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          //Considering registrationId as unique column here
          let formattedSmartTableCustomActionButtons = cloneDeep(
            this.smartTableCustomActionButtons,
          );
          formattedSmartTableCustomActionButtons.forEach((button) => {
            button['uniqueId'] = button?.id + '_' + row?.registrationId;
          });
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {
            smartTableCustomActionButtons:
              formattedSmartTableCustomActionButtons,
          };
          return JSON.stringify(displayVal);
        },

        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '12%',
        filter: false,
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
      perPage: 15,
    },
    mode: 'external',
    actions: false,
  };
  errorMsg: any;
  errorMsgDelete: any;
  @ViewChild('googleIntegration', { static: true })
  private googleIntegration: TemplateRef<any>;

  @ViewChild('deleteIntegration', { static: true })
  private deleteIntegration: TemplateRef<any>;

  actionButtons = [
    {
      id: 'add-integration',
      title: 'Add Integrations',
      buttonType: 'primary',
      nbTooltip: 'Add Integrations',
      disabled: false,
      loading: false,
      action: 'OAUTH_INTEGRATION_WRITE',
      onclick: () => {
        this.navigateToAddOrEditIntegrations();
      },
    },
  ];

  constructor(
    private oAuthControllerService: UiOAuthControllerService,
    private dialogService: NbDialogService,
    activatedRoute: ActivatedRoute,
    private router: Router,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  handlePageActions(event: string) {
    this.actionButtons?.find((button) => button?.id === event)?.onclick();
  }

  OnPageInit(): void {
    this.loading = true;
    this.oAuthControllerService.getAllIntegrationsUsingGET().subscribe(
      (response) => {
        this.integrations = response;
        this.loading = false;
        if (this.integrations?.length <= 0) {
          this.showEmptyState = true;
        }
      },
      (error) => {
        this.loading = false;
        this.showEmptyState = true;
        this.setErrorStates(error);
      },
    );
  }

  OnPageLoad(p: Params): void {}

  setErrorStates(error) {
    this.primaryMsg = '';
    this.secondaryMsg =
      error?.error?.message ||
      ' There is some error fetching integrations information';
    this.showBtn = false;
    this.status = 'error';
  }

  navigateToAddOrEditIntegrations(
    integration?: CustomOAuth2ClientRegistration,
  ) {
    if (integration) {
      this.router.navigate([
        'settings',
        'oauthClients',
        'edit',
        integration.registrationId,
      ]);
    } else {
      this.router.navigate(['settings', 'oauthClients', 'add']);
    }
  }

  deleteIntegrationPopup(rowData) {
    this.dialogService.open(this.deleteIntegration, {
      context: { id: rowData.registrationId, payload: rowData },
    });
  }

  deletingIntegration: boolean = false;
  removeIntegration(registrationId: string, ref: any) {
    this.deletingIntegration = true;
    this.oAuthControllerService
      .deleteIntegrationsUsingDELETE(registrationId)
      .subscribe(
        (response) => {
          this.integrations = response;
          this.customToastService.success(
            'Integration deleted - ' + this.registrationIdForDeletion,
          );
          ref.close();
        },
        (error) => {
          this.errorMsgDelete = error.error.message || error.message;
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Failed to delete integration',
            error,
          );
        },
      )
      .add(() => {
        this.deletingIntegration = false;
        this.registrationIdForDeletion = '';
      });
  }
}
