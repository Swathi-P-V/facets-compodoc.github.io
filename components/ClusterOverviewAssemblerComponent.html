              let error = JSON.parse(err?.error);

                error?.message || 'Unable to download credentials',

      );

  getEnvironmentConfig() {
    this.fetchingEnvironmentConfig = true;
    this.uiCommonClusterControllerService
      .getClusterInfoUsingGET(this.clusterId)
      .subscribe(
        async (r) => {
          this.cluster = r;
          //dependent Envs
          this.cluster?.dependentClusters?.forEach((cluster) => {
            this.dependentEnvList.push({
              name: cluster?.name,
              id: cluster?.id,
              url: `/${cluster?.stackName}/cluster/${cluster?.id}/overview`,
              disableLink: false,
            });
          });
          this.dependentEnvData = JSON.stringify({
            data: this.dependentEnvList,
            extraData: this.dependentEnvList,
            decorate: false,
            status: 'success',
            defaultDisplay: 1,
            showAsLink: true,
            dataObject: true,
            linkUnderline: true,
            linkClass: 'primary-text',
          });
          if (this.cluster?.azs?.length) {
            this.cluster.azs = this.cluster?.azs?.filter(
              (az) => az?.length > 0,
            );
          }
          // this.entityType = r?.entityType;
          this.cloud = r?.cloud;
          await this.getAccountDetails();
          this.initNavigationLinks();
          this.fetchingEnvironmentConfig = false;
          // This would override the previousComponent state and the onback event would always end up taking the user to the cluster-overview component
          if (this.redirectToEdit) {
            this.navigateTo('editEnvironment');
          }
        },
        (error) => {
          this.cluster = null;
          this.envErrorMessage =
            error?.error?.message ||
            ' There is some error fetching environment information';
          this.fetchingEnvironmentConfig = false;
        },

  getAlerts() {
    this.fetchingFiringAlerts = true;
    this.aWSClusterService.getAlertsUsingGET(this.clusterId).subscribe(
      (t) => {
        if (t['status'] === 'success') {
          let groups = t['data']['groups'];
          groups.forEach((group) => {
            let subRules = group['rules'];
            subRules.forEach((r) => {
              r.groupName = group['name'];
              if (r['state'] == 'firing') {
                if (r['alerts']?.length > 0) {
                  r['alerts'].forEach((alert) => {
                    if (this.rules.length < 5) {
                      this.rules.push({
                        message: alert['annotations']['message'],
                        groupName: r['groupName'],
                        lastEvaluation: r['lastEvaluation'],
                      });
                    }
                  });
                }
              }
            });
          });
        } else {
          this.rules = [];
          this.alertsErrorReason = '';
        }
        this.fetchingFiringAlerts = false;
      },
      (error) => {
        this.rules = null;
        this.alertsErrorReason =
          error?.error?.message ||
          'Error in fetching alerts for the environment';
        this.fetchingFiringAlerts = false;
        this.alertStatus = 'error';
      },
    );
  }

          this.deleteScheduleInProgress = false;
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Release schedule deletion failed',
            error,
          );
          ref.close();
        },
      );
  }

  getFormattedVal(val) {
    return JSON.stringify({
      value: val,
      truncateLength: 10,
      labelClass: 'weight-600',
    });
  }

    const url =
      '/' +
      this.stackName +
      '/cluster/' +
      this.clusterId +
      '/release-details/' +
      deployment?.id;
    this.router.navigate([url], {

        previousComponent: 'clusterOverview',
      },
    });
  }

  async updateActionButtonsProps() {
    await this.getClusterStateAndUpdateActionButtons(this.clusterId);
    this.getClusterSettingsAndUpdateActionButtons();
  }

  async getClusterSettingsAndUpdateActionButtons() {
    const actionButtons = cloneDeep(this.environmentActionButtons);
    const actionButton = actionButtons?.find((button) => button?.id === 'vpn');
    this.UiSettingsControllerService.getSettingValueUsingGET({
      entityType: 'CLUSTER',
      entityId: this.clusterId,
    }).subscribe(
      (clusterSettings) => {
        actionButton.disabled = !clusterSettings[VPN_ENABLE_DISABLE];
        actionButton.nbTooltip = clusterSettings[VPN_ENABLE_DISABLE]
          ? ''
          : 'Please enable VPN from environment settings';
        actionButton.buttonSuffixIconProps.hidden =
          clusterSettings[VPN_ENABLE_DISABLE];
        this.environmentActionButtons = cloneDeep(actionButtons);
      },
      (error) => {
        actionButton.disabled = true;
        actionButton.nbTooltip = 'Failed to fetch VPN setting!';
        this.environmentActionButtons = cloneDeep(actionButtons);
      },
    );
  }
  showVPNInstructions() {
    this.dialogService.open(this.vpnInstructionsModal, {
      context: {},
    });
  }
  downloadVPNProfile(id) {
    let downloadVPNProfile = this.environmentActionButtons.filter(
      (element) => element?.id === id,
    )?.[0];
    downloadVPNProfile.loading = true;
    downloadVPNProfile.disabled = true;
    this.uiCommonClusterControllerService
      .getVPNProfileUsingGET(this.clusterId)
      .subscribe(
        (fileContent) => {
          downloadFile(fileContent, this.cluster?.name + '-vpn-profile.conf');
          mixpanel.track(DOWNLOAD_VPN_PROFILE_SUCCESS);
          downloadVPNProfile.loading = false;
          downloadVPNProfile.disabled = false;
        },
        (error) => {
          downloadVPNProfile.loading = false;
          downloadVPNProfile.disabled = false;
          mixpanel.track(DOWNLOAD_VPN_PROFILE_FAIL, { error });
          let errorMessage = '';
          try {
            errorMessage = JSON.parse(error?.error || '{}')?.message;
          } catch (err) {}
          this.onAPIErrorService.onAPIError(
            errorMessage || 'Could not download vpn profile',
            error,
          );
        },
      );
  }

  pauseResumeReleaseScheduleConfirmPopup() {
    this.dialogService.open(this.pauseResumeScheduleConfirmUi, {
      context: this.releaseScheduleData,
    });
  }

  // Pause-Resume Release Schedule
  pauseResumeReleaseSchedule(ref) {
    this.pauseResumeScheduleInProgress = true;
    // isPaused is the state of release schedule, when pausing/resuming send pauseReleaseSchedule
    this.releaseScheduleCrudService
      .pauseResumeReleaseSchedule(this.releaseScheduleData, this.clusterId, ref)
      .then(
        (res) => {
          this.pauseResumeScheduleInProgress = false;
          this.releaseScheduleData = res[0];
        },
        (error) => {
          this.pauseResumeScheduleInProgress = false;
        },
      );
  }

  handleDropdownButtonClick(clickedBtn) {
    this.releaseScheduleDropdownButtons
      .find((btn) => btn?.id === clickedBtn?.id)
      ?.onClick();
  }

  getClusterStateAndUpdateActionButtons(clusterId) {
    return new Promise((resolve, reject) => {
      this.deploymentService.getClusterStateUsingGET(clusterId).subscribe(
        (state) => {
          this.clusterState = state;
          // Update disabled state of 'k8s-cred-download' and 'k8s-cred-refresh' button based on cluster state
          this.environmentActionButtons = updateK8sButtonsWithClusterState(
            this.environmentActionButtons,
            this.clusterState,
            ['k8s-cred-download', 'k8s-cred-refresh'],
            this.k8s_tooltips,
          );

          resolve(true);
        },
        (error) => {
          this.clusterState = null;
          resolve(false);
        },
      );
    });
  }
}
