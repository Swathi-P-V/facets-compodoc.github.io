import { Component, Input, ViewChild } from '@angular/core';
import { UiBlueprintDesignerControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params } from '@angular/router';
import { FormControl } from '@angular/forms';
import { NbPopoverDirective } from '@nebular/theme';
import { ClipboardService } from 'ngx-clipboard';
import {
  setInterfaceList,
  setAttributesList,
} from './resource-out-properties-utils';

@Component({
  selector: 'app-resource-out-properties',
  templateUrl: './resource-out-properties.component.html',
  styleUrls: ['./resource-out-properties.component.scss'],
})
export class ResourceOutPropertiesComponent extends FacetsPage {
  @ViewChild(NbPopoverDirective) popover: NbPopoverDirective; //copy popover
  blueprint: string;
  resourceName: string;
  resourceType: string;
  outProperties: any;
  interfacesList: any = [];
  attributesList: any = [];
  selectedInterface: any;
  interfaceOptions: { label: string; value: string }[] = [];
  attributeOptions: { label: string; value: string }[] = [];
  interfaceSelection = new FormControl('');
  attributeSelection = new FormControl('');
  loadingOutProps: boolean = false;
  resources: any = {};
  resourceSelection = new FormControl('');
  errorMessage: string = '';

  constructor(
    private blueprintDesignerController: UiBlueprintDesignerControllerService,
    private clipboardService: ClipboardService,
    activatedRoute: ActivatedRoute,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.blueprint = p?.stackName;
    this.resourceType = p?.resourceType;
    this.resourceName = p?.resourceName;
    this.resourceSelection.setValue(null);
    this.getOutProperties();
  }

  getOutProperties() {
    this.loadingOutProps = true;
    this.errorMessage = '';
    this.blueprintDesignerController
      .getAutocompleteDataUsingGET(this.blueprint)
      .subscribe(
        (data) => {
          const autoCompleteData = data;
          this.outProperties = autoCompleteData?.['outProperties'];
          this.resources = autoCompleteData?.['resources'];
          this.loadingOutProps = false;
        },
        (error) => {
          this.loadingOutProps = false;
          this.errorMessage = error?.error?.message || error?.message;
          this.outProperties = null;
        },
      );
  }

  processReferenceData() {
    this.setEmptyStates();
    if (this.outProperties) {
      const interfaces =
        this.outProperties?.[this.resourceType]?.out?.interfaces;
      const attributes =
        this.outProperties?.[this.resourceType]?.out?.attributes;
      if (interfaces) {
        this.interfaceOptions = Object.keys(interfaces).map((interfaceVal) => ({
          label: interfaceVal,
          value: interfaceVal,
        }));
      }
      if (attributes) {
        this.attributeOptions = Object.keys(attributes).map((attributeVal) => ({
          label: attributeVal,
          value: attributeVal,
        }));
      }
      this.setInterfaceList(true);
      this.setAttributesList();
      setTimeout(() => {
        this.interfaceSelection.patchValue(this.interfaceOptions?.[0]?.value);
      }, 100);
    }
  }

  setEmptyStates() {
    this.interfaceSelection.setValue(null);
    this.interfaceOptions = [];
    this.attributeSelection.setValue(null);
    this.attributeOptions = [];
  }

  setInterfaceList(defaultInterface = false) {
    this.interfacesList = setInterfaceList(
      this.outProperties,
      this.resourceType,
      this.resourceName,
      defaultInterface
        ? this.interfaceOptions?.[0]?.value
        : this.interfaceSelection?.value,
    );
  }

  setAttributesList() {
    this.attributesList = setAttributesList(
      this.outProperties,
      this.resourceType,
      this.resourceName,
    );
  }

  onResourceSelection(event, resourceType, resourceName) {
    this.resourceType = resourceType;
    this.resourceName = resourceName;
    this.processReferenceData();
  }

  onInterfaceChanged(attributeType) {
    this.setInterfaceList();
  }

  copyText(interfaceVal) {
    this.clipboardService.copy(interfaceVal);
  }
}
