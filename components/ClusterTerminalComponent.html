import {
  Component,
  Input,
  EventEmitter,
  Output,
  OnDestroy,
} from '@angular/core';
import { UiDeploymentControllerService } from 'src/app/cc-api/services';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { DeploymentLog } from 'src/app/cc-api/models';
import { logLines } from './logslines';
import { Subscription } from 'rxjs';
import { GetTailLogsHelperService } from 'src/app/services/get-tail-logs-helper.service';
import { processLogLines } from 'src/app/utils/process-logs-utils';
import { CustomToastService } from '../../custom-toast/custom-toast.service';

@Component({
  selector: 'app-cluster-terminal',
  templateUrl: './cluster-terminal.component.html',
  styleUrls: ['./cluster-terminal.component.scss'],
})
export class ClusterTerminalComponent extends FacetsPage implements OnDestroy {
  // Inputs and Outputs
  @Input() deploymentId: string = ''; //Deployment ID
  @Input() releaseStatus: string = ''; //Release Status
  @Output() closeEvent = new EventEmitter<boolean>();

  loadingLogs = false;
  emptyStatus: 'info' | 'error';
  errorMessage: string;
  stackName: any;
  clusterId = '';
  originalLogLines = [];
  processedLogLines: logLines[] = []; //This stores the final logs after processing and grouping

  // subscription
  controller: AbortController;
  deploymentLogsSubscription: Subscription;

  // Private
  private rootUrl: string = window.location.origin;

  // Readonly
  readonly IN_PROGRESS = 'IN_PROGRESS';
  readonly STARTED = 'STARTED';

  constructor(
    private deploymentService: UiDeploymentControllerService,
    activatedRoute: ActivatedRoute,
    private toastService: CustomToastService,
    private tailLogsService: GetTailLogsHelperService,
  ) {
    super(activatedRoute);
  }

  ngOnDestroy() {
    this.stopTailLogs();
  }

  OnPageLoad(p: Params): void {
    this.stackName = p?.stackName;
    this.clusterId = p?.clusterId;

    this.stopTailLogs();
    this.showLogDetails();
  }

  stopTailLogs() {
    this.controller?.abort();
    this.deploymentLogsSubscription?.unsubscribe();
    this.errorMessage = '';
  }

  showLogDetails() {
    // If cluster release is in progress or started make streaming API call.
    // For already completed releases make getLogs API call
    [this.IN_PROGRESS, this.STARTED].includes(this.releaseStatus)
      ? this.getStreamLogs(this.deploymentId)
      : this.getAllLogs(this.deploymentId);
  }

  getStreamLogs(deploymentId: string) {
    this.originalLogLines = [];
    this.processedLogLines = [];
    this.controller = new AbortController();
    const signal = this.controller.signal;
    this.loadingLogs = true;
    this.deploymentLogsSubscription =
      this.tailLogsService.logsUpdates.subscribe(
        (logs) => {
          let parsedLogs;
          try {
            parsedLogs = JSON.parse(logs);
          } catch {
            parsedLogs = {};
          }
          this.loadingLogs = false;
          if (+parsedLogs?.code >= 400) {
            this.errorMessage =
              'An error was encountered while tailing the deployment logs. Please retry or contact your admin. Error code: ' +
              parsedLogs.status +
              ' -- Error message: ' +
              parsedLogs.message +
              '';
          } else if (logs?.trim()?.length > 0) {
            this.originalLogLines = processLogLines(
              logs,
              this.originalLogLines,
            );
            this.processedLogLines = this.classifyAndGroupLogLines(
              this.originalLogLines,
            );
          }
        },
        (err) => {
          this.loadingLogs = false;
          this.errorMessage =
            'An error was encountered while tailing the deployment logs. Please retry or contact your admin. Error code: ' +
            err.status +
            ' -- Error message: ' +
            err.message +
            '';
        },
      );

    let logsUrl = `${this.rootUrl}/cc-ui/v1/clusters/${encodeURIComponent(
      String(this.clusterId),
    )}/deployments/${encodeURIComponent(String(deploymentId))}/logs/stream`;

    this.tailLogsService.getTailLogs({}, signal, logsUrl);
  }

  getAllLogs(deploymentId) {
    this.loadingLogs = true;
    this.processedLogLines = [];
    this.deploymentService
      .getDeploymentLogsUsingGET({
        clusterId: this.clusterId,
        deploymentId,
        nextToken: null,
      })
      .subscribe(
        (response) => {
          this.processedLogLines = this.classifyAndGroupLogLines(
            response?.logEventList,
          );
          this.loadingLogs = false;
        },
        (error) => {
          this.errorMessage = error?.error?.message || error?.statusText;
          if (
            (error.status === 500 || error.status === 403) &&
            error?.error?.message === 'Access is denied'
          ) {
            this.toastService.error({
              title: 'Access is forbidden',
              content: 'Please contact your administrator for more details',
            });
          } else {
            this.toastService.error(
              'Failed to fetch logs, ' + this.errorMessage,
            );
          }
          this.loadingLogs = false;
        },
      );
  }

  /**
   * classify and group log lines based on predefined patterns.
   * @param logLines - The array of log lines to be organized.
   * @returns An array of structured log objects.
   */
  classifyAndGroupLogLines(logLines) {
    let groupedLogLines: logLines[] = [];
    let isSectionStarted: boolean = false;
    let currentSectionStartIndex: number;
    const sectionPatterns = {
      preBuild: {
        start: /(##### PRE BUILD - STARTED ####)/,
        end: /(----- PRE BUILD - FINISHED)/,
      },
      postBuild: {
        start: /(##### POST BUILD - STARTED #####)/,
        end: /(----- POST BUILD - FINISHED)/,
      },
      module: {
        start: /(# module)/,
        end: /(Plan:)/,
      },
    };

    logLines?.forEach((logLine) => {
      if (this.isMessageNotEmpty(logLine)) {
        if (isSectionStarted) {
          let patternFound = false;
          Object.keys(sectionPatterns)?.forEach((key) => {
            const { start, end } = sectionPatterns?.[key];
            if (this.isPatternFound(logLine, start)) {
              groupedLogLines.push(this.getLogLine(logLine, true));
              patternFound = true;
              currentSectionStartIndex = groupedLogLines?.length - 1;
            } else if (this.isPatternFound(logLine, end)) {
              groupedLogLines.push(this.getLogLine(logLine, false));
              isSectionStarted = false;
              patternFound = true;
              currentSectionStartIndex = 0;
            }
          });
          if (!patternFound) {
            groupedLogLines[currentSectionStartIndex]?.children?.push(
              this.getLogLine(logLine, false),
            );
          }
        } else {
          let patternFound = false;
          Object.keys(sectionPatterns)?.forEach((key) => {
            const { start, end } = sectionPatterns?.[key];
            if (this.isPatternFound(logLine, start)) {
              groupedLogLines.push(this.getLogLine(logLine, true));
              isSectionStarted = true;
              patternFound = true;
              currentSectionStartIndex = groupedLogLines.length - 1;
            }
          });
          if (!patternFound) {
            groupedLogLines.push(this.getLogLine(logLine, false));
          }
        }
      }
    });
    return groupedLogLines;
  }

  /**
   * Creates a log object with timestamp and message.
   * @param logLine - The log line to extract data from.
   * @param withChild - Indicates whether to include a children array in the log object.
   * @returns A log object containing timestamp, message, and optional children array.
   */
  getLogLine(logLine, withChild: boolean) {
    let logObject = {
      timestamp: logLine['timestamp'],
      message: logLine['message'],
    };
    if (withChild) {
      logObject['children'] = new Array();
    }
    return logObject;
  }

  isMessageNotEmpty(logLine) {
    return logLine['message']?.trim()?.length > 0;
  }

  isPatternFound(logLine, regexp) {
    return regexp.test(logLine?.message);
  }

  closePopup() {
    this.closeEvent.emit(true);
  }
}
