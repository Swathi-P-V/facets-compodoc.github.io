import {
  Component,
  OnInit,
  ViewChild,
  TemplateRef,
  ElementRef,
} from '@angular/core';
import { FormGroup, Validators, FormBuilder } from '@angular/forms';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import { UiUserVcsTokenControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { UserVCSTokenRequest } from './../../../cc-api/models/user-vcstoken-request';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const emptyStateLogo = './assets/token-empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-github-credential',
  templateUrl: './github-credential.component.html',
  styleUrls: ['./github-credential.component.scss'],
})
export class GithubCredentialComponent extends FacetsPage {
  @ViewChild('deleteCredentialConfirm') deleteCredentialConfirm: ElementRef;
  @ViewChild('credentialCreateEdit', { static: true })
  private credentialCreateEdit: TemplateRef<any>;
  emptyStateLogo: string = emptyStateLogo;
  primaryMsg: string = 'No Github Credentials found';
  secondaryMsg: string =
    'Map credentials with respective host name to enable workspace creation';
  buttonText: string = 'Map Credentials';
  showBtn: boolean = true;
  status: 'info' | 'error' = 'info';
  showEmptyState: boolean = false;
  loadingCredentials: boolean = false;
  hostNameOptions: {
    label: string;
    value: string;
  }[] = [];
  deletionInProgress: boolean = false;
  createUpdateInProgress: boolean = false;
  githubCredentialsForm: FormGroup = this.fb.group({
    hostName: ['', [Validators.required]],
    userName: ['', [Validators.required]],
    accessToken: ['', [Validators.required]],
  });
  credentialsList: any = [];
  mode: 'create' | 'update' = 'create';
  credentialsDetails: UserVCSTokenRequest = {};

  smartTableCustomActionButtons = [
    {
      id: 'edit',
      icon: 'edit-2-outline',
      toolTip: 'Edit Credential',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: undefined,
      onClick: (data) => {
        this.editCredential(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete Credential',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: undefined,
      onClick: (data) => {
        this.openDeletePopup(data, this.deleteCredentialConfirm);
      },
    },
  ];

  tableSettings: any = {
    columns: {
      hostName: {
        title: 'Enterprise Host Name',
        filter: false,
        valuePrepareFunction: (value) => {
          return value || 'N/A';
        },
      },
      userName: {
        title: 'User Name',
        filter: false,
        valuePrepareFunction: (value) => {
          return value || 'N/A';
        },
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {
            smartTableCustomActionButtons: this.smartTableCustomActionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton?.id === data?.actionButton?.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '12%',
        filter: false,
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
      perPage: 15,
    },
    mode: 'external',
    actions: false,
  };

  constructor(
    private activatedroute: ActivatedRoute,
    private router: Router,
    private userVcsTokenController: UiUserVcsTokenControllerService,
    private dialogService: NbDialogService,
    private fb: FormBuilder,
    private customToast: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedroute);
  }

  get hostName() {
    return this.githubCredentialsForm.get('hostName');
  }

  get userName() {
    return this.githubCredentialsForm.get('userName');
  }

  get accessToken() {
    return this.githubCredentialsForm.get('accessToken');
  }

  OnPageLoad(p: Params): void {
    this.getCredentials();
  }

  getCredentials() {
    this.loadingCredentials = true;
    this.showEmptyState = false;
    this.credentialsList = [];
    this.userVcsTokenController.getAllUsingGET4().subscribe(
      (data) => {
        let filteredCredentials = data.filter((cred) => cred?.id);
        if (data?.length > 0) {
          this.processHostNames(data);
        }
        if (filteredCredentials?.length) {
          this.credentialsList = [...filteredCredentials];
        } else {
          this.showEmptyState = true;
        }
        this.loadingCredentials = false;
      },
      (error) => {
        this.loadingCredentials = false;
        this.showEmptyState = true;
        this.secondaryMsg =
          error?.error?.message || ' There is some error fetching credentials';
        this.primaryMsg = '';
        this.showBtn = false;
        this.emptyStateLogo = errorIcon;
        this.status = 'error';
      },
    );
  }

  processHostNames(data) {
    this.hostNameOptions = data.map((credential) => {
      return {
        label: credential?.hostName,
        value: credential?.hostName,
      };
    });
  }

  openCreatePopup(credentialCreate) {
    this.mode = 'create';
    this.githubCredentialsForm.patchValue({
      hostName: '',
      userName: '',
      accessToken: null,
    });
    this.dialogService.open(credentialCreate, {
      context: {},
    });
  }

  createEditCredentials(ref) {
    this.createUpdateInProgress = true;
    const isCreateMode = this.mode === 'create';

    let credentialsParams: UserVCSTokenRequest = {
      hostName: this.hostName?.value,
      userName: this.userName?.value,
      personalAccessToken: this.accessToken?.value,
    };

    let httpReq;
    if (isCreateMode) {
      httpReq = this.userVcsTokenController.addUsingPOST1(credentialsParams);
    } else if (this.mode === 'update') {
      httpReq = this.userVcsTokenController.updateUsingPUT1({
        id: this.credentialsDetails?.id,
        ...credentialsParams,
      });
    }

    const successText = isCreateMode ? 'added' : 'updated';
    const failedText = isCreateMode ? 'addition' : 'update';
    httpReq.subscribe(
      (data) => {
        this.customToast.success(`Credentials ${successText} successfully`);
        ref.close();
        this.createUpdateInProgress = false;
        this.githubCredentialsForm.reset();
        this.getCredentials();
      },
      (error) => {
        this.createUpdateInProgress = false;
        this.onAPIErrorService.onAPIError(
          error?.error?.message || `Credentials ${failedText} failed`,
          error,
        );
        ref.close();
      },
    );
  }

  editCredential(data) {
    this.mode = 'update';
    this.githubCredentialsForm.patchValue({
      hostName: data?.hostName,
      userName: data?.userName,
      accessToken: null,
    });
    this.credentialsDetails = { ...data };
    this.dialogService.open(this.credentialCreateEdit, {
      context: { data: data },
    });
  }

  openDeletePopup(data, deleteDialog) {
    this.dialogService.open(deleteDialog, {
      context: { data: data },
    });
  }

  deleteCredential(data, ref) {
    this.deletionInProgress = true;
    this.userVcsTokenController.deleteUsingDELETE2(data?.id).subscribe(
      (data) => {
        this.customToast.success(`Credential deleted successfully`);
        this.deletionInProgress = false;
        ref.close();
        this.getCredentials();
      },
      (error) => {
        this.onAPIErrorService.onAPIError(
          error?.error?.message || `Credentials deletion failed`,
          error,
        );
        this.deletionInProgress = false;
        ref.close();
      },
    );
  }
}
