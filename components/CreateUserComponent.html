import { Component, Input, OnDestroy, OnInit, ViewChild } from '@angular/core';
import { NbPopoverDirective } from '@nebular/theme';
import mixpanel from 'mixpanel-browser';
import { User } from 'src/app/cc-api/models/user';
import { ApplicationControllerService } from 'src/app/cc-api/services/application-controller.service';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  UiCustomRoleControllerService,
  UiStackControllerService,
  UiUserControllerService,
  UiUserGroupControllerService,
} from 'src/app/cc-api/services';
import { Subscription, distinctUntilChanged } from 'rxjs';
import { RoleMapping } from 'src/app/cc-api/models';
import { USER_WRITE } from 'src/app/constants/permissions';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
import {
  USER_CREATE_FAIL,
  USER_CREATE_SUCCESS,
  USER_EDIT_FAIL,
  USER_EDIT_SUCCESS,
} from 'src/app/constants/analytics-track-events';

@Component({
  templateUrl: './create-user.component.html',
  styleUrls: ['./create-user.component.scss'],
})
export class CreateUserComponent extends FacetsPage implements OnDestroy {
  mode: 'create' | 'edit' = 'create'; // Decides which mode is the component currently in use in
  pageHeading = {
    create: 'Create User',
    edit: 'Edit User',
  };

  baseRoles: Array<RoleMapping> = []; // All base roles for user form [passed in from parent through state]
  accessibleEnvironments: any = {}; // All [Stack -> Clusters] mapping, [passed in from parent through state]
  userGroupDropdownOptions;

  existingUser: User; // If the form is opened in 'edit' or 'changePassword' mode, this stores the existing user's data
  existingUserId: string;
  userModel: FormGroup; // Form Group for all the user related form events
  showPassword: boolean = false; // Toggle for showing and hiding password for 'create' and 'changePassword' modes
  loadingUserForm: boolean = true;
  showDefRoles: boolean = false;
  isUserAssociatedToGroup: boolean = false;
  accessPriviligesList;
  @ViewChild(NbPopoverDirective) popover: NbPopoverDirective;
  baseRoleMutationObserver: MutationObserver;
  cdkOverlayContainerNode: any;
  selectedRole: any;
  loading: boolean;
  userWrite = USER_WRITE;

  // Hint string for additionalRole desc
  additionalRoleHints = {
    K8S_READER: 'Can download K8s credentials and do read operations on K8s',
    K8S_DEBUGGER: 'Can download K8s credentials and do write operations on K8s',
    CLI_ARTIFACT_PUSH: 'Can push artifacts using CLI',
  };
  // private variable to hold all your subscriptions for the component
  private subscriptions: Subscription[] = [];

  // This component subscribes to the window.history.state API to catch the baseRole, additonalRoles, accessibleEnvs and user data from the
  // parent component to avoid making the same network calls - going back to the parent (user management) component, all of this is fetched again
  constructor(
    private applicationController: ApplicationControllerService,
    private router: Router,
    private _route: ActivatedRoute,
    private userGroupControllerService: UiUserGroupControllerService,
    private usersController: UiUserControllerService,
    private uiStackControllerService: UiStackControllerService,
    private customRoleControllerService: UiCustomRoleControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.asyncOnPageLoad(p);
  }

  async asyncOnPageLoad(params) {
    this.loadingUserForm = true;
    this.existingUserId = params?.userId;
    await this.getUserGroups();
    this.initUserModel();
    //If URL has userId then set mode to edit. Retrieve the user details and fill the form
    if (this.existingUserId) {
      this.mode = 'edit';
      //Remove all validators for password field in edit mode.
      //minLength and maxLength validators is no more needed for password based users as set password exceeds maxLength.
      //required validator is no more needed for oAuth users as their password field is empty.
      this.userModel.get('password').clearValidators();
      Promise.all([this.getAllEnvironments, this.getAllRoles]).then(
        async (values) => {
          await this.getAllUsers();
          this.dynamicGroupNameAndRoleValidation();
          this.patchUserModel(this.existingUser);
        },
      );
    }
  }

  // Initializes the userModel form for 'edit' and 'create' mode
  initUserModel() {
    this.userModel = new FormGroup({
      userName: new FormControl('', [Validators.required, Validators.email]),
      password: new FormControl('', [
        Validators.required,
        Validators.minLength(8),
        Validators.maxLength(20),
        Validators.pattern(
          '^(?=.*[a-z])(?=.*[A-Z])(?=.*[0-9])(?=.*[!@#$%^&*])[a-zA-Z0-9!@#$%^&*s]*',
        ),
      ]),
      groupName: new FormControl(
        '',
        this.mode === 'create' ? [Validators.required] : [],
      ),
      baseRole: new FormControl(''),
      accessibleClusters: new FormControl([]),
    });
    if (!this.existingUser) {
      this.loadingUserForm = false;
    }
  }

  // In 'edit' mode the user details is prepopulated into the form fields
  patchUserModel(user?) {
    this.userModel.patchValue({
      userName: user.userName,
      password: user.password,
      groupName: user.groupId,
      baseRole: user.roles?.filter(
        (r) => this.baseRoles.map((r2) => r2?.roleName).indexOf(r) != -1,
      )[0],
      accessibleClusters: user.clusterIds || [],
    });
    this.loadingUserForm = false;
  }

  //This function fetches all users and filters out the user to be edited
  getAllUsers() {
    return new Promise((resolve, reject) => {
      this.usersController.getAllUsersUsingGET().subscribe(
        (users) => {
          let allUsers = users;
          this.existingUser = allUsers.filter(
            (user) => user.id === this.existingUserId,
          )[0];
          this.existingUser.groupId
            ? (this.isUserAssociatedToGroup = true)
            : (this.isUserAssociatedToGroup = false);

          //Check if deployerOldRoles exists - capillary has some old roles set. Preserve these.
          let deployerOldRoles = this.existingUser.roles.filter(
            (r) => this.baseRoles.map((r2) => r2?.roleName).indexOf(r) == -1,
          );
          if (deployerOldRoles?.length > 0) {
            this.existingUser['deployerOldRoles'] = deployerOldRoles;
          }
          resolve(true);
        },
        (error) => {
          console.log('Error ', error);
          reject(error);
        },
      );
    });
  }

  //Used for group name dropdown
  getUserGroups() {
    return new Promise((resolve, reject) => {
      this.userGroupControllerService.getAllGroupUsingGET().subscribe(
        (data) => {
          this.userGroupDropdownOptions = data.map((group) => {
            return { label: group.groupName, value: group.id };
          });
          //Sort the user group dropdown
          this.userGroupDropdownOptions.sort((a, b) =>
            a?.label?.localeCompare(b?.label),
          );
          resolve(true);
        },
        (error) => reject(error),
      );
    });
  }

  //Get the base roles and additional roles dropdown values
  getAllRoles = new Promise((resolve, reject) => {
    this.customRoleControllerService.getAllRolesUsingGET().subscribe(
      (roles) => {
        this.baseRoles = roles.filter((r) => r.baseRole);
        resolve(true);
      },
      (error) => {
        console.log('Error fetching all roles ', error);
        reject(error);
      },
    );
  });

  //get all environments for selecting accessible envs
  getAllEnvironments = new Promise((resolve, reject) => {
    this.uiStackControllerService
      .getAllClustersUsingGET({
        page: 0,
        perPage: 200,
        sortBy: 'ID',
      })
      .subscribe(
        (envs) => {
          envs.content.forEach((cluster) => {
            let stackName = cluster.stackName;
            if (this.accessibleEnvironments.hasOwnProperty(stackName)) {
              this.accessibleEnvironments[stackName].push(cluster);
            } else {
              this.accessibleEnvironments[stackName] = [cluster];
            }
          });
          resolve(true);
        },
        (error) => {
          reject(error);
        },
      );
  });

  get groupName() {
    return this.userModel.get('groupName');
  }

  get baseRole() {
    return this.userModel.get('baseRole');
  }

  get password() {
    return this.userModel.get('password');
  }

  get userName() {
    return this.userModel.get('userName');
  }

  get accessibleClusters() {
    return this.userModel.get('accessibleClusters');
  }

  setMutationObserver() {
    this.cdkOverlayContainerNode = document.querySelector(
      '.cdk-overlay-container',
    );
    this.baseRoleMutationObserver = new MutationObserver((mutations) => {
      mutations.forEach((mutation) => {
        if (mutation.addedNodes.length == 0) {
          this.baseRoleMutationObserver.disconnect();
          //Close popover whenever select dropdown closes
          //Select dropdown closes if any option is selected or user clicks anywhere outside
          this.popover?.hide();
        }
      });
    });
    this.baseRoleMutationObserver.observe(this.cdkOverlayContainerNode, {
      attributes: true,
      childList: true,
      characterData: true,
    });
  }

  // Makes a call to get the hint for the baseRole whenever the info icon is hovered over
  // [Should probably be cached in the future]
  // [baseRoleHints$ takes care of cancelling an API call if the user moves to another base role's info icon]
  getUserAccessPrivileges(baseRole, event) {
    //Stop the click event from propogating to parent elements
    event.stopPropagation();
    this.popover.show();
    this.accessPriviligesList = [];
    this.selectedRole = baseRole?.label;
    this.accessPriviligesList = baseRole?.entities;
  }

  // Redirects back to the parent component [user management]
  backToUserManagement = () => {
    this.router.navigate(['/settings/users']);
  };

  // Maps password input box's type
  getInputType() {
    if (this.showPassword) {
      return 'text';
    }
    return 'password';
  }

  // Toggles show/hide for password input box
  toggleShowPassword() {
    this.showPassword = !this.showPassword;
  }

  // Maps the form back to the required user object and calls the needed function [editUser, createUser, updatePassword]
  submitForm() {
    this.loading = true;
    let user: any = new (class implements User {
      id: string;
      password: string;
      picture: string;
      roles: Array<string>;
      teams: Array<string>;
      userName: string;
      clusterIds: Array<string>;
      groupId: string;
    })();
    user.id = this.existingUser ? this.existingUser.id : null;
    user.password = this.userModel.get('password').value;
    user.userName = this.userModel.get('userName').value;
    user.picture = this.existingUser ? this.existingUser.picture : null;
    user.teams = this.existingUser ? this.existingUser.teams : null;
    //Remove null values that might have been added by clearing base role selection
    let roles = [this.userModel.get('baseRole').value].filter(
      (role) => role !== null && role !== undefined && role !== '',
    );
    user.roles = roles;
    if (this.mode === 'edit' && this.existingUser['deployerOldRoles']) {
      user.roles.push(...this.existingUser['deployerOldRoles']);
    }
    user.clusterIds = this.userModel.get('accessibleClusters').value;
    user.groupId = this.userModel.get('groupName').value;
    this.mode === 'create' ? this.createUser(user) : this.editUser(user);
  }

  // Calls backend to add new user
  createUser(user) {
    delete user['id'];
    this.applicationController.createUserCCUsingPOST(user).subscribe(
      (rsp) => {
        this.loading = false;
        mixpanel.track(USER_CREATE_SUCCESS);
        this.customToastService.success(
          `New user '${user?.userName}' created successfully`,
        );
        this.backToUserManagement();
      },
      (err) => {
        this.loading = false;
        mixpanel.track(USER_CREATE_FAIL, { error: err });
        this.onAPIErrorService.onAPIError(
          err?.error?.message || 'Could not create new user',
          err,
        );
      },
    );
  }

  // Calls backend to edit exisiting user
  editUser(user) {
    delete user['password'];
    user['userId'] = user.id;

    //If you are in edit mode and user chooses groupId, then reset clusterIds
    if (user['groupId']) {
      if (!user['roles']?.length) {
        user['roles'] = ['CC-GUEST'];
      }
      user['clusterIds'] = [];
    }

    this.applicationController
      .updateUserUsingPUT({ user: user, userId: user.id })
      .subscribe(
        (rsp) => {
          this.loading = false;
          mixpanel.track(USER_EDIT_SUCCESS);
          this.customToastService.success(
            `Updated '${user.userName}' successfully`,
          );
          this.backToUserManagement();
        },
        (err) => {
          this.loading = false;
          mixpanel.track(USER_EDIT_FAIL);
          this.onAPIErrorService.onAPIError(
            err?.error?.message || 'Failed to update user',
            err,
          );
        },
      );
  }

  dynamicGroupNameAndRoleValidation() {
    this.subscriptions.push(
      this.userModel
        .get('groupName')
        .valueChanges.pipe(distinctUntilChanged())
        .subscribe((value) => {
          if (value) {
            this.userModel.get('baseRole').clearValidators();
            this.showDefRoles = false;
          } else {
            this.userModel.get('baseRole').setValidators([Validators.required]);
            this.showDefRoles = true;
          }
          this.userModel.get('baseRole').updateValueAndValidity();
        }),
    );
    this.subscriptions.push(
      this.userModel
        .get('baseRole')
        .valueChanges.pipe(distinctUntilChanged())
        .subscribe((value) => {
          if (value) {
            this.userModel.get('groupName').clearValidators();
          } else {
            this.userModel
              .get('groupName')
              .setValidators([Validators.required]);
          }
          this.userModel.get('groupName').updateValueAndValidity();
        }),
    );
  }

  ngOnDestroy() {
    this.baseRoleMutationObserver?.disconnect();
    if (this.subscriptions && this.subscriptions.length > 0) {
      this.subscriptions.forEach((s) => s.unsubscribe());
    }
  }
}
