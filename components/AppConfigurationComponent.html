import { Component } from '@angular/core';
import { ActivatedRoute, Params } from '@angular/router';
import { LocalDataSource } from 'ng2-smart-table';
import { UiApplicationControllerService } from 'src/app/cc-api/services';
import { OverrideStatusBadgeComponent } from 'src/app/components/override-status-badge/override-status-badge.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { getEffectiveValue, getUserVars } from 'src/app/utils/app-config-utils';
const emptyIcon = './assets/empty-override.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-app-configuration',
  templateUrl: './app-configuration.component.html',
  styleUrls: ['./app-configuration.component.scss'],
})
export class AppConfigurationComponent extends FacetsPage {
  objectKeys = Object.keys;
  clusterId: string;
  resourceName: string;
  stackName: string;
  resourceType: string;
  userVarsCount: number = 0;
  addOverrideSpinner = false;
  appJson: any;
  appJsonVersion2 = true;
  overrideJson: any;
  userDefinedVariables: LocalDataSource = new LocalDataSource();
  configData: any = {};
  containerConfigurationModel: any = {};
  fetchingApplicationDetails = false;
  fetchingOverrides = false;

  icon = emptyIcon;
  primaryMsg: string = 'No variables found!';
  errMsg: string = '';
  status: 'info' | 'error' = 'info';
  settings = {
    columns: {
      name: {
        title: 'Variable Name',
        filter: true,
        width: '40%',
        editable: false,
        sort: true,
      },
      value: {
        title: 'Value',
        filter: true,
        width: '40%',
        editable: false,
      },
      overridden: {
        type: 'custom',
        renderComponent: OverrideStatusBadgeComponent,
        title: 'Override Status',
        filter: {
          type: 'list',
          config: {
            selectText: 'Status',
            list: [
              { value: 'NO CHANGE', title: 'No change' },
              { value: 'OVERRIDDEN', title: 'Overridden' },
            ],
          },
        },
        width: '20%',
        editable: false,
      },
    },
    noDataMessage: 'No variables found',
    pager: {
      display: false,
      perPage: 10,
    },
    edit: {
      inputClass: '',
      editButtonContent: '<i class="eva-edit-outline eva"></i>',
      saveButtonContent: '<i class="eva-checkmark-outline eva"></i>',
      cancelButtonContent: '<i class="eva-close-outline eva"></i>',
      confirmSave: false,
    },
    actions: {
      add: false,
      edit: false,
      delete: false,
      position: 'right',
    },
  };

  configKeyPaths = {
    disabled: { paths: ['disabled'], defaultValue: false },
    size: { paths: ['size', 'spec.runtime.size.value'], defaultValue: 'NA' },
    resourceAllocationStrategy: {
      paths: ['resourceAllocationStrategy', 'spec.runtime.size.namespace'],
      defaultValue: 'NA',
    },
    hpaMinReplicas: {
      paths: ['scaling.hpaMinReplicas', 'spec.runtime.autoscaling.min'],
      defaultValue: 'NA',
    },
    hpaMaxReplicas: {
      paths: ['scaling.hpaMaxReplicas', 'spec.runtime.autoscaling.max'],
      defaultValue: 'NA',
    },
    hpaMetricThreshold: {
      paths: [
        'scaling.hpaMetricThreshold',
        'spec.runtime.autoscaling.cpuThreshold',
      ],
      defaultValue: 'NA',
    },
  };

  constructor(
    _route: ActivatedRoute,
    private uiApplicationControllerService: UiApplicationControllerService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p.clusterId;
    this.resourceName = p.resourceName;
    this.stackName = p.stackName;
    this.resourceType = p.resourceType;
    this.fetchingApplicationDetails = true;
    this.configData = {};
    this.containerConfigurationModel = {};
    this.overrideJson = undefined;
    this.appJson = undefined;
    this.userDefinedVariables = new LocalDataSource();
    this.userVarsCount = 0;
    this.uiApplicationControllerService
      .getResourceByNameV2UsingGET({
        resourceName: this.resourceName,
        clusterId: this.clusterId,
        resourceType: this.resourceType,
      })
      .subscribe(
        (json) => {
          this.appJson = json;
          if ('v2' !== this.appJson?.apiVersion) {
            this.appJsonVersion2 = false;
          }
          this.fetchingOverrides = true;
          this.fetchingApplicationDetails = false;
          this.uiApplicationControllerService
            .getResourceOverrideObjectUsingGET({
              resourceName: this.resourceName,
              resourceType: this.resourceType,
              clusterId: this.clusterId,
            })
            .subscribe(
              (json) => {
                this.overrideJson = json?.overrides || {};
                this.fetchingOverrides = false;
              },
              (err) => (this.fetchingOverrides = false),
            )
            .add(() => {
              this.loadDataForConfiguration(this.appJson, this.overrideJson);
              this.populateUserVars(this.appJson, this.overrideJson);
            });
        },
        (error) => {
          this.loadDataForConfiguration(this.appJson, this.overrideJson);
          this.populateUserVars(this.appJson, this.overrideJson);
          this.fetchingApplicationDetails = false;
          this.fetchingOverrides = false;
          this.icon = errorIcon;
          this.primaryMsg = '';
          this.errMsg =
            error?.error?.message || ' There is some error fetching resource';
          this.status = 'error';
        },
      );
  }

  loadDataForConfiguration(json: any, overrides: any) {
    Object.keys(this.configKeyPaths).forEach((key) => {
      this.configData[key] = getEffectiveValue(
        this.configKeyPaths[key].paths,
        json,
        overrides,
        this.configKeyPaths[key].defaultValue,
      );
    });
  }

  populateUserVars(json: any, overrides: any) {
    const { varsCount, dataSource = [] } = getUserVars(
      json,
      overrides,
      this.appJsonVersion2,
    );
    this.userVarsCount = varsCount;
    this.userDefinedVariables.load(dataSource);
  }
}
