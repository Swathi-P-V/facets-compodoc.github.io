import { Component, ViewChild, TemplateRef, ElementRef } from '@angular/core';
import { LocalDataSource } from 'ng2-smart-table';
import { NbDialogService } from '@nebular/theme';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CanDeactivateComponent } from 'src/app/can-deactivate.guard';
import { SaveChangesFooterComponent } from 'src/app/components/save-changes-footer/save-changes-footer.component';
import { STACK_WRITE } from 'src/app/constants/permissions';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { variableSettings } from 'src/app/constants/secrets-variables-config';
import {
  UiStackControllerService,
  UiBlueprintDesignerControllerService,
} from 'src/app/cc-api/services';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { Stack } from 'src/app/cc-api/models';
import { getVariablesChanged } from './stack-vars-utils';
import { ContextUpdatesService } from 'src/app/services/context-updates.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const emptyStateLogo = './assets/secret-empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-stack-vars',
  templateUrl: './stack-vars.component.html',
  styleUrls: ['./stack-vars.component.scss'],
})
export class StackVarsComponent
  extends FacetsPage
  implements CanDeactivateComponent
{
  @ViewChild('deleteVariableConfirm') deleteVariableConfirm: ElementRef;
  stackVarsSource: LocalDataSource = new LocalDataSource();
  variablesChanged: {
    oldValue: any;
    newValue: any;
    entity: string;
    name: string;
  }[] = [];
  variablesChangedSource: LocalDataSource = new LocalDataSource();
  settings = variableSettings;
  addSpinner: boolean = false;
  stackName: any;
  originalData: any;
  emptyStateLogo = emptyStateLogo;
  errorLogo = errorIcon;
  deletionInProgress: boolean = false;
  showVariableUpdateSpinner: boolean = false;
  actionButtons = [
    {
      id: 'add-variable',
      title: 'Add Secret',
      buttonType: 'primary',
      status: 'primary',
      nbTooltip: 'Add new secret or variable to the stack',
      icon: 'plus-outline',
      loading: false,
      hidden: false,
      onClick: (id) => {
        this.goToAddSecret(id);
      },
    },
    {
      id: 'sync-git',
      title: 'Sync with Git',
      buttonType: 'secondary',
      status: 'primary',
      nbTooltip:
        'Sync to view the newly added secrets and variables in the blueprint',
      icon: 'sync-outline',
      loading: false,
      hidden: false,
      onClick: (id) => {
        this.refreshStack(id);
      },
    },
  ];
  changesTableSettings = {
    columns: {
      name: {
        title: 'Variable name',
        filter: true,
        width: '15%',
      },
      entity: {
        title: 'Entity',
        filter: false,
        width: '20%',
      },
      oldValue: {
        title: 'Old Value',
        filter: false,
        width: '20%',
        valuePrepareFunction: (value, row, cell) => {
          return this.getGlobalStatus(row, value);
        },
      },
      newValue: {
        title: 'New Value',
        width: '15%',
        filter: false,
        valuePrepareFunction: (value, row, cell) => {
          return this.getGlobalStatus(row, value);
        },
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
    },
    actions: false,
  };
  stack: Stack;
  stackConfigurePermission = STACK_WRITE;
  loadingStackDetails: boolean = false;

  @ViewChild(SaveChangesFooterComponent, { static: false })
  private footerComponent: SaveChangesFooterComponent;
  constructor(
    route: ActivatedRoute,
    private dialogService: NbDialogService,
    private router: Router,
    private uiStackControllerService: UiStackControllerService,
    private blueprintDesignerService: UiBlueprintDesignerControllerService,
    private contextUpdatesService: ContextUpdatesService,
    private customToast: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(route);
  }

  OnPageLoad(p: Params): void {
    if (p?.stackName) {
      this.stackName = p.stackName;
      this.setStackSecretsData();
    }
  }

  getGlobalStatus(row, value) {
    if (row?.entity === 'global') {
      return value ? 'Yes' : 'No';
    } else {
      return value;
    }
  }

  goToAddSecret(id) {
    this.router.navigate(['/', 'stack', this.stackName, 'variables', 'create']);
  }

  setStackSecretsData() {
    this.cleanSecretsVariables();
    this.loadingStackDetails = true;
    this.uiStackControllerService.getStackUsingGET(this.stackName).subscribe(
      (stack) => {
        this.stack = stack;
        const dataSource = [];
        const stackVarsData = this.stack['clusterVariablesMeta'];
        Object.keys(stackVarsData).forEach((key) => {
          stackVarsData[key]['name'] = key;
          stackVarsData[key]['type'] = stackVarsData?.[key]?.['secret']
            ? 'SECRET'
            : 'VARIABLE';
          stackVarsData[key]['globalStatus'] = stackVarsData?.[key]?.['global']
            ? 'Yes'
            : 'No';
          dataSource.push(stackVarsData?.[key]);
        });
        //this.originalData = dataSource;
        dataSource?.forEach((stackVar) => {
          this.originalData[stackVar?.name] = { ...stackVar };
        });
        this.stackVarsSource = new LocalDataSource();
        this.stackVarsSource.load(dataSource);
        this.loadingStackDetails = false;
      },
      (err) => {
        this.loadingStackDetails = false;
      },
    );
  }

  validateSecrets(variableChangeData) {
    /* 
      Check if event data's new value is not white space, then only process the value as valid
    */
    if (whiteSpaceInputPattern.test(variableChangeData?.newData?.value)) {
      this.variablesChanged = getVariablesChanged(
        variableChangeData,
        this.variablesChanged,
        this.originalData,
      );
      this.variablesChangedSource.load(this.variablesChanged);
      variableChangeData.confirm.resolve(variableChangeData?.newData);
    } else {
      variableChangeData.confirm.reject();
    }
  }

  saveStackVariables = async (ref: any) => {
    this.stackVarsSource?.getAll().then((variableList) => {
      this.showVariableUpdateSpinner = true;
      if (variableList?.length > 0) {
        const addVarsParams = {
          stackName: this.stackName,
          variables: {},
        };
        variableList?.forEach((variable) => {
          let varValue = variable?.value;
          if (
            !this.variablesChanged.filter(
              (val) => val?.name === variable?.name && val?.entity === 'value',
            )?.length
          ) {
            varValue = null;
          }
          addVarsParams.variables[variable['name']] = {
            value: varValue,
            global: variable['global'],
            description: variable['description'],
            secret: variable['type'] === 'SECRET',
          };
        });
        this.blueprintDesignerService
          .updateVariablesUsingPUT(addVarsParams)
          .subscribe(
            (data) => {
              this.customToast.success(`Variables updated successfully`);
              this.showVariableUpdateSpinner = false;
              ref.close();
              this.setStackSecretsData();
            },
            (error) => {
              this.showVariableUpdateSpinner = false;
              this.onAPIErrorService.onAPIError(
                error?.error?.message || `Variable updation failed`,
                error,
              );
              ref.close();
            },
          );
      }
    });
  };

  openConfirmation(confirmation, ref = null) {
    if (ref) {
      ref.close('confirm');
    }
    this.dialogService.open(confirmation, {});
  }

  openConfirmRemove(confirmRemove) {
    this.dialogService.open(confirmRemove, {});
  }

  cleanSecretsVariables = (ref?) => {
    ref?.close('discard');
    this.variablesChanged = [];
    this.originalData = {};
  };

  async canDeactivate(): Promise<boolean> {
    if (this.variablesChanged.length === 0) {
      return true;
    }

    return this.footerComponent.onDeactivate();
  }

  handlePageActions(event) {
    try {
      this.actionButtons.find((button) => button?.id === event)?.onClick(event);
    } catch (error) {
      console.log('Error while handling button click', error);
    }
  }

  deleteSecretsConfirm(event, deleteVarPopup) {
    const data = event?.data;
    data.promiseFn = event?.confirm;
    this.dialogService.open(deleteVarPopup, {
      context: {
        data: event?.data,
      },
    });
  }

  refreshStack(id) {
    let gitSyncButton = this.actionButtons.find((button) => button?.id === id);
    gitSyncButton.loading = true;
    this.uiStackControllerService
      .reloadStackUsingGET1(this.stackName)
      .subscribe(
        (stack) => {
          gitSyncButton.loading = false;
          this.customToast.success(`Synced with git`);
          this.cleanSecretsVariables();
          this.setStackSecretsData();
        },
        (error) => {
          gitSyncButton.loading = false;
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Syncing with git failed',
            error,
          );
        },
      );
  }

  handleBack = () => {
    this.router.navigate([
      '/',
      'stack',
      this.stackName,
      'blueprint-designer',
      this.contextUpdatesService.getLastBranchForBlueprint(this.stackName),
      'resource-list',
    ]);
  };

  deleteVariable(event, ref) {
    const deleteParams = {
      stackName: this.stackName,
      variables: [event?.name],
    };
    this.deletionInProgress = true;
    this.blueprintDesignerService
      .deleteVariablesUsingDELETE(deleteParams)
      .subscribe(
        (data) => {
          this.customToast.success(
            `Variable ${event?.name} deleted successfully`,
          );
          this.deletionInProgress = false;
          ref.close();
          this.setStackSecretsData();
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            error?.error?.message || `Variable ${event?.name} deletion failed`,
            error,
          );
          this.deletionInProgress = false;
          ref.close();
        },
      );
  }
}
