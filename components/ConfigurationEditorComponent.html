import {
  Component,
  EventEmitter,
  Input,
  OnChanges,
  OnDestroy,
  Output,
  SimpleChanges,
  ViewChild,
} from '@angular/core';
import {
  MonacoDiffEditorConstructionOptions,
  MonacoEditorConstructionOptions,
} from '@materia-ui/ngx-monaco-editor';
import { cloneDeep } from 'lodash';
import { UiApplicationControllerService } from 'src/app/cc-api/services';
import { ENVIRONMENT_CONFIGURE } from 'src/app/constants/permissions';
import { findDifferences } from '../../../resource-config-edit/components/simple-resource-config-edit/config-diff-utils';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { ActionEventEmitterService } from 'src/app/services/action-event-emitter.service';
import { actionEvents } from '../../../../../constants/action-events';

@Component({
  selector: 'app-configuration-editor',
  templateUrl: './configuration-editor.component.html',
  styleUrls: ['./configuration-editor.component.scss'],
})
export class ConfigurationEditorComponent implements OnChanges {
  @Input() resourceName: string;
  @Input() stackName: string;
  @Input() clusterId: string;
  @Input() resourceType: string;
  @Output() byClose: EventEmitter<any> = new EventEmitter<any>();

  blueprintCode: any;
  overrides: any;
  environmentCode: any;
  environmentCodeBackup: any;
  schemaLink: string;
  fetchingBlueprintCode: boolean = false;
  fetchingOverrides: boolean = false;
  changesMade: boolean = false;
  environmentConfigurePermission = ENVIRONMENT_CONFIGURE;
  savingOverrides: boolean = false;
  showAppConfig: boolean = false;
  diffEditorOptions: MonacoDiffEditorConstructionOptions = {
    theme: 'vs-light',
    readOnly: true,
    minimap: {
      enabled: false,
    },
  };
  readOnlyEditorOptions: MonacoEditorConstructionOptions = {
    readOnly: true,
  };

  constructor(
    private appController: UiApplicationControllerService,
    private customToasterService: CustomToastService,
    private actionEventEmitterService: ActionEventEmitterService,
  ) {}

  ngOnChanges(changes: SimpleChanges): void {
    if (changes) {
      this.stackName = changes?.stackName?.currentValue;
      this.clusterId = changes?.clusterId?.currentValue;
      this.resourceType = changes?.resourceType?.currentValue;
      this.resourceName = changes?.resourceName?.currentValue;
      this.fetchConfig();
    }
  }

  fetchConfig() {
    this.showAppConfig =
      this.resourceType === 'application' ||
      this.resourceType === 'windows-application';
    this.fetchingBlueprintCode = true;
    this.fetchingOverrides = true;
    this.appController
      .getResourceByNameV2UsingGET({
        resourceName: this.resourceName,
        resourceType: this.resourceType,
        clusterId: this.clusterId,
      })
      .subscribe((data: any) => {
        this.blueprintCode = JSON.stringify(data, null, 2);
        this.schemaLink = data['$schema'];
        this.fetchingBlueprintCode = false;
        this.fetchingOverrides = true;
        this.appController
          .getResourceOverrideObjectUsingGET({
            resourceName: this.resourceName,
            resourceType: this.resourceType,
            clusterId: this.clusterId,
          })
          .subscribe(
            (json) => {
              this.overrides = json?.overrides;
            },
            (error) => {
              this.overrides = {};
            },
          )
          .add(() => {
            this.environmentCode = JSON.stringify(
              this.mergeOverrides(
                cloneDeep(this.parseJSON(this.blueprintCode)),
                cloneDeep(this.overrides),
              ),
              null,
              2,
            );
            this.environmentCodeBackup = cloneDeep(this.environmentCode);
            this.fetchingOverrides = false;
          });
      });
  }

  mergeOverrides(originalCode: any, overrides: any) {
    if (typeof originalCode !== 'object' || typeof overrides !== 'object') {
      return overrides;
    }

    if (
      JSON.stringify(overrides) === '[]' ||
      Array.isArray(overrides) ||
      JSON.stringify(overrides) === '{}'
    ) {
      return overrides;
    }

    for (const key in overrides) {
      if (originalCode.hasOwnProperty(key)) {
        originalCode[key] = this.mergeOverrides(
          originalCode[key],
          overrides[key],
        );
      } else {
        originalCode[key] = overrides[key];
      }
    }

    return originalCode;
  }

  mergedEnvironmentState: any;
  codeUpdates: number = 0;
  invalidCode: boolean = false;
  updateEnvironmentCode(code) {
    if (!code) {
      this.invalidCode = true;
      return;
    }
    this.invalidCode = false;
    this.codeUpdates++;
    this.environmentCode = code;
    this.changesMade =
      JSON.stringify(this.environmentCodeBackup) !==
      JSON.stringify(this.environmentCode);
  }

  onDiscard(ref) {
    if (ref) {
      ref.close();
    }
    this.byClose.emit();
  }

  submittingOverrides: boolean = false;
  environmentModified: any = {};
  submitOverrides(ref: any) {
    try {
      if (!this.overrides) {
        this.overrides = {
          resourceName: this.resourceType,
          resourceType: this.resourceType,
        };
      } else {
        this.overrides.resourceName = this.resourceType;
        this.overrides.resourceType = this.resourceType;
      }
      this.overrides.overrides = findDifferences(
        this.parseJSON(this.blueprintCode),
        this.parseJSON(this.environmentCode),
      );
      this.submittingOverrides = true;
      this.appController
        .postResourceOverrideObjectUsingPOST({
          resourceName: this.resourceName,
          resourceType: this.resourceType,
          overrideRequest: this.overrides,
          clusterId: this.clusterId,
        })
        .subscribe(
          (data) => {
            this.submittingOverrides = false;
            this.customToasterService.success(
              'Updated overrides, changes will be applied after next release',
            );
            this.sendOverrideCompleteEvent();
            ref.close();
            this.byClose.emit();
          },
          (err) => {
            this.submittingOverrides = false;
            if (
              (err.status === 500 || err.status === 403) &&
              err?.error?.message === 'Access is denied'
            ) {
              this.customToasterService.error(
                'Please contact your administrator for more details',
              );
            } else {
              this.customToasterService.error(
                err?.error?.message || 'There is some error while overriding',
              );
            }
          },
        );
    } catch (err) {
      this.submittingOverrides = false;
      this.customToasterService.error(err?.error?.message);
      ref.close();
    }
  }

  sendOverrideCompleteEvent() {
    this.actionEventEmitterService.emitActionEvent(
      actionEvents?.ON_SAVE_ENV_OVERRIDES,
    );
  }

  parseJSON(object) {
    let parsedResult;
    try {
      parsedResult = JSON.parse(object);
    } catch {
      parsedResult = {};
      this.customToasterService.error('An error occurred while parsing JSON');
    }
    return parsedResult;
  }
}

/*
Whenever we have an array present inside the configuration. What currently happens is,
that we take the overrides and just let the user to only edit the overrides object without 
merging it in the original configuration. 

What we need to do is: 
  1. blueprintCode = Fetch blueprint configuration.
  2. overrides = Fetch overrides.
  3. Merge blueprintCode and overrides.
  4. Let the user make changes to the environment configuration now.
*/
