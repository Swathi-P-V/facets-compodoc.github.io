import { Component, OnDestroy, TemplateRef, ViewChild } from '@angular/core';

@Component({

export class CreateEditSubscriptionNewComponent
  extends FacetsPage
  implements OnDestroy
{
  //Form groups
  subscriptionForm: FormGroup = this.fb.group({
    name: new FormControl('', [
      Validators.required,
      Validators.pattern(/[\S]/),
      Validators.minLength(4),
      Validators.maxLength(20),
    ]),
    notificationType: new FormControl('', [Validators.required]),
    stackName: new FormControl('', [Validators.required]),
    channelType: new FormControl('', [Validators.required]),
    channelId: new FormControl('', [Validators.required]),
    payloadJson: new FormControl(null),
  });
  additionalFiltersForm: FormGroup = new FormGroup({});
  //dropdowns
  notificationTypeDropdown: { label: string; value: string }[] = [];
  channelTypesDropdown: { label: string; value: string }[] = [];
  allChannelNamesMap: {
    label: string;
    value: string;
    type: string;
  }[] = [];
  blueprintsDropdown: { label: string; value: string }[] = [];
  channelNamesByTypeDropdown: { label: string; value: string }[] = [];
  // maps
  channelNameMap: {} = {};
  channelAddressMap: {} = {};
  integrationKeyMap: {} = {};
  availableFilters = [];
  filterNameIdMap = {};
  channelEmailMap = {};
  selectedFilters: {
    id: string;
    label: string;
    placeholder: string;
    options: [];
  }[] = [];
  mode: 'create' | 'edit' = 'create';
  editId: any;
  //Flags
  filtersLoaded: boolean = false;
  isDropdownOpen: boolean;
  //Loaders
  subCreationInProgress: boolean = false;
  notificationTestInProgress: boolean = false;
  loadingSubscriptionForm: boolean = false;
  addingFilterInProgress: boolean = false;
  //Constants
  filterIdNameMap = filterIdNameMap;
  filterIdErrorNameMap = filterIdErrorNameMap;
  headingOptions = headingOptions;
  //This is for setting monaco editor
  payload;
  editorOptions: MonacoEditorConstructionOptions = {
    theme: 'vs-light',
    language: 'json',
    readOnly: true,
    minimap: {
      enabled: false,
    },
  };
  fetchingResources: boolean = false;

  additionalFiltersDropdownObserver: MutationObserver;

  private currentPayloadDialog: TemplateRef<any>;
  skipPopulateEdit: boolean;
  subscriptionWrite = SUBSCRIPTION_WRITE;

  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private fb: FormBuilder,
    private uiStackControllerService: UiStackControllerService,
    private uiNotificationControllerService: UiNotificationControllerService,
    private dialogBoxService: NbDialogService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  get name() {
    return this.subscriptionForm.get('name');
  }

  get channelType() {
    return this.subscriptionForm.get('channelType');
  }

  get channelId() {
    return this.subscriptionForm.get('channelId');
  }

  get notificationType() {
    return this.subscriptionForm.get('notificationType');
  }

    return this.subscriptionForm.get('stackName');
  }

  get payloadJson() {
    return this.subscriptionForm.get('payloadJson');
  }

  OnPageLoad(params: Params): void {
    this.loadingSubscriptionForm = true;
    if (params) {
      this.mode = params?.mode;
      this.editId = params?.editId;
    }
    //Before loading form get all the dropdown values.
    Promise.all([
      this.getAllBlueprints(),
      this.getAllChannelTypes(),
      this.getAllNotificationTypes(),
      this.getAllChannels(),
    ]).then(async (data) => {
      this.setRouterStateListener();
      if (this.mode === 'edit' && !this.skipPopulateEdit) {
        await this.populateValuesForEdit();
      }
      this.loadingSubscriptionForm = false;
    });
  }

  populateValuesForEdit() {
    return new Promise((resolve, reject) => {
      this.uiNotificationControllerService
        .getSubscriptionUsingGET(this.editId)
        .subscribe(async (res) => {
          if (res) {
            let subscription: Subscription = res;
            //Populate the subscription form
            await this.patchSubscriptionForm(subscription);

            //Set payload
            this.payload = this.payloadJson.value;

            //Populate the additional filters dropdown
            await this.getNotificationTags(this.notificationType.value);

            //If there are additional filters selected then render it in form.
            if (Object.keys(subscription.filters).length > 1) {
              //Remove filters with empty values
              subscription.filters = this.removeEmptyFilters(
                subscription.filters,
              ) as { [key: string]: string[] };
              Object.entries(subscription.filters).forEach(
                ([key, value], index) => {
                  if (key !== 'STACK_NAME') {
                    this.getDropdownOptionsForFilter(
                      key,
                      this.notificationType.value,
                      subscription.filters,
                    ).then((data: []) => {
                      this.selectedFilters.push({
                        id: key,
                        label: this.filterIdNameMap[key],
                        placeholder: `Select ${this.filterIdNameMap[key]}`,
                        options: data,
                      });
                      this.additionalFiltersForm.addControl(
                        key,
                        new FormControl(value, [Validators.required]),
                      );
                      this.availableFilters = this.availableFilters.filter(
                        (f) => f.id !== key,
                      );
                      //Set disabled states for remaining filters
                      this.availableFilters.forEach((eachFilter) => {
                        eachFilter.disabled = this.isDisabled(eachFilter.id);
                      });
                      //Sort array
                      if (
                        subscription.filters.hasOwnProperty('CLUSTER_NAME') &&
                        subscription.filters.hasOwnProperty('CLUSTER_TYPE')
                      ) {
                        this.selectedFilters = this.sortFiltersForEditMode(
                          this.selectedFilters,
                        );
                      }
                    });
                  }
                  if (index === Object.keys(subscription.filters).length - 1) {
                    resolve(true);
                  }
                },
              );
            } else {
              resolve(true);
            }
          }
        });
    });
  }

  //This function is used to cache and pass data between create-edit-subscription component and add-payload component
  setRouterStateListener() {
    this.activatedRoute.paramMap
      .pipe(map(() => window.history.state))
      .subscribe(async (state) => {
        this.skipPopulateEdit =
          state.previousComponent === 'edit-subscription' ? true : false;
        if (state.subscription) {
          //Populate the subscription form
          await this.patchSubscriptionForm(state.subscription);

          if (this.stackName.value && this.notificationType.value) {
            //Populate the additional filters dropdown
            await this.getNotificationTags(this.notificationType.value);

            //If there are additional filters selected then render it in form.
            if (state.additionalFilters) {
              this.selectedFilters = state.selectedFilters;
              this.selectedFilters.forEach((filter) => {
                this.additionalFiltersForm.addControl(
                  filter.id,
                  new FormControl(state.additionalFilters[filter.id], [
                    Validators.required,
                  ]),
                );

                this.availableFilters = this.availableFilters.filter(
                  (f) => f.id !== filter.id,
                );
              });

              //Set disabled states for remaining filters
              this.availableFilters.forEach((eachFilter) => {
                eachFilter.disabled = this.isDisabled(eachFilter.id);
              });

              //If both alert and environment is added in form and environment is not set OR
              //If alert is added and environment is not added then reset and disable alert.
              let additionalFormValue =
                this.additionalFiltersForm.getRawValue();
              if (
                (additionalFormValue.hasOwnProperty('ALERT_NAME') &&
                  !additionalFormValue.hasOwnProperty('CLUSTER_NAME')) ||
                (additionalFormValue.hasOwnProperty('ALERT_NAME') &&
                  additionalFormValue.hasOwnProperty('CLUSTER_NAME') &&
                  additionalFormValue['CLUSTER_NAME'].length === 0)
              ) {
                this.setAlertsFormControlDisabled();
              }
            }
          }
        }

        if (state.payload != this.payloadJson.value) {
          this.subscriptionForm.patchValue({
            payloadJson: state.payload,
          });
          this.customToastService.success(`Payload updated successfully`);
        }
        this.payload = this.payloadJson.value;
      });
  }

  //Fetch all the Blueprints available for the control plane

  createSubscription() {
    this.subCreationInProgress = true;
    let httpReq;

    const subscriptionReq: Subscription = {
      ...this.subscriptionForm.value,
      channelAddress: this.channelAddressMap[this.channelId.value],
      test: false,
      filters: {
        ...this.removeEmptyFilters(this.additionalFiltersForm.value),
        STACK_NAME: [this.stackName.value],
      },
    };

    if (subscriptionReq.channelType !== 'WEBHOOK') {
      subscriptionReq.payloadJson = null;
    }

    if (this.mode === 'create') {
      httpReq =
        this.uiNotificationControllerService.createSubscriptionUsingPOST(
          subscriptionReq,
        );
    } else if (this.mode === 'edit') {
      httpReq = this.uiNotificationControllerService.editSubscriptionUsingPUT({
        subscription: subscriptionReq,
        subscriptionId: this.editId,
      });
    }

    const { successSubText, failureSubText } = getToastSubTexts(this.mode);
    httpReq.subscribe(
      (response) => {
        this.customToastService.success(
          `Successfully ${successSubText} subscription`,
        );
        this.subCreationInProgress = false;
        this.handleBack();
      },
      (error) => {
        this.onAPIErrorService.onAPIError(
          error?.error?.message || `Failed to ${failureSubText} subscription.`,
          error,
        );
        this.subCreationInProgress = false;
      },
    );
  }

  testNotifications() {
    this.notificationTestInProgress = true;

    let testNotificationReq: TestNotificationRequest = {
      channelAddress: this.channelAddressMap[this.channelId.value],
      channelType: this.channelType.value,
      integrationKey:
        this.channelType.value === 'PAGER_DUTY'

          : '',
      notificationType: this.notificationType.value,

        this.channelType.value == 'WEBHOOK' ? this.payloadJson.value : null,
      emailAddresses: this.channelEmailMap[this.channelId.value],
    };

    this.uiNotificationControllerService

      .subscribe(
        (response) => {
          this.notificationTestInProgress = false;
          if (response) {

  patchSubscriptionForm(subscription: Subscription) {
    return new Promise((resolve, reject) => {
      //Populate the subscription form
      this.subscriptionForm.patchValue({

        stackName: subscription.stackName,
        notificationType: subscription.notificationType,
        channelType: subscription.channelType,
        payloadJson: subscription.payloadJson,
      });

      //If channelType is set then fetch channelNames for given channelType and set its value
      if (this.channelType.value) {
        this.onChannelTypeChange();
        this.subscriptionForm.patchValue({
          channelId: subscription.channelId,
        });
      }
      resolve(true);
    });
  }

  //Whenever notificationType changes, the filters should be reloaded.
  //Filters should be loaded only when blueprint and notificationType has been set.
  async getFilters(load?: boolean) {
    this.addingFilterInProgress = load;
    if (this.notificationType.value && this.stackName.value) {
      await this.getNotificationTags(this.notificationType.value);
      this.addingFilterInProgress = false;
    }
  }

  //This function gets all the filters/tags for given notification type
  getNotificationTags(notificationType) {
    return new Promise((resolve, reject) => {
      this.uiNotificationControllerService
        .getNotificationTagsForNotificationTypeUsingGET(notificationType)
        .subscribe(
          (res) => {
            this.availableFilters = [];
            res?.forEach((tag) => {
              this.availableFilters.push({
                label: tag.label,
                id: tag.id,
                disabled: this.isDisabled(tag.id),
              });
              this.filterNameIdMap[tag.label] = tag.id;
            });
            resolve(true);
          },
          (error) => {
            //TODO - what to do if we do not get filters ?
            reject(error);
          },
        );
    });
  }

  //This function creates a form control for selected filter
  addFilterFormControl(id, label) {
    this.addingFilterInProgress = true;

    this.getDropdownOptionsForFilter(id, this.notificationType.value, {
      ...this.additionalFiltersForm.value,
      STACK_NAME: [this.stackName.value],
    }).then(
      (data: []) => {
        //Add this filter to selectedFilters array
        this.selectedFilters.push({
          id: id,
          label: label,
          placeholder: `Select ${label}`,
          options: data,
        });
        //Add form control to additionalFiltersForm
        this.additionalFiltersForm.addControl(
          id,
          new FormControl([], [Validators.required]),
        );
        //Remove this filter from availableFilters ie; it should no more be shown in add filters dropdown.
        this.availableFilters = this.availableFilters.filter(
          (eachFilter) => eachFilter.id !== id,
        );
        //Set disabled states for remaining availableFilters
        if (['CLUSTER_NAME', 'CLUSTER_TYPE'].includes(id)) {
          this.availableFilters.forEach((eachFilter) => {
            if (
              ['CLUSTER_NAME', 'CLUSTER_TYPE', 'ALERT_NAME'].includes(
                eachFilter.id,
              )
            ) {
              eachFilter.disabled = this.isDisabled(eachFilter.id);
            }
          });
        }

        this.addingFilterInProgress = false;
      },
      (error) => {
        this.addingFilterInProgress = false;
      },
    );
  }

  //This function removes form control when its corresponding delete button is clicked
  removerFilterFormControl(filter) {
    //Remove the form control
    this.additionalFiltersForm.removeControl(filter.id);

    //Remove from selectedFilters array
    this.selectedFilters = this.selectedFilters.filter(
      (eachFilter) => eachFilter.id !== filter.id,
    );

    //Add it back to available filters array
    this.availableFilters.push({
      label: filter.label,
      id: filter.id,
    });

    //Set disabled states for remaining availableFilters
    if (['CLUSTER_NAME', 'CLUSTER_TYPE', 'ALERT_NAME'].includes(filter.id)) {
      this.availableFilters.forEach((eachFilter) => {
        if (
          ['CLUSTER_NAME', 'CLUSTER_TYPE', 'ALERT_NAME'].includes(eachFilter.id)
        ) {
          eachFilter.disabled = this.isDisabled(eachFilter.id);
        }
      });
    }
    //If environment is removed and we have alerts selected then reset alerts and disable it.
    if (
      filter.id === 'CLUSTER_NAME' &&
      this.additionalFiltersForm.value.hasOwnProperty('ALERT_NAME')
    ) {
      this.setAlertsFormControlDisabled();
    }
    //change applications back to normal if user deletes the environment or release stream filter
    if (
      !this.availableFilters.find((filter) => filter?.id === 'APPLICATION_NAME')
    ) {
      this.populateNewResources();
    }
  }

  //Get options for selected filters
  getDropdownOptionsForFilter(filterId, notificationType, selectedFilters) {
    return new Promise((resolve, reject) => {
      let dropdownOptions: { key: string; value: string }[] = [];
      let validSelectedFilters = this.removeEmptyFilters(selectedFilters) as {
        [key: string]: string[];
      };
      this.uiNotificationControllerService
        .getFiltersForSubscriptionsUsingPOST({
          tagName: filterId,
          notificationType: notificationType,
          filterParams: validSelectedFilters,
        })
        .subscribe(
          (data) => {
            dropdownOptions = data['values'];
            resolve(dropdownOptions);
          },
          (error) => {
            if (error?.status === 404) {
              resolve([]);
            }
            reject(error);
          },
        );
    });
  }

  isDisabled(filterId) {
    switch (filterId) {
      case 'CLUSTER_NAME':
        return this.additionalFiltersForm?.value.hasOwnProperty('CLUSTER_TYPE');
      case 'CLUSTER_TYPE':
        return this.additionalFiltersForm?.value.hasOwnProperty('CLUSTER_NAME');
      case 'ALERT_NAME':
        return (
          Object.keys(this.additionalFiltersForm?.value).length === 0 ||
          !this.additionalFiltersForm.value.hasOwnProperty('CLUSTER_NAME') ||
          this.additionalFiltersForm.value['CLUSTER_NAME'].length === 0
        );
    }
  }

  resetAdditionalFilters() {
    //Reset the additionalFiltersForm
    this.additionalFiltersForm = new FormGroup({});

    //Reset the selectedFilters array
    this.selectedFilters = [];
  }

  getSortedAvailableFilters() {
    if (this.availableFilters?.length) {
      this.availableFilters.sort(function (a, b) {
        if (a.label < b.label) return -1;
        if (a.label > b.label) return 1;
        return 0;
      });
      return this.availableFilters;
    }
    return this.availableFilters;
  }

  debouncedHandleSelectionChange = debounce((event, filter) => {
    this.onAdditionalFiltersChange(filter);
  }, 500);

  //If environment changes then refetch alerts and preserve previous values
  //Also if environment is set for first time then enable alert name in dropdown
  onAdditionalFiltersChange(filter) {
    if (
      filter.id === 'CLUSTER_NAME' &&
      this.notificationType.value === 'ALERT'
    ) {
      let clusterCount =
        this.additionalFiltersForm.value['CLUSTER_NAME'].length;
      let hasAlertInForm = this.additionalFiltersForm
        .getRawValue()
        .hasOwnProperty('ALERT_NAME');
      if (clusterCount > 0) {
        if (!hasAlertInForm) {
          this.availableFilters.find(
            (filter) => filter.id === 'ALERT_NAME',
          ).disabled = false;
        } else {
          //Refetch and reassign
          let previousAlerts =
            this.additionalFiltersForm.getRawValue()['ALERT_NAME'];
          this.getDropdownOptionsForFilter(
            'ALERT_NAME',
            this.notificationType.value,
            {
              STACK_NAME: [this.stackName.value],
              ...this.additionalFiltersForm.value,
            },
          ).then((data: []) => {
            //Assign new dropdown values
            this.selectedFilters.find(
              (eachFilter) => eachFilter.id === 'ALERT_NAME',
            ).options = data;
            setTimeout(() => {
              if (data.length > 0) {
                this.additionalFiltersForm.patchValue({
                  ALERT_NAME: previousAlerts,
                });
              }
            }, 0);
            this.additionalFiltersForm.get('ALERT_NAME').enable();
          });
        }
      } else if (clusterCount === 0) {
        if (!hasAlertInForm) {
          this.availableFilters.find(
            (filter) => filter.id === 'ALERT_NAME',
          ).disabled = true;
        } else {
          this.setAlertsFormControlDisabled();
        }
      }
    } else if (
      (filter.id === 'CLUSTER_NAME' || filter.id === 'CLUSTER_TYPE') &&
      ['APP_DEPLOYMENT', 'APPLICATION_DEPLOYMENT_COMPLETE'].includes(
        this.notificationType.value,
      ) &&
      this.selectedFilters.find((filter) => filter.id === 'APPLICATION_NAME')
    ) {
      this.populateNewResources();
    }
  }

  populateNewResources() {
    this.fetchingResources = true;
    this.getDropdownOptionsForFilter(
      'APPLICATION_NAME',
      this.notificationType.value,
      {
        STACK_NAME: [this.stackName.value],
        ...this.additionalFiltersForm.value,
      },
    ).then(
      (data: any) => {
        const applicationFilter = this.selectedFilters.find(
          (filter) => filter.id === 'APPLICATION_NAME',
        );
        if (applicationFilter?.options) {
          applicationFilter.options = data;
        }
        const selectedAppName =
          this.additionalFiltersForm.get('APPLICATION_NAME').value;
        let selectedCommonOptions = data
          .filter((resource) => selectedAppName.includes(resource?.value))
          .map((resource) => resource?.value);
        this.additionalFiltersForm.get('APPLICATION_NAME').setValue([]);
        setTimeout(() => {
          this.additionalFiltersForm.patchValue({
            APPLICATION_NAME: selectedCommonOptions,
          });
        }, 0);
        this.fetchingResources = false;
      },
      (err) => {
        this.fetchingResources = false;
      },
    );
  }

  toggleDropdown() {
    const node = document.querySelector('.dropdown-menu');

    this.additionalFiltersDropdownObserver = new MutationObserver(
      (mutations) => {
        mutations.forEach((mutation) => {
          if (mutation?.target['classList'].contains('show')) {
            this.isDropdownOpen = true;
          } else {
            this.isDropdownOpen = false;
          }
        });
      },
    );

    this.additionalFiltersDropdownObserver.observe(node, {
      attributes: true,
      childList: true,
      characterData: true,
    });
  }

  // Opens another component to add/edit payload for WEBHOOK and sends state of current form fields to let it stay cached
  addPayloadConfig() {
    let url = this.editId
      ? `/settings/notification-center/subscription/edit/${this.editId}/edit/payload`
      : `/settings/notification-center/subscription/create/payload`;
    let subscription: Subscription = this.subscriptionForm.value;
    let additionalFilters = this.additionalFiltersForm.value;
    subscription.channelAddress = this.channelAddressMap[this.channelId.value];
    this.router.navigate([url], {
      state: {
        subscription: subscription,
        channelName: this.channelNameMap[this.channelId.value],
        additionalFilters: additionalFilters,
        selectedFilters: this.selectedFilters,
        previousComponent: this.editId
          ? 'edit-subscription'
          : 'create-subscription',
      },
    });
  }

  // Displays the current payload JSON inside a dialog box
  showCurrentPayload() {
    this.dialogBoxService.open(this.currentPayloadDialog, {
      hasBackdrop: true,
      autoFocus: false,
    });
  }

  removePayload() {
    this.subscriptionForm.patchValue({
      payloadJson: null,
    });
    this.payload = null;
  }

  handleBack = () => {
    this.router.navigate(['/settings/notification-center/subscriptions']);
  };

  //Extra function
  removeEmptyFilters(filters) {
    return Object.fromEntries(
      Object.entries(filters)
        .filter(([_, values]) => values && values['length'] > 0)
        .map(([key, values]) => [key, values]),
    );
  }

  sortFiltersForEditMode(filters) {
    const sortedFilters = filters.sort((a, b) => {
      if (a.id === 'CLUSTER_NAME') {
        return -1;
      } else if (b.id === 'CLUSTER_TYPE') {
        return 1;
      } else {
        return a.id.localeCompare(b.id);
      }
    });
    return sortedFilters;
  }

  //This function sets alerts dropdown to empty, resets alerts and make it disabled.
  setAlertsFormControlDisabled() {
    this.selectedFilters.find(
      (eachFilter) => eachFilter.id === 'ALERT_NAME',
    ).options = [];
    this.additionalFiltersForm.patchValue({ ALERT_NAME: [] });
    this.additionalFiltersForm.get('ALERT_NAME').disable();
  }

  onDropdownClick(event) {
    if (event.target.innerHTML.includes('disabled')) {
      event.stopPropagation();
    }
  }

  ngOnDestroy() {
    this.additionalFiltersDropdownObserver?.disconnect();
  }
}
