import { Component, OnInit, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import { UiOpaControllerService } from 'src/app/cc-api/services';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { SmartTableToggleComponent } from 'src/app/components/smart-table-toggle/smart-table-toggle.component';
import { SmartTableLongStringEllipsisComponent } from 'src/app/components/smart-table-long-string-ellipsis/smart-table-long-string-ellipsis.component';
import { SmartTableRouterlinkComponent } from 'src/app/components/smart-table-routerlink/smart-table-routerlink.component';
import { SmartTableTimeagoTooltipComponent } from 'src/app/components/smart-table-timeago-tooltip/smart-table-timeago-tooltip.component';
import { SmartTableWithBadgeComponent } from 'src/app/components/smart-table-with-badge/smart-table-with-badge.component';
import { OPA_DELETE, OPA_WRITE } from 'src/app/constants/permissions';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';

      title: 'Create Policy',
      buttonType: 'primary',
      icon: 'plus-outline',

      loading: false,
      action: OPA_WRITE,
      click: () => {
        this.navigateToCreatePolicy();
      },
    },
  ];

  smartTableCustomActionButtons = [
    {
      id: 'clone',
      icon: 'clone-outline',
      isFciPack: true,
      toolTip: 'Clone Policy',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: OPA_WRITE,
      onClick: (data) => {
        this.navigateToClonePolicy(data);
      },
    },
    {
      id: 'edit',
      icon: 'edit-outline',
      isFciPack: true,
      toolTip: 'Edit Policy',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: OPA_WRITE,
      onClick: (data) => {
        this.navigateToEditPolicy(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-outline',
      isFciPack: true,
      toolTip: 'Delete Policy',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: OPA_DELETE,
      onClick: (data) => {
        this.deletePolicyConfirmPopup(data);
      },
    },
  ];

  policyList = [];

  policyTableSettings = {
    columns: {
      policyName: {
        title: 'Policy Name',
        filter: true,
        type: 'text',
      },
      description: {
        title: 'Description',
        filter: false,
        type: 'custom',
        valuePrepareFunction: (cell: any) => {
          return JSON.stringify({
            value: cell,
            truncateLength: 35,
          });
        },
        renderComponent: SmartTableLongStringEllipsisComponent,
      },
      stackName: {
        title: 'Blueprint',
        valuePrepareFunction: (value) => {
          const displayVal = {
            url: ['/', 'stack', value],
            display: value,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: SmartTableRouterlinkComponent,
        type: 'custom',
      },
      policyCode: {
        title: 'Code',
        filter: false,
        type: 'custom',
        valuePrepareFunction: (value, row, cell) => {
          const data = {
            url: null,
            display: 'View Code',
          };
          return JSON.stringify(data);
        },
        renderComponent: SmartTableRouterlinkComponent,
        onComponentInitFunction: (instance: any) => {
          instance.byLinkClick.subscribe((data: any) => {
            this.openViewCodeDialog(data);
          });
        },
      },
      severity: {
        title: 'Severity',
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [
              { value: 'WARNING', title: 'WARNING' },
              { value: 'ERROR', title: 'ERROR' },
            ],
          },
        },

          return JSON.stringify({

            badgeClass: 'p-2 text-capitalize mb-0',
          });
        },

      },

            this.openEnableDisablePolicyConfirmPopup(data);

        },

        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance: any) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(

                return actionButton?.id === data?.actionButton?.id;

    actions: false,
  };

  @ViewChild('viewRegoCodeDialog', { static: true })
  viewRegoCodeDialog: TemplateRef<any>;

  @ViewChild('deletePolicyUI', { static: true })
  deletePolicyUI: TemplateRef<any>;

  @ViewChild('confirmPolicyEnableDisable', { static: true })
  confirmPolicyEnableDisable: TemplateRef<any>;

  loading: boolean = false;
  errorMessage = '';
  deletionInProgress: boolean = false;
  policyUpdateInProgress: boolean = false;

  constructor(
    private _route: ActivatedRoute,
    private router: Router,
    private uiOpaControllerService: UiOpaControllerService,
    private dialogService: NbDialogService,
    private customToastService: CustomToastService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.getPolicies();
  }

  handlePageActions(event) {
    this.guardrailsActionButtons?.find((btn) => btn?.id === event)?.click();
  }

  navigateToCreatePolicy() {
    this.router.navigate(['/settings/guardrails/create']);
  }

  navigateToEditPolicy(data) {
    this.router.navigate([`/settings/guardrails/edit/${data?.policyName}`]);
  }

  navigateToClonePolicy(data) {
    this.router.navigate(['/settings/guardrails/create'], {
      queryParams: { clonePolicy: data?.policyName },
    });
  }

  getPolicies() {
    this.loading = true;
    this.errorMessage = '';
    this.policyList = [];
    this.uiOpaControllerService.getAllPoliciesUsingGET().subscribe(
      (res: any) => {
        this.policyList = res;
        this.loading = false;
      },
      (error) => {
        console.log(error);
        this.loading = false;
        this.errorMessage =
          error?.error?.message || 'There was some error fetching the policies';
      },
    );
  }

  openViewCodeDialog(policy) {
    this.dialogService.open(this.viewRegoCodeDialog, {
      context: policy,
    });
  }

  deletePolicyConfirmPopup(rowData) {
    this.dialogService.open(this.deletePolicyUI, { context: rowData });
  }

  deletePolicy(data, ref) {
    this.deletionInProgress = true;
    this.uiOpaControllerService
      .deletePolicyUsingDELETE(data?.policyName)
      .subscribe(
        (data) => {
          this.deletionInProgress = false;
          ref.close();
          this.customToastService.success('Policy deleted successfully');
          this.getPolicies();
        },
        (error) => {
          this.deletionInProgress = false;
          ref.close();
          this.customToastService.error(
            error?.error?.message || 'Policy delete failed',
          );
        },
      );
  }

  openEnableDisablePolicyConfirmPopup(data) {
    this.dialogService.open(this.confirmPolicyEnableDisable, {
      context: { rowData: data?.rowData, value: data?.value },
      closeOnBackdropClick: false,
    });
  }

  onPolicyEnableDisable(data, ref) {
    this.policyUpdateInProgress = true;
    const { rowData, value } = data;
    // These fields are not needed in edit request object
    const {
      policyCode,
      createdBy,
      creationDate,
      lastModifiedBy,
      lastModifiedDate,
      ...restRowData
    } = rowData || {};
    let reqObj = {
      ...restRowData,
      disabled: !data?.value,
      rawPolicy: policyCode,
      policyType: 'RAW',
    };
    this.uiOpaControllerService
      .editPolicyUsingPUT({
        policyName: reqObj?.policyName,
        opaRequest: reqObj,
      })
      .subscribe(
        (res) => {
          this.policyUpdateInProgress = false;
          ref.close();
          this.customToastService.success(
            `The policy ${data?.rowData?.policyName} has been ${
              data?.value ? 'enabled' : 'disabled'
            } successfully. 'Run validations' on the releases page to see the changes.`,
          );
          // Update the value in policyList
          this.policyList.find(
            (policy) => policy?.policyName === res?.policyName,
          ).disabled = res?.disabled;
        },
        (error) => {
          this.policyUpdateInProgress = false;
          ref.close();
          this.customToastService.error(
            error?.error?.message ||
              `The policy ${data?.rowData?.policyName} could not be ${
                data?.value ? 'enabled' : 'disabled'
              }`,
          );
        },
      );
  }

  onPolicyEnableDisableCancel(data, ref) {
    ref.close();
    this.policyList = [...this.policyList];
  }
}
