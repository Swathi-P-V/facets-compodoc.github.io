import { Component, ViewChild } from '@angular/core';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { UiSettingsControllerService } from 'src/app/cc-api/services';
import { Setting } from 'src/app/cc-api/models';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import {
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import { CanDeactivateComponent } from 'src/app/can-deactivate.guard';
import { UserAuthoritiesService } from 'src/app/services/user-authorities.service';
import { SaveChangesFooterComponent } from 'src/app/components/save-changes-footer/save-changes-footer.component';
import { ENVIRONMENT_CONFIGURE } from 'src/app/constants/permissions';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
const errLogo = './assets/error.svg';
import { ROLE_CC_ADMIN } from 'src/app/constants/permissions';

@Component({
  selector: 'app-control-plane-config',
  templateUrl: './control-plane-config.component.html',
  styleUrls: ['./control-plane-config.component.scss'],
})
export class ControlPlaneConfigComponent
  extends FacetsPage
  implements CanDeactivateComponent
{
  controlSettingsForm: FormGroup;
  clusterId: string;
  stackName: any;
  fetchingSetting: Boolean = false;
  entityType: 'CLUSTER' | 'BLUE_PRINT' | 'TEMPLATE_INPUT' | 'CONTROL_PLANE' =
    'CONTROL_PLANE';
  cpDefaultSettings: Setting[];
  cpSettings: {};
  cpSettingOriginal: {} = {};
  cpSettingChanged: {} = {};
  selectOptions: {} = {};
  reviewComments: string = '';
  changeSettings: boolean = false;
  settingsError: boolean = false;
  cpError: boolean = false;
  errIcon: any = errLogo;
  status: 'info' | 'error' = 'info';
  primaryMsg: string = 'No settings available';
  secondaryMsg: string = '';
  showBtn: boolean = true;
  buttonText: string = 'Reload';
  tooltip: string = 'Load Settings';
  showTooltip: boolean = true;
  isUserAdmin: boolean = false;
  objectKeys = Object.keys;
  environmentConfigurePermission = ENVIRONMENT_CONFIGURE;

  @ViewChild(SaveChangesFooterComponent)
  private footerComponent: SaveChangesFooterComponent;
  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private uiSettingService: UiSettingsControllerService,
    private userAuthoritiesService: UserAuthoritiesService,
    private fb: FormBuilder,
    private toastService: CustomToastService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.stackName = p.stackName;
    this.clusterId = 'global';
    this.entityType = 'CONTROL_PLANE';
    this.getControlPlaneSettings();
    this.isUserAdmin =
      !this.userAuthoritiesService.isActionDisabled(ROLE_CC_ADMIN);
  }

  getControlPlaneSettings() {
    this.cpDefaultSettings = [];
    this.selectOptions = {};
    this.cpSettingChanged = {};
    this.cpSettingOriginal = {};
    this.cpSettings = {};
    this.fetchingSetting = true;
    let settingParams = {
      entityType: this.entityType,
      entityId: this.clusterId,
    };
    this.settingsError = false;
    this.uiSettingService.getSettingValueUsingGET(settingParams).subscribe(
      (r) => {
        this.cpSettings = r;
        this.uiSettingService
          .getAllSettingsForEntityUsingGET(this.entityType)
          .subscribe(
            (t) => {
              this.cpDefaultSettings = t;
              this.cpDefaultSettings.forEach((setting) => {
                this.prepareOptions(setting);
                if (
                  this.cpSettings &&
                  Object.keys(this.cpSettings).indexOf(setting.id) > -1
                ) {
                  this.cpSettingOriginal[setting.id] =
                    this.cpSettings[setting.id];
                } else {
                  this.cpSettings[setting.id] = setting.defaultValue;
                  this.cpSettingOriginal[setting.id] = setting.defaultValue;
                }
              });
              this.setCpSettingsForm();
              this.fetchingSetting = false;
            },
            (error) => {
              this.fetchingSetting = false;
              this.settingsError = true;
            },
          );
      },
      (error) => {
        this.fetchingSetting = false;
        this.settingsError = true;
      },
    );
  }

  setCpSettingsForm = () => {
    const group: any = {};
    Object.keys(this.cpSettings).forEach((setting) => {
      group[setting] = new FormControl(
        this.cpSettings[setting],
        Validators.pattern(whiteSpaceInputPattern),
      );
    });
    this.controlSettingsForm = new FormGroup(group);
  };

  prepareOptions(setting) {
    let selectOption = [];
    if (setting.settingType === 'ENUM') {
      setting.allowedValues.forEach((value) => {
        selectOption.push({
          label: value,
          value: value,
        });
      });
      this.selectOptions[setting.id] = selectOption;
    } else if (setting.settingType === 'BOOLEAN') {
      selectOption.push({
        label: true,
        value: true,
      });
      selectOption.push({
        label: false,
        value: false,
      });
      this.selectOptions[setting.id] = selectOption;
    }
  }

  onChange(event, setting, type) {
    if (event?.target?.value) {
      if (event.target.value != this.cpSettingOriginal[setting.id]) {
        this.cpSettingChanged[setting.id] = {
          value:
            type === 'num' ? parseInt(event.target.value) : event.target.value,
          name: setting.name,
        };
      } else {
        delete this.cpSettingChanged[setting.id];
      }
    }
  }

  onChangeSelect(value, setting) {
    if (value != this.cpSettingOriginal[setting.id]) {
      this.cpSettingChanged[setting.id] = {
        value: value,
        name: setting.name,
      };
    } else {
      delete this.cpSettingChanged[setting.id];
    }
  }

  submitcpSettings = (ref) => {
    this.changeSettings = true;
    let insertSettingParams: {
      settingValueRequest: {};
      entityType: 'CLUSTER' | 'BLUE_PRINT' | 'TEMPLATE_INPUT' | 'CONTROL_PLANE';
      entityId: string;
    } = {
      entityId: this.clusterId,
      entityType: this.entityType,
      settingValueRequest: {},
    };
    const settingValues = {};
    for (const setting in this.cpSettingChanged) {
      settingValues[setting] = this.cpSettingChanged[setting].value;
    }
    insertSettingParams['settingValueRequest'] = settingValues;
    this.uiSettingService
      .addSettingValueUsingPUT(insertSettingParams)
      .subscribe(
        (settings: any) => {
          this.toastService.success('Settings changed successfully ');
          ref.close();
          this.getControlPlaneSettings();
          this.changeSettings = false;
        },
        (error) => {
          ref.close();
          this.changeSettings = false;
          if (
            (error.status === 500 || error.status === 403) &&
            error?.error.message === 'Access is denied'
          ) {
            this.toastService.error(
              'Please contact your administrator for more details',
            );
          } else {
            this.toastService.error(
              'Settings change failed. ' + error?.error?.message,
            );
          }
        },
      );
  };

  onDiscard = (ref: any) => {
    this.cpSettingChanged = {};
    this.cpSettings = this.cpSettingOriginal;
    this.setCpSettingsForm();
    ref.close('discard');
  };

  async canDeactivate(): Promise<boolean> {
    if (this.objectKeys(this.cpSettingChanged)?.length == 0) {
      return true;
    }
    return this.footerComponent.onDeactivate();
  }
}
