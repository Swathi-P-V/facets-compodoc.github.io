import { Component, OnInit } from '@angular/core';
import { FormControl } from '@angular/forms';
import { Observable, of, startWith, map } from 'rxjs';
import { BlueprintDesignerUtilService } from '../../blueprint-designer.service';

export class Group {
  resourceType: string;
  resourceNames: string[] = [];
  children: any[] = [];
  toggle: boolean = false;
}

@Component({
  selector: 'app-blueprint-search',
  templateUrl: './search.component.html',
  styleUrls: ['./search.component.scss'],
})
export class SearchComponent implements OnInit {
  groups: Group[];
  filteredGroups$: Observable<Group[]>;
  inputFormControl: FormControl;

  constructor(
    public blueprintDesignerUtilService: BlueprintDesignerUtilService,
  ) {}

  ngOnInit() {
    this.initAllData();
  }

  initAllData() {
    this.blueprintDesignerUtilService.resourceMapReady.subscribe((ready) => {
      const resourceNodeData =
        this.blueprintDesignerUtilService.getAllNodesMap();
      this.groups = [];
      for (let resourceType of resourceNodeData.keys()) {
        if (resourceType.toLowerCase() != 'unknown') {
          const resourceList = resourceNodeData.get(resourceType);
          this.groups.push({
            resourceType: resourceType,
            resourceNames: resourceList.map(
              (resource) => resource.resourceName,
            ),
            children: resourceList,
            toggle: false,
          });
        }
      }
      this.filteredGroups$ = of(this.groups);
      this.inputFormControl = new FormControl();
      this.filteredGroups$ = this.inputFormControl.valueChanges.pipe(
        startWith(''),
        map((filterString) => this.filter(filterString)),
      );
    });
  }

  private filterChildren(children: string[], filterValue: string) {
    return children.filter((optionValue) =>
      optionValue.toLowerCase().includes(filterValue),
    );
  }

  private filter(value: string): Group[] {
    const filterValue = value.toLowerCase();
    return this.groups
      .map((group) => {
        return {
          resourceType: group.resourceType,
          resourceNames: this.filterChildren(group.resourceNames, filterValue),
          children: group.children,
          toggle: group.toggle,
        };
      })
      .filter((group) => group.resourceNames.length);
  }

  trackDataByKey(index, item) {
    return item.resourceType;
  }

  toggleGroup(group, event) {
    event.stopPropagation();
    group.toggle = !group.toggle;
  }

  showNode(resourceGroup, resourceName) {
    let resourceMap = this.blueprintDesignerUtilService.getAllNodesMap();
    let resourceNode = resourceMap
      .get(resourceGroup.resourceType)
      .find((resource) => resource.resourceName === resourceName);
    this.blueprintDesignerUtilService.setActiveNode(resourceNode);
    this.blueprintDesignerUtilService.bringNodeInFocus(resourceNode);
  }

  closeWrapper() {
    this.inputFormControl.setValue('');
  }
}
