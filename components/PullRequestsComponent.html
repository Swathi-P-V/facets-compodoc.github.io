import { Component, OnDestroy } from '@angular/core';
import { ResourceDetailsService } from '../resource-details/resource-details.service';
import { ActivatedRoute, Params } from '@angular/router';
import { HttpClient } from '@angular/common/http';
import { FormControl, Validators } from '@angular/forms';
import { UiStackControllerService } from 'src/app/cc-api/services';
import {
  animate,
  state,
  style,
  transition,
  trigger,
} from '@angular/animations';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CustomServerDataSource } from 'src/app/services/custom-server-data-source';
import { SmartTableIconSubheadingComponent } from '../../../components/smart-table-icon-subheading/smart-table-icon-subheading.component';
import { FormatDateService } from 'src/app/services/format-date.service';
import { Stack } from '../../../cc-api/models/stack';
const emptyStateLogo = './assets/artifact-empty.svg';
import { interval, Subscription } from 'rxjs';

@Component({
  selector: 'app-pull-requests',
  templateUrl: './pull-requests.component.html',
  styleUrls: ['./pull-requests.component.scss'],
  animations: [
    trigger('menuSlide', [
      state('hidden', style({ transform: 'translateX(-100%)' })),
      state('visible', style({ transform: 'translateX(0)' })),
      transition('hidden => visible', animate('300ms ease-in')),
      transition('visible => hidden', animate('300ms ease-out')),
    ]),
  ],
})
export class PullRequestsComponent extends FacetsPage implements OnDestroy {
  prStatusIconMap = {
    OPEN: {
      icon: 'open-pr',
      tooltip: 'Open Pull Request',
    },
    CLOSED: {
      icon: 'closed-pr',
      tooltip: 'Closed Pull Request',
    },
    MERGED: {
      icon: 'merged-pr',
      tooltip: 'Merged Pull Request',
    },
  };
  emptyBtnText: string = 'Create Pull Requests';
  loadingStackDetails: boolean = false;
  prEmptyIcon = emptyStateLogo;
  secondaryMsg = 'No pull requests available';

  emptyStateLogo = emptyStateLogo;
  fetchingPR: boolean = false;
  prData: CustomServerDataSource;
  currentPrResponse;
  stack: Stack;
  stackName: string = '';
  resourceType: string;
  resourceName: string;
  pullRequestOptions: {
    label: string;
    value: string;
    checked: boolean;
  }[] = [
    { label: 'All Requests', value: 'ALL', checked: false },
    { label: 'Open Requests', value: 'OPEN', checked: true },
    { label: 'Closed Requests', value: 'CLOSED', checked: false },
  ];
  statusFormControl: FormControl = new FormControl('OPEN', [
    Validators.required,
  ]);
  prRefreshInterval: number = 10000;
  tableSettings: any = {
    columns: {
      title: {
        title: 'title',
        type: 'custom',
        filter: true,
        valuePrepareFunction: (value, row, cell) => {
          let prStatus = '';
          row?.labels.forEach((label) => {
            label.textColor = this.getContrast('#' + label?.color);
          });
          if (row?.state === 'closed' && row?.mergedAt) {
            prStatus = 'MERGED';
          } else {
            prStatus = row?.state?.toUpperCase();
          }
          return JSON.stringify({
            icon: this.prStatusIconMap[prStatus]?.icon,
            tooltipText: this.prStatusIconMap[prStatus]?.tooltip,
            pack: 'fci',
            status: 'primary',
            title: value,
            labels: row?.labels,
            subtitle: `#${
              row?.number
            } opened ${this.formatDateService.getRelativeTime(
              row?.createdAt,
            )} by ${row?.user?.login}`,
          });
        },
        filterFunction(cell?: any, search?: any): boolean {
          return cell.id === Number(search);
        },
        renderComponent: SmartTableIconSubheadingComponent,
        onComponentInitFunction: (instance) => {
          instance?.byLinkClick?.subscribe((data) => {
            window.open(data?.htmlUrl, '_blank');
          });
        },
      },
    },
    noDataMessage: '',
    pager: {
      display: true,
      perPage: 10,
    },
    mode: 'external',
    actions: false,
  };
  intervalSub: Subscription;
  pullRequestsError;

  constructor(
    private resourceDetailsService: ResourceDetailsService,
    private uiStackControllerService: UiStackControllerService,
    private formatDateService: FormatDateService,
    private http: HttpClient,
    private activatedroute: ActivatedRoute,
  ) {
    super(activatedroute);
  }

  OnPageLoad(p: Params) {
    this.fetchingPR = true;
    this.resetPrFields();
    this.stackName = p?.stackName;
    this.resourceType = p?.resourceType;
    this.resourceName = p?.resourceName;
    this.currentPrResponse = [];
    this.getPrData('OPEN');
    this.getStackDetails();
  }

  ngOnDestroy() {
    this.intervalSub?.unsubscribe();
  }

  resetPrFields() {
    this.prData = null;
  }

  getPrData(status) {
    this.intervalSub?.unsubscribe();
    let customServerUrl = '';
    if (this.resourceType?.length) {
      customServerUrl = `${
        window.location.origin
      }/cc-ui/v1/designer/${encodeURIComponent(
        String(this.stackName),
      )}/resourceType/${encodeURIComponent(
        String(this.resourceType),
      )}/resourceName/${encodeURIComponent(String(this.resourceName))}/pulls`;
    } else {
      customServerUrl = `${
        window.location.origin
      }/cc-ui/v1/designer/${encodeURIComponent(String(this.stackName))}/pulls`;
    }
    this.prData = new CustomServerDataSource(
      this.http,
      {
        endPoint: customServerUrl,
        dataKey: 'data.pullRequests', // path in the response which holds array of deployments
        totalKey: 'data.totalCount', // this key from response holds total number of prs present, smart table shows the pagination numbers based on this
        pagerLimitKey: 'pageSize', // will be passed as queryparam to the api, set 15 in table settings config
        pagerPageKey: 'pageNumber',
        filterFieldKey: '#field#', // Takes the field name dynamically for filtering purpose
      },
      {
        customQueryParams: {
          state: status,
        },
      },
      (prList: [] = []) => {
        this.pullRequestsError = null;
        this.fetchingPR = false;
        this.currentPrResponse = [...prList];
        return prList;
      },
      true,
      (error) => {
        this.fetchingPR = false;
        this.pullRequestsError = error;
      },
    );
    this.intervalSub = interval(this.prRefreshInterval).subscribe((x) => {
      this.prData.refresh();
    });
  }

  parsePrData(prList) {
    this.prData = prList;
  }

  getStackDetails() {
    this.loadingStackDetails = true;
    this.uiStackControllerService.getStackUsingGET(this.stackName).subscribe(
      (s) => {
        this.stack = s;
        this.loadingStackDetails = false;
      },
      (err) => {
        this.loadingStackDetails = false;
      },
    );
  }

  get menuState(): string {
    return 'visible';
  }

  onCreateClick(event) {
    const compareUrl = this.stack?.vcsUrl?.slice(0, -4) + '/compare';
    window.open(compareUrl, '_blank');
  }

  onStatusChange(event) {
    this.getPrData(event);
  }

  getContrast(bgColor) {
    var color = bgColor.charAt(0) === '#' ? bgColor.substring(1, 7) : bgColor;
    var r = parseInt(color.substring(0, 2), 16); // hexToR
    var g = parseInt(color.substring(2, 4), 16); // hexToG
    var b = parseInt(color.substring(4, 6), 16); // hexToB
    return r * 0.299 + g * 0.587 + b * 0.114 > 186 ? '#000' : '#fff';
  }
}
