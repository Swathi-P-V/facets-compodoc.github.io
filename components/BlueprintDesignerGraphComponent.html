import {
  Component,
  EventEmitter,
  Injector,
  Input,
  OnDestroy,
  Output,
} from '@angular/core';
import { ActivatedRoute, Params } from '@angular/router';
import { DagreLayout, Layout } from '@antv/layout';
import { Graph } from '@antv/x6';
import { uniqBy } from 'lodash';
import {
  UiBlueprintDesignerControllerService,
  UiDropdownsControllerService,
} from 'src/app/cc-api/services';
import {
  BlueprintDesignerUtilService,
  highlightStates,
} from '../blueprint-designer.service';
import { groupByEdges } from '../blueprint-designer.utils';
import { ResourceNodeComponent } from '../nodes/resource-node/resource-node.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { BlueprintDesignerGraphService } from './blueprint-designer-graph.service';
import { Subscription } from 'rxjs';
import { SimpleResourceNodeComponent } from '../nodes/simple-resource-node/simple-resource-node.component';
import { BrowserDetectionService } from 'src/app/services/browser-detection.service';
import { cloneDeep } from 'lodash';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
const errorStateIcon = './assets/error.svg';
const emptyIcon = './assets/empty.svg';

@Component({
  selector: 'app-blueprint-designer-graph',
  templateUrl: './blueprint-designer-graph.component.html',
  styleUrls: ['./blueprint-designer-graph.component.scss'],
})
export class BlueprintDesignerGraphComponent implements OnDestroy {
  @Input() showSingleResourceConnections: boolean = false;
  @Input() singleResourceID: string = '';
  @Input() useSimpleNodes: boolean = false;
  @Input() hideMiniMap: boolean = false;
  @Input() hideOtherNodes: boolean = false;
  @Input() hideResourceConfigViewer: boolean = false;
  @Output() graphLoaded: EventEmitter<any> = new EventEmitter<any>();
  singleResourceNode: any;

  blueprint: string; // Name of the current blueprint
  branchName: string; // Name of the current branch
  componentLoaded: boolean; // Makes sure that the graph modules are only registered once while the component is alive
  fetchingData: boolean; // Status for fetching the data from the API for the blueprint's configuration
  isFirefox: boolean = false;

  graph: Graph;
  configTypeResources = [];
  resourcesWithConnection = [];
  graphLayout: any;
  graphData: any = {
    nodes: [],
    edges: [],
  };
  resourcesWithoutConnection = [];
  zoom = 100;
  loadingResources = false;
  uniqueEdges;
  uniqueNodes;
  errorStateIcon = errorStateIcon;
  emptyIcon = emptyIcon;
  showEmptyState: boolean = false;
  errorWhileFetchingResources: boolean = false;

  otherNodeIntents = new Set([
    'alert_group',
    'log_collector',
    'kubernetes_node_pool',
    'iam_policy',
    'grafana_dashboard',
    'vpc_peering',
    'config_map',
    'unknown',
  ]);
  routeParamSubscription: Subscription;

  constructor(
    private injector: Injector,
    private uiDropdownControllerService: UiDropdownsControllerService,
    private blueprintDesignerUtilService: BlueprintDesignerUtilService,
    private blueprintDesignerGraphService: BlueprintDesignerGraphService,
    private customToasterService: CustomToastService,
    private browserDetectionService: BrowserDetectionService,
    private blueprintDesignerService: UiBlueprintDesignerControllerService,
    private activatedRoute: ActivatedRoute,
  ) {
    this.isFirefox = browserDetectionService.isFirefox();
    this.routeParamSubscription = this.activatedRoute.params.subscribe((p) =>
      this.CustomOnPageLoad(p),
    );
  }

  previousParams: Params;

  CustomOnPageLoad(p: Params): void {
    if (JSON.stringify(p) === JSON.stringify(this.previousParams)) {
      return;
    } else {
      this.previousParams = p;
    }

    this.showEmptyState = false;
    this.errorWhileFetchingResources = false;
    this.blueprint = p?.stackName;
    this.branchName = p?.branchName;
    if (this.showSingleResourceConnections) {
      this.singleResourceID = `${p?.resourceType}:${p?.resourceName}`;
    }

    this.checkForBlockOverlay();

    /*
      Currently, when we come back from blueprint designer resource details page to resources list graph page,
      the page shows blank as x6 library takes some time to load the new graph after component instantiation
      The below settimeout gives that time to load the component and html elements and then fire the loading of the graph in those elements
    */
    setTimeout(() => {
      this.initGraphComponent();
    }, 0);
  }

  async initGraphComponent() {
    // Remove all the data and unregister everything which we might have
    // previously added
    this.graph?.dispose();
    this.unregisterGraphModules();
    this.registerGraphNodes();
    this.resetGraph();
    this.initializeVariables();

    // Unsubscribe from the subscriptions and resubscribe to them again
    // to only have one single channel of signals
    this.zoomEventSubscription?.unsubscribe();
    this.headerEventSubscription?.unsubscribe();
    this.listenToHeaderEvents();

    // Initialize the graph object with the required settings and the layout
    // and then finally call re
    this.initializeGraph();
    this.initializeLayout();
    await this.getResourcesInfo();
  }

  ngOnDestroy(): void {
    let columns: any = document.querySelector('.columns');
    columns.style.overflow = 'auto';
    if (this.isFirefox) {
      let nbLayoutElement: any = document.querySelector(
        '.columns > nb-layout-column:nth-child(1)',
      );
      // nbLayoutElement.css('padding',this.fireFoxPaddingBackup);
      nbLayoutElement.classList.remove('column-container-firefox');
    }
    this.zoomEventSubscription?.unsubscribe();
    this.headerEventSubscription?.unsubscribe();
    this.routeParamSubscription?.unsubscribe();
  }

  registerGraphNodes = () => {
    Graph.registerAngularContent('resource-node', {
      injector: this.injector,
      content: ResourceNodeComponent,
    });
    Graph.registerAngularContent('simple-resource-node', {
      injector: this.injector,
      content: SimpleResourceNodeComponent,
    });
  };

  unregisterGraphModules() {
    Graph.unregisterAngularContent('resource-node');
    Graph.unregisterAngularContent('simple-resource-node');
  }

  initializeVariables() {
    this.graph = undefined;
    this.configTypeResources = [];
    this.resourcesWithConnection = [];
    this.graphLayout = undefined;
    this.graphData = {
      nodes: [],
      edges: [],
    };
    this.resourcesWithoutConnection = [];
  }

  resetGraph() {
    this.graph?.clearCells();
    this.graph?.clearBackground();
    this.graph?.clearGrid();
    const minimapContainer = document.querySelector('#minimap-container');
    while (minimapContainer?.firstChild) {
      minimapContainer.removeChild(minimapContainer?.firstChild);
    }
    this.blueprintDesignerUtilService.closeConfigurationTool();
  }

  initializeGraph = () => {
    // Check whether the graph element exists in the template or not
    // If it's not there, just return from here
    if (
      !document.getElementById(
        !this.showSingleResourceConnections
          ? 'graph-container'
          : 'graph-container-small',
      )
    ) {
      return;
    }

    // If the graph container exists in the container,
    // initialize the graph object with the following configuration
    this.graph = new Graph({
      container: document.getElementById(
        !this.showSingleResourceConnections
          ? 'graph-container'
          : 'graph-container-small',
      ),
      connecting: {
        connector: 'smooth',
      },
      autoResize: true,
      grid: {
        visible: !this.showSingleResourceConnections,
      },
      minimap: {
        enabled: !this.hideMiniMap,
        container: document.getElementById('minimap-container'),
      },
      panning: true,
      mousewheel: {
        enabled: true,
        modifiers: ['ctrl', 'meta'],
        zoomAtMousePosition: true,
        minScale: 0.5,
        maxScale: 2,
      },
    });
  };

  fireFoxPaddingBackup: any;
  initializeLayout = () => {
    // Initialize the DAGRE layout for the nodes' arrangement
    // in the graph
    this.graphLayout = new DagreLayout({
      type: 'dagre',
      rankdir: 'LR',
      align: 'DR',
      ranksep: 85,
      nodesep: 15,
      begin: [50, 50],
    });

    // If we are loading this component for all the nodes then we need
    // to set the overflow property to hidden because the canvas would be
    // loading in the full height of the page
    if (!this.showSingleResourceConnections) {
      let columns: any = document.querySelector('.columns');
      columns.style.overflow = 'hidden';
    }

    if (this.isFirefox) {
      let nbLayoutElement: any = document.querySelector(
        '.columns > nb-layout-column:nth-child(1)',
      );
      this.fireFoxPaddingBackup = nbLayoutElement?.style?.padding;
      // nbLayoutElement.css('padding', '0px !important');
      nbLayoutElement.classList.add('column-container-firefox');
    }
  };

  // Make a call to get all the resource nodes
  // and add them to the graph
  getResourcesInfo = async () => {
    return new Promise((resolve) => {
      this.loadingResources = true;
      this.showEmptyState = false;
      this.uiDropdownControllerService
        .getAllResourcesByStackUsingGET({
          stackName: this.blueprint,
          includeContent: true,
          branch: this.branchName,
        })
        .subscribe(
          (data) => {
            if (
              !cloneDeep(data).filter(
                (resource) => resource?.resourceType !== 'UNKNOWN',
              )?.length
            ) {
              this.showEmptyState = true;
            }
            // If we are opening the component in focus mode for a particular resource,
            // then we can override the data with the singleResourceID's connections only
            if (this.showSingleResourceConnections) {
              data = this.getConnectedNodes(data, this.singleResourceID);
            }
            this.parseAndSetResourcesData(data);
            this.generateGraphNodesAndEdges();
            this.renderGraph();
            this.hideUnusedGraphContainer();
            this.blueprintDesignerUtilService.setGraph(this.graph);
            this.blueprintDesignerGraphService.setGraph(this.graph);
            this.blueprintDesignerUtilService.setAllNodesMap(data);

            // ====== If we are opening the graph in the resource specific mode
            // ====== we set that node as active
            if (this.showSingleResourceConnections) {
              this.blueprintDesignerUtilService.setActiveNode(
                this.singleResourceNode,
              );
            }
            resolve(1);
          },
          (err) => {
            this.errorWhileFetchingResources = true;
            this.errorStateIcon = this.emptyIcon;
            this.showEmptyState = true;
            resolve(1);
          },
        )
        .add(() => {
          this.loadingResources = false;
          if (this.resourceToFocus && !this.blockOverlayOn) {
            this.bringNodeToFocus();
          }
          this.graphLoaded.emit(true);
        });
    });
  };

  hideUnusedGraphContainer() {
    const containerName = !this.showSingleResourceConnections
      ? 'graph-container-small'
      : 'graph-container';
    setTimeout(() => {
      if (document.querySelector(`#${containerName}`)) {
        document.querySelector(`#${containerName}`)['style'].display = 'none';
      }
    }, 0);
  }

  // Processes an array of resourcesInfo objects to extract and collect
  // unique edge IDs and resource node IDs into two separate sets.
  getUnitEdges = (resourcesInfo) => {
    let edgesSet = new Set();
    let nodeSet = new Set();
    resourcesInfo.forEach((element) => {
      nodeSet = new Set([
        ...nodeSet,
        this.blueprintDesignerUtilService.getResourceID(element),
      ]);
      if (element?.edges?.length) {
        edgesSet = new Set([
          ...edgesSet,
          ...element?.edges.map((edge) => edge.id),
        ]);
      }
    });
    this.uniqueEdges = edgesSet;
    this.uniqueNodes = nodeSet;
  };

  cleanOtherNodeIntents() {
    // uniqueEdges has all the nodes to or from which a connection goes through
    // We should remove any intent from the otherNodesList which is originally supposed to be
    // a config node but has some connection present
    Array.from(this.uniqueEdges).forEach((resourceEdge: any) => {
      const intent = resourceEdge.split(':')[0];
      if (this.otherNodeIntents.has(intent)) {
        this.otherNodeIntents.delete(intent);
      }
    });
  }

  parseAndSetResourcesData(resourcesInfo) {
    // Set the unique edges and nodes
    this.getUnitEdges(resourcesInfo);

    // Create an array of resources with IDs from the resourceNodes
    let resourcesWithId = resourcesInfo.map((resource) => ({
      ...resource,
      id: this.blueprintDesignerUtilService.getResourceID(resource),
    }));

    // Get all the unique resources based on the IDs
    resourcesWithId = uniqBy(resourcesWithId, 'id');

    this.cleanOtherNodeIntents();

    resourcesWithId.forEach((resource) => {
      if (resource.resourceType === 'UNKNOWN') return;

      if (
        !resource?.edges?.length &&
        !this.uniqueEdges.has(
          this.blueprintDesignerUtilService.getResourceID(resource),
        )
      ) {
        this.resourcesWithoutConnection.push(resource);
      }
      if (
        resource?.edges?.length ||
        this.uniqueEdges.has(
          this.blueprintDesignerUtilService.getResourceID(resource),
        )
      ) {
        this.resourcesWithConnection.push(resource);
      }
    });
  }

  getConnectedNodes(resources: any[], requiredResourceID: string) {
    let allOutgoingConnections = new Set<any>();
    allOutgoingConnections = new Set(
      resources
        .find(
          (resource) =>
            this.blueprintDesignerUtilService.getResourceID(resource) ===
            requiredResourceID,
        )
        ?.edges?.map((resource) => resource.id),
    );
    return resources?.filter((resource) => {
      const resourceID =
        this.blueprintDesignerUtilService.getResourceID(resource);
      if (resourceID === requiredResourceID) {
        this.singleResourceNode = resource;
      }
      return (
        resourceID === requiredResourceID ||
        // Resources which point to our requiredResource
        resource?.edges?.find(
          (dependencyID) => dependencyID.id === requiredResourceID,
        ) ||
        // Resources which our requiredResource points to`
        allOutgoingConnections.has(resourceID)
      );
    });
  }

  renderGraph() {
    const model = this.graphLayout.layout(this.graphData);
    this.graph?.fromJSON(model);
    this.graph?.scaleContentToFit({
      padding: 10,
      maxScale: 1,
      minScale: 0.3,
    });
    this.graph?.centerContent();
    // Since we have added 100vw to the graph container, the centre function
    // is not able to exactly horizontally centre the graph and thus we
    // need to translate it a little to the left
    this.graph?.translateBy(-150, 30);
  }

  graphNodesForUnConnectedResources() {
    this.resourcesWithoutConnection.forEach((node: any) => {
      this.graphData.nodes.push({
        id: this.blueprintDesignerUtilService.getResourceID(node),
        shape: 'angular-shape',

        // ========= We might need to change the resource node component based on
        // ========= the mode we are opening the graph component in

        // componentName: this.showSingleResourceConnections
        //   ? 'simple-resource-node'
        //   : 'resource-node',

        componentName: 'resource-node',
        height: 80,
        width: 200,
        data: {
          ngArguments: {
            nodeInfo: node,
            resourceSpecificGraph: !!this.singleResourceNode,
          },
        },
      });
    });
  }

  generateGraphNodesAndEdges = () => {
    this.nodeAndEdgesWithoutGrouping();
    this.graphNodesForUnConnectedResources();
    // this.nodeAndEdgesAfterGrouping();
  };

  nodeAndEdgesWithoutGrouping() {
    this.resourcesWithConnection.forEach((node: any) => {
      this.graphData.nodes.push({
        id: this.blueprintDesignerUtilService.getResourceID(node),
        shape: 'angular-shape',

        // ========= We might need to change the resource node component based on
        // ========= the mode we are opening the graph component in

        // componentName: this.showSingleResourceConnections
        //   ? 'simple-resource-node'
        //   : 'resource-node',

        componentName: 'resource-node',
        height: 69,

        // ========== We might need to change the resource node width based on
        // the mode we are opening the graph component in
        // width: !this.showSingleResourceConnections ? 195 : 140,

        width: 195,
        data: {
          ngArguments: {
            nodeInfo: node,
            resourceSpecificGraph: !!this.singleResourceNode,
          },
        },
      });
      node.edges.forEach((edge: any) => {
        if (
          !this.showSingleResourceConnections &&
          node.resourceType === 'service' &&
          edge.resourceType === 'service'
        ) {
          return;
        }

        if (!edge?.resourceName) return;

        if (this.uniqueNodes?.has(edge.id)) {
          this.graphData.edges.push({
            source: node.id,
            target: edge.id,
            attrs: {
              line: {
                stroke: '#A2B1C3',
                strokeWidth: 1,
              },
            },
          });
        }
      });
    });
  }

  nodeAndEdgesAfterGrouping() {
    const nodeMap = {};
    this.resourcesWithConnection.forEach((node: any) => {
      nodeMap[node.id] = node;
    });

    const edges = [];
    this.resourcesWithConnection.forEach((node: any) => {
      node.edges.forEach((edge: any) => {
        if (!edge?.resourceName) return;
        edges.push({
          source: edge.id,
          target: node.id,
        });
      });
    });
    const groupedEdges = groupByEdges(edges);
    Object.entries(groupedEdges).forEach(([key, value]: [string, string[]]) => {
      value.forEach((nodeId) => {
        const node = nodeMap[nodeId];
        if (!node) return;

        this.graphData.nodes.push({
          id: this.blueprintDesignerUtilService.getResourceID(node),
          shape: 'angular-shape',
          componentName: this.showSingleResourceConnections
            ? 'simple-resource-node'
            : 'resource-node',
          height: 69,

          // ========== We might need to change the resource node width based on
          // the mode we are opening the graph component in
          // width: !this.showSingleResourceConnections ? 195 : 140,

          width: 195,
          data: {
            ngArguments: {
              nodeInfo: node,
              resourceSpecificGraph: !!this.singleResourceNode,
            },
          },
        });
        node.edges.forEach((edge: any) => {
          if (
            node.resourceType === 'service' &&
            edge.resourceType === 'service'
          ) {
            return;
          }
          if (edge?.resourceName) return;

          this.graphData.edges.push({
            source: node.id,
            target: edge.id,
            attrs: {
              line: {
                stroke: '#A2B1C3',
                strokeWidth: 1,
              },
              padding: 102,
            },
          });
        });
      });
    });
  }

  zoomEventSubscription: Subscription;
  headerEventSubscription: Subscription;
  listenToHeaderEvents() {
    this.zoomEventSubscription =
      this.blueprintDesignerGraphService.zoomEvent.subscribe((zoomType) => {
        this.onZoom(zoomType);
      });
    this.headerEventSubscription =
      this.blueprintDesignerGraphService.eventFromHeader.subscribe(
        (eventType) => {
          let action = eventType?.action;
          let resource = eventType?.extras;
          switch (action) {
            case 'addedResource':
              this.resourceToFocus = resource;
              this.initGraphComponent();
              break;
          }
        },
      );
  }

  // If we are creating a new resource node, we should be bringing that into focus
  // once the node gets added to the graph. This resource node will only be
  // coming into focus once and then we will be setting this back to null so that
  // we don't end up centering the same node again when it's not really needed
  resourceToFocus: any;
  bringNodeToFocus() {
    let resourceMap = this.blueprintDesignerUtilService.getAllNodesMap();
    let resourceNode = resourceMap
      .get(this.resourceToFocus?.resourceType)
      .find(
        (resource) =>
          resource.resourceName === this.resourceToFocus?.resourceName,
      );
    this.blueprintDesignerUtilService.setActiveNode(resourceNode);
    this.blueprintDesignerUtilService.bringNodeInFocus(resourceNode);
    this.resourceToFocus = null;
  }

  onZoom(zoomType) {
    const zoom = this.graph.zoom();
    if (
      (zoom <= 0.5 && zoomType === 'out') ||
      (zoom >= 2 && zoomType === 'in')
    ) {
      return;
    }
    if (zoomType === 'in') {
      this.graph.zoom(0.05);
    } else {
      this.graph.zoom(-0.05);
    }
    this.zoom = Math.round(100 * this.graph.zoom());
  }

  getZoom() {
    return Math.round(100 * this.graph.zoom());
  }

  blockOverlayOn: boolean = false; // Blocks the graph component by adding an overlay on top
  // whenever the branch is anything other than the default one
  async checkForBlockOverlay() {
    const defaultBranchName = await this.getDefaultBranch(this.blueprint);
    this.blockOverlayOn = defaultBranchName !== this.branchName;
    if (this.blockOverlayOn && !this.singleResourceID) {
      this.customToasterService.info({
        content:
          'The graph view is only available in the default branch. Please select the default branch from the top to view the architecture graph.',
        duration: 5000,
      });
    }
  }

  async getDefaultBranch(blueprint: string) {
    return new Promise((resolve) => {
      this.blueprintDesignerService.listBranchesUsingGET(blueprint).subscribe(
        (data) => {
          resolve(data.defaultBranch);
        },
        (error) => {
          console.log(error);
        },
      );
    });
  }
}
