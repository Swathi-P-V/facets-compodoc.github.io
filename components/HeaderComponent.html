import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, NavigationEnd, Params, Router } from '@angular/router';
import { SimpleOauth2User } from '../../cc-api/models/simple-oauth-2user';
import { ApplicationControllerService } from '../../cc-api/services/application-controller.service';
import { UiStackControllerService } from '../../cc-api/services/ui-stack-controller.service';
import { AbstractCluster } from '../../cc-api/models/abstract-cluster';
import { NbMenuService, NbSidebarService } from '@nebular/theme';
import { filter } from 'rxjs';
import { HttpClient } from '@angular/common/http';
import { CookieUtilWrapperService } from 'src/app/services/cookie-util-wrapper.service';
import { SessionContextStoreService } from 'src/app/services/session-context-store.service';

@Component({
  selector: 'app-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss'],
})
export class HeaderComponent implements OnInit {
  isClusterContext = true;
  user: SimpleOauth2User;
  currentNav = '';
  private BASE_URL = '/';
  currentClusterURL;
  clusters: AbstractCluster[];
  isDemo = '';
  userItems: any = [
    { title: 'Personal Tokens', link: '/users/tokens' },
    { title: 'Log out' },
  ];
  settingsItems: any = [
    { title: 'Users', link: '/users' },
    { title: 'Notifications', link: '/notification-center' },
    { title: 'Artifactories', link: '/artifactories' },
    { title: 'OAuth Integrations', link: '/users/oauthClients' },
  ];
  stackClusters = {};
  newNav = [];
  stackName: string;
  userName: any;
  logo: any;
  blueprints = [];

  constructor(
    private route: ActivatedRoute,
    private applicationControllerService: ApplicationControllerService,
    private router: Router,
    private uiStackControllerService: UiStackControllerService,
    private cookieService: CookieUtilWrapperService,
    private sessionContextStoreService: SessionContextStoreService,
    private nbMenuService: NbMenuService,
    protected http: HttpClient,
    public sidebarService: NbSidebarService,
  ) {}

  ngOnInit(): void {
    this.http
      .get('/public/v1/logo', { responseType: 'text' })
      .subscribe((x) => {
        this.logo = x;
      });
    this.router.events
      .pipe(filter((event) => event instanceof NavigationEnd))
      .subscribe((x) => {
        this.findRealRoot();
      });

    this.route.params.subscribe((p) => {
      this.findRealRoot();
    });
    this.isDemo = this.cookieService.get('isDemo');
    this.currentNav = window.location.pathname;
    this.applicationControllerService
      .meUsingGET()
      .subscribe((x: SimpleOauth2User) => {
        this.user = x;
        this.userName = x?.attributes['name'] || x?.name;
      });
    this.addCallbacksToUserMenu();
  }

  private findRealRoot() {
    let root = this.router.routerState.snapshot.root;
    while (root) {
      if (root?.children && root?.children?.length) {
        root = root?.children[0];
      } else if (root?.params) {
        this.currentNav = window.location.pathname;
        const p = root?.params;
        this.initializeStackPage(p);
        return;
      }
    }
  }

  private initializeStackPage(p: Params) {
    if (!p.clusterId) {
      this.isClusterContext = false;
    } else {
      this.isClusterContext = true;
      this.currentClusterURL = p.stackName + '/cluster/' + p.clusterId;
    }
    if (p.stackName && p.stackName !== this.stackName) {
      this.stackName = p.stackName;
      //When a new blueprint or environment is created the page reloads and  this.stackClusters={}
      if (Object.keys(this.stackClusters)?.length == 0) {
        this.getAllEnvironments();
      }
    }
  }

  addCallbacksToUserMenu(): void {
    this.nbMenuService
      .onItemClick()
      .pipe(
        filter(({ tag }) => tag === 'user-menu'),
        filter(({ item: { title } }) => title === 'Log out'),
      )
      .subscribe((title) => {
        this.cookieService.delete('selection', '/');
        const currentContext = this.cookieService.get('currentContext');
        this.sessionContextStoreService.setSessionContext(
          currentContext,
          this.userName,
          'false',
        );
        this.http.get(`/perform_logout`).subscribe(
          (x: SimpleOauth2User) => {},
          (error) => {
            window.location.reload();
          },
        );
      });
  }

  navToPage(): void {
    const navParts = this.currentNav.split('/');
    const trail = navParts.slice(5).join('/');
    const url = this.BASE_URL + this.currentClusterURL + '/' + trail;
    this.router.navigate([url]);
  }

  navToPageLeft($event: any) {
    this.router.navigate([$event]);
  }

  private getAllEnvironments() {
    this.uiStackControllerService.getStacksUsingGET1().subscribe(
      (blueprints) => {
        this.blueprints = blueprints;
        this.blueprints.forEach((b) => {
          this.uiStackControllerService
            .getClustersUsingGET1(b.name)
            .subscribe((c) => {
              this.stackClusters[b.name] = c;
            });
        });
      },
      (err) => {
        console.log(err);
      },
    );
  }
}
