import {
  Component,
  Input,
  Output,
  EventEmitter,
  SimpleChanges,
  OnChanges,
} from '@angular/core';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params } from '@angular/router';
import { FormControl, FormGroup, FormGroupDirective } from '@angular/forms';
import { Options } from './single-options.model';

@Component({
  selector: 'app-select-single-search',
  templateUrl: './select-single-search.component.html',
  styleUrls: ['./select-single-search.component.scss'],
  providers: [FormGroupDirective],
})
export class SelectSingleSearchComponent implements OnChanges {
  //events
  @Output() selectedOptionEvent = new EventEmitter<any>();
  @Output() removeOptionEvent = new EventEmitter<string>();
  @Output() infoClick = new EventEmitter<any>();

  //inputs
  @Input() options: Options[] = [];
  @Input() optionsOriginal: Options[] = [];
  @Input() searchInputId: string = '';
  @Input() selectInputPlaceholder: string = 'Select options';
  @Input() searchInputPlaceholder: string = 'Search options..';
  @Input() selectInputLabel: string = '';
  @Input() multiple: boolean = false;
  @Input() required: boolean = false;
  @Input() showClearAllOptionsIcon: boolean = true;
  @Input() selectTooltipText: string = '';
  @Input() showLimitedTags: boolean = false;
  @Input() tagLimit: number = 2;
  @Input() showSelectionsOutside: boolean = true;
  @Input() inputColSize: number = 5;
  @Input() labelColSize: number = 3;
  @Input() showSpinner: boolean = false;
  @Input() selectionHeight: string = '15vh';
  @Input() showRemovableOptionTags: boolean = true;
  @Input() showCombinedText: boolean = false;
  @Input() disableSelectInput: boolean = false;
  @Input() selectionTextLabel: string = '';
  @Input() disabledInputTooltip: string = '';
  @Input() showDisabledInputTooltip: boolean = false;
  @Input() showUserComponent: boolean = false;
  @Input() showOptionsIcon: boolean = false;
  @Input() showSelectAll: boolean = false;
  @Input() parentClass: string = '';
  @Input() showTags: boolean = true;
  @Input() showInfoIcon: boolean = false;
  @Input() infoIcon: string = 'info-outline';

  //local variables
  selectedOptionsList: {
    label: string;
    value: string;
    icon?: string;
    iconPack?: string;
  }[] = [];
  openOptionsPopover: boolean = false;
  selectAllControl: FormControl = new FormControl(false);
  searchString: FormControl = new FormControl('');

  constructor(private activatedRoute: ActivatedRoute) {}

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {}

  ngOnChanges(changes: SimpleChanges) {
    this.selectedOptionsList = [];
    //show spinner for dropdown before data loads
    this.showSpinner = changes?.showSpinner?.currentValue || false;
    if (changes?.options?.currentValue) {
      this.optionsOriginal = changes?.options?.currentValue;
      changes?.options?.currentValue.forEach((option) => {
        if (option?.checked) {
          this.selectedOptionsList.push({
            label: option?.label,
            value: option?.value,
          });
        }
      });
      this.setSelectAllCheckboxStatus();
    }
  }

  changeSelectOptionValue(event, selectedOption) {
    this.selectedOptionsList = [];
    if (!this.multiple) {
      this.options.forEach((option) => {
        if (option?.value !== selectedOption?.value) {
          option.checked = false;
        }
      });
    }
    this.optionsOriginal.forEach((option) => {
      if (option?.checked) {
        this.selectedOptionsList.push({
          label: option?.label,
          value: option?.value,
        });
      }
    });
    this.emitOptionEvents();
  }

  /*
      Filter function
    */
  getFilteredOptions() {
    if (this.searchString.value?.length > 0) {
      this.options = [
        ...this.searchOptions(
          this.optionsOriginal,
          'label',
          this.searchString.value,
        ),
      ];
    } else {
      this.options = [...this.optionsOriginal];
    }
  }

  searchOptions(options, key, searchStr) {
    // Use Array.filter() to only keep objects whose labels match the search string
    const filteredArr = options.filter((option) =>
      option[key].toLowerCase().includes(searchStr.toLowerCase()),
    );
    return filteredArr;
  }

  removeClicked(filterOption) {
    this.options.find((option) => option.value === filterOption.value).checked =
      false;
    this.selectedOptionsList = this.selectedOptionsList.filter(
      (option) =>
        !(
          option.label === filterOption?.label &&
          option.value === filterOption?.value
        ),
    );
    this.emitOptionEvents();
    this.removeOptionEvent.emit(filterOption);
  }

  /*
   * Clear all selected options when user clicks on close button on top level select input
   */
  clearAllSelectedOptions() {
    this.selectedOptionsList = [];
    this.emitOptionEvents();
    this.options.forEach((option) => (option.checked = false));
  }

  setSelectAllCheckboxStatus() {
    this.selectAllControl.setValue(
      this.options.every((option) => option.checked),
    );
  }

  emitOptionEvents() {
    this.setSelectAllCheckboxStatus();
    this.selectedOptionEvent.emit(this.selectedOptionsList);
  }

  showStateChange(event) {
    this.openOptionsPopover = event.isShown;
    if (!event.isShown) {
      this.searchString.setValue('');
      this.options = [...this.optionsOriginal];
    } else {
      this.openOptionsPopover = true;
      let selectBoundingRect = document
        .querySelector('.input-parent')
        .getBoundingClientRect();
      let popoverElem = document.querySelector('nb-popover');
      let dropdownElement = popoverElem?.parentElement;
      if (dropdownElement) {
        dropdownElement.style.width = selectBoundingRect.width + 'px';
        dropdownElement.style.left = selectBoundingRect.left + 'px';
      }
    }
  }

  onSelectAllClick(event) {
    this.selectedOptionsList = [];
    this.options.forEach((option) => {
      option.checked = event;
    });
    this.optionsOriginal.forEach((option) => {
      if (option?.checked) {
        this.selectedOptionsList.push({
          label: option?.label,
          value: option?.value,
        });
      }
    });
    this.emitOptionEvents();
    this.searchString.setValue('');
  }

  onInfoBtnClick(event, option) {
    this.infoClick.emit(option);
    event.preventDefault();
  }
}
