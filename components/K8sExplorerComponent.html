  constructor(
    private menuService: NbMenuService,
    private k8sService: UiKubernetesExplorerControllerService,
    private activatedRoute: ActivatedRoute,
    private dialogService: NbDialogService,
    private httpClient: HttpClient,
    private uiApplicationControllerService: UiApplicationControllerService,
    private clipboardService: ClipboardService,
    private router: Router,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
    this.menuItemSubscriber = this.menuService
      .onItemClick()
      .subscribe(this.onMenuItemClick);
    this.handleClipboardResponse();
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p?.clusterId;
    this.resourceName = p?.resourceName;
    this.resourceType = p?.resourceType;

    this.onMenuItemClick({ item: { data: { id: this.selectedMenuItemId } } });
  }

  getFormattedAPIResponse = (response) => {
    return this.selectedMenuItemId === 'autoscaling_hpa'
      ? response.map(({ metric, replicaStatus, ...rest }) => ({
          ...rest,
          ...metric,
          ...replicaStatus,
        }))
      : response;
  };

  getFinalListOfColumns = (apiResponse, parentId) => {
    let columns = tableColumnOrderMap[this.selectedMenuItemId];
    columns = columns.filter((column) => {
      if (
        (this.selectedMenuItemId === 'workload_daemonSets' &&
          column === 'nodeSelector') ||
        (this.selectedMenuItemId === 'workload_statefulsets' &&
          (column === 'podCount' || column === 'status'))
      ) {
        return false;
      }
      return true;
    });

    columns.push('actions');

    return columns;
  };

  setTableSettings = (apiResponse, parentId) => {
    const tableSettings = cloneDeep(defaultTableSettings);
    let columns = this.getFinalListOfColumns(apiResponse, parentId);
    columns.forEach((column) => {
      tableSettings.columns[column] = {
        title: this.tableColumnNameMap[column] || camelToPascal(column),
        filter: false,
        width: column === 'ageInSeconds' ? '12%' : undefined,
        valuePrepareFunction: (val, row) => {
          const valueMap = {
            ageInSeconds: secondsToDuration(val),
            durationInSeconds: secondsToDuration(val),
            podStatus: row.podStatus?.completions,
            lastSchedule: JSON.stringify({
              user: '',
              label: row.lastSchedule,
            }),
            actions: JSON.stringify({
              smartTableCustomActionButtons: this.actionsColumn,
            }),
          };

          if (valueMap[column]) {
            return valueMap[column];
          }

          switch (column) {
            case 'cpuUtilization':
              return row?.cpuUtilization
                ? `${row?.cpuUtilization?.current}% / ${row?.cpuUtilization?.target}%`
                : '';
            case 'memoryUtilization':
              return row?.memoryUtilization
                ? `${row?.memoryUtilization?.current}% / ${row?.memoryUtilization?.target}%`
                : '';
            case 'replicas':
              return this.selectedMenuItemId === 'workload_statefulsets'
                ? `${row.podCount}/${row.replicas}`
                : val;
          }

          if (
            this.selectedMenuItemId === 'workload_pods' &&
            column === 'name'
          ) {
            const displayVal = {
              display: val,
              isRouterLink: true,
              showImg: false,
              iconName: iconAndTextMap[row.role]?.icon || '',
              iconTooltip: iconAndTextMap[row.role]?.text || '',
              iconStatus: iconAndTextMap[row.role]?.status,
              iconFontSize: '10px',
              iconClass: iconAndTextMap[row?.role]?.iconClass,
            };
            return JSON.stringify(displayVal);
          }
          return val;
        },
        onComponentInitFunction: (instance: any) => {
          instance.byCustomActionButtonClick?.subscribe((data) => {
            let customActionButton = this.actionsColumn.filter(
              (actionButton) => {
                return actionButton?.id === data?.actionButton?.id;
              },
            )[0];
            customActionButton.onClick(data?.rowData);
          });
          instance?.byLinkClick?.subscribe((data) => {
            this.router.navigate(['pod', data?.name, 'containers'], {
              relativeTo: this.route,
            });
          });
        },
        renderComponent:
          tableColumnComponentMap[column] ||
          tableColumnComponentMap[`${this.selectedMenuItemId}_${column}`],
        type:
          Object.keys(tableColumnComponentMap).includes(column) ||
          tableColumnComponentMap[`${this.selectedMenuItemId}_${column}`]
            ? 'custom'
            : undefined,
      };
    });
    this.tableSettings = Object.assign({}, tableSettings, {
      noDataMessage: `No ${
        this.selectedMenuItemNameMap[this.selectedMenuItemId]
      } found`,
    });
  };

  async loadDataForSelectedMenu() {
    this.emptyMessage = '';
    this.httpSubscriber?.unsubscribe();
    const [parentId, childMenuItemId] = this.selectedMenuItemId?.split('_');
    const apiResource = this.apiResourceMap[childMenuItemId];
    const url = `${this.rootUrl}/cc-ui/v1/clusters/${this.clusterId}/k8s-explorer/${apiResource}?resourceType=${this.resourceType}&resourceName=${this.resourceName}`;

    this.httpSubscriber = this.httpClient.get(url).subscribe(
      (response) => {
        this.tableLoader = false;
        const apiResponse = this.getFormattedAPIResponse(response);

        if (
          this.selectedMenuItemId === 'workload_pods' &&
          this.resourceType === 'service'
        ) {
          console.log('inside loadDataForSelectedMenu');
          this.byGetPods.emit(apiResponse);
        }
        this.emptyMessageState();
        this.setTableSettings(apiResponse, parentId);
        this.tableData = apiResponse?.slice();
      },
      (error) => {
        this.tableLoader = false;
        this.showEmptyTable = false;
        this.errorMessage =
          error?.error?.message ||
          `Error while fetching ${
            this.selectedMenuItemNameMap[this.selectedMenuItemId]
          }`;
      },
    );
  }
  onMenuItemClick = (event) => {
    this.tableLoader = true;
    this.emptyMessage = '';
    this.errorMessage = '';
    this.tableData = [];
    this.tableSettings = cloneDeep(defaultTableSettings);

    this.selectedMenuItemId = event?.item?.data?.id;

    this.showEmptyTable = menuItemsForEmptyTable?.includes(
      this.selectedMenuItemId,
    );

    this.setSelectedMenuStatus();
    this.loadDataForSelectedMenu();
  };

  setSelectedMenuStatus() {
    const [parentId] = this.selectedMenuItemId?.split('_') || [];
    this.menuItems = cloneDeep(K8sSidebarMenuItems);
    const parentMenu = this.menuItems?.find((m) => m?.data?.id === parentId);
    if (parentMenu) {
      parentMenu.expanded = true;
      const childMenu = parentMenu?.children?.find(
        (m) => m?.data?.id === this.selectedMenuItemId,
      );
      if (childMenu) {
        childMenu.selected = true;
      }
    }
  }

  onClickActionsIcon(rowData) {
    this.selectedItemForDropdown = rowData.name;
    this.logsUrl = `${this.rootUrl}/cc-ui/v1/clusters/${this.clusterId}/pods/${this.selectedItemForDropdown}/logs?tail=${this.tailLogsParams?.tail}&resourceName=${this.resourceName}&resourceType=${this.resourceType}`;
    this.dialogService.open(this.actionsModal, {
      context: {
        ...rowData,
      },
    });
  }

  getEvents() {
    this.getEventsSubscriber?.unsubscribe();
    this.eventsLoader = true;
    this.eventsEmptyMessage = '';
    this.eventsErrorMessage = '';
    const eventsTableSettings = cloneDeep(defaultTableSettings);
    this.eventsTableData = [];
    const menuItemId = this.selectedMenuItemId?.split('_')?.[1];

    const url = `${this.rootUrl}/cc-ui/v1/clusters/${this.clusterId}/k8s-explorer/${this.k8sResourceTypeMapping[menuItemId]}/${this.selectedItemForDropdown}/events?resourceName=${this.resourceName}&resourceType=${this.resourceType}`;
    // listEventsByK8sResourceUsingGET
    this.getEventsSubscriber = this.httpClient.get(url).subscribe(
      (response: any) => {
        this.eventsLoader = false;

        if (!response?.length) {
          this.eventsEmptyMessage = 'No Events Found';
          return;
        }

        const columns = Object.keys(response?.[0] || {});
        columns.forEach((column) => {
          if (column !== 'object') {
            eventsTableSettings.columns[column] = {
              title: camelToPascal(column),
              filter: false,
              valuePrepareFunction: (cell, row) => {
                if (column === 'lastSeen') {
                  return JSON.stringify({
                    user: '',
                    label: row.lastSeen,
                  });
                }

                return cell;
              },
              renderComponent: tableColumnComponentMap[column],
              type: Object.keys(tableColumnComponentMap)?.includes(column)
                ? 'custom'
                : undefined,
            };
          }
        });
        this.eventsTableSettings = eventsTableSettings;
        this.eventsTableData = response;
      },
      (error) => {
        this.eventsLoader = false;
        this.eventsErrorMessage =
          error?.error?.message || 'Error while fetching events';
      },
    );
  }

  getManifest() {
    this.getManifestLoader = true;
    this.fileContent = null;
    const menuItemId = this.selectedMenuItemId?.split('_')?.[1];
    const url = `${this.rootUrl}/cc-ui/v1/clusters/${this.clusterId}/k8s-explorer/${this.k8sResourceTypeMapping[menuItemId]}/${this.selectedItemForDropdown}/manifest?resourceName=${this.resourceName}&resourceType=${this.resourceType}`;
    //  k8sService.getManifestUsingGET
    this.httpClient.get(url, { responseType: 'text' }).subscribe(
      (response: any) => {
        this.getManifestLoader = false;
        this.fileContent = response;
      },
      (error) => {
        this.getManifestLoader = false;
        this.onAPIErrorService.onAPIError(
          error?.error?.message || 'Error fetching manifest file',
          error,
        );
      },
    );
  }

  getDataBasedOnTab() {
    switch (this.selectedTabId) {
      case 'logs':
        break;
      case 'events':
        this.getEvents();
        break;
      case 'manifest':
        this.getManifest();
        break;
    }
  }
  onChangeTab(data) {
    this.selectedTabId = data?.tabId;
    this.getDataBasedOnTab();
  }

  onChangeDropdown(event) {
    this.selectedItemForDropdown = event;
    this.getDataBasedOnTab();
  }

  onClickRefresh() {
    this.tableLoader = true;
    this.loadDataForSelectedMenu();
  }
  confirmRollingRestart(confirmRollingRestartUI) {
    this.dialogService.open(confirmRollingRestartUI);
  }
  rollingRestartAll(ref) {
    this.rollingRestartInProgress = true;
    const params = new HttpParams()
      .set('resourceName', this.resourceName)
      .set('resourceType', this.resourceType);
    // uiApplicationControllerService.rollingRestartUsingPOST
    this.httpClient
      .post(
        `${this.rootUrl}/cc-ui/v1/clusters/${this.clusterId}/restart/${this.resourceName}`,
        null,
        { params: params },
      )
      .subscribe(
        (data) => {
          this.rollingRestartInProgress = false;
          ref.close();
          this.customToastService.success('Rolling Restart Started');
          this.onClickRefresh();
        },
        (error) => {
          this.rollingRestartInProgress = false;
          ref.close();
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Failed to trigger Rolling Restart',
            error,
          );
        },
      );
  }
  copyLogsToClipboard() {
    let logLines = '';
    const logs = this.tailLogsComponent.getLogsData();
    logs.forEach((logLine) => {
      logLines += `${logLine.message}\n`;
    });
    this.clipboardService.copy(logLines);
  }
  handleClipboardResponse() {
    this.clipboardServiceSubscriber =
      this.clipboardService.copyResponse$.subscribe((res: any) => {
        if (res.isSuccess) {
          this.customToastService.show({
            title: 'Copied!',
            content: 'Logs copied to clipboard',
            duration: 1000,
            type: 'success',
          });
        }
      });
  }
  emptyMessageState() {
    const menuItemName = this.selectedMenuItemNameMap[this.selectedMenuItemId];
    this.emptyMessage = `No ${menuItemName} found`;
    const newTableSettings = cloneDeep(this.tableSettings);

    this.tableSettings = {
      ...newTableSettings,
      noDataMessage: this.emptyMessage,
    };
  }
  ngOnDestroy() {
    this.httpSubscriber?.unsubscribe();
    this.menuItemSubscriber?.unsubscribe();
    this.getEventsSubscriber?.unsubscribe();
    this.clipboardServiceSubscriber?.unsubscribe();
  }
}
