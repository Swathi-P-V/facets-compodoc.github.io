import { Component } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { ENVIRONMENT_CONFIGURE } from 'src/app/constants/permissions';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { IntentDetailsService } from 'src/app/services/intent-details.service';

@Component({
  selector: 'app-resource-configuration',
  templateUrl: './resource-configuration.component.html',
  styleUrls: ['./resource-configuration.component.scss'],
})
export class ResourceConfigurationComponent extends FacetsPage {
  objectKeys = Object.keys;
  clusterId: string;
  appName: string;
  stackName: string;
  resourceType: string;
  resourceName: string;
  editorMode: boolean = false;
  editorModeName: 'json' | 'form';

  editButton = {
    id: 'edit',
    title: 'Edit',
    icon: 'edit-outline',
    buttonType: 'primary',
    nbTooltip: 'Edit resource configuration',
    action: ENVIRONMENT_CONFIGURE,
    click: () => {
      this.toggleEditor('on');
    },
  };

  jsonModeButton = {
    id: 'editor',
    title: 'JSON Mode',
    buttonType: 'primary',
    nbTooltip: 'Switch to JSON editor mode',
    click: () => {
      this.toggleEditorMode('json');
    },
  };

  formModeButton = {
    id: 'editor',
    title: 'Form Mode',
    buttonType: 'primary',
    nbTooltip: 'Switch to Form editor mode',
    click: () => {
      this.toggleEditorMode('form');
    },
  };

  actionButtons: any[] = [
    {
      id: 'history',
      title: 'History',
      buttonType: 'secondary',
      icon: 'file-text-outline',
      status: 'primary',
      nbTooltip: 'View version history for resource configurations',
      click: () => {
        this.navigateToHistory();
      },
    },
  ];

  documentationButton: any = {
    id: 'schema-documentation',
    title: 'Spec Documentation',
    buttonType: 'secondary',
    icon: 'external-link',
    status: 'primary',
    nbTooltip: 'Check resource schema documentation',
    click: () => {
      this.navigateToDocumentation();
    },
  };

  showAppConfig = false;
  documentationURL: string = '';

  constructor(
    _route: ActivatedRoute,
    private router: Router,
    private intentDetailsService: IntentDetailsService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p?.clusterId;
    this.appName = p?.resourceName;
    this.stackName = p?.stackName;
    this.resourceType = p?.resourceType;
    this.resourceName = p?.resourceName;
    this.editorMode = false;
    this.documentationURL = '';
    this.showAppConfig =
      this.resourceType === 'application' ||
      this.resourceType === 'windows-application';
    this.editorModeName = this.showAppConfig ? 'form' : 'json';
    this.actionButtons = this.actionButtons.filter(
      (actionButton) => actionButton.id !== 'editor',
    );
    if (
      this.actionButtons.findIndex(
        (actionButton) => actionButton.id === 'edit',
      ) === -1
    ) {
      this.actionButtons.unshift(this.editButton);
    }
    if (!this.intentDetailsService?.intentLoaded) {
      this.intentDetailsService.updateIntentDataSubject.subscribe(
        (intentStatus) => {
          if (intentStatus) {
            this.getDocumentationUrl();
          }
        },
      );
    } else {
      this.getDocumentationUrl();
    }
  }

  getDocumentationUrl() {
    this.documentationURL = this.intentDetailsService.getIntentDetails(
      this.resourceType,
    )?.documentation;
    if (
      this.documentationURL &&
      !this.actionButtons?.find(
        (button) => button?.id === 'schema-documentation',
      )
    ) {
      this.actionButtons.push(this.documentationButton);
    } else if (!this.documentationURL?.length) {
      this.actionButtons = this.actionButtons.filter(
        (button) => button?.id !== 'schema-documentation',
      );
    }
  }

  navigateToDocumentation() {
    window.open(this.documentationURL, '_blank');
  }

  navigateToEdit() {
    this.router.navigate(['edit'], {
      relativeTo: this.route,
    });
  }

  toggleEditor(toggle: 'on' | 'off') {
    if (toggle === 'on') {
      this.editorMode = true;
      this.actionButtons = this.actionButtons.filter(
        (actionButton) => actionButton.id !== 'edit',
      );
      this.setEditorModeButton();
    } else {
      this.editorMode = false;
      this.actionButtons = this.actionButtons.filter(
        (actionButton) => actionButton.id !== 'editor',
      );
      this.actionButtons.unshift(this.editButton);
    }
  }

  toggleEditorMode(mode: 'json' | 'form') {
    this.editorModeName = mode;
    this.setEditorModeButton();
  }

  setEditorModeButton() {
    this.actionButtons = this.actionButtons.filter(
      (actionButton) => actionButton.id !== 'editor',
    );
    if (!this.showAppConfig) {
      this.actionButtons.filter((actionButton) => actionButton.id !== 'editor');
    } else {
      this.actionButtons.unshift(
        this.editorModeName === 'form'
          ? this.jsonModeButton
          : this.formModeButton,
      );
    }
  }

  navigateToHistory() {
    this.router.navigate(['version-history'], {
      relativeTo: this.route,
    });
  }

  handleBack = () => {
    this.router.navigate([
      '/',
      this.stackName,
      'cluster',
      this.clusterId,
      'resources',
    ]);
  };

  handleActions(event) {
    this.actionButtons?.find((button) => button.id === event)?.click();
  }
}

<app-page-header
  [heading]="showAppConfig && !editorMode ? 'Effective Configuration' : ''"
  [actionButtons]="actionButtons"
  (click)="handleActions($event)"
  [isTabHeader]="true"
></app-page-header>

<ng-container *ngIf="!editorMode">
  <ng-container *ngIf="showAppConfig">
    <app-app-configuration></app-app-configuration>
  </ng-container>
  <ng-container *ngIf="!showAppConfig">
    <app-simple-resource-config></app-simple-resource-config>
  </ng-container>
</ng-container>
<ng-container *ngIf="editorMode">
  <app-configuration-editor
    *ngIf="editorModeName === 'json'"
    [stackName]="stackName"
    [clusterId]="clusterId"
    [resourceType]="resourceType"
    [resourceName]="resourceName"
    (byClose)="toggleEditor('off')"
  ></app-configuration-editor>
  <app-app-config-edit
    *ngIf="editorModeName === 'form'"
    (byClose)="toggleEditor('off')"
  ></app-app-config-edit>
</ng-container>
