  OnPageLoad(p: Params): void {
    this.baseEnvOptions = [];
    this.stackName = p?.stackName;
    this.clusterId = p?.clusterId;
    const URLFragments = this.router.url.split('/');

    this.clusterRequestObject = {};
    this.cloudAccountId?.reset();
    if (!this.clusterId) {
      this.cloud = URLFragments[URLFragments?.length - 1];

      //get the dropdown values for regions and availability zones
      this.getRegionsDropdown();
    }

    this.activatedroute.paramMap
      .pipe(map(() => window.history.state))
      .subscribe((data) => {
        this.previousComponent = data.previousComponent;
      });

    //Get environment list for CD parent dropdown
    this.getCDParentDropdown();

    this.getReleaseStreamDropdown();

    //Set default values for the form fields if create environment
    if (!this.clusterId) {
      this.getAccountDetails();
      this.getAllEnvironments();
      this.clusterRequestObject.tz = 'GMT';
      this.clusterRequestObject.instanceTypes =
        this.spotInstanceDefaults[this.cloud];
      this.clusterRequestObject.requireSignOff = false;
      this.clusterRequestObject.isEphemeral = false;
      this.clusterRequestObject.namespace =
        this.cloud === 'kubernetes' ? 'default' : '';
      this.clusterRequestObject.k8sRequestsToLimitsRatio = 1;
    } else if (p.clusterId) {
      this.initCloudSpecificClusterObject();
    }

    //Initialise cluster request object
  }

      .getReleaseStreamsUsingGET()

          return { value: releaseStream, label: releaseStream };
        });
      });
  }

  getAccountDetails(accountId = '') {
    this.pageLoading = true;
    this.awsAccounts = [];
    this.azureAccounts = [];
    this.gcpAccounts = [];
    this.selectAccounts = [];
    this.accountControllerService.getAccountsByTypeUsingGET('CLOUD').subscribe(
      (response) => {
        this.accounts = response;
        this.pageLoading = false;
        this.accounts = response;
        this.pageLoading = false;
        const cloudAccounts = this.accounts?.filter(
          (account) =>
            account?.provider?.toLowerCase() === this.cloud?.toLowerCase(),
        );
        this.selectAccounts = cloudAccounts?.map((account) => ({
          label: account?.name,
          value: account?.id,

  incrementCIDR(existingCIDRs) {
    // Helper function to extract second value after dot(.) from a CIDR range.
    function extractXXValue(cidr) {
      return parseInt(cidr?.split('.')?.[1]);
    }

    // Start with 10.0.0.0/16.
    let xxValue = -1;

    while (true) {
      xxValue++;

      if (xxValue === 256) {
        xxValue = 0;
      }

      // Format the XX value as a single digit if it's between 0 and 9,
      // or as a double digit if it's greater than 9.
      const formattedXXValue = `${xxValue}`;

      let nextCIDR = `10.${formattedXXValue}.0.0/16`;

      // Check if the nextCIDR conflicts with any existing CIDRs.
      const conflict = existingCIDRs.some((existingCIDR) => {
        const existingXXValue = extractXXValue(existingCIDR);
        return xxValue === existingXXValue;
      });

      if (!conflict) {
        return nextCIDR;
      }
    }
  }

    const cidrs = this.allEnvs.map((cluster) => cluster?.vpcCIDR);
    const nextCIDR = this.incrementCIDR(cidrs);

      httpReq =

          cloudRequestObject,
        );
    } else if (this.cloud === 'gcp') {
      httpReq =
        this.gcpClusterController.createGCPClusterUsingPOST(cloudRequestObject);
    } else if (this.cloud === 'kubernetes') {
      httpReq =
        this.kubernetesClusterController.createK8sClusterUsingPOST(

              Validators.pattern(whiteSpaceInputPattern),

            .get('resourceGroupName')
            .setValidators([Validators.pattern(whiteSpaceInputPattern)]);
        }
        this.azureAdvancedForm
          .get('resourceGroupName')
          .updateValueAndValidity();
      });

    this.azureAdvancedForm
      .get('resourceGroupName')
      .valueChanges.pipe(distinctUntilChanged())
      .subscribe((value) => {
        if (value) {
          this.azureAdvancedForm
            .get('vnetName')
            .setValidators([
              Validators.required,
              Validators.pattern(whiteSpaceInputPattern),
            ]);

          this.azureAdvancedForm
