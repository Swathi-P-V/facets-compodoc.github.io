import {
  Component,
  Input,
  OnChanges,
  OnDestroy,
  SimpleChanges,
} from '@angular/core';
import { SafeUrl, DomSanitizer } from '@angular/platform-browser';
import { ActivatedRoute, Params } from '@angular/router';
import { Subscription } from 'rxjs';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CookieUtilWrapperService } from 'src/app/services/cookie-util-wrapper.service';
import { SidebarStateService } from 'src/app/services/sidebar-state.service';
import { UiCommonClusterControllerService } from 'src/app/cc-api/services';

@Component({
  selector: 'app-tools-grafana',
  templateUrl: './tools-grafana.component.html',
  styleUrls: ['./tools-grafana.component.scss'],
})
export class ToolsGrafanaComponent
  extends FacetsPage
  implements OnDestroy, OnChanges
{
  @Input() grafanaURL: string;
  @Input() clusterId: string;

  grafanaTunnel: string;
  grafanaTunnelSafe: SafeUrl;
  previousState: string;

  expandIframe: boolean = false;
  sidebarStateSubscription: Subscription;
  showButton: boolean = false;

  observer: MutationObserver;
  iframeReady: boolean = false;
  cluster: any = null;

  constructor(
    private activatedRoute: ActivatedRoute,
    private sanitizer: DomSanitizer,
    private sidebarStateService: SidebarStateService,
    private cookieService: CookieUtilWrapperService,
    private uiCommonClusterControllerService: UiCommonClusterControllerService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    if (p.clusterId) {
      this.clusterId = p.clusterId;
    }

    this.sidebarStateSubscription =
      this.sidebarStateService.sidebarState.subscribe(
        (state: 'collapse' | 'expand') => {
          if (state == 'expand') {
            this.expandIframe = false;
          }
        },
      );

    this.showButton = true;
    if (!this.grafanaURL) {
      this.getEnvironmentConfig();
    }
  }

  getEnvironmentConfig() {
    this.uiCommonClusterControllerService
      .getClusterCommonUsingGET(this.clusterId)
      .subscribe(
        async (cluster) => {
          this.cluster = cluster;
          this.updateGrafanaURL();
        },
        (error) => {
          this.cluster = null;
          this.updateGrafanaURL();
        },
      );
  }

  ngOnChanges(changes: SimpleChanges): void {
    if (changes?.grafanaURL?.currentValue?.length) {
      this.grafanaURL = changes?.grafanaURL?.currentValue;
      this.clusterId = changes?.clusterId?.currentValue;
      this.updateGrafanaURL();
    }
  }

  updateGrafanaURL() {
    this.iframeReady = false;
    this.previousState = this.cookieService.get('grafanaState_tools');
    this.grafanaTunnel = this.grafanaURL
      ? this.grafanaURL
      : `tunnel/${this.cluster?.baseClusterId || this.clusterId}/grafana/`;
    this.grafanaTunnel = window.location.origin + '/' + this.grafanaTunnel;
    if (
      this.previousState &&
      this.previousState != 'undefined' &&
      !this.grafanaURL?.length
    ) {
      this.grafanaTunnel += `${this.previousState}`;
    }
    this.grafanaTunnelSafe = this.makeSafe(this.grafanaTunnel);
    this.iframeReady = true;
  }

  makeSafe(url: string) {
    return this.sanitizer.bypassSecurityTrustResourceUrl(url);
  }

  toggleExpand() {
    this.expandIframe = !this.expandIframe;
    this.sidebarStateService.sendSidebarStateUpdate(
      this.expandIframe ? 'collapse' : 'expand',
    );
  }

  ngOnDestroy(): void {
    this.sidebarStateSubscription?.unsubscribe();
  }

  // This function is called on onload event of iframe.
  // This will only be called when iframe mounts a fresh instance in page or
  // iframe url domain changes or iframe content refreshes entire window to show some page
  // onload event does not get called during normal single page application journey
  setLatestUrl(frames) {
    this.setMutationalObserver(frames);
  }

  // For normal url change detection inside iframe for a single page application, we need to use mutationObserver
  // mutationObserver observes dom changes inside the element we observe the changes on.
  // In this case frames input to the function is an iframe instance with contentWindow property on it which is an window
  // instance of iframe.
  setMutationalObserver(frames) {
    this.observer = new MutationObserver((mutations) => {
      if (frames?.location?.href !== this.grafanaTunnel) {
        let url = frames?.location?.href;
        this.grafanaTunnel = frames?.location?.href;
        if (url) {
          let newState = url.split(`${this.clusterId}/grafana/`)[1];
          if (this.previousState != newState) {
            this.previousState = newState;
            this.storeGrafanaStateInCookies();
          }
        }
      }
    });
    let config = {
      attributes: true,
      childList: true,
      characterData: true,
      subtree: true,
    };
    this.observer.observe(frames?.document?.body, config);
  }

  // Store the current state in the cookies to be later retrieved
  storeGrafanaStateInCookies() {
    if (this.grafanaURL?.length) {
      return;
    }
    if (this.cookieService.get('grafanaState_tools')) {
      this.cookieService.delete('grafanaState_tools', '/');
    }
    this.cookieService.set(
      'grafanaState_tools',
      this.previousState,
      undefined,
      '/',
    );
  }

  openInNewTab() {
    window.open(this.grafanaTunnel, '_blank').focus();
  }
}

        size="small"
        ghost
        *ngIf="showButton"
        (click)="toggleExpand()"
        id="expandBtn"
        [nbTooltip]="!expandIframe ? 'Maximize View' : 'Minimize'"
      >
        <nb-icon
          [icon]="expandIframe ? 'collapse-outline' : 'expand-outline'"
        ></nb-icon>
      </button>
      <iframe
        #iframe
        [src]="this.grafanaTunnelSafe"
        srcdoc="<p>Loading...<p>"
        onload="this.removeAttribute('srcdoc');"
        frameborder="0"
        width="100%"
        height="100%"
        (load)="setLatestUrl(iframe.contentWindow)"
      ></iframe>
      <button
        [disabled]="!grafanaTunnel?.length"
        ghost
        nbButton
        size="small"
        id="openNewTabBtn"
        (click)="openInNewTab()"
        nbTooltip="Open in new tab"
      >
        <nb-icon icon="external-link-outline"></nb-icon>
      </button>
    </ng-container>
  </div>
</div>
