import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  Component,
  EventEmitter,
  Input,
  Output,
  ViewChild,
  OnDestroy,
  OnChanges,
  SimpleChanges,
} from '@angular/core';
import {
  MonacoEditorComponent,
  MonacoEditorConstructionOptions,
  MonacoEditorLoaderService,
  MonacoStandaloneCodeEditor,
} from '@materia-ui/ngx-monaco-editor';
import { take, filter } from 'rxjs';
import { ActivatedRoute, Params } from '@angular/router';
import { AutocompleteResponse } from './../../cc-api/models/autocomplete-response';
import { UiBlueprintDesignerControllerService } from 'src/app/cc-api/services';
import { filterObjectKeys } from 'src/app/utils/object-utils';
import { detectKeyKeywordPresence } from './schema-utils';

@Component({
  selector: 'app-monaco-editor-schema-based',
  templateUrl: './monaco-editor-schema-based.component.html',
  styleUrls: ['./monaco-editor-schema-based.component.scss'],
})
export class MonacoEditorSchemaBasedComponent
  extends FacetsPage
  implements OnDestroy, OnChanges
{
  @Input() code: any;
  @Input() disableChanges: boolean = false;
  @Input() schemaLink: string;
  @Output() byValidityChange: EventEmitter<boolean> = new EventEmitter();

  @ViewChild(MonacoEditorComponent, { static: false })
  monacoComponent: MonacoEditorComponent;
  completionProviderRegistered: boolean = false;
  autoCompleteData: AutocompleteResponse = {};
  refreshing: boolean = false;
  editorOptions: MonacoEditorConstructionOptions = {
    theme: 'vs',
    language: 'json',
    minimap: {
      enabled: true,
    },
    lineNumbers: 'on',
    automaticLayout: true,
    inlineSuggest: {
      enabled: true,
    },
    quickSuggestions: true,
    suggestOnTriggerCharacters: true,
    suggest: {
      snippetsPreventQuickSuggestions: false,
    },
    formatOnPaste: true,
    formatOnType: true,
  };
  stackName: string = '';
  model: any;
  modelUri: any = null;
  filteredOptions: any;
  completionProvider: any;

  constructor(
    private monacoLoaderService: MonacoEditorLoaderService,
    private blueprintDesignerController: UiBlueprintDesignerControllerService,
    private activatedRoute: ActivatedRoute,
  ) {
    super(activatedRoute);
    this.monacoLoaderService.isMonacoLoaded$
      .pipe(
        filter((isLoaded) => isLoaded),
        take(1),
      )
      .subscribe(() => {
        monaco.languages.json.jsonDefaults.setDiagnosticsOptions({
          enableSchemaRequest: true,
          validate: true,
        });
      });
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.stackName = p?.stackName;
  }

  ngOnDestroy(): void {
    if (this.completionProvider) {
      this.completionProvider.dispose();
    }
  }

  ngOnChanges(changes: SimpleChanges): void {
    if (changes?.disableChanges?.currentValue) {
      this.editorOptions.readOnly = true;
    }
  }

  editor: any;
  editorInit(editor: MonacoStandaloneCodeEditor) {
    this.editor = editor;
    this.getAutosuggestData();
    this.listenToCodeChanges();
  }

  getAutosuggestData() {
    this.blueprintDesignerController
      .getAutocompleteDataUsingGET(this.stackName)
      .subscribe((data) => {
        this.autoCompleteData = data;
        if (data?.outProperties?.blueprint) {
          this.autoCompleteData.resources['blueprint'] = ['self'];
        }
        if (data?.resources && Object.keys(data?.resources).length > 0) {
          this.initCompletionProvider();
        }
      });
  }

  //initialize custom completion provider for monaco editor
  initCompletionProvider() {
    /*
     * If any existing instance of completion provider is active, discard it.
     */
    if (this.completionProvider) {
      this.completionProvider.dispose();
    }
    /*
     * Register a new completion provider.
     * provide triggerCharacters so that a single character or combination of characters can be detected when user types them
     * define provideCompletionItems function which will be trigger when trigger character is typed
     * get the entire entire word typed by the user
     * Separate the word from trigger character and detect if trigger character is '${'
     * If trigger character is '${' then show resource type options to the user
     * Otherwise if trigger character is . then show resource name options to the user
     * For subsequent options, map the resource type to the outOptions from the api response and show the options
     */
    if (!this.completionProviderRegistered) {
      let monaco = window['monaco'];
      this.completionProvider = monaco.languages.registerCompletionItemProvider(
        'json',
        {
          triggerCharacters: ['{', '.', '$'], //trigger characters to detect
          provideCompletionItems: (
            model: monaco.editor.ITextModel,
            position: monaco.Position,
          ): monaco.languages.ProviderResult<monaco.languages.CompletionList> => {
            const wordUntilPosition = model.getWordUntilPosition(position);
            const filteredWords = wordUntilPosition?.word
              .split('.')
              .filter((word) => word !== '');
            const lineUntilPosition = model.getValueInRange({
              startLineNumber: position?.lineNumber,
              startColumn: 1,
              endLineNumber: position?.lineNumber,
              endColumn: position?.column,
            });
            const fullLineAtPosition = model.getValueInRange({
              startLineNumber: position?.lineNumber,
              startColumn: 1,
              endLineNumber: position?.lineNumber,
              endColumn: position?.column + 1,
            });
            let selectedResourceType = '';
            /*
             * Get the entire word typed by the user in lineUntilPosition
             * if the word ends with '${' then it is the first trigger character
             * In this case, show the resource type options to the user
             */
            if (lineUntilPosition.endsWith('${')) {
              this.filteredOptions = null;
              const suggestionsList = Object.keys(
                this.autoCompleteData?.resources,
              );
              return this.getAutoSuggestions(
                suggestionsList,
                '${',
                position?.lineNumber,
                wordUntilPosition?.startColumn - 1,
                position?.column,
                ['default'],
                fullLineAtPosition,
              );
            } else if (wordUntilPosition?.word?.endsWith('.')) {
              let suggestionsList = [];
              let nextSuggestions = [];
              /*
               * Check if current word typed by the user ends with '.' trigger character
               * If only one word is typed by the user before . character, then only resource type is selected by the user
               * In this case, we need to show resource name options to the user
               */
              if (filteredWords.length === 1) {
                selectedResourceType = wordUntilPosition.word
                  .substring(0, wordUntilPosition?.word?.lastIndexOf('.'))
                  .trim();
                suggestionsList = this.autoCompleteData?.resources?.[
                  selectedResourceType
                ]
                  ? [
                      ...this.autoCompleteData?.resources?.[
                        selectedResourceType
                      ],
                    ]
                  : [];
                const outOptions =
                  this.autoCompleteData?.outProperties?.[selectedResourceType];
                nextSuggestions = outOptions ? ['default'] : [];
              } else {
                /*
                 * Check if current word typed by the user ends with '.' trigger character
                 * If more than one word is typed by the user before . character, then resource type and name is already selected
                 * In this case, we need to find outOptions from the api response
                 * and based on what is selected by the user before . character, we show the out options
                 */
                const outOptions =
                  this.autoCompleteData?.outProperties[filteredWords[0]];
                const resourceNames =
                  this.autoCompleteData?.resources[filteredWords[0]];
                let wordToProcess = wordUntilPosition?.word;

                /*
                 * For KEY as option, we don't show any suggestion and let user type the value
                 * But after user types key value, we need to show next option in the series
                 * For that, Get last suggestion before user types the value.
                 * If the last value is 'KEY', then replace user typed value in the filteredWords with 'KEY'
                 * This will give us the proper next entires/suggestions in the order
                 */

                // slice resource name and resource type from the list of words typed by the user to send it
                let slicedSuggestionsList = detectKeyKeywordPresence(
                  filteredWords,
                  outOptions,
                );

                /*
                 * If resource names includes 2nd filtered options, then get the next options by passing out options object
                 */

                if (resourceNames?.includes(filteredWords[1])) {
                  suggestionsList = filterObjectKeys(
                    outOptions,
                    filteredWords?.length === 2
                      ? null
                      : slicedSuggestionsList.join('.'),
                  );

                  /*
                   * If suggestion list includes 'KEY' as option, then send nothing and let user type
                   */
                  if (suggestionsList[0] === 'KEY') {
                    suggestionsList = [];
                  }
                  nextSuggestions =
                    filteredWords?.length === 2
                      ? ['out']
                      : filterObjectKeys(
                          outOptions,
                          `${slicedSuggestionsList.join('.')}.${
                            suggestionsList[0]
                          }`,
                        );
                }
              }
              return this.getAutoSuggestions(
                suggestionsList,
                '.',
                position?.lineNumber,
                position?.column,
                position?.column,
                nextSuggestions,
                fullLineAtPosition,
              );
            }

            return null;
          },
        },
      );

      this.completionProviderRegistered = true;
    }
  }

  /*
   * Get Suggestions based on input properties
   * label - is what is shown to the user
   * kind - type of a field which is getting auto completed
   * insertText - text to be inserted when user selected particular option
   * range - Column and row position on which property needs to be inserted
   * For insertText, if next options are empty, then it signifies the end of options journey and we add '}' at the end
   */
  getAutoSuggestions(
    suggestionsList,
    specialChar,
    startLineNumber,
    startColumn,
    endColumn,
    nextSuggestions = [],
    fullLineAtPosition,
  ) {
    const suggestions: monaco.languages.CompletionItem[] = suggestionsList.map(
      (word) => ({
        label: `${specialChar}${word}`,
        kind: monaco.languages.CompletionItemKind.Property,
        insertText: `${specialChar === '${' ? '{' : ''}${word}${
          nextSuggestions?.length > 0 || fullLineAtPosition?.endsWith('}')
            ? ''
            : '}'
        }`,
        range: {
          startLineNumber: startLineNumber,
          endLineNumber: startLineNumber,
          startColumn: startColumn,
          endColumn: endColumn + specialChar === '${' ? 0 : word?.length,
        },
      }),
    );
    return {
      suggestions: suggestions,
      incomplete: false,
    };
  }

  isCodeValid() {
    const model = this.editor.getModel();
    if (model) {
      const value = model.getValue();
      try {
        const code = JSON.parse(value);
        this.updateSchemaLink(code);
        return code;
      } catch {
        return false;
      }
    }
    return true;
  }

  updateSchemaLink(config) {
    if (config['$schema']) {
      this.schemaLink = config['$schema'];
    }
  }

  listenToCodeChanges() {
    this.editor.onDidChangeModelContent((event) => {
      setTimeout(() => {
        this.emitUpdatedCode();
      }, 50);
    });
  }

  emitUpdatedCode() {
    const isValid = this.isCodeValid();
    return this.byValidityChange.emit(isValid);
  }

  @Output() byCodeChange: EventEmitter<any> = new EventEmitter<any>();
  codeUpdate(code) {
    if (this.isCodeValid()) {
      this.byCodeChange.emit(code);
    } else {
      this.byCodeChange.emit(false);
    }
  }
}
