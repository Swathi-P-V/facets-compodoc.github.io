import { Component } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { GrafanaServiceService } from 'src/app/services/grafana-service.service';
import {
  ClusterUsageStatistic,
  ResourcesStatistic,
} from '../cluster-statistics';
const alertEmptyLogo = './assets/alert-empty.svg';
const errorLogo = './assets/error.svg';

@Component({
  selector: 'app-cluster-resource-allocation',
  templateUrl: './cluster-resource-allocation.component.html',
  styleUrls: ['./cluster-resource-allocation.component.scss'],
})
export class ClusterResourceAllocationComponent extends FacetsPage {
  clusterId: any;
  fetchingClusterUsageStats: boolean;
  cpuMemErrLogo: string = './assets/hazard.svg';
  cpuError: boolean = false;
  memoryError: boolean = false;
  showClusterAllocationEmpty: boolean = false;
  isClusterAllocationError: boolean = false;
  fetchingResourceStats: boolean = false;
  resourceStats: ResourcesStatistic = {};
  clusterUsageStats: ClusterUsageStatistic = {};
  errorLogo = errorLogo;
  alertEmptyLogo: any = alertEmptyLogo;

  constructor(
    private activatedroute: ActivatedRoute,
    private grafanaServiceService: GrafanaServiceService,
  ) {
    super(activatedroute);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p?.clusterId;
    this.getClusterUsageStatistics();
  }

  getClusterUsageStatistics() {
    this.fetchingClusterUsageStats = true;
    this.cpuError = false;
    this.memoryError = false;
    this.showClusterAllocationEmpty = false;
    this.isClusterAllocationError = false;

    //using promise allsettled here as we need all four data points (cpu request, cores, memory used and total memory)
    //to generate the final data and states.
    Promise.allSettled([
      this.grafanaServiceService.getTotalCpuRequests(this.clusterId),
      this.grafanaServiceService.getTotalCores(this.clusterId),
      this.grafanaServiceService.getTotalMemoryUsed(this.clusterId),
      this.grafanaServiceService.getTotalMemory(this.clusterId),
    ]).then((data) => {
      // assign data to cluster usage variable after all promises responds.
      this.clusterUsageStats.coresInUse =
        data[0]?.status === 'fulfilled'
          ? this.getFormattedString(data[0]?.['value']?.['coresInUse'])
          : '';

      this.clusterUsageStats.totalCores =
        data[1]?.status === 'fulfilled'
          ? this.getFormattedString(data[1]?.['value']?.['totalCores'])
          : '';

      this.clusterUsageStats.memoryInUse =
        data[2]?.status === 'fulfilled'
          ? this.convertBytesToGbInString(data[2]?.['value']?.['memoryInUse'])
          : '';

      this.clusterUsageStats.totalMemory =
        data[3]?.status === 'fulfilled'
          ? this.convertBytesToGbInString(data[3]?.['value']?.['totalMemory'])
          : '';

      //check if promise if fullfilled for all of the promises and cluster data is not present for all the responses.
      //if the cluster data is absent, then show empty state message and icon
      if (
        data.every(this.checkIfFulfilled) &&
        !this.clusterUsageStats?.coresInUse &&
        !this.clusterUsageStats?.totalCores &&
        !this.clusterUsageStats?.memoryInUse &&
        !this.clusterUsageStats?.totalMemory
      ) {
        this.showClusterAllocationEmpty = true;
        this.isClusterAllocationError = false;
      }

      //check if promise if rejected for all of the requests. If it is rejected,
      //then check if state of rejection is empty data or not, if empty data , then show empty state icon
      //If reason for rejection is not empty, then it must be error. Show error icon and error state in that context
      if (data.every(this.checkIfReject)) {
        if (data.every(this.checkIfEmptyState)) {
          this.showClusterAllocationEmpty = true;
          this.isClusterAllocationError = false;
        } else {
          this.showClusterAllocationEmpty = true;
          this.isClusterAllocationError = true;
        }
      }

      //Using setTimeout as accessing the elements within ngIf is giving null as the DOM is not rendered yet
      setTimeout(() => {
        this.drawStroke();
      }, 1000);
    });
  }

  //check if state is empty for all items in data
  checkIfEmptyState(data) {
    return data?.reason?.state === 'empty';
  }

  //check if status is fullfilled for all items in a list
  checkIfFulfilled(data) {
    return data?.status === 'fulfilled';
  }

  //check if status is rejected for all items in a list.
  checkIfReject(data) {
    return data?.status === 'rejected';
  }

  //set the values for circular stroke for memory and cpu stats
  drawStroke() {
    let cpuProgress = document.querySelector<HTMLElement>(
      '#cpu-usage-circular-progress',
    );
    let cpuPer = document.querySelector<HTMLElement>(
      '#cpu-usage-percent-value',
    );
    let memProgress = document.querySelector<HTMLElement>(
      '#mem-usage-circular-progress',
    );
    let memPer = document.querySelector<HTMLElement>(
      '#mem-usage-percent-value',
    );

    //Check if cores data and total cores are present, then only show the data and set the data,
    //or set the cpuerror to true, which in turn will show the error icon for cpu stats.
    this.populatePercentages(
      +this.clusterUsageStats?.coresInUse,
      +this.clusterUsageStats?.totalCores,
      'cpu',
      cpuProgress,
      cpuPer,
    );

    //check if memory in use and total memory is greater than 0, then only show the data
    //or set the memoryerror to true, which will load the error icon for memory.
    this.populatePercentages(
      +this.clusterUsageStats?.memoryInUse,
      +this.clusterUsageStats?.totalMemory,
      'mem',
      memProgress,
      memPer,
    );

    this.fetchingClusterUsageStats = false;
  }

  populatePercentages(numer, denom, type, progressElem, percentElem) {
    let circleConst = 3.6;
    if (numer && numer > 0 && denom && denom > 0) {
      let percent = ((numer / denom) * 100).toFixed(0);
      let numPercent = +percent;
      if (numPercent < 100) {
        let progressBarColor = this.getProgressBarColor(numPercent);
        if (progressElem) {
          progressElem.style.background = `conic-gradient( ${progressBarColor} ${
            circleConst * numPercent
          }deg, #ededed 0deg)`;
        }
        if (percentElem) {
          percentElem.style.color = progressBarColor;
          percentElem.textContent = percent + '%';
        }
      } else {
        type === 'cpu' ? (this.cpuError = true) : (this.memoryError = true);
        if (progressElem) {
          progressElem.style.background = `conic-gradient(rgba(227, 77, 77, 1), 360deg, #ededed 0deg)`;
        }
      }
    } else {
      type === 'cpu' ? (this.cpuError = true) : (this.memoryError = true);
      if (progressElem) {
        progressElem.style.background = `conic-gradient(rgba(227, 77, 77, 0.08), 360deg, #ededed 0deg)`;
      }
    }
  }

  getProgressBarColor(value: number) {
    let colorCode = '#FDB11F';
    if (value >= 0 && value <= 50) {
      colorCode = '#da1414';
    } else if (value > 50 && value <= 70) {
      colorCode = '#FDB11F';
    } else if (value > 70 && value <= 100) {
      colorCode = '#1eb75a';
    }
    return colorCode;
  }

  convertBytesToGbInString(bytes: string) {
    return (Number(bytes) / 10 ** 9).toFixed(2);
  }

  getFormattedString(value: string) {
    return parseFloat(value).toFixed(2);
  }
}
