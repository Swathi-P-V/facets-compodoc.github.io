import { Component } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { UiCommonClusterControllerService } from 'src/app/cc-api/services';
import { ScheduleRequest } from 'src/app/cc-api/models/schedule-request';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { AccessDeniedService } from 'src/app/services/access-denied.service';
import { map } from 'rxjs';
import timezones from 'src/app/constants/timezones';
import { FormGroup, Validators, FormBuilder } from '@angular/forms';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-cluster-release-scheduler',
  templateUrl: './cluster-release-scheduler.component.html',
  styleUrls: ['./cluster-release-scheduler.component.scss'],
})
export class ClusterReleaseSchedulerComponent extends FacetsPage {
  repeatIntervalOptions = [
    { value: 'PER_MINUTE', label: 'Minutes' },
    { value: 'HOURLY', label: 'Hours' },
    { value: 'DAILY', label: 'Days' },
    { value: 'WEEKLY', label: 'Weeks' },
  ];

  timeZones = timezones;

  releaseDaysOptions = [
    { value: 'MON', label: 'Monday' },
    { value: 'TUE', label: 'Tuesday' },
    { value: 'WED', label: 'Wednesday' },
    { value: 'THU', label: 'Thursday' },
    { value: 'FRI', label: 'Friday' },
    { value: 'SAT', label: 'Saturday' },
    { value: 'SUN', label: 'Sunday' },
  ];

  intervalMinMaxLimits = {
    PER_MINUTE: { min: 1, max: 59 },
    HOURLY: { min: 1, max: 23 },
    DAILY: { min: 1, max: 31 },
    WEEKLY: { min: 1, max: 1 },
  };

  timeZoneOptions = [
    {
      value: Intl.DateTimeFormat().resolvedOptions().timeZone,
      label: Intl.DateTimeFormat().resolvedOptions().timeZone,
    },
  ];

  intervalMinLimit;
  intervalMaxLimit;
  timeZone: any;
  clusterId: string;
  stackName: string;
  releaseScheduleRequestObject: ScheduleRequest;
  releaseScheduleData = [];
  scheduleRequest: boolean = false;
  errorIcon = errorIcon;
  fetchScheduleErrorMessage = '';
  fetchingSchedule: boolean = true;
  previousComponent: string;
  releaseScheduleForm: FormGroup = this.fb.group({
    frequency: ['', [Validators.required]],
    interval: [1, [Validators.required]],
    byTime: ['', [Validators.required]],
    byDay: [[], [Validators.required]],
    timezone: ['', [Validators.required]],
  });

  constructor(
    private activatedroute: ActivatedRoute,
    private router: Router,
    private commonClusterController: UiCommonClusterControllerService,
    private fb: FormBuilder,
    private accessDeniedService: AccessDeniedService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedroute);
  }

  get frequency() {
    return this.releaseScheduleForm.get('frequency');
  }

  get interval() {
    return this.releaseScheduleForm.get('interval');
  }

  get byTime() {
    return this.releaseScheduleForm.get('byTime');
  }

  get byDay() {
    return this.releaseScheduleForm.get('byDay');
  }

  get timezone() {
    return this.releaseScheduleForm.get('timezone');
  }

  OnPageLoad(p: Params): void {
    this.releaseScheduleForm.reset();
    this.clusterId = p.clusterId;
    this.stackName = p.stackName;
    this.releaseScheduleData = [];
    this.fetchingSchedule = true;

    this.activatedroute.paramMap
      .pipe(map(() => window.history.state))
      .subscribe((data) => {
        if (data?.previousComponent) {
          this.previousComponent = data.previousComponent;
        }
      });

    this.commonClusterController
      .getClusterSchedulesUsingGET(this.clusterId)
      .subscribe(
        (data) => {
          //Filter out only RELEASE releasetype object: Returns an array
          this.releaseScheduleData = data.filter(
            (schedule) => schedule.releaseType === 'RELEASE',
          );
          if (this.releaseScheduleData[0]?.timeZone === 'UTC') {
            this.releaseScheduleData[0].timeZone = 'GMT';
          }
          if (this.releaseScheduleData?.length) {
            this.releaseScheduleForm.patchValue({
              frequency: this.releaseScheduleData[0].frequency,
              interval: this.releaseScheduleData[0].interval,
              byTime: this.releaseScheduleData[0].byTime,
              byDay: this.releaseScheduleData[0].byDay,
              timezone: this.releaseScheduleData[0].timeZone,
            });
            this.setIntervalMinMaxLimit(this.releaseScheduleData[0].frequency);
          } else {
            this.frequency.setValue('PER_MINUTE');
            this.interval.setValue(1);
            this.setIntervalMinMaxLimit('PER_MINUTE');
          }
          this.fetchingSchedule = false;
        },
        (error) => {
          this.fetchingSchedule = false;
          this.releaseScheduleData = [];
          this.fetchScheduleErrorMessage =
            error?.error?.message || 'There was error fetching schedule';
        },
      );
  }

  setIntervalMinMaxLimit(event) {
    this.intervalMinLimit = `${this.intervalMinMaxLimits[event].min}`;
    this.intervalMaxLimit = `${this.intervalMinMaxLimits[event].max}`;
  }

  saveReleaseSchedule() {
    const mode = this.releaseScheduleData?.length ? 'edit' : 'create';

    this.scheduleRequest = true;
    this.releaseScheduleRequestObject = {
      byDay: this.byDay.value,
      byTime: this.byTime.value,
      frequency: this.frequency.value,
      interval: this.interval.value,
      timeZone: this.timezone.value,
      releaseType: 'RELEASE',
      pauseReleaseSchedule:
        mode === 'create' ? false : this.releaseScheduleData?.[0]?.isPaused,
    };
    let freqLabel = '';
    this.repeatIntervalOptions.forEach((option) => {
      if (option?.value === this.frequency.value) {
        freqLabel = option?.label;
      }
    });

    let requestService =
      this.commonClusterController.addClusterSchedulesUsingPOST({
        request: this.releaseScheduleRequestObject,
        clusterId: this.clusterId,
      });
    if (mode === 'edit') {
      requestService =
        this.commonClusterController.updateClusterSchedulesUsingPUT({
          request: this.releaseScheduleRequestObject,
          clusterScheduleId: this.releaseScheduleData[0].id,
          clusterId: this.clusterId,
        });
    }

    requestService.subscribe(
      (data) => {
        this.scheduleRequest = false;
        this.customToastService.success(
          `Schedule ${mode === 'create' ? 'created' : 'updated'} successfully`,
        );
        this.cancelReleaseScheduler();
      },
      (error) => {
        this.scheduleRequest = false;
        this.onAPIErrorService.onAPIError(
          error?.error?.message ||
            `Release schedule ${
              mode === 'create' ? 'creation' : 'update'
            } failed`,
          error,
        );
      },
    );
  }

  cancelReleaseScheduler = () => {
    let url = '';
    if (this.previousComponent === 'environmentOverview') {
      url = '/' + this.stackName + '/cluster/' + this.clusterId + '/overview';
    } else if (this.previousComponent === 'clusterReleases') {
      url = '/' + this.stackName + '/cluster/' + this.clusterId + '/releases';
    } else if (!this.previousComponent) {
      url = '/' + this.stackName + '/cluster/' + this.clusterId + '/overview';
    }
    this.router.navigate([url]);
  };

  freqChange(event) {
    this.releaseScheduleForm.reset();
    this.setIntervalMinMaxLimit(event);
    if (
      this.releaseScheduleData?.length &&
      event === this.releaseScheduleData[0].frequency
    ) {
      this.releaseScheduleForm.patchValue({
        frequency: this.releaseScheduleData[0].frequency,
        timezone: this.releaseScheduleData[0].timeZone,
        interval: this.releaseScheduleData[0].interval,
        byTime: this.releaseScheduleData[0].byTime,
        byDay: this.releaseScheduleData[0].byDay,
      });
    } else {
      this.interval.setValue(1);
    }
  }

  isFormValid() {
    if (
      (this.frequency.value === 'WEEKLY' &&
        this.byDay.value?.length > 0 &&
        this.byTime.value?.length > 0 &&
        this.timezone.value?.length > 0) ||
      (this.frequency.value === 'DAILY' &&
        this.interval.value >= this.intervalMinLimit &&
        this.interval.value <= this.intervalMaxLimit &&
        this.byTime.value?.length > 0 &&
        this.timezone.value?.length > 0) ||
      ((this.frequency.value === 'HOURLY' ||
        this.frequency.value === 'PER_MINUTE') &&
        this.interval.value >= this.intervalMinLimit &&
        this.interval.value <= this.intervalMaxLimit)
    ) {
      return true;
    }
    return false;
  }
}
