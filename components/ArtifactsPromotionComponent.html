import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { SmartTableWithIconComponent } from 'src/app/components/smart-table-with-icon/smart-table-with-icon.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { AbstractCluster, PromotionWorkflow } from 'src/app/cc-api/models';
import {
  UiPromotionWorkflowControllerService,
  UiArtifactCiControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { SmartTableHierarchyComponent } from './smart-table-hierarchy.component';
import { NbDialogService } from '@nebular/theme';
import {
  PROMOTIONAL_WORKFLOW_WRITE,
  PROMOTIONAL_WORKFLOW_DELETE,
  ARTIFACTS_WRITE,
} from '../../../constants/permissions';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-artifacts-promotion',
  templateUrl: './artifacts-promotion.component.html',
  styleUrls: ['./artifacts-promotion.component.scss'],
})
export class ArtifactsPromotionComponent extends FacetsPage {
  stackName: string = ''; //blueprint bame
  fetchingPromotion: boolean = false; //promotion workflow loader
  emptyStateLogo = './assets/promotion-empty.svg'; //empty state logo
  errorLogo = './assets/error.svg'; //empty state error logo
  primaryMsg: string = ''; //empty state primary message
  secondaryMsg: string = ''; //empty state secondary message
  status: 'info' | 'error' = 'info'; //empty state status
  showBtn: boolean = false; // empty state show button
  buttonText: string = ''; //empty state button text
  emptyTooltip: string = ''; //empty state button tooltip
  showEmptyState: boolean = false; //show empty state condition variable
  promotionWorkflows: PromotionWorkflow[] = []; //promotion workflow list
  envs: AbstractCluster[] = []; //environments list
  deleteWorkflowInProgress: boolean = false; //delete in progress loader variable
  @ViewChild('workflowDetails', { static: true })
  private workflowDetails: TemplateRef<any>; //workflow details popup
  @ViewChild('confirmDelete', { static: true }) //delete workflow popup
  private confirmDelete: TemplateRef<any>;
  fetchingArtifacts: boolean = false; //fetching artifacts CI loader
  artifacts: any = []; //artifacts list;
  createPermission = PROMOTIONAL_WORKFLOW_WRITE;

  smartTableCustomActionButtons = [
    {
      id: 'edit',
      icon: 'edit-2-outline',
      isFciPack: true,
      toolTip: 'Edit workflow',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: this.createPermission,
      onClick: (data) => {
        this.editWorkflow(data);
      },
    },
    {
      id: 'default-inactive',
      icon: 'pin-outline',
      isFciPack: true,
      toolTip: 'Set as default',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: ARTIFACTS_WRITE,
      onClick: (data) => {
        this.promptUpdateDefaultStatus(data, true);
      },
    },
    {
      id: 'default-active',
      icon: 'pin-solid',
      isFciPack: true,
      toolTip: 'Unset as default',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: ARTIFACTS_WRITE,
      onClick: (data) => {
        this.promptUpdateDefaultStatus(data, false);
      },
    },
    {
      id: 'delete',
      icon: 'trash-outline',
      isFciPack: true,
      toolTip: 'Delete workflow',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: PROMOTIONAL_WORKFLOW_DELETE,
      onClick: (data) => {
        this.deleteWorkflowConfirmPopup(data);
      },
    },
  ];
  tableSettings: any = {
    columns: {
      workflowName: {
        title: 'Promotion Workflow',
        filter: true,
        width: '40%',
        type: 'custom',
        valuePrepareFunction: (cell, row) => {
          let tooltip = '';
          return JSON.stringify({
            isTextLink: true,
            url: null,
            textContent: cell,
            showIcon: false,
            displayText:
              row.registrationType == 'RELEASE_STREAM'
                ? 'RELEASE STREAM DEFAULT'
                : 'ENVIRONMENT DEFAULT',
            showBadge: row.defaultWorkflow,
            noTooltip: true,
            badgeClass: 'mb-0 selected-text cursor-pointer',
          });
        },
        renderComponent: SmartTableWithIconComponent,
        onComponentInitFunction: (instance) => {
          instance.byTitleClick.subscribe((data) => {
            this.openDetails(data.rowData);
          });
          instance.byIconClick.subscribe((data) => {
            this.promptUpdateDefaultStatus(
              data.rowData,
              !data.rowData.defaultWorkflow,
            );
          });
        },
      },
      hierarchies: {
        title: 'Hierarchy',
        type: 'custom',
        filter: false,
        valuePrepareFunction: (cell, row) => {
          let dataToSend = this.getHierarchy(cell, row);
          return JSON.stringify({
            data: dataToSend,
            registrationType: row?.registrationType,
          });
        },
        renderComponent: SmartTableHierarchyComponent,
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {};
          displayVal['smartTableCustomActionButtons'] =
            this.smartTableCustomActionButtons;
          //If release stream is PROD or registration type is clusterID, hide promote action button
          if (row.releaseStream === 'PROD' || row.clusterId) {
            displayVal['smartTableCustomActionButtons'] =
              this.smartTableCustomActionButtons.filter(
                (actionButton) => actionButton.id === 'delete',
              );
          }
          displayVal['smartTableCustomActionButtons'] =
            this.smartTableCustomActionButtons.filter(
              (actionButton) =>
                actionButton.id !==
                (row.defaultWorkflow ? 'default-inactive' : 'default-active'),
            );
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance: any) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '15%',
        filter: false,
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
      perPage: 10,
    },
    mode: 'external',
    actions: false,
  };

  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private promotionWorkflow: UiPromotionWorkflowControllerService,
    private uiStackController: UiStackControllerService,
    private dialogService: NbDialogService,
    private uiArtifactCiController: UiArtifactCiControllerService,
    private promotionWorkflowService: UiPromotionWorkflowControllerService,
    private customToastService: CustomToastService,
    private onApiErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    if (p.stackName) {
      this.stackName = p?.stackName;
    }
    this.getEnvs();
    this.fetchArtifacts();
  }

  fetchArtifacts() {
    this.fetchingArtifacts = true;
    this.uiArtifactCiController
      .getArtifactCisByStackUsingGETResponse(this.stackName)
      .subscribe(
        (r) => {
          this.artifacts = r.body;
          if (this.artifacts?.length > 0) {
            this.showEmptyState = false;
          } else {
            this.showEmptyState = true;
            this.showBtn = true;
          }
          this.showBtn;
          this.fetchingArtifacts = false;
        },
        (error) => {
          this.fetchingArtifacts = false;
          this.showEmptyState = true;
          this.emptyStateLogo = this.errorLogo;
          this.secondaryMsg =
            error?.error?.message ||
            'There is some error fetching CI integrations';
          this.primaryMsg = '';
          this.status = 'error';
          this.showBtn = false;
        },
      );
  }

  getHierarchy(cell, row) {
    let dataToSend = [];
    let sortedList = cell.sort((a, b) => {
      return a?.sequence - b?.sequence;
    });
    if (row?.registrationType === 'ENVIRONMENT') {
      sortedList.forEach((val) => {
        this.envs.forEach((env) => {
          if (env?.id === val?.registrationValue) {
            dataToSend.push({
              envName: env?.name,
              envId: env?.id,
              stackName: env?.stackName,
              url: ['/', env?.stackName, 'cluster', env?.id, 'overview'],
            });
          }
        });
      });
    } else {
      sortedList.forEach((val) => {
        dataToSend.push(val?.registrationValue);
      });
    }
    return dataToSend;
  }

  getEnvs() {
    this.fetchingPromotion = true;
    this.uiStackController
      .getAllClustersUsingGET({
        page: 0,
        perPage: 10000,
        sortBy: 'ID',
      })
      .subscribe(
        (res) => {
          this.envs = res?.content;
          this.getAllWorkflows();
        },
        (err) => {
          this.fetchingPromotion = false;
          this.primaryMsg = '';
        },
      );
  }

  getAllWorkflows() {
    this.fetchingPromotion = true;
    this.promotionWorkflow
      .getWorkflowsByStackUsingGETResponse(this.stackName)
      .subscribe(
        (res) => {
          this.fetchingPromotion = false;
          this.promotionWorkflows = res?.body;
          if (!this.promotionWorkflows?.length) {
            this.showEmptyState = true;
            this.secondaryMsg =
              'Create your own promotion workflow to promote your artifacts';
            this.primaryMsg = 'No Promotion Workflows';
            this.status = 'info';
            this.showBtn = true;
            this.emptyTooltip = 'Create promotion workflow';
            this.buttonText = 'Create Promotion Workflow';
          }
        },
        (error) => {
          this.fetchingPromotion = false;
          this.showEmptyState = true;
          this.emptyStateLogo = this.errorLogo;
          this.secondaryMsg =
            error?.error?.message ||
            'There is some error fetching promotion workflows';
          this.primaryMsg = '';
          this.status = 'error';
          this.showBtn = false;
        },
      );
  }

  openDetails(data) {
    this.dialogService.open(this.workflowDetails, { context: data });
  }

  deleteWorkflowConfirmPopup(rowData) {
    let idPresent = false;
    this.artifacts.forEach((artifact) => {
      if (artifact?.promotionWorkflowId === rowData?.id) {
        idPresent = true;
      }
    });
    rowData['showMessage'] = idPresent;
    this.dialogService.open(this.confirmDelete, { context: rowData });
  }

  deleteWorkflow(data, ref) {
    let workflowId = data?.id;
    if (workflowId) {
      this.deleteWorkflowInProgress = true;
      this.promotionWorkflow.deleteWorkflowUsingDELETE(workflowId).subscribe(
        (res) => {
          this.deleteWorkflowInProgress = false;
          this.customToastService.success(
            `Promotion workflow ${data?.name} deleted successfully`,
          );
          ref.close();
          this.getAllWorkflows();
        },
        (error) => {
          this.deleteWorkflowInProgress = false;
          this.onApiErrorService.onAPIError(
            error?.error?.message ||
              `Failed to delete promotion workflow ${data?.name}`,
            error,
          );
          ref.close();
        },
      );
    }
  }

  createPromotion() {
    this.router.navigate(
      ['/', 'stack', this.stackName, 'artifacts', 'create-workflow'],
      {
        state: {
          previousComponent: 'artifactsPromotion',
          stackName: this.stackName,
        },
      },
    );
  }

  editWorkflow(data) {
    this.router.navigate(
      ['/', 'stack', this.stackName, 'artifacts', 'edit-workflow', data?.id],
      {
        state: {
          previousComponent: 'artifactsPromotion',
          stackName: this.stackName,
        },
      },
    );
  }

  promptUpdateDefaultStatus(workflow, setAsDefault) {
    if (this.defaultAlreadyAvailable(workflow)?.length) {
      this.dialogService.open(this.blockDefaultSetActionDialog, {
        hasBackdrop: true,
        closeOnBackdropClick: false,
        context: {
          registrationType: workflow.registrationType,
        },
      });
      return;
    }

    this.dialogService.open(this.updateDefaultDialog, {
      context: { workflow, setAsDefault },
    });
  }

  defaultAlreadyAvailable(workflow) {
    return this.promotionWorkflows.filter(
      (pwf) =>
        pwf.id != workflow.id &&
        pwf.registrationType === workflow.registrationType &&
        pwf.defaultWorkflow,
    );
  }

  updatingDefaultStatus: boolean = false;
  @ViewChild('updateDefaultDialog', { static: true }) //delete workflow popup
  private updateDefaultDialog: TemplateRef<any>;

  @ViewChild('blockDefaultSetActionDialog', { static: true })
  private blockDefaultSetActionDialog: TemplateRef<any>;

  setDefaultStatus(workflow, setAsDefault, ref?) {
    this.updatingDefaultStatus = true;
    const fallBackOption = workflow.defaultWorkflow;
    workflow.defaultWorkflow = setAsDefault;
    const subtext = setAsDefault ? 'set' : 'unset';
    this.promotionWorkflowService
      .updateWorkflowUsingPUT({
        workflowId: workflow.id,
        promotionWorkflow: workflow,
      })
      .subscribe(
        (rsp) => {
          this.getAllWorkflows();
          this.customToastService.success(
            `Promotion workflow ${subtext} as default successfully`,
          );
        },
        (error) => {
          this.onApiErrorService.onAPIError(
            error?.error?.message ||
              `Failed to ${subtext} this promotion workflow`,
            error,
          );
          workflow.defaultWorkflow = fallBackOption;
        },
      )
      .add(() => {
        this.updatingDefaultStatus = false;
        ref.close();
      });
  }
}
