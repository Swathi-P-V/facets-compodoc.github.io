import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { Component, ViewChild, TemplateRef } from '@angular/core';
import { ActivatedRoute, Router, Params } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { UiDomainMappingControllerService } from 'src/app/cc-api/services';
import { DomainDTO } from './../../../cc-api/models/domain-dto';
import { NbDialogService } from '@nebular/theme';
import { GrafanaServiceService } from 'src/app/services/grafana-service.service';
import { getTimeStringFromDurationValue } from 'src/app/utils/calendar-utils';
import { ENVIRONMENT_CONFIGURE } from 'src/app/constants/permissions';
import cloneDeep from 'lodash/cloneDeep';
import moment from 'moment';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-resource-ingress-domain',
  templateUrl: './resource-ingress-domain.component.html',
  styleUrls: ['./resource-ingress-domain.component.scss'],
})
export class ResourceIngressDomainComponent extends FacetsPage {
  @ViewChild('deleteDomainUI', { static: true })
  private deleteDomainUI: TemplateRef<any>;
  clusterId: any;
  clusterState: string;
  resourceName: string;
  resourceType: string;
  loadingCustomDomains: boolean = true;
  showEmptyState: boolean = false;
  stackName: string = '';
  customDomains: any = [];
  tableData: DomainDTO[] = [];
  deletionInProgress: boolean = false;
  readonly ENVIRONMENT_CONFIGURE = ENVIRONMENT_CONFIGURE;

  //empty state variables
  errorIcon: string = errorIcon;
  primaryMsg: string = 'No Custom Domain Mapped';
  secondaryMsg: string = 'Create a new custom domain';
  buttonText: string = 'Map Custom Domains';
  status: 'info' | 'error' = 'info';
  showEmptyBtn: boolean = true;

  headerActionButtons = [
    {
      id: 'map-domain',
      title: 'Map Custom Domain',
      buttonType: 'secondary',
      status: 'primary',
      appearance: 'outline',
      nbTooltip: 'Map a new custom domain',
      loading: false,
      action: ENVIRONMENT_CONFIGURE,
      hidden: false,
    },
  ];

  smartTableCustomActionButtons = [
    {
      id: 'edit',
      icon: 'edit-2-outline',
      toolTip: 'Edit Domain',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: ENVIRONMENT_CONFIGURE,
      onClick: (data) => {
        this.editDomain(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete Domain',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: ENVIRONMENT_CONFIGURE,
      onClick: (data) => {
        this.deleteDomainConfirmPopup(data);
      },
    },
  ];

  domainTableSettings = {
    columns: {
      domain: {
        title: 'Domain Name',
        filter: true,
        type: 'text',
      },
      alias: {
        title: 'Alias',
        filter: true,
        type: 'text',
      },
      domainExpiry: {
        title: 'Certification Expiry',
        type: 'text',
        valuePrepareFunction: (value, row) => {
          if (value?.length > 0) {
            const duration = moment.duration(value, 'days');
            return getTimeStringFromDurationValue(duration);
          } else {
            return 'N/A';
          }
        },
      },
      certificateReference: {
        title: 'Certificate Reference',
        filter: false,
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          let formattedSmartTableCustomActionButtons = cloneDeep(
            this.smartTableCustomActionButtons,
          );

          formattedSmartTableCustomActionButtons.forEach((button) => {
            button['uniqueId'] = button.id + '_' + row.name;
          });
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {
            smartTableCustomActionButtons:
              formattedSmartTableCustomActionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '12%',
        filter: false,
      },
    },
    noDataMessage: 'No custom domains mapped.',
    pager: {
      display: false,
    },
    actions: false,
  };
  errorMessage: string = '';

  constructor(
    _route: ActivatedRoute,
    private router: Router,
    private domainMappingService: UiDomainMappingControllerService,
    private grafanaService: GrafanaServiceService,
    private customToast: CustomToastService,
    private dialogService: NbDialogService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p?.clusterId;
    this.resourceName = p?.resourceName;
    this.resourceType = p?.resourceType;
    this.stackName = p?.stackName;
    this.errorMessage = '';
    this.getDomainData();
  }

  getDomainExpiration(domainData) {
    let promises: [] = this.customDomains?.map((data) => {
      if (data?.domain?.length > 0) {
        return this.grafanaService.getDomainExpiration(this.clusterId, {
          domainURL: data?.domain,
        });
      }
    });
    // Execute the promises using Promise.allSettled
    Promise.allSettled(promises).then((results) => {
      results.forEach((result, index) => {
        if (result?.['status'] === 'fulfilled' && result?.['value']) {
          const domainResultURL = Object.keys(result?.['value'])?.[0];
          this.customDomains.find(
            (domain) => domain?.domain === domainResultURL,
          ).domainExpiry = result?.['value']?.[domainResultURL];
        }
      });
      this.tableData = [...this.customDomains];
      this.loadingCustomDomains = false;
    });
  }

  getDomainData() {
    this.loadingCustomDomains = true;
    this.domainMappingService
      .getAllDomainsUsingGET({
        resourceName: this.resourceName,
        resourceType: this.resourceType,
        clusterId: this.clusterId,
      })
      .subscribe(
        (domains) => {
          this.customDomains = domains;
          this.customDomains?.forEach((data) => (data.domainExpiry = ''));
          if (this.customDomains?.length > 0) {
            this.getDomainExpiration(this.customDomains);
          } else {
            this.tableData = [...this.customDomains];
            this.loadingCustomDomains = false;
          }
        },
        (err) => {
          this.errorMessage =
            err?.error?.message ||
            'There is some error while fetching the custom domains';
          this.loadingCustomDomains = false;
        },
      );
  }

  listenToActionButtonEvents() {
    this.router.navigate(
      [
        '/',
        this.stackName,
        'cluster',
        this.clusterId,
        'resources',
        'type',
        this.resourceType,
        'resource',
        this.resourceName,
        'ingress-domains',
        'create',
      ],
      {
        queryParams: { from: 'overview' },
      },
    );
  }

  deleteDomainConfirmPopup(data) {
    this.dialogService.open(this.deleteDomainUI, { context: data });
  }

  editDomain(data) {
    this.router.navigate(
      [
        '/',
        this.stackName,
        'cluster',
        this.clusterId,
        'resources',
        'type',
        this.resourceType,
        'resource',
        this.resourceName,
        'ingress-domains',
        'edit',
        data?.alias,
      ],
      {
        queryParams: { from: 'overview' },
      },
    );
  }

  deleteDomain(domainData, ref) {
    this.deletionInProgress = true;
    this.domainMappingService
      .deleteDomainMappingUsingDELETE({
        clusterId: this.clusterId,
        resourceType: this.resourceType,
        resourceName: this.resourceName,
        alias: domainData?.alias,
      })
      .subscribe(
        (data) => {
          this.customToast.success(
            `Custom domain ${domainData?.alias} deleted successfully.`,
          );
          ref.close();
          this.getDomainData();
          this.deletionInProgress = false;
        },
        (err) => {
          this.customToast.error(
            err?.error?.message || 'Domain deletion failed.',
          );
          ref.close();
          this.deletionInProgress = false;
        },
      );
  }
}

<nb-card
  [ngClass]="{ 'py-5 background-grey': loadingCustomDomains }"
  [nbSpinner]="loadingCustomDomains"

  <nb-card-header

  </nb-card-body>

<ng-template #deleteDomainUI let-data let-ref="dialogRef">
  <app-delete-confirmation
    [dialogRef]="ref"
    [deleteItem]="data"
    [showSpinner]="deletionInProgress"
    headerText="Domain"
    (deleteConfirmEvent)="deleteDomain($event, ref)"
    inputProperty="alias"
    [deleteBtnPermission]="ENVIRONMENT_CONFIGURE"
  >
  </app-delete-confirmation>
</ng-template>
