import {
  Component,
  Input,
  OnDestroy,
  TemplateRef,
  ViewChild,
} from '@angular/core';
import { ActivatedRoute, Params } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import { Subscription } from 'rxjs';
import {
  UiBlueprintDesignerControllerService,
  UiCommonClusterControllerService,
} from 'src/app/cc-api/services';
import { ENVIRONMENT_CONFIGURE } from 'src/app/constants/permissions';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { BulkEnableDisableResourcesService } from 'src/app/services/bulk-enable-disable-resources.service';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';

@Component({
  selector: 'app-enable-disable-resource',
  templateUrl: './enable-disable-resource.component.html',
  styleUrls: ['./enable-disable-resource.component.scss'],
})
export class EnableDisableResourceComponent
  extends FacetsPage
  implements OnDestroy
{
  @Input() clusterId: string = '';
  @Input() stackName: string = '';

  selectedResources: {
    resourceName: string;
    resourceType: string;
    disabled: boolean;
  }[] = [];
  status;
  differentResStatusError: string = '';
  bulkResourceOpsFooterButtons;
  footerDataSubscription: Subscription;

  @ViewChild('bulkEnableDisableResources', { static: true })
  private bulkEnableDisableResources: TemplateRef<any>;

  enableDisableBulkResourcesInProgress: boolean;
  readonly ENVIRONMENT_CONFIGURE = ENVIRONMENT_CONFIGURE;
  constructor(
    private _route: ActivatedRoute,
    private uiCommonClusterService: UiCommonClusterControllerService,
    private toastService: CustomToastService,
    private bulkEnableDisableResourceService: BulkEnableDisableResourcesService,
    private uiBlueprintDesignerControllerService: UiBlueprintDesignerControllerService,
    private dialogService: NbDialogService,
  ) {
    super(_route);
    this.footerDataSubscription =
      this.bulkEnableDisableResourceService.updateFooterData.subscribe(
        (data) => {
          this.updateFooterState(data);
        },
      );
  }

  OnPageLoad(p: Params): void {}

  bulkEnableDisableAtClusterLevel(selectedResources, status, ref) {
    this.enableDisableBulkResourcesInProgress = true;
    let resourceEnableDisableRequestList = selectedResources.map((res) => {
      return {
        resourceName: res.resourceName,
        resourceType: res.resourceType,
        disabled: status === 'Enabled',
      };
    });
    return this.uiCommonClusterService
      .bulkEnableDisableResourcesUsingPUT({
        clusterId: this.clusterId,
        resourceEnableDisableRequestList: resourceEnableDisableRequestList,
      })
      .subscribe(
        (data) => {
          ref.close();
          this.enableDisableBulkResourcesInProgress = false;
          this.toastService.success(
            `All selected resources have been ${
              status === 'Enabled' ? 'disabled' : 'enabled'
            } successfully.`,
          );
          this.resetFootersProps();
          this.bulkEnableDisableResourceService.sendMessage(true);
        },
        (error) => {
          this.enableDisableBulkResourcesInProgress = false;
          this.toastService.error(
            `Failed to ${
              status === 'Enabled' ? 'disable' : 'enable'
            } resources.`,
          );
          ref.close();
          this.bulkEnableDisableResourceService.sendMessage(false);
        },
      );
  }

  bulkEnableDisableAtStackLevel(selectedResources, status, ref) {
    this.enableDisableBulkResourcesInProgress = true;
    let resourceEnableDisableRequestList = selectedResources.map((res) => {
      return {
        resourceName: res.resourceName,
        resourceType: res.resourceType,
        disabled: status === 'Enabled',
        filename: res.filename,
        directory: res.directory,
      };
    });
    this.uiBlueprintDesignerControllerService
      .bulkEditDisabledForResourcesUsingPUT1({
        stackName: this.stackName,
        resourceEnableDisableRequestList: resourceEnableDisableRequestList,
      })
      .subscribe(
        async () => {
          this.enableDisableBulkResourcesInProgress = false;
          ref?.close();
          this.toastService.success(
            `All selected resources have been ${
              status === 'Enabled' ? 'disabled' : 'enabled'
            } successfully.`,
          );
          this.resetFootersProps();
          this.bulkEnableDisableResourceService.sendMessage(true);
        },
        (error) => {
          this.enableDisableBulkResourcesInProgress = false;
          this.toastService.error(
            error?.error?.message ||
              `Failed to ${
                status === 'Enabled' ? 'disable' : 'enable'
              } resources.`,
          );
          ref.close();
        },
      );
  }

  handleFooterButtons(clickedButton) {
    if (clickedButton === 'cancel-bulk-enable-disable-resources') {
      this.resetFootersProps();
      this.bulkEnableDisableResourceService.sendMessage(false);
    } else {
      if (!['Enabled', 'Disabled'].includes(this.status)) {
        this.status = this.status === 'true' ? 'Disabled' : 'Enabled';
      }
      this.dialogService.open(this.bulkEnableDisableResources, {
        context: {
          status: this.status,
          selectedResources: this.selectedResources,
        },
      });
    }
  }

  updateFooterState(state) {
    this.differentResStatusError = '';

    if (typeof state === 'boolean' && !state) {
      this.differentResStatusError =
        'Please select resources which are either enabled or disabled';
      if (this.bulkResourceOpsFooterButtons?.length) {
        this.bulkResourceOpsFooterButtons[0].disabled = true;
      }
      return;
    } else if (typeof state === 'object') {
      this.selectedResources = state['selectedEntitiesForBulkOps'];
      this.status = state['statusOfSelectedEntities'];
      console.log('Status of selected resources is ', this.status);
      this.bulkResourceOpsFooterButtons =
        this.bulkEnableDisableResourceService.getBulkEnableDisableActionButtons(
          this.status,
        );
    }
  }

  resetFootersProps() {
    this.selectedResources = [];
    this.differentResStatusError = '';
    this.bulkResourceOpsFooterButtons = [];
    this.status = '';
  }

  ngOnDestroy() {
    this.footerDataSubscription.unsubscribe();
  }
}

<app-page-footer
  *ngIf="
    (selectedResources?.length > 0 &&
      status?.length > 0 &&
      status !== 'undefined') ||
    differentResStatusError?.length > 0
  "
  [showPageFooter]="true"
  [footerButtons]="bulkResourceOpsFooterButtons"
  infoText="{{ selectedResources?.length || 'No' }} {{
    selectedResources?.length | pluralize: 'Resource'
  }} Selected"
  (byClickFooterButtons)="handleFooterButtons($event)"
  errorText="{{ differentResStatusError }}"
></app-page-footer>

        </label>

      <app-rbac-button

        [status]="data?.status === 'Enabled' ? 'danger' : 'primary'"
        [action]="ENVIRONMENT_CONFIGURE"
        (onClick)="
          clusterId?.length
            ? bulkEnableDisableAtClusterLevel(
                data?.selectedResources,
                data?.status,
                ref
              )
            : bulkEnableDisableAtStackLevel(
                data?.selectedResources,
                data?.status,
                ref
              )
        "
        [disabled]="
          enableDisableBulkResourcesInProgress ||
          confirmEnableDisableInput?.value?.toLowerCase() !== 'confirm'
        "
        class="mr-3"
        [loading]="enableDisableBulkResourcesInProgress"
      ></app-rbac-button>
      <button
        nbButton
        size="small"
        (click)="ref.close()"
        [disabled]="enableDisableBulkResourcesInProgress"
      >
        Cancel
      </button>
    </nb-card-footer>
  </nb-card>
</ng-template>
