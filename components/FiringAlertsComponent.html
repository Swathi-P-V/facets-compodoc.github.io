          let imageUrl;
          let tooltip = '';
          let severityMap = {
            critical: [criticalAlertSmall, 'Critical alert'],
            warning: [warningAlertSmall, 'Warning alert'],
            unknown: [unknownAlertSmall, 'Unknown Alert'],
          };
          if (row?.['labels']?.['severity']?.length) {
            imageUrl = severityMap[row?.['labels']?.['severity']][0];
            tooltip = severityMap[row?.['labels']?.['severity']][1];
          }
          return JSON.stringify({
            textContent: cell,
            imageSVG: imageUrl,
            is_reverse: true,
            toolTipText: tooltip,
            imgClass: 'pb-0',
            showIcon: true,

  ngOnChanges(changes: SimpleChanges) {
    this.loading = true;
    if (changes?.alertGroups?.currentValue) {
      this.processAlerts().then((data) => {
        this.loading = false;
      });
    }
  }

  processAlerts() {
    return new Promise((resolve, reject) => {
      let tempFiringAlerts = [];
      let tempSilencedAlerts = [];
      this.alertGroups.forEach((group) => {
        let rules = group['rules'];
        if (rules?.length > 0) {
          rules.forEach((rule) => {
            if (rule['state'] === 'firing') {
              let alerts = rule['alerts'];
              alerts.forEach((alert) => {
                if (alert?.status?.state === 'active') {
                  let tableRow = {};
                  tableRow['alertMessage'] =
                    alert['annotations']['message'] || '--';
                  tableRow['resourceType'] =
                    alert['labels']['resourceType'] || '--';
                  tableRow['resourceName'] =
                    alert['labels']['resourceName'] || '--';
                  tableRow['severity'] = alert['labels']['severity'] || '--';
                  tableRow['activeSince'] =
                    this.formatDateService.getFormattedDate(alert['startsAt']);

                  tableRow['labels'] = alert['labels'];
                  tableRow['details'] = alert['labels'];
                  tableRow['details']['query'] = rule['query'];
                  tableRow['status'] = alert['status'];
                  tempFiringAlerts.push(tableRow);
                }
                if (alert?.status?.state === 'suppressed') {
                  let silenceRow = {};
                  const silenceEndDate = new Date(
                    alert?.['silenceDetails'][0]['endsAt'],
                  ).getTime();
                  const currentDate = new Date().getTime();
                  if (silenceEndDate > currentDate) {
                    silenceRow['alertMessage'] =
                      alert?.['annotations']['message'] || '--';
                    // console.log(
                    //   'moment ends at',
                    //   moment(alert['silenceDetails'][0]['endsAt']).fromNow(
                    //     true,
                    //   ),
                    // );
                    silenceRow['silencedTill'] = alert?.['silenceDetails'][0]?.[
                      'endsAt'
                    ]
                      ? moment(
                          alert?.['silenceDetails'][0]?.['endsAt'],
                        ).fromNow(true)
                      : '--';
                    silenceRow['comments'] =
                      alert?.['silenceDetails'][0]?.['comment'] || '--';
                    silenceRow['labels'] = alert['labels'];
                    silenceRow['details'] = alert['labels'];
                    silenceRow['details']['query'] = rule['query'];
                    silenceRow['status'] = alert['status'];
                    silenceRow['silenceDetails'] = alert?.['silenceDetails'];
                    tempSilencedAlerts.push(silenceRow);
                  }
                }
              });
            }
          });
        }
      });
      this.firingAlerts = tempFiringAlerts;
      this.silencedAlerts = tempSilencedAlerts;
      if (this.firingAlerts.length === 0) {
        this.primaryFiring = 'No firing alerts available';
      }
      if (this.silencedAlerts.length === 0) {
        this.primarySilence = 'No silenced alerts available';
      }
      resolve(true);
    });
  }

  silenceAlertPost(data: any, ref: any) {
    if (Number(this.interval) <= 0) {
      this.customToastService.error('Enter a valid time interval');
      return;
    }
    this.silencingInProgress = true;
    let snoozeInMinutes = 0;
    let alertLabels = {};

    if (data['labels'] && Object.keys(data['labels']?.length > 0)) {
      Object.keys(data['labels']).forEach((key) => {
        if (key !== 'query') {
          alertLabels[key] = data['labels'][key];
        }
      });
    }
    if (this.duration === 'minutes') {
      snoozeInMinutes = Number(this.interval);
    } else if (this.duration === 'hours') {
      snoozeInMinutes = Number(this.interval) * 60;
    } else if (this.duration === 'days') {
      snoozeInMinutes = Number(this.interval) * 24 * 60;
    }
    let request: SilenceAlarmRequest = {
      comment: this.comments,
      snoozeInMinutes: snoozeInMinutes,
      labels: alertLabels,
      id: null,
    };
    this.aWSClusterService
      .silenceAlertsUsingPOST({
        request: request,
        clusterId: this.clusterId,
      })
      .subscribe(
        (result) => {
          this.customToastService.success(
            'Alert silenced successfully for ' +
              this.interval +
              ' ' +
              this.duration,
          );
          this.silencingInProgress = false;
          this.silenceEvent.emit(true);
          ref.close();
        },
        (err) => {
          this.onAPIErrorService.onAPIError(
            err?.error?.message || 'Failed to silence the alert',
            err,
          );
        },
      );
  }

  editSilenceAlertPost(data: any, ref: any) {
    // console.log("data", data);
    if (Number(this.interval) <= 0) {
      this.customToastService.error('Enter a valid time interval');
      return;
    }
    this.silencingInProgress = true;
    let snoozeInMinutes = 0;
    let alertLabels = {};

    if (data['labels'] && Object.keys(data['labels']?.length > 0)) {
      Object.keys(data['labels']).forEach((key) => {

    } else if (this.duration === 'days') {
      snoozeInMinutes = Number(this.interval) * 24 * 60;
    }
    let request: SilenceAlarmRequest = {

    this.aWSClusterService

        clusterId: this.clusterId,
      })
      .subscribe(
        (result) => {
          this.customToastService.success(
            'Silence configuration edited successfully',
          );
          this.silencingInProgress = false;
          this.silenceEvent.emit(true);
          ref.close();
        },
        (err) => {
          this.onAPIErrorService.onAPIError(
            err?.error?.message || 'Silence edit failed',
            err,
          );
        },
      );
  }

  removeSilence(data) {
    this.unsilencingInProgress = true;
    if (data?.silenceDetails?.[0]?.id) {
      let silenceRemoveRequest = {
        silenceId: data?.silenceDetails?.[0]?.id,
        clusterId: this.clusterId,
      };
      this.loading = true;
      this.aWSClusterService
        .removeSilenceUsingDELETE(silenceRemoveRequest)
        .subscribe(
          (response) => {
            if (response?.status === 'success') {
              this.customToastService.success('Alert unsilenced successfully');
              this.unsilencingInProgress = false;
              this.silenceEvent.emit(true);
            } else if (response?.status === 'fail') {
              this.customToastService.success('Unsilencing alert failed');
            }
          },
          (err) => {
            this.unsilencingInProgress = false;
            this.onAPIErrorService.onAPIError(
              err?.error?.message || 'Unsilencing alert failed',
              err,
            );
          },
        );
    } else {
      this.customToastService.error('Invalid alert id');
    }
  }

  getMoreInfo(data) {
    this.dialogService.open(this.createArtifactory, {
      context: { ...data },
    });
  }

  openSilencePopup(data) {
    this.commentsEmpty = false;
    if (data?.silenceDetails) {
      let startDate = moment(data?.silenceDetails[0]?.startsAt);
      let endDate = moment(data?.silenceDetails[0]?.endsAt);
      let difference = Math.round(
        moment.duration(endDate.diff(startDate)).asMinutes(),
      );
      // console.log("difference", difference);
      this.comments = data?.silenceDetails[0]?.comment;
      if (difference >= 1440 && difference % (24 * 60) === 0) {
        this.interval = Math.round(difference / (24 * 60));
        this.duration = 'days';
      } else if (difference >= 60 && difference % 60 === 0) {
        this.interval = Math.round(difference / 60);
        this.duration = 'hours';
      } else {
        this.interval = difference;
        this.duration = 'minutes';
      }
    }
    this.dialogService
      .open(this.silenceAlert, {
        context: { ...data },
      })
      .onClose.subscribe((data) => {
        this.comments = '';
        this.interval = null;
        this.duration = '';
      });
  }

  openUnsilencePopup(data) {
    this.dialogService.open(this.unsilence, {
      context: { ...data },
    });
  }

  changed() {
    this.commentsEmpty = !this.comments;
  }

  blurLost() {
    this.commentsEmpty = !this.comments;
  }

  getAlertCount(severity) {
    let count = 0;
    this.firingAlerts.forEach((alert) => {
      if (alert['severity'] === severity) {
        count++;
      }
    });
    return count;
  }
}
