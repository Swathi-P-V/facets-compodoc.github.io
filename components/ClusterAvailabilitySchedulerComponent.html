import { Component } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { UiCommonClusterControllerService } from 'src/app/cc-api/services';
import { ScheduleRequest } from 'src/app/cc-api/models/schedule-request';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { AvailabilityTaskSchedule, TimeZone } from 'src/app/cc-api/models';
import { ScheduleStringService } from 'src/app/services/schedule-string.service';
import timezones from 'src/app/constants/timezones';
import {
  LAUNCH,
  DESTROY,
  SCALE_UP,
  SCALE_DOWN,
} from 'src/app/constants/releaseTypes';
import { CustomToastService } from '../../custom-toast/custom-toast.service';

const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-cluster-availability-scheduler',
  templateUrl: './cluster-availability-scheduler.component.html',
  styleUrls: ['./cluster-availability-scheduler.component.scss'],
})
export class ClusterAvailabilitySchedulerComponent extends FacetsPage {
  availableDaysOptions = [
    { value: 'MON', label: 'Monday' },
    { value: 'TUE', label: 'Tuesday' },
    { value: 'WED', label: 'Wednesday' },
    { value: 'THU', label: 'Thursday' },
    { value: 'FRI', label: 'Friday' },
    { value: 'SAT', label: 'Saturday' },
    { value: 'SUN', label: 'Sunday' },
  ];

  timeZones = timezones;

  byDay: ('MON' | 'TUE' | 'WED' | 'THU' | 'FRI' | 'SAT' | 'SUN')[] = [];
  interval: number = 1;
  byTime;
  startTimeZone: any;
  endTimeZone: any;
  clusterId: string;
  stackName: string;
  scheduleRequest: boolean = false;
  errorIcon = errorIcon;
  fetchScheduleErrorMessage = '';
  fetchingSchedule: boolean = false;
  startTime: any;
  stopTime: any;
  frequency: string = 'DAILY';
  startDay: any;
  endDay: any;
  fetchingAvailabilityRule: boolean = false;
  availabilitySchedule: AvailabilityTaskSchedule[] = [];
  availableRule: string;
  availabilityScheduleRequestList: ScheduleRequest[];
  availabilityString: string = '';
  getTimezoneText = '';
  scheduleId: string = '';
  launchTimeZone: TimeZone;
  destroyTimeZone: TimeZone;
  selectedRule:
    | `LAUNCH_DESTROY`
    | `SCALE_UP_SCALE_DOWN` = `SCALE_UP_SCALE_DOWN`;

  constructor(
    route: ActivatedRoute,
    private router: Router,
    private commonClusterController: UiCommonClusterControllerService,
    private scheduleStringService: ScheduleStringService,
    private uiCommonClusterControllerService: UiCommonClusterControllerService,
    private customToastService: CustomToastService,
  ) {
    super(route);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p.clusterId;
    this.stackName = p.stackName;
    this.getAvailableRules();
  }

  isDependedEnvironment: boolean = false;
  rulesRadioOptions = [];
  defaultRuleOption;
  getEnvironmentDetails() {
    return new Promise((resolve, reject) => {
      this.uiCommonClusterControllerService
        .getClusterCommonUsingGET(this.clusterId)
        .subscribe(
          (response) => {
            this.isDependedEnvironment = response?.baseClusterId?.length > 0;
            this.rulesRadioOptions = [
              {
                value: 'SCALE_UP_SCALE_DOWN',
                label: 'Scale up / Scale down',
                tooltipText: this.isDependedEnvironment
                  ? 'Scaling is not supported in dependent environments.'
                  : 'Scales down deployments and statefulsets, and disables cronjobs.',
                disabled: this.isDependedEnvironment,
              },
              {
                value: 'LAUNCH_DESTROY',
                label: 'Launch / Destroy',
                tooltipText:
                  'Launch and Destroy will delete the cluster and its dependent infra resources.',
                disabled: false,
              },
            ];
            this.selectedRule = this.isDependedEnvironment
              ? 'LAUNCH_DESTROY'
              : 'SCALE_UP_SCALE_DOWN';

            resolve(true);
          },
          (error) => {
            resolve(true);
          },
        );
    });
  }

  async getAvailableRules() {
    this.setDefaultValue();
    await this.getEnvironmentDetails();
    this.commonClusterController
      .getAvailabilitySchedulesUsingGET(this.clusterId)
      .subscribe(
        (response) => {
          if (response?.schedules?.length > 0) {
            this.availabilitySchedule = response.schedules;
            this.scheduleId = response.id;
            this.populateSchedule();
            this.availabilityString =
              this.scheduleStringService.constructSchedule(
                this.availabilitySchedule,
              );
            this.getTimezoneText = this.scheduleStringService.getTimezoneText(
              this.availabilitySchedule,
            );
            this.fetchingAvailabilityRule = false;
          }
        },
        (error) => {
          this.availabilitySchedule = [];
          this.fetchingAvailabilityRule = false;
          if (error?.error?.message === 'No schedules found') {
            this.fetchScheduleErrorMessage = '';
          } else {
            this.fetchScheduleErrorMessage =
              error?.error?.message || 'There was error fetching schedule';
          }
          this.availableRule =
            'Add availability rules see the active time frame here...';
        },
      );
  }

  setDefaultValue() {
    this.fetchingAvailabilityRule = true;
    this.scheduleId = '';
    this.availabilityString = '';
    this.getTimezoneText = '';
    this.frequency = 'WEEKLY';
    this.startTime = '';
    this.startTimeZone = '';
    this.endTimeZone = '';
    this.endDay = '';
    this.byDay = [];
    this.startDay = '';
    this.startTime = '';
    this.stopTime = '';
  }

  populateSchedule() {
    this.byDay = this.availabilitySchedule[0].byDay;
    let isDaily = false;
    let startTime;
    let stopTime;
    this.selectedRule =
      this.availabilitySchedule[0]?.releaseType === LAUNCH
        ? `${LAUNCH}_${DESTROY}`
        : `${SCALE_UP}_${SCALE_DOWN}`;
    if (
      this.availabilitySchedule[0]?.byDay?.length === 1 &&
      this.availabilitySchedule[0]?.byDay[0] ===
        this.availabilitySchedule[1]?.byDay[0]
    ) {
      this.availabilitySchedule.forEach((schedule) => {
        if ([LAUNCH, SCALE_UP].includes(schedule.releaseType)) {
          startTime = schedule.byTime;
          this.launchTimeZone = schedule.timeZone;
        }
        if ([DESTROY, SCALE_DOWN].includes(schedule.releaseType)) {
          stopTime = schedule.byTime;
          this.destroyTimeZone = schedule.timeZone;
        }
      });
      if (startTime > stopTime) {
        isDaily = false;
      } else {
        isDaily = true;
      }
    }
    if (this.availabilitySchedule[0]?.byDay?.length === 1 && !isDaily) {
      this.frequency = 'WEEKLY';
    } else {
      this.frequency = 'DAILY';
    }
    this.availabilitySchedule.forEach((schedule) => {
      if ([LAUNCH, SCALE_UP].includes(schedule.releaseType)) {
        this.startTime = schedule.byTime;
        this.startTimeZone = schedule.timeZone;
        this.launchTimeZone = schedule.timeZone;
        if (schedule.byDay.length === 1) {
          this.startDay = schedule.byDay[0];
        }
      }
      if ([DESTROY, SCALE_DOWN].includes(schedule.releaseType)) {
        this.stopTime = schedule.byTime;
        this.endTimeZone = schedule.timeZone;
        this.destroyTimeZone = schedule.timeZone;
        if (schedule.byDay.length === 1) {
          this.endDay = schedule.byDay[0];
        }
      }
    });
  }

  getAvailabilitySchedulePayload = (): ScheduleRequest[] => {
    const availabilityScheduleRequestList: ScheduleRequest[] = [
      {
        byDay: this.frequency === 'DAILY' ? this.byDay : [this.startDay],
        byTime: this.startTime,
        frequency: 'WEEKLY',
        timeZone: this.startTimeZone,
        releaseType:
          this.selectedRule === `${LAUNCH}_${DESTROY}` ? LAUNCH : SCALE_UP,
      },
      {
        byDay: this.frequency === 'DAILY' ? this.byDay : [this.endDay],
        byTime: this.stopTime,
        frequency: 'WEEKLY',
        timeZone: this.endTimeZone,
        releaseType:
          this.selectedRule === `${LAUNCH}_${DESTROY}` ? DESTROY : SCALE_DOWN,
      },
    ];
    return availabilityScheduleRequestList;
  };

  createUpdateAvailabilitySchedule() {
    this.scheduleRequest = true;
    if (
      this.frequency === 'WEEKLY' &&
      this.startDay === this.endDay &&
      this.startTime < this.stopTime
    ) {
      this.customToastService.error(
        'Please select stop time less than start time for weekly frequency selection of same day',
      );
      this.scheduleRequest = false;
    } else if (this.frequency === 'DAILY' && this.startTime >= this.stopTime) {
      this.customToastService.error(
        'Please select stop time greater than start time for daily frequency',
      );
      this.scheduleRequest = false;
    } else {
      let apiRequest;
      let toasterText = 'updated';
      let errorText = 'updation';
      if (this.availabilitySchedule.length) {
        apiRequest =
          this.commonClusterController.updateAvailabilitySchedulesUsingPUT({
            schedules: this.getAvailabilitySchedulePayload(),
            clusterId: this.clusterId,
            scheduleId: this.scheduleId,
          });
      } else {
        toasterText = 'created';
        errorText = 'creation';
        apiRequest =
          this.commonClusterController.createAvailabilityScheduleUsingPOST({
            schedules: this.getAvailabilitySchedulePayload(),
            clusterId: this.clusterId,
          });
      }

      apiRequest.subscribe(
        (data) => {
          this.scheduleRequest = false;
          this.customToastService.success(
            `Schedule ${toasterText} successfully`,
          );
          this.cancelAvailabilityScheduler();
        },
        (error) => {
          this.scheduleRequest = false;
          this.customToastService.error(`Schedule ${errorText} failed`);
        },
      );
    }
  }

  cancelAvailabilityScheduler = () => {
    const url =
      '/' + this.stackName + '/cluster/' + this.clusterId + '/overview';
    this.router.navigate([url]);
  };
}
