import * as Convert from 'ansi-to-html';
import {
  Component,
  ElementRef,
  Input,
  OnChanges,
  OnDestroy,
  SimpleChanges,
  ViewChild,
} from '@angular/core';
import { debounce } from 'lodash';
const errorStateIcon = './assets/error.svg';
@Component({
  selector: 'app-terminal',
  templateUrl: './terminal.component.html',
  styleUrls: ['./terminal.component.scss'],
})
export class TerminalComponent implements OnChanges, OnDestroy {
  @Input() logLines: any[]; // Logs in form of an array of {timestamp, message} type
  @Input() errorMessage: string; // Error message in case there are no log lines
  @ViewChild('terminal') private terminal: ElementRef; // HTML element reference for terminal div
  @ViewChild('endOfStream') private endOfStream: ElementRef; // HTML element reference for the end of stream line from the bottom of the terminal
  convert: Convert; // Used for converting string to html type
  contentPresent: boolean = false; // If content is present, the terminal doesn't have to automatically scroll to the bottom
  // Used at first load

  searchKey: string; // Search keyword
  noOfResults: number = -1; // Number of logs consisting of N number of search keyword
  resultInFocus: number = -1; // The index of the log which the user is currently at (while navigating through up and down arrow buttons)
  matchingResults: any[]; // The list of log elements of which consist of the search keyword
  matchingResultsOriginalCopy: any[]; // A copy of the original log elements before the 'highlight' span was added to it after processing

  showNewLogLinesNotif: boolean = false; // Flag for checking if the new log notification should pop up or not
  cancelTimeout: boolean = false; // Flag for cancelling the timeout which was set to remove the ne log notification pop up

  errorStateIcon = errorStateIcon;

  shouldScrollToBottom: boolean = true;

  wheelEventListener: any;

  constructor() {
    this.convert = new Convert();
  }

  // If there's some content on the terminal already, we'll check if the user is already at the bottom of the
  // terminal - if they're not, we will show the pop up for them to see that a more recent version of logs is present at the bottom now
  // (if the user had some active search results, we need to reapply that filter)
  // Else if the content is being loaded for the first time, we'll just  automatically scroll to the bottom of the div
  ngOnChanges(changes: SimpleChanges) {
    if (changes) {
      if (changes?.logLines?.currentValue?.length) {
        if (!this.contentPresent) {
          this.scrollToBottom();
          this.contentPresent = true;
        } else {
          if (this.shouldScrollToBottom) {
            this.scrollToBottom();
          }
          // this.bringSearchResultIntoFocus(); [NOT IN USE RIGHT NOW BUT DO NOT REMOVE]
        }
      } else if (changes.errorMessage?.currentValue?.length) {
        this.errorMessage = changes.errorMessage.currentValue;
      }

      // Remove the previous wheel event listener if it exists
      this.removeWheelEventListener();

      // Add a new wheel event listener
      const logLinesElement = document?.getElementById('logLines');
      if (logLinesElement) {
        logLinesElement.addEventListener('wheel', this.handleWheelEvent);
      }
    }
  }

  ngOnDestroy(): void {
    this.removeWheelEventListener();
  }

  updateScrollBehavior = () => {
    this.shouldScrollToBottom = this.atBottom10Percent();
  };

  scrollingTimeout: any;

  // Function to remove the wheel event listener
  removeWheelEventListener() {
    if (this.wheelEventListener) {
      document
        ?.getElementById('logLines')
        ?.removeEventListener('wheel', this.handleWheelEvent);
      clearTimeout(this.scrollingTimeout); // Clear any existing timeout
      this.wheelEventListener = null;
    }
  }

  // Function to add the wheel event listener
  addWheelEventListener() {
    const logLinesElement = document?.getElementById('logLines');
    if (logLinesElement) {
      logLinesElement.addEventListener('wheel', this.handleWheelEvent);
    }
  }

  // Clear any existing timeout
  // Remove the current listener after a pause in wheel events
  // Add a new listener for the next wheel event
  handleWheelEvent = () => {
    clearTimeout(this.scrollingTimeout);
    this.scrollingTimeout = setTimeout(() => {
      this.removeWheelEventListener();
      this.updateScrollBehavior();
      this.addWheelEventListener();
    }, 100);
  };

  // Bring search result into focus
  bringSearchResultIntoFocus() {
    if (this.searchKey && this.noOfResults > 0) {
      setTimeout(() => {
        const resultInFocus = this.resultInFocus;
        this.search(this.searchKey);
        this.scrollToResult(resultInFocus);
        this.resultInFocus = resultInFocus;
      }, 500);
    }
  }

  // Checks if the user is at the bottom 10% area of the div or has scrolled up
  atBottom10Percent() {
    let el = this.terminal?.nativeElement;
    return el?.offsetHeight + el?.scrollTop >= el?.scrollHeight * 0.9;
  }

  // Scrolls to the bottom of the terminal
  scrollToBottom() {
    setTimeout(() => {
      this.endOfStream.nativeElement.scrollIntoView({
        behavior: 'smooth',
        block: 'end',
      });
      this.cancelTimeout = true;
      this.showNewLogLinesNotif = false;
    }, 500);
  }

  // Resets if the search state of the previous search keyword (if required)
  // Searches for all the log lines which have more than 0 occurrences of the search keyword and adds a highlight span wrapper around all the matches
  search(key: string) {
    if (this.matchingResultsOriginalCopy) {
      this.removeSearchState();
    }
    this.removeFocusedLogHighlights();

    key = key?.toLowerCase();

    if (!this.searchKey || !this.searchKey?.length) {
      this.matchingResultsOriginalCopy = null;
      this.resultInFocus = -1;
      this.noOfResults = -1;
      return;
    }

    let allLogItems = Array.from(document.querySelectorAll('.logItem'));
    this.matchingResults = allLogItems.filter((logItem) =>
      logItem.innerHTML?.toLowerCase().includes(key),
    );
    this.storeOriginalState();
    this.noOfResults = this.matchingResults?.length;
    this.resultInFocus = -1;

    this.removeCurrentHighlights();
    this.highlightResults(this.matchingResults, key);
  }

  // While searching and highlighting, the matching log lines are converted to lowercase
  // Once the search key word is updated or removed, these logs should go back to their original state
  removeSearchState() {
    for (let i = 0; i < this.matchingResults?.length; i++) {
      this.matchingResults[i].innerHTML = this.matchingResultsOriginalCopy[i];
    }
  }

  // Storing original innerHTML data for all the logs (used in removeSearchState() while resetting)
  storeOriginalState() {
    this.matchingResultsOriginalCopy = [];
    for (let i = 0; i < this.matchingResults?.length; i++) {
      this.matchingResultsOriginalCopy[i] = this.matchingResults[i].innerHTML;
    }
  }

  // Highlights all the occurrences of the matching keyword
  highlightResults(results, key) {
    for (let i = 0; i < results?.length; i++) {
      let HTMLLowerCase: string = results[i].innerHTML?.toLowerCase();
      // @ts-ignore
      let newHTML = HTMLLowerCase.replaceAll(
        new RegExp(key, 'ig'),
        '<span class="highlight__">' + key + '</span>',
      );
      results[i].innerHTML = newHTML;
    }
  }

  // Removes highlight from all the occurrences of the search keyword
  removeCurrentHighlights() {
    var highlights = document.getElementsByClassName('highlight__');
    while (highlights?.length) highlights[0].classList.remove('highlight__');
  }

  // Moves up and down through the log lines which have match(s) for the search keyword
  // Highlights the currently focused log line with a gray background
  // (only happens once the use starts navigating through the search results using the arrow buttons)
  moveToResult(move: 'next' | 'previous') {
    if (this.resultInFocus === -1) {
      this.resultInFocus = 0;
    } else {
      let step = move === 'next' ? +1 : -1;
      this.matchingResults[this.resultInFocus].classList.remove(
        'activeTerminalLogLine',
      );
      this.removeFocusedLogHighlights();
      this.resultInFocus =
        (this.resultInFocus + this.noOfResults + step) % this.noOfResults;
    }
    this.scrollToResult(this.resultInFocus);
  }

  scrollToResult(result: any) {
    this.matchingResults[result].scrollIntoView({
      behavior: 'smooth',
      block: 'center',
      inline: 'nearest',
    });
    this.matchingResults[result].classList.add('activeTerminalLogLine');
  }

  // Removes any other highlighted log lines
  removeFocusedLogHighlights() {
    var activeLogs = document.getElementsByClassName('activeTerminalLogLine');
    while (activeLogs?.length)
      activeLogs[0].classList.remove('activeTerminalLogLine');
  }
}
