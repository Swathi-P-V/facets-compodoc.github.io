import { BlueprintFile } from './../../../../cc-api/models/blueprint-file';
import { AbstractCluster } from 'src/app/cc-api/models';
import { Component } from '@angular/core';
import { ActivatedRoute, Router, Params } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  UiDomainMappingControllerService,
  UiCommonClusterControllerService,
  UiDropdownsControllerService,
} from 'src/app/cc-api/services';
import { DomainDTO } from './../../../../cc-api/models/domain-dto';
import {
  FormBuilder,
  FormGroup,
  Validators,
  FormControl,
} from '@angular/forms';
import {
  whiteSpaceInputPattern,
  domainURLValidationRegex,
} from 'src/app/constants/inputPatterns';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';

@Component({
  selector: 'app-resource-domain-create-edit',
  templateUrl: './resource-domain-create-edit.component.html',
  styleUrls: ['./resource-domain-create-edit.component.scss'],
})
export class ResourceDomainCreateEditComponent extends FacetsPage {
  clusterId: any;
  resourceInfo: BlueprintFile;
  clusterState: string;
  resourceName: string;
  resourceType: string;
  stackName: string = '';
  urlParamAlias: string = '';
  selectedDomain: DomainDTO = {};
  loadingDomains: boolean = false;
  cluster: AbstractCluster = {};
  cloud: 'AWS' | 'AZURE' | 'LOCAL' | 'GCP' | 'KUBERNETES';
  loadingInitialData: boolean = false;
  cloudTooltipMap = {
    AWS: 'Select an ACM certificate that includes your domain and subdomains. Enter the ACM certificate ARN.',
  };
  certTooltip: string = '';
  allowedFlavours = {
    AWS: ['nlb_nginx', 'nginx_ingress_controller', 'aws_alb'],
    GCP: ['nlb_nginx', 'nginx_ingress_controller', 'gcp_alb'],
    AZURE: ['nlb_nginx', 'nginx_ingress_controller', 'azure_alb'],
  };
  flavorTooltipMap = {
    nlb_nginx: `We support certificates through Kubernetes secrets, follow these instructions to 
                  <a class="white-link" target="_blank" href="https://github.com/nabsul/k8s-letsencrypt#step-6-create-a-kubernetes-secret">
                  create k8s-based secret certificate</a>.`,
    nginx_ingress_controller: `We support certificates through Kubernetes secrets, follow these instructions to 
                  <a class="white-link" target="_blank" href="https://github.com/nabsul/k8s-letsencrypt#step-6-create-a-kubernetes-secret">
                  create k8s-based secret certificate</a>.`,
    aws_alb: `You need to pass the base domain acm certificate arn from AWS
                (<a class="white-link" target="_blank" style="color: #fff" href="https://docs.aws.amazon.com/acm/latest/userguide/gs-acm-request-public.html">Public</a>,
                  <a class="white-link" target="_blank" href="https://docs.aws.amazon.com/acm/latest/userguide/gs-acm-request-private.html">Private</a>
                )`,
    gcp_alb: `You need to pass pre-shared certificate, follow these instruction to 
                <a class="white-link" target="_blank" href="https://cloud.google.com/kubernetes-engine/docs/how-to/ingress-multi-ssl#pre-shared-certs">
                create pre-shared-certificates</a>.`,
    azure_alb: `You need to pass the secret key id of the secret in
                <a class="white-link" target="_blank" href="https://learn.microsoft.com/en-us/azure/key-vault/certificates/tutorial-import-certificate?tabs=azure-portal">
                azure key vault.</a>`,
  };

  domainForm: FormGroup = this.fb.group({
    domainURL: [
      '',
      [
        Validators.required,
        Validators.pattern(whiteSpaceInputPattern),
        this.validateUrl.bind(this),
      ],
    ],
    alias: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    certReference: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
  });

  creationInProgress: boolean = false;
  updateInProgress: boolean = false;
  isCreate: boolean = true;
  domainRequestInProgress: boolean;

  constructor(
    _route: ActivatedRoute,
    private router: Router,
    private domainMappingService: UiDomainMappingControllerService,
    private uiCommonClusterService: UiCommonClusterControllerService,
    private uiDropdownsControllerService: UiDropdownsControllerService,
    private fb: FormBuilder,
    private customToast: CustomToastService,
  ) {
    super(_route);
  }

  get domainURL() {
    return this.domainForm.get('domainURL');
  }

  get alias() {
    return this.domainForm.get('alias');
  }

  get certReference() {
    return this.domainForm.get('certReference');
  }

  validateUrl(control: FormControl): { [key: string]: boolean } | null {
    const url = control.value;
    if (url) {
      const urlParts = url.split('.');
      const startsOrEndsWithHyphen = urlParts.some(
        (part) => part.startsWith('-') || part.endsWith('-'),
      );
      if (urlParts.length >= 2 && urlParts[urlParts.length - 2].length > 63) {
        return { urlLengthError: true };
      }
      if (startsOrEndsWithHyphen) {
        return { hyphenError: true };
      }
      if (url && !domainURLValidationRegex.test(url)) {
        return { invalidUrl: true };
      }
    }
    return null;
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p?.clusterId;
    this.resourceName = p?.resourceName;
    this.resourceType = p?.resourceType;
    this.stackName = p?.stackName;
    this.certTooltip = '';
    this.domainForm.reset();
    if (p?.resourceType !== 'ingress') {
      this.navigateBack('configuration');
    } else {
      this.getIngressResourceDetails();
      if (p?.alias) {
        this.urlParamAlias = p?.alias;
        this.isCreate = false;
        this.alias.disable();
        this.getSelectedDomainDetails();
      }
    }
  }

  getIngressResourceDetails() {
    this.loadingInitialData = true;
    Promise.allSettled([this.getEnvInformation(), this.getResourceInfo()]).then(
      (data) => {
        /*
          Take user back to configuration Page if
          * If cloud is GCP and AZURE and flavor is empty
          * If resource flavour is not part of allowed flavours for the resource
        */

        this.checkResourceAndNavigateBack();
        this.setTooltipText();
        this.loadingInitialData = false;
      },
      (err) => {
        this.loadingInitialData = false;
      },
    );
  }

  checkResourceAndNavigateBack() {
    if (
      !this.allowedFlavours[this.cloud]?.includes(
        this.resourceInfo?.info?.flavour,
      )
    ) {
      this.navigateBack('configuration');
    }
  }

  setTooltipText() {
    if (this.cloud === 'AWS') {
      let additionalTooltip = [
        'nginx_ingress_controller',
        'nlb_nginx',
      ].includes(this.resourceInfo?.info?.flavour)
        ? 'Along with this we also support ACM certificate'
        : '';
      this.certTooltip =
        this.resourceInfo?.info?.flavour?.length > 0
          ? `${
              this.flavorTooltipMap[this.resourceInfo?.info?.flavour]
            } ${additionalTooltip}`
          : this.cloudTooltipMap['AWS'];
    } else {
      this.certTooltip =
        this.flavorTooltipMap[this.resourceInfo?.info?.flavour];
    }
  }

  getEnvInformation = () => {
    return new Promise((resolve, reject) => {
      this.uiCommonClusterService
        .getClusterCommonUsingGET(this.clusterId)
        .subscribe(
          (data) => {
            this.cluster = data;
            this.cloud = this.cluster?.cloud;
            resolve(true);
          },
          (err) => {
            reject(err);
          },
        );
    });
  };

  getResourceInfo = () => {
    return new Promise((resolve, reject) => {
      this.uiDropdownsControllerService
        .getResourceByClusterIdUsingGET({
          clusterId: this.clusterId,
          resourceType: this.resourceType,
          resourceName: this.resourceName,
        })
        .subscribe(
          (data) => {
            this.resourceInfo = data;
            resolve(true);
          },
          (err) => {
            reject(err);
          },
        );
    });
  };

  navigateBack = (pageType = 'ingress-domains') => {
    this.router.navigate([
      '/',
      this.stackName,
      'cluster',
      this.clusterId,
      'resources',
      'type',
      this.resourceType,
      'resource',
      this.resourceName,
      pageType,
    ]);
  };

  getEnv() {
    this.uiCommonClusterService
      .getClusterCommonUsingGET(this.clusterId)
      .subscribe(
        (data) => {
          this.cluster = data;
          this.cloud = this.cluster?.cloud;
        },
        (err) => {
          console.log('err', err);
        },
      );
  }

  getSelectedDomainDetails() {
    this.loadingDomains = true;
    this.domainMappingService
      .getAllDomainsUsingGET({
        resourceName: this.resourceName,
        resourceType: this.resourceType,
        clusterId: this.clusterId,
      })
      .subscribe(
        (domains) => {
          let domainData = domains;
          this.selectedDomain = domainData.filter(
            (domain) => domain.alias === this.urlParamAlias,
          )[0];
          if (!this.selectedDomain) {
            this.navigateBack('configuration');
          } else {
            this.domainForm.patchValue({
              domainURL: this.selectedDomain?.domain,
              alias: this.selectedDomain?.alias,
              certReference: this.selectedDomain.certificateReference,
            });
          }
          this.loadingDomains = false;
        },
        (err) => {
          this.loadingDomains = false;
          this.customToast.error(
            err?.error?.message || 'There is some error fetching domain data',
          );
          this.navigateBack();
        },
      );
  }

  createOrUpdateCustomDomain() {
    this.domainRequestInProgress = true;
    let httpReq = this.isCreate
      ? this.domainMappingService.addDomainMappingUsingPOST(
          this.getDomainParams(),
        )
      : this.domainMappingService.updateDomainMappingUsingPUT(
          this.getDomainParams(),
        );
    httpReq.subscribe(
      (response) => {
        this.domainRequestInProgress = false;
        this.customToast.success(
          `Custom Domain has been ${
            this.isCreate ? 'created' : 'updated'
          } successfully.`,
        );
        this.navigateBack();
      },
      (error) => {
        this.domainRequestInProgress = false;
        this.customToast.error(
          error?.error?.message ||
            `Domain ${this.isCreate ? 'creation' : 'update'} failed.`,
        );
      },
    );
  }

  getDomainParams() {
    return {
      resourceName: this.resourceName,
      resourceType: this.resourceType,
      clusterId: this.clusterId,
      domain: {
        domain: this.domainURL.value,
        alias: this.alias.value,
        certificateReference: this.certReference.value,
      },
    };
  }
}
