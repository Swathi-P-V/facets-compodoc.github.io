import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';

import { BulkSelectTableEntriesService } from 'src/app/services/bulk-select-table-entries.service';

  };

  smartTableCustomActionButtons = [

              '/',

              this.blueprint,

              this.branchName,
              'resource',
              'type',
              row?.resourceType,
              'name',
              row?.resourceName,
              'overview',

            display: value,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: SmartTableRouterlinkComponent,
        type: 'custom',
      },
      flavour: {
        title: 'Flavor',
        filter: true,
        width: '18%',
        type: 'text',
        valuePrepareFunction(flavour, row) {
          return flavour?.length > 0 ? flavour : 'N/A';

      version: {
        title: 'Version',
        filter: true,

          return version?.length > 0 ? version : 'N/A';

  errorStateIcon = errorStateIcon;
  blueprint: string;
  branchName: string;
  deleteLoader: boolean = false;
  emptyIcon = emptyIcon;

  // Bulk-ops variables
  isIndeterminate = false;
  isSelectAll = false;
  source: LocalDataSource = new LocalDataSource();
  filteredResources: any[] = [];
  bulkResourceOpsSubscription: Subscription;
  routeParamSubscription: Subscription;

  constructor(
    private uiDropdownControllerService: UiDropdownsControllerService,
    private uiBlueprintDesignerControllerService: UiBlueprintDesignerControllerService,
    private blueprintDesignerTableService: BlueprintDesignerTableService,
    private resourceDetailsService: ResourceDetailsService,
    private dialogService: NbDialogService,
    private router: Router,
    private activatedRoute: ActivatedRoute,
    private bulkSelectTableEntriesService: BulkSelectTableEntriesService,
    private bulkEnableDisableResourceService: BulkEnableDisableResourcesService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    this.routeParamSubscription = this.activatedRoute.params.subscribe((p) =>
      this.CustomOnPageLoad(p),
    );

    // Subscribe to data changes in ng2-smart-table's 'LocalDataSource'(this.source).
    // Update 'filteredResources' and recalculate selection state on each change.
    this.source.onChanged().subscribe((change) => {
      this.filteredResources = change.elements;
      this.setSelectAllCheckboxState();
    });

    // Subscribe to the 'bulkResourceOpsComplete$' observable which emits status of completion of bulk resource operations.
    this.bulkResourceOpsSubscription =
      this.bulkEnableDisableResourceService.bulkResourceOpsComplete.subscribe(
        (status) => {
          this.onBulkEnableDisableResourceComplete(status);
        },
      );
  }

  CustomOnPageLoad(p: Params): void {
    this.headerEventSubscription?.unsubscribe();

    this.blueprint = p?.stackName;
    this.branchName = p?.branchName;

    this.getAllResources();
    this.subscribeToHeaderEvents();
    this.resourceDetailsService.clearContextStack();
  }

  ngOnDestroy(): void {
    this.headerEventSubscription?.unsubscribe();
    this.bulkResourceOpsSubscription.unsubscribe();
    this.routeParamSubscription?.unsubscribe();
  }

  headerEventSubscription: Subscription;
  subscribeToHeaderEvents() {
    this.headerEventSubscription =
      this.blueprintDesignerTableService.eventFromHeader.subscribe(
        (eventType) => {
          let action = eventType?.action;
          let extras = eventType?.extras;
          switch (action) {
            case 'addedResource':
              this.getAllResources();
              break;
            case 'bulkUpload':
              this.openBulkUploadDialog();
              break;
          }
        },
      );
  }

  // This function is called when the component is booting up
  // we need to get all the resources for the table
  loadingResources: boolean = false;
  errorWhileFetchingResources: boolean = false;
  allResources: any[];
  getAllResources() {
    this.allResources = [];
    const allResourceTypes: Set<any> = new Set<any>();
    this.errorWhileFetchingResources = false;
    this.loadingResources = true;
    this.uiDropdownControllerService
      .getAllResourcesByStackUsingGET({
        stackName: this.blueprint,
        includeContent: true,
        branch: this.branchName,
      })
      .subscribe(
        (resources) => {
          resources.forEach((resource) => {
            if (resource.resourceType?.toLowerCase() != 'unknown') {
              this.allResources.push({
                resourceName: resource.resourceName,
                resourceType: resource.resourceType,
                status: resource?.info?.disabled,
                modifiedBy: null,
                providedStatus: resource?.info?.provided
                  ? 'Provided'
                  : 'Normal',
                ...(resource?.info || {}),
                selected: false,
                id: `${resource.resourceType}:${resource.resourceName}`,
                filename: resource?.filename,
                directory: resource?.directory,
              });
              allResourceTypes.add(resource.resourceType);
            }
          });
        },
        (error) => {
          this.errorWhileFetchingResources = true;
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Failed to fetch the resources',
            error,
          );
        },
      )
      .add(() => {
        this.resourceListTableConfig.columns.resourceType.filter.config.list = [
          ...Array.from(allResourceTypes)
            .sort()
            .map((name) => {
              return {
                title: name,
                value: name,
              };
            }),
        ];
        this.resourceListTableConfig = { ...this.resourceListTableConfig };
        this.source.load([...this.allResources]);
        this.loadingResources = false;
      });
  }

  // Redirect the user to the resource overview
  // page. This might change for the edit button later.
  redirectToResourceOverview(resource: any) {
    this.router.navigate([
      `/stack/${this.blueprint}/blueprint-designer/${this.branchName}/resource/type/${resource.resourceType}/name/${resource.resourceName}/spec`,
    ]);
  }

  // Open a dialog to confirm with the user
  // if they want to delete the resource
  openDeleteResourceModal(resource: any) {
    this.dialogService.open(this.deleteConfirmation, {
      context: {
        id: resource.resourceName,
        name: resource.resourceName,
        ...resource,
      },
    });
  }

  @ViewChild('bulkUploadDialog') bulkUploadDialog: TemplateRef<any>;
  openBulkUploadDialog() {
    this.dialogService.open(this.bulkUploadDialog, {
      closeOnBackdropClick: false,
      hasBackdrop: true,
    });
  }

  onDeleteResource({ id, name, ...resource }, ref) {
    this.deleteLoader = true;
    this.uiBlueprintDesignerControllerService
      .deleteResourcesUsingDELETE({
        stackName: this.blueprint,
        branch: this.branchName,
        resourceFileRequests: [
          {
            content: {},
            resourceName: resource.resourceName,
            resourceType: resource.resourceType,
          },
        ],
      })
      .subscribe(
        (response) => {
          this.customToastService.success(
            `Resource ${resource.resourceName} deleted successfully`,
          );
          this.getAllResources();
          ref?.close();
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            error?.error?.message ||
              `Failed to delete Resource ${resource.resourceName}`,
            error,
          );
        },
      )
      .add(() => {
        this.deleteLoader = false;
      });
  }

  onSelectAllCheckboxClicked(checked) {
    // Update isSelectAll
    this.isSelectAll = checked;

    //Update the selected property of all resources which are currently loaded in table
    let { updatedSource, filteredEntities } =
      this.bulkSelectTableEntriesService.updateSelectedPropOnSelectAllClick(
        this.source,
        this.filteredResources,
        checked,
        'id',
      );
    this.source.load([...updatedSource]);
    this.filteredResources = cloneDeep(filteredEntities);

    // Update the page footer based on currentSelectedEntities
    this.updateFooterState();
  }

  onCheckboxClicked(data) {
    // Update the selected property of resource whose checkbox is clicked
    let updatedSourceData =
      this.bulkSelectTableEntriesService.updateSelectedPropOnCheckboxClick(
        data,
        this.source,
        this.bulkEnableDisableResourceService.findResource(data),
      );
    this.source.load([...updatedSourceData]);

    // Determine the state of 'selectAll' checkbox
    this.setSelectAllCheckboxState();

    // Update the page footer based on currentSelectedEntities
    this.updateFooterState();
  }

  updateFooterState() {
    let selectedEntitiesState =
      this.bulkSelectTableEntriesService.determineSelectedEntityState(
        this.source,
        'disabled',
      );
    this.bulkEnableDisableResourceService.sendFooterData(selectedEntitiesState);
  }

  // False is emitted when user clicks on cancel button. If enable/disable is successful then true is emitted.
  onBulkEnableDisableResourceComplete(complete) {
    this.isIndeterminate = false;
    this.isSelectAll = false;
    complete
      ? this.getAllResources()
      : this.source.load([...this.allResources]);
  }

  setSelectAllCheckboxState() {
    const { isIndeterminate, isSelectAll } =
      this.bulkSelectTableEntriesService.getSelectAllCheckboxState(
        this.source,
        'id',
        this.filteredResources,
      );
    this.isIndeterminate = isIndeterminate;
    this.isSelectAll = isSelectAll;
  }
}
