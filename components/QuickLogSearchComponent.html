import { Component } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { SafeUrl, DomSanitizer } from '@angular/platform-browser';
import { ClusterState } from '../../../enums/cluster-state';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  UiCommonClusterControllerService,
  UiDeploymentControllerService,
  UiDropdownsControllerService,
} from 'src/app/cc-api/services';
const emptyStateLogo = './assets/empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-quick-log-search',
  templateUrl: './quick-log-search.component.html',
  styleUrls: ['./quick-log-search.component.scss'],
})
export class QuickLogSearchComponent extends FacetsPage {
  clusterId: any;
  stackName: any;
  clusterState: string;
  logSearchUrl = '';
  showEmptyState: boolean = false;
  secondaryMsg: string;
  primaryMsg: string =
    'This page is available when environment is in running state.';
  buttonText: string = 'Manage Your Environment';
  fetchingClusterState = false;
  grafanaTunnelSafe: SafeUrl = this.sanitizer.bypassSecurityTrustUrl(
    this.logSearchUrl.toString(),
  );
  emptyStateLogo = emptyStateLogo;
  showBtn: boolean = true;
  status: 'info' | 'error' = 'info';
  emptyStateClusterStates = ClusterState;
  fetchingCluster: boolean = false;
  cluster: any;
  resourceType: string = '';
  resourceName: string = '';
  resourceDetails;
  fetchingResourceDetails: boolean = true;
  namespace: string = '';

  constructor(
    activateRoute: ActivatedRoute,
    private uiDeploymentControllerService: UiDeploymentControllerService,
    private uiCommonClusterControllerService: UiCommonClusterControllerService,
    private router: Router,
    private sanitizer: DomSanitizer,
    private dropdownControllerService: UiDropdownsControllerService,
  ) {
    super(activateRoute);
  }

  OnPageLoad(p: Params): void {
    this.stackName = p?.stackName;
    this.clusterId = p?.clusterId;
    this.resourceName = p?.resourceName;
    this.resourceType = p?.resourceType;
    this.showEmptyState = false;
    this.namespace = '';
    this.asyncOnPageLoad();
  }

  async asyncOnPageLoad() {
    await this.getClusterState();
    if (
      !Object.keys(this.emptyStateClusterStates)?.includes(this.clusterState)
    ) {
      Promise.all([
        this.getEnvironmentConfig(),
        this.getResourceDetails(),
      ]).then(
        (data) => {
          this.namespace =
            this.resourceDetails?.info?.namespace ||
            this.cluster?.namespace ||
            'default';
          this.setLogsUrl();
        },
        (error) => {
          this.setEmptyStateValues(
            '',
            error?.error?.message ||
              ' There is some error fetching environment or resource information',
            'error',
          );
        },
      );
    }
  }

  getClusterState() {
    this.fetchingClusterState = true;
    return new Promise((resolve, reject) => {
      this.uiDeploymentControllerService
        .getClusterStateUsingGET(this.clusterId)
        .subscribe(
          (cluster) => {
            this.clusterState = cluster;
            if (
              Object.keys(this.emptyStateClusterStates)?.includes(
                this.clusterState,
              )
            ) {
              this.setEmptyStateValues(
                'This page is available when environment is in running state.',
                'Environment is in ' +
                  this.emptyStateClusterStates[this.clusterState] +
                  ' state!',
                'info',
              );
            }
            this.fetchingClusterState = false;
            resolve(true);
          },
          (error) => {
            this.clusterState = null;
            this.setEmptyStateValues(
              '',
              error?.error?.message ||
                ' There is some error fetching environment information',
              'error',
            );
            this.fetchingClusterState = false;
            resolve(true);
          },
        );
    });
  }

  getEnvironmentConfig = () => {
    this.fetchingCluster = true;
    return new Promise((resolve, reject) => {
      this.uiCommonClusterControllerService
        .getClusterCommonUsingGET(this.clusterId)
        .subscribe(
          async (response) => {
            this.cluster = response;
            this.fetchingCluster = false;
            resolve(true);
          },
          (error) => {
            this.cluster = null;
            this.fetchingCluster = false;
            reject(error);
          },
        );
    });
  };

  getResourceDetails = () => {
    this.fetchingResourceDetails = true;
    return new Promise((resolve, reject) => {
      this.dropdownControllerService
        .getResourceByClusterIdUsingGET({
          clusterId: this.clusterId,
          resourceType: this.resourceType,
          resourceName: this.resourceName,
          includeContent: true,
        })
        .subscribe(
          (data) => {
            this.resourceDetails = data;
            this.fetchingResourceDetails = false;
            resolve(true);
          },
          (error) => {
            this.fetchingResourceDetails = false;
            reject(error);
          },
        );
    });
  };

  setLogsUrl() {
    this.logSearchUrl = `/tunnel/${
      this.cluster?.baseClusterId || this.clusterId
    }/grafana/d/liz0yRCZRas/quick-log-search?orgId=1&theme=light&refresh=30s&var-namespace=${
      this.namespace
    }&var-app=${this.resourceName}&var-pod=All&var-container=All`;
    this.grafanaTunnelSafe = this.makeSafe(this.logSearchUrl);
  }

  setEmptyStateValues(primaryMsg, secondaryMsg, status) {
    this.secondaryMsg = secondaryMsg;
    this.showBtn = status === 'error' ? false : true;
    this.primaryMsg = primaryMsg;
    this.emptyStateLogo = status === 'error' ? errorIcon : emptyStateLogo;
    this.showEmptyState = true;
    this.status = status;
  }

  makeSafe(url: string) {
    return this.sanitizer.bypassSecurityTrustResourceUrl(url);
  }

  navigate(event) {
    let urlToNav =
      '/' + this.stackName + '/cluster/' + this.clusterId + '/releases/';
    this.router.navigate([urlToNav]);
  }

  handleBack = () => {
    this.router.navigate([
      '/',
      this.stackName,
      'cluster',
      this.clusterId,
      'resources',
    ]);
  };
}
