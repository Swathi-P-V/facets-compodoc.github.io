import { Component } from '@angular/core';
import { SnapshotInfo } from 'src/app/cc-api/models';
import {
  UiCommonClusterControllerService,
  UiDropdownsControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { ActivatedRoute, Params } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import { PinSnapshotDialogComponent } from './pin-snapshot-dialog/pin-snapshot-dialog.component';
import { CreateSnapshotDialogComponent } from './create-snapshot-dialog/create-snapshot-dialog.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { Observable, of } from 'rxjs';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const emptyStateLogo = './assets/empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-cluster-disaster-recovery',
  templateUrl: './cluster-disaster-recovery.component.html',
  styleUrls: ['./cluster-disaster-recovery.component.scss'],
})
export class ClusterDisasterRecoveryComponent extends FacetsPage {
  settings = {
    columns: {
      name: {
        title: 'Snapshot ID',
        editable: false,
      },
      startTime: {
        title: 'Timestamp',
        valuePrepareFunction: function (startTime: string) {
          return new Date(startTime).toString();
        },
        editable: false,
      },
      pinned: {
        title: 'Pinned',
        valuePrepareFunction: function (pinned: boolean) {
          return pinned ? 'YES' : 'NO';
        },
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
      perPage: 5,
    },
    add: false,
    actions: {
      edit: false,
      delete: false,
      add: false,
      position: 'right',
      custom: [],
    },
    hideSubHeader: true,
  };

  clusterId = '';
  resourceType = '';
  instanceName = '';

  drData: Array<SnapshotInfo> = [];

  fetchSnapshotsSpinner = false;
  pinSnapshotSpinner = false;
  createSnapshotSpinner = false;
  stackName: any;
  secondaryMsg: string = 'Create snapshots for resources';
  emptyStateLogo = emptyStateLogo;
  errorLogo = errorIcon;
  primaryMsg: string = 'No backups found!';
  showBtn: boolean = false;
  status: 'info' | 'error' = 'info';
  resourceTypes: string[];
  fetchingResourceTypes = false;
  fetchingResources = false;
  filteredResourceTypes$: Observable<string[]>;
  resources: string[];
  filteredResources$: Observable<string[]>;
  resourceNamePlaceholder: string = 'Select Resource';
  constructor(
    private uiCommonClusterController: UiCommonClusterControllerService,
    route: ActivatedRoute,
    private dialogService: NbDialogService,
    private uiStackController: UiStackControllerService,
    private uiDropdownsControllerService: UiDropdownsControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(route);
  }

  OnPageLoad(p: Params): void {
    this.drData = [];
    this.clusterId = p.clusterId;
    this.stackName = p.stackName;
    this.resourceType = '';
    this.instanceName = '';
    this.filteredResourceTypes$ = null;
    this.filteredResources$ = null;
    this.fetchingResourceTypes = true;
    this.uiStackController.getResourceTypesUsingGET(this.stackName).subscribe(
      (data: Array<string>) => {
        this.resourceTypes = data;
        this.filteredResourceTypes$ = of(this.resourceTypes);
        this.fetchingResourceTypes = false;
      },
      (error) => {
        this.fetchingResourceTypes = false;
      },
    );
  }

  pinSnapshot(event) {
    if (event.action === 'Pin') {
      if (event.data['pinned']) {
        this.customToastService.error('Snapshot is already pinned');
        return;
      }

      this.dialogService
        .open(PinSnapshotDialogComponent)
        .onClose.subscribe((proceed) => {
          if (proceed) {
            this.pinSnapshotSpinner = true;
            this.uiCommonClusterController
              .pinSnapshotUsingPOST1({
                snapshotInfo: event.data,
                resourceType: this.resourceType,
                instanceName: this.instanceName,
                clusterId: this.clusterId,
              })
              .subscribe((snap) => {
                console.log('pinned snapshot');
                this.fetchSnapshots();
                this.pinSnapshotSpinner = false;
              });
          }
        });
    }
  }

  private async getPinnedSnapshot() {
    try {
      let pinnedSnapshot = await this.uiCommonClusterController
        .getPinnedSnapshotUsingGET1({
          resourceType: this.resourceType,
          instanceName: this.instanceName,
          clusterId: this.clusterId,
        })
        .toPromise();

      return pinnedSnapshot;
    } catch (err) {
      console.log(err);
    }

    return null;
  }

  private filterPinnedSnapshot(
    pinnedSnapshot: SnapshotInfo,
    snapshots: Array<SnapshotInfo>,
  ) {
    if (!pinnedSnapshot) {
      return null;
    }

    let pinnedSnapshotPos = -1;
    for (let idx = 0; idx < snapshots?.length; idx++) {
      const element = snapshots[idx];
      if (element?.cloudSpecificId === pinnedSnapshot?.cloudSpecificId) {
        pinnedSnapshotPos = idx;
        break;
      }
    }

    return pinnedSnapshotPos;
  }

  async fetchSnapshots() {
    this.fetchSnapshotsSpinner = true;
    try {
      let snapshots = await this.uiCommonClusterController
        .listSnapshotsUsingGET1({
          resourceType: this.resourceType,
          instanceName: this.instanceName,
          clusterId: this.clusterId,
        })
        .toPromise();

      let pinnedSnapshot = await this.getPinnedSnapshot();
      let pinnedSnapshotPos = this.filterPinnedSnapshot(
        pinnedSnapshot,
        snapshots,
      );

      if (pinnedSnapshotPos === -1) {
        snapshots.unshift(pinnedSnapshot);
      } else if (pinnedSnapshot != null && pinnedSnapshotPos >= 0) {
        snapshots.splice(pinnedSnapshotPos, 1);
        snapshots.unshift(pinnedSnapshot);
      }

      if (snapshots?.length == 0) {
        this.customToastService.error('No snapshots found');
      } else {
        this.customToastService.success('Success');
      }

      this.drData = snapshots;
      this.fetchSnapshotsSpinner = false;
    } catch (err) {
      console.log(err);
      this.drData = [];
      this.fetchSnapshotsSpinner = false;
      this.setErrorValues(err);
    }
  }

  setErrorValues(error) {
    this.secondaryMsg =
      error?.error?.message ||
      ' There is some error fetching snapshot information';
    this.primaryMsg = '';
    this.emptyStateLogo = this.errorLogo;
    this.status = 'error';
  }

  createSnapshots() {
    this.dialogService
      .open(CreateSnapshotDialogComponent)
      .onClose.subscribe((proceed) => {
        if (proceed) {
          this.fetchSnapshotsSpinner = true;
          try {
            this.uiCommonClusterController
              .createSnapshotUsingPOST({
                resourceType: this.resourceType,
                instanceName: this.instanceName,
                clusterId: this.clusterId,
              })
              .subscribe(
                (result) => {
                  if (!result) {
                    this.customToastService.error(
                      'Error: Could not create snapshot',
                    );
                  } else {
                    this.customToastService.error(
                      'Snapshot creation triggered',
                    );
                  }
                  this.fetchSnapshots();
                  this.fetchSnapshotsSpinner = false;
                },
                (err) => {
                  this.onAPIErrorService.onAPIError(
                    err?.error?.message ||
                      'There is some error while creating snapshot',
                    err,
                  );
                  this.fetchSnapshotsSpinner = false;
                },
              );
          } catch (err) {
            this.fetchSnapshotsSpinner = false;
            this.customToastService.error('Error: Could not create snapshot');
          }
        }
      });
  }

  onResourceTypeSelect($event: string) {
    this.resourceType = $event;
    // fetch resources
    this.resourceNamePlaceholder = '';
    this.instanceName = '';
    this.fetchingResources = true;
    this.uiDropdownsControllerService
      .getApplicationListByClusterIdAndResourceNameUsingGET({
        clusterId: this.clusterId,
        resourceType: this.resourceType,
      })
      .subscribe(
        (resources: Array<string>) => {
          this.resources = resources;
          this.filteredResources$ = of(this.resources);
          this.resourceNamePlaceholder = 'Select Resource';
          this.fetchingResources = false;
        },
        (err) => {
          this.resources = [];
          this.resourceNamePlaceholder = 'Resources unavailable';
          this.fetchingResources = false;
        },
      );
  }

  onResourceSelect($event: string) {
    this.instanceName = $event;
  }
}
