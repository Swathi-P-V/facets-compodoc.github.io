import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import mixpanel from 'mixpanel-browser';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  UiStackControllerService,
  UiArtifactCiControllerService,
  UiPromotionWorkflowControllerService,
  UiArtifactRoutingRuleControllerService,
} from 'src/app/cc-api/services';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { NbDialogService, NbSelectComponent } from '@nebular/theme';
import { ArtifactCI, PromotionWorkflow, Stack } from 'src/app/cc-api/models';
import { UserAuthoritiesService } from 'src/app/services/user-authorities.service';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import {
  ARTIFACT_CI_WRITE,
  PROMOTIONAL_WORKFLOW_WRITE,
} from 'src/app/constants/permissions';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
import {
  CI_INTEGRATION_CREATE_FAIL,
  CI_INTEGRATION_CREATE_SUCCESS,
  CI_INTEGRATION_EDIT_FAIL,
  CI_INTEGRATION_EDIT_SUCCESS,
} from 'src/app/constants/analytics-track-events';

const newTabSvg = './assets/create-account.svg';

@Component({
  selector: 'app-artifacts-create-ci',
  templateUrl: './artifacts-create-ci.component.html',
  styleUrls: ['./artifacts-create-ci.component.scss'],
})
export class ArtifactsCreateCIComponent extends FacetsPage {
  @ViewChild('workflowDetails', { static: true }) //promotional workflow details popover
  private workflowDetails: TemplateRef<any>;

  @ViewChild('workflowSelect') workflowSelectElement: NbSelectComponent;

  stackName: string = ''; //blueprint name
  stacks: Stack[] = []; //blueprint list
  formType: string = 'create'; //formtype takes 2 values, 'create' or 'edit' based on page action
  ciId: string = ''; //CI integration id in case of edit operation
  callInProgress: boolean = false; //in progress status variable for create and edit ci
  artifactCIForm: FormGroup = this.fb.group({
    name: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    registrationType: ['RELEASE_STREAM', Validators.required],
    blueprint: [false],
    workflow: [''],
    ruleId: [null],
  }); //CI integration create form
  typeSelectOptions: { label: string; value: string }[] = [
    { label: 'Environment', value: 'ENVIRONMENT' },
    { label: 'Release Stream', value: 'RELEASE_STREAM' },
  ]; //options for registration type selection
  advanceOpenLogo = './assets/advance-open.svg';
  advanceCloseLogo = './assets/advance-close.svg';
  showAdvance: boolean = true;
  blueprintOptionsList: { label: string; value: string }[] = []; //blueprint select menu options list
  promotionWorkflows: PromotionWorkflow[] = [];
  environmentWorkflows: PromotionWorkflow[] = []; //environment stack specific workflows
  releaseStreamWorkflows: PromotionWorkflow[] = []; //release stream registration type workflows
  selectedPromotionWorkflow: any; // The PWF user has selected or the one which was preselected
  defaultWorkflow: any; // Current default WF
  artifactCiWrite = ARTIFACT_CI_WRITE;

  //promotion workflow select menu options list
  workflowOptionsList: {
    label: string;
    value: string;
    hierarchy: string;
    isDefault: boolean;
  }[] = [];

  ruleOptionsList: { label: string; value: string }[] = [];
  envIdNameMapping: any = {}; // Maps environment names to their IDs
  allEnvs: any; //environment list
  fetchingPromotion: boolean = false; //promotion api call status
  fetchingCIInfo: boolean = false; //CI integration edit ci info api call status
  fetchingAllEnvs: boolean = false; //environment fetch api call info
  fetchingCIRules: boolean = false;
  loading: boolean = false; //Loading page
  currentArtifactCI: ArtifactCI = {}; //current CI integration object in case of edit operation
  newTabSvg = newTabSvg;
  isHybrid: boolean = false; // Indicates if the current form is open for an HYBRID object

  constructor(
    private activatedRoute: ActivatedRoute,
    private uiStackController: UiStackControllerService,
    private uiArtifactCiController: UiArtifactCiControllerService,
    private promotionWorkflow: UiPromotionWorkflowControllerService,
    private router: Router,
    private fb: FormBuilder,
    private dialogService: NbDialogService,
    public userAuthoritiesService: UserAuthoritiesService,
    private uiArtifactRoutingRuleController: UiArtifactRoutingRuleControllerService,
    private customToastService: CustomToastService,
    private onApiErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  get name() {
    return this.artifactCIForm.get('name');
  }

  get registrationType() {
    return this.artifactCIForm.get('registrationType');
  }

  get blueprint() {
    return this.artifactCIForm.get('blueprint');
  }

  get workflow() {
    return this.artifactCIForm.get('workflow');
  }

  get ruleId() {
    return this.artifactCIForm.get('ruleId');
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params) {
    this.loading = true;
    this.resetForm();
    this.asyncPageLoad(p);
  }

  async asyncPageLoad(param) {
    this.stackName = param?.stackName;
    await this.getAllEnvironments();
    if (param?.ciId) {
      this.ciId = param?.ciId;
      this.formType = 'edit';
      this.getCIInfo();
    } else {
      this.getRegistrationTypeSpecificWorkflows(
        this.registrationType.value,
        this.stackName,
        null,
        true,
      );
      this.getCIRuleByRegistrationTypeAndStackName(
        this.registrationType.value,
        this.stackName,
      );
    }
    this.loading = false;
  }

  resetForm() {
    this.environmentWorkflows = [];
    this.releaseStreamWorkflows = [];
    this.artifactCIForm.patchValue({
      name: '',
      registrationType: 'RELEASE_STREAM',
      blueprint: false,
      workflow: '',
      ruleId: null,
    });
    this.isHybrid = false;
  }

  async getAllEnvironments(populateVals: boolean = false) {
    this.fetchingAllEnvs = true;
    return new Promise((resolve) => {
      this.uiStackController
        .getAllClustersUsingGET({
          page: 0,
          perPage: 10000,
          sortBy: 'ID',
        })
        .subscribe(
          (envs) => {
            this.allEnvs = envs?.content;
            this.allEnvs.forEach((env) => {
              this.envIdNameMapping[env?.id] = env?.name;
            });
            this.fetchingAllEnvs = false;
          },
          (err) => {
            this.fetchingAllEnvs = false;
          },
        )
        .add(() => {
          resolve(true);
        });
    });
  }

  getAllWorkflows(registrationType, stackName, artifactCI) {
    this.fetchingPromotion = true;
    this.workflowOptionsList = [];
    this.promotionWorkflow.getAllWorkflowsUsingGET().subscribe(
      (res) => {
        this.fetchingPromotion = false;
        this.promotionWorkflows = res;
        this.promotionWorkflows.forEach((workflow) => {
          if (
            (stackName?.length > 0 &&
              workflow.stackName === stackName &&
              workflow.registrationType === 'ENVIRONMENT') ||
            workflow.registrationType === 'RELEASE_STREAM' ||
            workflow.registrationType === 'HYBRID'
          ) {
            this.workflowOptionsList.push({
              label: workflow.workflowName,
              value: workflow.id,
              hierarchy: this.getStreamValue(
                workflow,
                workflow.registrationType,
              ),
              isDefault: workflow.defaultWorkflow,
            });
            if (!this.ciId && workflow.defaultWorkflow) {
              this.setPresetDefaultWorkflow(
                workflow.registrationType != 'HYBRID',
                workflow,
              );
            }
          }
          if (this.ciId && artifactCI?.promotionWorkflowId === workflow.id) {
            this.setPresetDefaultWorkflow(false, workflow);
          }
        });
      },
      (err) => {
        this.fetchingPromotion = false;
      },
    );
  }

  getRegistrationTypeSpecificWorkflows(
    registrationType,
    stackName = '',
    artifactCI = null,
    start = false,
  ) {
    this.fetchingPromotion = true;
    this.workflowOptionsList = [];
    this.promotionWorkflows = [];
    let workflowParams = {
      registrationType: registrationType,
      stackName: stackName?.length > 0 ? stackName : null,
    };
    this.promotionWorkflow
      .getRegistrationSpecificWorkflowsUsingGET(workflowParams)
      .subscribe(
        (res) => {
          this.fetchingPromotion = false;
          this.promotionWorkflows = res;
          if (registrationType === 'ENVIRONMENT') {
            this.environmentWorkflows = [...res];
          } else if (registrationType === 'RELEASE_STREAM') {
            this.releaseStreamWorkflows = [...res];
            this.releaseStreamWorkflows = this.releaseStreamWorkflows.filter(
              (wf) => !wf.stackName?.length || wf.stackName === this.stackName,
            );
          }
          this.setWorkflows(registrationType, artifactCI?.promotionWorkflowId);
          if (this.ciId) {
            if (artifactCI?.promotionWorkflowId) {
              this.workflow.setValue(artifactCI?.promotionWorkflowId);
              this.selectedPromotionWorkflow = this.workflow;
            }
          }
        },
        (error) => {
          this.fetchingPromotion = false;
        },
      );
  }

  getStreamValue(workflow, registrationType) {
    const names = workflow?.hierarchies?.map((hierarchy) => {
      return registrationType === 'ENVIRONMENT'
        ? this.envIdNameMapping[hierarchy.registrationValue]
        : hierarchy.registrationValue;
    });
    return `(${names.join(' -> ')})`;
  }

  getCIInfo() {
    this.fetchingCIInfo = true;
    this.typeSelectOptions = [
      { label: 'Environment', value: 'ENVIRONMENT' },
      { label: 'Release Stream', value: 'RELEASE_STREAM' },
    ];
    this.uiArtifactCiController.getArtifactCIUsingGET(this.ciId).subscribe(
      async (res) => {
        if (res) {
          this.currentArtifactCI = res;
          //Once you get registrationType fetch the ci rules specific to stackName and registrationType
          await this.getCIRuleByRegistrationTypeAndStackName(
            this.currentArtifactCI?.registrationType,
            this.stackName,
          );
          //If registraionType is Hybrid then add one more radio option Hybrid.
          if (this.currentArtifactCI?.registrationType === 'HYBRID') {
            this.typeSelectOptions.push({
              label: 'Hybrid',
              value: 'HYBRID',
            });
            this.isHybrid = true;
          }
          //Populate form values
          if (this.currentArtifactCI) {
            this.artifactCIForm.patchValue({
              name: this.currentArtifactCI?.ciName,
              registrationType: this.currentArtifactCI?.registrationType,
              blueprint: !this.currentArtifactCI?.stackName?.length,
              ruleId: this.currentArtifactCI?.ruleId,
              workflow: '',
            });
          }
          //Based on registrationType populate and set promotion workflow
          if (this.currentArtifactCI?.registrationType === 'HYBRID') {
            this.getAllWorkflows(
              res?.registrationType,
              res?.stackName,
              this.currentArtifactCI,
            );
          } else {
            this.getRegistrationTypeSpecificWorkflows(
              res?.registrationType,
              res?.stackName,
              this.currentArtifactCI,
            );
          }
          this.name.disable();
          this.registrationType.disable();
          this.blueprint.disable();
        }
        this.fetchingCIInfo = false;
      },
      (err) => {
        this.fetchingCIInfo = false;
      },
    );
  }

  goBack = () => {
    this.router.navigate(['/', 'stack', this.stackName, 'artifacts', 'ci']);
  };

  openAdvance() {
    this.showAdvance = !this.showAdvance;
  }

  createArtifactCI() {
    const stackName =
      this.registrationType.value === 'ENVIRONMENT' ||
      (this.registrationType.value === 'RELEASE_STREAM' &&
        !this.blueprint.value)
        ? this.stackName
        : null;
    let ciParams: ArtifactCI = {
      ciName: this.name.value ? this.name.value.trim() : '',
      stackName,
      registrationType: this.registrationType.value,
      integrationType: 'EXTERNAL',
      entityType: 'ARTIFACT_CI',
      promotionWorkflowId:
        this.workflow?.value?.length > 0 ? this.workflow.value : null,
      ruleId: this.ruleId.value,
    };
    this.callInProgress = true;
    this.uiArtifactCiController.createArtifactCIUsingPOST(ciParams).subscribe(
      (res) => {
        this.callInProgress = false;
        mixpanel.track(CI_INTEGRATION_CREATE_SUCCESS);
        this.customToastService.success(
          `CI Integration ${res?.ciName} created successfully`,
        );
        this.goBack();
      },
      (error) => {
        mixpanel.track(CI_INTEGRATION_CREATE_FAIL, { error });
        this.onApiErrorService.onAPIError(
          'Failed: ' + error?.error?.message ||
            `Failed to create CI Integration ${ciParams?.ciName}`,
          error,
        );
        this.callInProgress = false;
      },
    );
  }

  editArtifactCI() {
    let ciParams: ArtifactCI = {
      ciName: this.currentArtifactCI?.ciName,
      stackName: this.currentArtifactCI?.stackName,
      registrationType: this.currentArtifactCI?.registrationType,
      integrationType: 'EXTERNAL',
      entityType: 'ARTIFACT_CI',
      promotionWorkflowId:
        this.workflow?.value?.length > 0 ? this.workflow.value : null,
      createdBy: this.currentArtifactCI.createdBy,
      creationDate: this.currentArtifactCI.creationDate,
      ruleId: this.ruleId.value,
    };
    this.callInProgress = true;
    this.uiArtifactCiController
      .updateArtifactCIUsingPUT({
        ciId: this.currentArtifactCI.id,
        artifactCI: ciParams,
      })
      .subscribe(
        (res) => {
          this.callInProgress = false;
          mixpanel.track(CI_INTEGRATION_EDIT_SUCCESS);
          this.customToastService.success(
            `CI Integration ${ciParams?.ciName} updated successfully`,
          );
          this.goBack();
        },
        (error) => {
          mixpanel.track(CI_INTEGRATION_EDIT_FAIL, { error });
          this.onApiErrorService.onAPIError(
            'Failed: ' + error?.error?.message ||
              `Failed to update CI Integration ${ciParams?.ciName}`,
            error,
          );
          this.callInProgress = false;
        },
      );
  }

  openPopover(event, value) {
    event.stopPropagation();
    let workflow = this.promotionWorkflows.find(
      (workflow) => workflow?.id === value,
    );
    this.dialogService.open(this.workflowDetails, {
      context: { rowData: workflow },
    });
  }

  isLoading() {
    return this.loading || this.fetchingCIInfo;
  }

  setWorkflows(registrationType, presetWorkflowID?) {
    this.defaultWorkflow = null;
    if (registrationType === 'ENVIRONMENT') {
      this.promotionWorkflows = [...this.environmentWorkflows];
    } else if (registrationType === 'RELEASE_STREAM') {
      this.promotionWorkflows = [...this.releaseStreamWorkflows];
    }
    this.promotionWorkflows?.forEach((workflow) => {
      this.workflowOptionsList.push({
        label: workflow.workflowName,
        value: workflow.id,
        hierarchy: this.getStreamValue(workflow, registrationType),
        isDefault: workflow.defaultWorkflow,
      });
      if (
        workflow.id === presetWorkflowID ||
        (!presetWorkflowID && workflow.defaultWorkflow)
      ) {
        this.defaultWorkflow =
          this.workflowOptionsList[this.workflowOptionsList.length - 1];
        this.artifactCIForm.get('workflow').setValue(this.defaultWorkflow);
        this.setPresetDefaultWorkflow(!presetWorkflowID, workflow);
      }
    });
  }

  typeChange(event) {
    this.workflowOptionsList = [];
    this.ruleOptionsList = [];
    this.workflow.setValue('');
    this.ruleId.setValue(null);
    this.name.markAsUntouched();
    this.name.markAsPristine();
    //Refetch ci rules
    this.getCIRuleByRegistrationTypeAndStackName(event, this.stackName);
    if (
      (event === 'RELEASE_STREAM' &&
        this.releaseStreamWorkflows?.length <= 0) ||
      (event === 'ENVIRONMENT' && this.environmentWorkflows?.length <= 0)
    ) {
      this.getRegistrationTypeSpecificWorkflows(event, this.stackName);
    } else {
      this.setWorkflows(event);
    }
  }

  openCreateWorkflow(event) {
    event.stopPropagation();
    this.router.navigate(['../create-workflow'], {
      relativeTo: this.route,
    });
  }

  disablePromotionWorkflowOption(): boolean {
    return this.userAuthoritiesService.isActionDisabled(
      PROMOTIONAL_WORKFLOW_WRITE,
    );
  }

  setPresetDefaultWorkflow(settingDefault: boolean, defaultWorkflow) {
    if (settingDefault && this.formType == 'edit') {
      return;
    }
    setTimeout(() => {
      const defaultWorkflow_option: any = this.workflowOptionsList.find(
        (workflow) => workflow.value == defaultWorkflow.id,
      );
      this.artifactCIForm.get('workflow').setValue(defaultWorkflow.id);
      this.selectedPromotionWorkflow = defaultWorkflow_option;
    }, 500);
  }

  getCIRuleByRegistrationTypeAndStackName(registrationType, stackName) {
    return new Promise((resolve, reject) => {
      this.fetchingCIRules = true;
      this.ruleOptionsList = [];
      this.uiArtifactRoutingRuleController
        .getAllArtifactRoutingRulesUsingGET({
          stackName: stackName,
          registrationType: registrationType,
        })
        .subscribe(
          (res) => {
            if (res?.length > 0) {
              this.ruleOptionsList.push({
                label: 'Clear selection',
                value: null,
              });
              res.forEach((rule) => {
                this.ruleOptionsList.push({
                  label: rule.ruleName,
                  value: rule.id,
                });
              });
            }
            resolve(true);
            this.fetchingCIRules = false;
          },
          (error) => {
            this.fetchingCIRules = false;
            resolve(false);
          },
        );
    });
  }

  navigateToCreateCIRule() {
    this.router.navigate([
      '/',
      'stack',
      this.stackName,
      'artifacts',
      'ci-rules',
      'create',
    ]);
  }
}
