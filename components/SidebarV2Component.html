import { Component } from '@angular/core';
import { ContextUpdatesService } from 'src/app/services/context-updates.service';
import { BreadCrumbConfig } from 'src/app/models/breadcrumb.model';
import { UserService } from 'src/app/services/user.service';
import { SidebarService } from './sidebar.util.service';
import {
  UiBlueprintDesignerControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { Router } from '@angular/router';
import { NbMenuService } from '@nebular/theme';
import { Subject, takeUntil } from 'rxjs';
import { SessionContextStoreService } from 'src/app/services/session-context-store.service';

@Component({
  selector: 'app-sidebar-v2',
  templateUrl: './sidebar.component.html',
  styleUrls: ['./sidebar.component.scss'],
})
// implements OnChanges
export class SidebarV2Component {
  private destroy$ = new Subject<void>(); // Used for fetching the active menu item from the nb-menu component (tag: 'mainMenu' here)

  sidebarInitialized: boolean = false; // Used to hold the rendering of the sidebar till the context is ready
  initiatedFirstContext: boolean = false; // Used to check if the sidebar has been initialized with the first context
  skipFirstUpdateFromBreadcrumbs: boolean = true; // First context is set from cookies/session storage, optimistic update should not be processed

  allowedBlueprints: Set<string>; // Set of all the allowed blueprints for the user
  allowedEnvironmentIDs: Set<string>; // Set of all the allowed environments IDs
  allowedStackNames: Set<string> = new Set<string>(); //Stores stackNames of all the stacks a user has access to.
  allBlueprints: any[]; // List of all the blueprints
  blueprintEnvironmentMap = new Map<string, any[]>(); // Map of all the blueprints and their environments

  currentContext: BreadCrumbConfig = {} as BreadCrumbConfig; // Current context set inside the sidebar
  currentClusterURL: string; // Current cluster URL from the URL (stack/stackName/cluster/clusterID)

  currentSidebarConfig: any; // Entire sidebar object which gets rendered

  constructor(
    private sidebarUtilService: SidebarService,
    private uiStackControllerService: UiStackControllerService,
    private blueprintDesignerService: UiBlueprintDesignerControllerService,
    private contextUpdatesService: ContextUpdatesService,
    private sessionContextStoreService: SessionContextStoreService,
    private userService: UserService,
    private menuService: NbMenuService,
    private router: Router,
  ) {
    if (!this.userService.userLoaded) {
      this.userService.userObserver.subscribe(async (userLoaded) => {
        if (userLoaded) {
          this.allowedEnvironmentIDs = new Set(
            this.userService.user.allowedClusterIds,
          );
          await this.getAllBlueprints();
          this.subscribeToChangesInData();
          this.checkForLastUserSession();
          this.checkForPreviousContextInCookies();
          this.checkForMissingContext();
          this.startListeningToContextUpdates();
          this.sidebarInitialized = true;
        }
      });
    }
  }

  sidebarReloading: boolean = false;
  skipFirstReloadSignal: boolean = true;
  subscribeToChangesInData() {
    this.contextUpdatesService.contextSwitcherUpdated.subscribe(
      async (signal: any) => {
        if (signal.updated && !this.skipFirstReloadSignal) {
          this.sidebarReloading = true;
          await this.getAllBlueprints();
          this.checkForLastUserSession(signal.extras);
          this.sidebarReloading = false;
        }
        this.skipFirstReloadSignal = false;
      },
    );
  }

  async getAllBlueprints() {
    this.blueprintEnvironmentMap?.clear();
    return new Promise((resolve) => {
      this.allBlueprints = [];
      this.uiStackControllerService
        .getAllClustersUsingGET({
          page: 0,
          perPage: 20000,
          sortBy: 'ID',
        })
        .subscribe((envs) => {
          this.allowedStackNames = new Set(
            this.userService?.user?.allowedStackNames,
          );
          let envsList = envs?.content;
          if (this.allowedStackNames.size !== 0) {
            envsList = envsList?.filter((env) =>
              this.allowedStackNames?.has(env?.stackName),
            );
          }
          if (this.allowedEnvironmentIDs.size !== 0) {
            envsList?.forEach((cluster) => {
              const stackName = cluster?.stackName;
              if (this.allowedEnvironmentIDs.has(cluster?.id)) {
                // Two blueprints can have same environment name. So concatenating blueprint with environment name to get uniqueID
                if (!this.blueprintEnvironmentMap.has(stackName)) {
                  this.blueprintEnvironmentMap.set(stackName, []);
                }
                this.blueprintEnvironmentMap.get(stackName).push(cluster);
              }
            });
          } else {
            envsList?.forEach((cluster) => {
              const stackName = cluster?.stackName;
              if (!this.blueprintEnvironmentMap.has(stackName)) {
                this.blueprintEnvironmentMap.set(stackName, []);
              }
              this.blueprintEnvironmentMap.get(stackName).push(cluster);
            });
          }
        })
        .add(async () => {
          // Need to fill in the remaining blueprints which were not returned in the
          // above API call because they didn't have any environments
          await this.getRemainingBlueprints();
          this.allowedBlueprints = new Set(this.blueprintEnvironmentMap.keys());
          this.allBlueprints = Array.from(this.allowedBlueprints.keys()).sort();
          resolve(true);
        });
    });
  }

  async getRemainingBlueprints() {
    return new Promise((resolve) => {
      this.uiStackControllerService
        .getStacksUsingGET1()
        .subscribe((stacks) => {
          stacks.forEach((stack) => {
            if (!this.blueprintEnvironmentMap.has(stack.name)) {
              this.blueprintEnvironmentMap.set(stack.name, []);
            }
            if(!this.contextUpdatesService.getLastBranchForBlueprint(stack.name)) {
              this.contextUpdatesService.setLastOpenedBranchInCookies(
                stack?.name,
                stack?.branch,
              );
            }
          });
        })
        .add(() => {
          resolve(true);
        });
    });
  }

  // Fetch the context from the localStorage
  // if it is there, initialize the sidebar with it
  checkForLastUserSession(options?: any) {
    let hostName = window.location.hostname.split('.')[0];
    const ITEM_NAME = `${this.userService?.user?.name}_${hostName}_context`;
    const ITEM_STATUS_NAME = `${this.userService?.user?.name}_${hostName}_context_used`;
    if (localStorage.getItem(ITEM_NAME)) {
      const newContext = JSON.parse(
        localStorage.getItem(ITEM_NAME),
      ) as BreadCrumbConfig;
      if (options?.extras) {
        if (options?.extras?.entityType === 'CLUSTER') {
          newContext.environment = options.extras?.id;
          newContext.resource = '';
          newContext.resourceType = '';
        } else if (options?.extras?.entityType === 'BLUE_PRINT') {
          newContext.blueprint = options?.extras?.stackName;
          newContext.environment = '';
          newContext.resource = '';
          newContext.resourceType = '';
        }
      }
      this.updateCurrentBasedOnLatest(newContext);
      this.initiatedFirstContext = true;
      localStorage.setItem(ITEM_STATUS_NAME, 'true');
      this.sessionContextStoreService.setUsedStatus(
        this.userService?.user?.name,
        'true',
      );
    }
  }

  // Check if the cookieService has got context
  // if it is there, initialize the sidebar with it
  // ONLY IF, the context already has not been initialized with
  // the last user session context
  checkForPreviousContextInCookies() {
    let currentFromCookies =
      this.contextUpdatesService.getContextFromCookies('currentContext');
    let previousFromCookies =
      this.contextUpdatesService.getContextFromCookies('previousContext');
    if (!this.initiatedFirstContext && currentFromCookies?.blueprint?.length) {
      if (!currentFromCookies.blueprint && previousFromCookies.blueprint) {
        currentFromCookies = previousFromCookies;
      }
      const newContext = currentFromCookies;
      this.updateCurrentBasedOnLatest(newContext);
      this.initiatedFirstContext = true;
    }
  }

  checkForMissingContext() {
    if (!this.currentContext?.blueprint) {
      this.setFirstBlueprintAndEnvironment();
      this.updateCurrentBasedOnLatest(this.currentContext);
    }
  }

  // Subscribe to the breadcrumb's updates
  // and update the currentContext based on it
  // (Exception: the initial load)
  startListeningToContextUpdates() {
    this.contextUpdatesService.optimisticContextUpdates.subscribe((context) => {
      if (!this.sidebarReloading) {
        const nextContext = context.currentContext as BreadCrumbConfig;
        this.updateCurrentBasedOnLatest(nextContext);
      }
    });
  }

  getClusterURL(currentContext: BreadCrumbConfig) {
    if (!currentContext?.environment) {
      currentContext.environment = this.blueprintEnvironmentMap
        .get(this.currentContext.blueprint)
        ?.sort(this.nameComparator)[0]?.id;
    }
    return `/${this.currentContext.blueprint}/cluster/${this.currentContext.environment}`;
  }

  updateCurrentBasedOnLatest(newContext: BreadCrumbConfig) {
    let keys = Array.from(Object.keys(newContext));
    keys.forEach((key) => {
      if (newContext[key]) {
        this.currentContext[key] = newContext[key];
      }
    });

    if (this.invalidBlueprint(this.currentContext)) {
      if (this.allBlueprints.length) {
        this.setFirstBlueprintAndEnvironment();
      } else {
        this.currentContext = {
          blueprint: '',
          environment: '',
        } as any as BreadCrumbConfig;
      }
    } else if (this.invalidEnvironment(this.currentContext)) {
      const allEnvironments = this.blueprintEnvironmentMap
        .get(this.currentContext.blueprint)
        ?.sort(this.nameComparator);
      if (allEnvironments?.length) {
        this.currentContext.environment = allEnvironments[0].id;
      } else if (this.currentContext.blueprint?.length) {
        this.currentContext.environment = '';
      } else {
        this.setFirstBlueprintAndEnvironment();
      }
    }
    // Handle conditions for when the currently set context
    // is either not available in the API response or
    // is not available for the user to access because of user's restricted access
    if (
      this.allowedEnvironmentIDs?.size != 0 ||
      !this.currentContext?.blueprint
    ) {
      if (!this.allowedBlueprints.has(this.currentContext.blueprint)) {
        // Make the first available blueprint from the
        // list of all blueprints the current blueprint
        this.setFirstBlueprintAndEnvironment();
      } else if (
        this.currentContext?.environment?.length &&
        !this.allowedEnvironmentIDs.has(this.currentContext.environment)
      ) {
        // Make the first available environment from the
        // currently chosen blueprint the current environment
        this.setFirstBlueprintAndEnvironment();
      }
    }

    this.currentClusterURL = this.getClusterURL(this.currentContext);
    this.sessionContextStoreService.setSessionContext(
      JSON.stringify(this.currentContext),
      this.userService?.user?.name,
      'true',
    );

    // By this point, the currentContext has
    // 1. Checked for previous user session's context
    // 2. Checked for the last session from cookies
    // 3. Has subscribed to the breadcrumbs' optimistic updates
    // 4. Has made sure that the currentContext is:
    // ----> Available in the list fetched from the API
    // ----> Available to the user if they have restricted access

    this.currentSidebarConfig = this.sidebarUtilService.getSidebarConfig(
      this.currentContext.blueprint,
      this.currentContext.environment,
      this.currentClusterURL,
      this.contextUpdatesService.getLastBranchForBlueprint(
        this.currentContext.blueprint,
      ),
      this.allBlueprints,
      !this.blueprintEnvironmentMap.has(this.currentContext.blueprint) ||
        this.blueprintEnvironmentMap.get(this.currentContext.blueprint)
          ?.length === 0,
      this.router.url,
    );
    this.contextUpdatesService.setCurrentContext(this.currentContext);
    this.setActiveState();
  }

  setFirstBlueprintAndEnvironment() {
    if (this.allBlueprints.length === 0) {
      this.currentContext = {
        blueprint: '',
        environment: '',
      } as BreadCrumbConfig;
      return;
    }
    this.currentContext.blueprint = this.allBlueprints[0];
    const allEnvironments = this.blueprintEnvironmentMap
      .get(this.currentContext.blueprint)
      ?.sort(this.nameComparator);
    this.currentContext.environment = allEnvironments[0]?.id;
  }

  nameComparator = (a, b) => {
    const nameA = a.name.toLowerCase();
    const nameB = b.name.toLowerCase();
    if (nameA < nameB) {
      return -1;
    } else if (nameA > nameB) {
      return 1;
    } else {
      return 0;
    }
  };

  // Helper functions
  // On click event from the mainMenu nb-menu
  // Used in disabling the active menu item from the other nb-menu (tag: 'footerMenu') in app-layout
  updateFooterMenuWithState() {
    // Removing all the active menu items by passing some random string to removeAllOtherActiveTabs
    this.removeAllOtherActiveTabs('_____remove-all-menu-items____');
    setTimeout(() => {
      this.menuService
        .getSelectedItem('mainMenu')
        .pipe(takeUntil(this.destroy$))
        .subscribe((menuItem) => {
          if (
            menuItem?.item ||
            this.router.url.includes(this.currentClusterURL)
          ) {
            this.removeAllOtherActiveTabs(menuItem?.item?.title);
          } else if (this.router.url.includes('settings')) {
            // patch: re-mark settings as active if environment was clicked on
            this.markSettingsActive();
          }
        });
    }, 3000);
  }

  // If the current blueprint set in the context
  // does not belong to the application (maybe after being deleted or something like that)
  // this function will return true and false otherwise
  invalidBlueprint(context: BreadCrumbConfig) {
    return !this.blueprintEnvironmentMap.has(context.blueprint);
  }

  // If the current environment set in the context
  // does not belong to the blueprint which is set, this function
  // will return true and false otherwise
  invalidEnvironment(context: BreadCrumbConfig) {
    const environmentID = context.environment;
    const blueprint = context.blueprint;
    return !this.blueprintEnvironmentMap
      ?.get(blueprint)
      ?.find((env) => env.id === environmentID);
  }

  // DOM manipulation because nb-menu doesn't have APIs to update the state for active menu items
  // Goes through the anchor tabs of the other menu and updates their classList
  removeAllOtherActiveTabs(currentlyActive: string) {
    const allMenus: any = document.querySelectorAll('nb-menu');
    const allAnchorTags = [...allMenus[1]?.getElementsByTagName('a')];
    for (let i = 0; i < allAnchorTags?.length; i++) {
      const item = allAnchorTags[i];
      item?.classList.remove('active');
    }
  }

  markSettingsActive() {
    const allMenus: any = document.querySelectorAll('nb-menu');
    if (allMenus?.length) {
      const allAnchorTags = [...allMenus?.[1]?.getElementsByTagName('a')];
      allAnchorTags?.[1]?.classList?.add('active');
    }
  }

  setActiveState() {
    let urlParams;
    const currentURL = this.router.url;
    if (currentURL?.length > 0 && currentURL.indexOf('/') >= 0) {
      urlParams = currentURL.split('/');
      if (
        urlParams?.length > 1 &&
        (urlParams[1] === 'home' || urlParams[1] === 'stack') &&
        !(urlParams[3] === 'artifacts') &&
        !(urlParams[3] === 'dashboards')
      ) {
        this.currentSidebarConfig.forEach((item) => {
          if (item.title === 'Blueprint') {
            item.selected = true;
          }
        });
      } else if (urlParams?.length > 1 && urlParams[3] === 'artifacts') {
        this.currentSidebarConfig.forEach((item) => {
          if (item.title === 'Artifacts') {
            item.selected = true;
          }
        });
      } else if (urlParams?.length > 1 && urlParams[1] === 'settings') {
        this.currentSidebarConfig.forEach((item) => {
          if (item.title === 'Settings') {
            item.selected = true;
          }
        });
      } else if (urlParams?.length > 2 && urlParams[2] === 'cluster') {
        this.currentSidebarConfig.forEach((item) => {
          if (item.title === 'Environment') {
            item.selected = true;
          }
        });
      }
    }
  }
}
