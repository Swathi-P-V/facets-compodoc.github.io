import {
  Component,
  Input,
  Output,
  EventEmitter,
  OnChanges,
  SimpleChanges,
} from '@angular/core';
import { ActivatedRoute, Params } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { UserAuthoritiesService } from 'src/app/services/user-authorities.service';
declare var gtag;

@Component({
  selector: 'app-page-header',
  templateUrl: './page-header.component.html',
  styleUrls: ['./page-header.component.scss'],
})
export class PageHeaderComponent extends FacetsPage implements OnChanges {
  @Input()
  heading: string;

  @Input() pageHeadingTextAsLink: {
    link: string;
    openInNewTab: boolean;
    tooltip?: string;
  };

  @Input()
  subheading: string;

  @Input()
  onBack: () => {};

  @Input()
  isTabHeader: boolean = false;

  @Input()
  buttonClass = '';

  @Input() actionButtons: {
    id: string;
    title: string;
    buttonType: string;
    nbTooltip: string;
    icon: any;
    disabled: boolean;
    hidden: boolean;
    loading?: boolean;
    gtagEvent: {
      action: string;
      category: string;
    };
    action: string;
    actionsList?: string[];
    dataAttrActionId?: string;
  }[] = [];

  tertiaryButtons: {
    id: string;
    title: string;
    buttonType: string;
    nbTooltip: string;
    icon: any;
    disabled: boolean;
    hidden: boolean;
    loading?: boolean;
    gtagEvent: {
      action: string;
      category: string;
    };
    action: string | string[];
    actionsList?: string[];
    buttonSuffixIconProps: {
      icon: string;
      status: string;
    };
    dataAttrActionId?: string;
  }[] = [];

  clusterId: string;

  @Input() headingIcons: {
    id: string;
    icon: string;
    pack: string;
    status: string;
    iconSize: string;
    class: string;
    iconTooltip: string;
  }[] = [];

  @Input() primaryIcon: {
    id: string;
    icon: string;
    pack: string;
    status: string;
    iconTooltip: string;
    iconSize: string;
  };
  @Input() showHighlighterForMoreActions: boolean = false;

  // eslint-disable-next-line @angular-eslint/no-output-native
  @Output() click = new EventEmitter<string>();

  @Output() tertiaryButtonSuffixIconClick = new EventEmitter<string>();

  constructor(
    private activatedRoute: ActivatedRoute,
    public userAuthoritiesService: UserAuthoritiesService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    if (p) {
      this.clusterId = p?.clusterId;
    }
  }

  ngOnChanges(change: SimpleChanges) {
    if (change.actionButtons?.currentValue) {
      const actionButtons = change.actionButtons.currentValue;
      this.tertiaryButtons = actionButtons.filter((button) => {
        return button.buttonType === 'tertiary';
      });
    }
  }
  onButtonClick(value: any) {
    if (value.gtagEvent) {
      this.trackOnClick(value.gtagEvent);
    }
    this.click.emit(value.id);
  }

  trackOnClick(gtagEventObject) {
    gtag('event', gtagEventObject.action, {
      eventCategory: gtagEventObject.category,
    });
  }
  onClickSuffixIcon(button) {
    this.tertiaryButtonSuffixIconClick?.emit(button?.id);
  }
}
