import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute } from '@angular/router';
import { v4 as uuid } from 'uuid';
import { AwsAccount } from 'src/app/cc-api/models';
import {
  UiAccountsControllerService,
  UiMetaControllerService,
} from 'src/app/cc-api/services';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { getToastSubTexts } from 'src/app/utils/toast.utils';

@Component({
  selector: 'app-aws-account-create-edit',
  templateUrl: './aws-account-create-edit.component.html',
  styleUrls: ['./aws-account-create-edit.component.scss'],
})
export class AwsAccountCreateEditComponent extends FacetsPage {
  @Input() mode: 'create' | 'edit';
  @Input() passedRef: any;

  @Input() accountDetails: AwsAccount;

  @Output() afterSaveAccount: EventEmitter<any> = new EventEmitter<any>();

  awsAccountForm: FormGroup = this.formBuilder.group({
    name: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    externalId: [
      uuid(),
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    role: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
  });

  readonly ACCOUNTS_WRITE = 'ACCOUNTS_WRITE';
  saveAccountInProgress: boolean = false;
  verifyInProgress: boolean = false;
  accountId: string = '';
  fetchingAccountId: boolean = false;

  constructor(
    private activatedRoute: ActivatedRoute,
    private formBuilder: FormBuilder,
    private accountControllerService: UiAccountsControllerService,
    private customToast: CustomToastService,
    private uiMetaControllerService: UiMetaControllerService,
  ) {
    super(activatedRoute);
  }

  get name() {
    return this.awsAccountForm.get('name');
  }
  get externalId() {
    return this.awsAccountForm.get('externalId');
  }
  get role() {
    return this.awsAccountForm.get('role');
  }

  OnPageLoad() {
    this.getAccountId();
    if (this.mode === 'edit') {
      this.awsAccountForm.patchValue({
        name: this.accountDetails?.name,
        externalId: this.accountDetails?.externalId,
        role: this.accountDetails?.iamRole,
      });
    }
  }

  getRequestPayload() {
    let accountName = this.awsAccountForm?.controls?.name?.value?.trim() || '';
    let externalId =
      this.awsAccountForm?.controls?.externalId?.value?.trim() || '';
    let role = this.awsAccountForm?.controls?.role?.value?.trim() || '';
    let accountParams: AwsAccount = {
      name: accountName,
      externalId: externalId,
      iamRole: role,
      accountType: 'CLOUD',
      provider: 'AWS',
    };
    return accountParams;
  }

  createEditAwsAccount() {
    let requestPayload = this.getRequestPayload();
    this.saveAccountInProgress = true;

    let httpReq;
    if (this.mode === 'create') {
      httpReq =
        this.accountControllerService.createAwsAccountUsingPOST(requestPayload);
    } else if (this.mode === 'edit') {
      httpReq = this.accountControllerService.updateAwsAccountUsingPUT({
        id: this.accountDetails?.id,
        awsAccountRequest: {
          ...requestPayload,
          id: this.accountDetails?.id,
          createdBy: this.accountDetails?.createdBy,
          creationDate: this.accountDetails?.creationDate,
        },
      });
    }

    let { successSubText, failureSubText } = getToastSubTexts(this.mode);
    httpReq.subscribe(
      (data) => {
        this.customToast.success(
          `Aws account ${data?.name} ${successSubText} successfully`,
        );
        this.afterSaveAccount.emit(data);
        this.passedRef.close();
        this.saveAccountInProgress = false;
        this.awsAccountForm.reset();
      },
      (err) => {
        this.saveAccountInProgress = false;
        this.customToast.error(
          err?.error?.message || `Failed to ${failureSubText} Aws account`,
        );
      },
    );
  }

  verifyAccount() {
    this.verifyInProgress = true;
    const requestPayload = this.getRequestPayload();
    this.accountControllerService
      .validateAwsAccountUsingPOST(requestPayload)
      .subscribe(
        (data) => {
          this.verifyInProgress = false;
          this.customToast.success(`This is a verified account`);
        },
        (error) => {
          this.verifyInProgress = false;
          this.customToast.error(
            error?.error?.message || `Failed to verify account`,
          );
        },
      );
  }
  getUrl() {
    return (
      'https://console.aws.amazon.com/cloudformation/home?region=ap-south-1#/stacks/create/review?stackName=facets-' +
      'cluster-role&param_ExternalId=' +
      this.externalId.value +
      '&param_AccountId=' +
      this.accountId +
      '&templateURL=https://facets-cf-templates.s3.amazonaws.com/facets-othercluster-iam-policy.json'
    );
  }
  getAccountId() {
    this.fetchingAccountId = true;
    this.uiMetaControllerService.getCPAccountUsingGET().subscribe({
      next: (res) => {
        this.accountId = res?.accountId;
        this.fetchingAccountId = false;
      },
      error: (res) => {
        this.accountId = '';
        this.fetchingAccountId = false;
      },
    });
  }
}
