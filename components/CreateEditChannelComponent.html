import { Component, OnDestroy } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { NbTagComponent } from '@nebular/theme';
import { distinctUntilChanged } from 'rxjs';
import { TestNotificationRequest, User } from 'src/app/cc-api/models';
import { UiNotificationControllerService } from 'src/app/cc-api/services/ui-notification-controller.service';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import { UiUserControllerService } from 'src/app/cc-api/services';
import { EMAIL_REGEX, CHANNEL_URL_REGEX } from 'src/app/constants/regex';
import { CHANNEL_WRITE } from './../../../constants/permissions';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';

@Component({
  selector: 'app-create-edit-channel',
  templateUrl: './create-edit-channel.component.html',
  styleUrls: ['./create-edit-channel.component.scss'],
})
export class CreateEditChannelComponent
  extends FacetsPage
  implements OnDestroy
{
  headingOptions = {
    create: 'Create Channel',
    edit: 'Edit Channel',
  };
  mode: 'create' | 'edit' = 'create';
  //Loaders
  loadingChannelForm: boolean = true;
  channelActionInProgress: boolean = false;
  testNotificationInprogress: boolean;
  loadingUser: boolean;

  channelOptions = [];
  facetsUserList: User[] = [];
  usersSelectOptions: {
    label: string;
    value: string;
    icon: string;
    checked?: boolean;
  }[];
  facetsEmailList: string[] = [];
  customEmails: Set<string> = new Set([]);
  customEmailError: string;
  //Edit mode vars
  channelEditID: any;
  editChannelDetails;
  //Channel form
  channelForm: FormGroup = this.fb.group({
    name: new FormControl('', [
      Validators.required,
      Validators.pattern(/[\S]/),
      Validators.minLength(4),
      Validators.maxLength(20),
    ]),
    channelType: new FormControl('', [Validators.required]),
    integrationKey: new FormControl(''),
    channelAddress: new FormControl(''),
    authorizationHeader: new FormControl(''),
    facetsUsers: new FormControl([]),
    otherUsers: new FormControl(''),
  });
  dynamicValidationSubscription: any;
  channelWrite = CHANNEL_WRITE;

  constructor(
    private uiNotificationControllerService: UiNotificationControllerService,
    private router: Router,
    private _route: ActivatedRoute,
    private customToast: CustomToastService,
    private fb: FormBuilder,
    private userController: UiUserControllerService,
  ) {
    super(_route);
  }

  OnPageLoad(params: Params) {
    this.loadingChannelForm = true;
    this.mode = params?.mode;
    this.channelEditID = params?.channelID;

    this.asyncOnPageLoad();
  }

  async asyncOnPageLoad() {
    if (this.mode === 'create') {
      await this.getAllChannelTypes();
      this.loadingChannelForm = false;
    } else if (this.mode === 'edit') {
      Promise.all([
        this.getAllChannelTypes(),
        this.getChannelDetailsById(this.channelEditID),
      ]).then(async (promiseData) => {
        this.editChannelDetails = promiseData[1];

        //If channelType is email then getAllUsers
        if (this.editChannelDetails.channelType === 'EMAIL') {
          await this.getAllUsers();
        }

        const [facetsUsers, otherUsers] = this.sortUsers(
          this.editChannelDetails.emailAddresses || [],
          this.facetsEmailList || [],
        );
        this.customEmails = new Set(otherUsers);

        //Patch channelForm
        this.channelForm.patchValue({
          name: this.editChannelDetails.name,
          channelAddress: this.editChannelDetails.channelAddress,
          channelType: this.editChannelDetails.channelType,
          integrationKey: this.editChannelDetails.integrationKey,
          emailAddresses: this.editChannelDetails.emailAddresses,
          facetsUsers: facetsUsers,
          otherUsers: '',
          authorizationHeader: this.editChannelDetails?.authorizationHeader,
        });

        this.loadingChannelForm = false;
      });
    }
    this.dynamicFormValidation();
    //We need to get all channel types, then get channel details, then if needed get all users.
  }

  //Getters
  get name() {
    return this.channelForm.get('name');
  }

  get channelType() {
    return this.channelForm.get('channelType');
  }

  get channelAddress() {
    return this.channelForm.get('channelAddress');
  }

  get integrationKey() {
    return this.channelForm.get('integrationKey');
  }

  get facetsUsers() {
    return this.channelForm.get('facetsUsers');
  }

  get otherUsers() {
    return this.channelForm.get('otherUsers');
  }

  get authorizationHeader() {
    return this.channelForm?.get('authorizationHeader');
  }

  testNotification() {
    this.testNotificationInprogress = true;
    let payload: TestNotificationRequest = {
      channelAddress: this.channelAddress.value,
      channelType: this.channelType.value,
      integrationKey: this.integrationKey.value,
      emailAddresses: [
        ...(this.facetsUsers.value || []),
        ...(this.customEmails || []),
      ],
      notificationType: 'TEST',
      payload: null,
    };

    this.uiNotificationControllerService
      .testNotificationChannelUsingPOSTResponse(payload)
      .subscribe(
        (response) => {
          this.testNotificationInprogress = false;
          this.customToast.success(
            `A test notification has been successfully delivered to the specified ${
              this.channelType.value === 'EMAIL' ? 'email IDs' : 'channel URL'
            }.`,
          );
        },
        (error) => {
          this.testNotificationInprogress = false;
          this.customToast.error(
            error?.error?.message ||
              `Test Notification sending failed. Verify the ${
                this.channelType.value === 'EMAIL' ? 'email IDs' : 'channel URL'
              } and try again.`,
          );
        },
      );
  }

  addNotificationChannel() {
    this.channelActionInProgress = true;
    let channelPayload = {
      name: this.name.value,
      channelType: this.channelType.value,
      channelAddress:
        this.channelType.value === 'EMAIL' ? '' : this.channelAddress.value,
      authorizationHeader:
        this.channelType?.value === 'WEBHOOK'
          ? this.authorizationHeader?.value?.trim()
          : undefined,
      integrationKey:
        this.channelType.value === 'PAGER_DUTY'
          ? this.integrationKey.value
          : '',
      emailAddresses: [...this.facetsUsers.value, ...this.customEmails],
    };

    this.uiNotificationControllerService
      .createNotificationChannelUsingPOST(channelPayload)
      .subscribe(
        (response) => {
          this.channelActionInProgress = false;
          this.customToast?.success(
            `Channel ${channelPayload?.name} created successfully`,
          );
          this.router.navigate(['/settings/notification-center']);
        },
        (error) => {
          this.channelActionInProgress = false;
          this.customToast.error(
            error?.error?.message ||
              `Channel ${channelPayload?.name} creation failed`,
          );
        },
      );
  }

  editNotificationChannel() {
    this.channelActionInProgress = true;
    let channelPayload = {
      name: this.name.value,
      channelType: this.channelType.value,
      id: this.editChannelDetails.id,
      channelAddress:
        this.channelType.value === 'EMAIL' ? '' : this.channelAddress.value,
      authorizationHeader:
        this.channelType?.value === 'WEBHOOK'
          ? this.authorizationHeader?.value?.trim()
          : undefined,
      integrationKey:
        this.channelType.value === 'PAGER_DUTY'
          ? this.integrationKey.value
          : '',
      emailAddresses:
        this.channelType.value === 'EMAIL'
          ? [...(this.facetsUsers.value || []), ...(this.customEmails || [])]
          : undefined,
    };

    this.uiNotificationControllerService
      .editNotificationChannelUsingPUT({
        nc: channelPayload,
        channelId: this.channelEditID,
      })
      .subscribe(
        (response) => {
          this.channelActionInProgress = false;
          this.customToast?.success(
            `Successfully updated channel ${channelPayload?.name}`,
          );
          this.router.navigate(['/settings/notification-center']);
        },
        (error) => {
          this.customToast?.error(
            `Failed to update the channel ${channelPayload?.name}`,
          );
          this.channelActionInProgress = false;
        },
      );
  }

  getAllChannelTypes() {
    return new Promise((resolve, reject) => {
      this.uiNotificationControllerService
        .getAllChannelTypesUsingGET()
        .subscribe(
          (res) => {
            this.channelOptions = res.map((channel) => {
              return { label: channel?.label, value: channel?.type };
            });
            resolve(this.channelOptions);
          },
          (error) => {
            reject(error);
          },
        );
    });
  }

  getChannelDetailsById(channelID) {
    return new Promise((resolve, reject) => {
      this.uiNotificationControllerService
        .getChannelUsingGET(channelID)
        .subscribe(
          (res) => {
            resolve(res);
          },
          (error) => {
            reject(error);
          },
        );
    });
  }

  getAllUsers() {
    return new Promise((resolve, reject) => {
      this.loadingUser = true;
      this.userController.getAllUsersUsingGETResponse().subscribe(
        (users) => {
          this.facetsUserList = users.body;
          let editEmailAddresses = [];
          if (this.editChannelDetails) {
            editEmailAddresses = this.editChannelDetails?.emailAddresses;
          }
          this.usersSelectOptions = this.facetsUserList.map((user) => {
            return {
              label: user?.userName,
              value: user?.userName,
              icon: user?.picture,
              checked: editEmailAddresses?.includes(user?.userName),
            };
          });
          this.facetsEmailList = this.facetsUserList.map(
            (user) => user?.userName,
          );
          this.loadingUser = false;
          resolve(true);
        },
        (error) => {
          this.facetsUserList = [];
          this.facetsEmailList = [];
          this.loadingUser = false;
          reject(error);
        },
      );
    });
  }

  sortUsers(allEmailAddress = [], facetsEmailAddress = []) {
    if (allEmailAddress?.length > 0) {
      const facetsUsers = allEmailAddress?.filter((email) =>
        facetsEmailAddress.includes(email),
      );
      const otherUsers = allEmailAddress?.filter(
        (email) => !facetsEmailAddress.includes(email),
      );
      return [facetsUsers, otherUsers];
    }
    return [];
  }

  changeUserSelection(event) {
    if (!event?.length) return;
    const selectedUsersList = event.map((user) => user.value);
    this.facetsUsers.setValue(selectedUsersList);
  }

  onChannelTypeChange(event) {
    if (event === 'EMAIL' && this.facetsUserList?.length <= 0) {
      this.getAllUsers();
    }
  }

  onFacetsUserRemove(tagToRemove: NbTagComponent): void {
    let clonedFacetsUsers = new Set(this.facetsUsers.value);
    clonedFacetsUsers.delete(tagToRemove.text);
    this.channelForm.patchValue({
      facetsUsers: Array.from(clonedFacetsUsers),
    });
  }

  onCustomEmailRemove(tagToRemove: NbTagComponent): void {
    this.customEmails.delete(tagToRemove.text);
  }

  isValidEmail(email: string): boolean {
    return EMAIL_REGEX.test(email);
  }

  handleKeyDown(event) {
    let value = this.otherUsers.value;
    if (event.key === 'Enter' || value?.endsWith('.com')) {
      if (this.isValidEmail(value)) {
        this.customEmailError = '';
        this.customEmails.add(value);
        this.otherUsers.setValue('');
      } else {
        this.customEmailError = 'Enter a valid email address';
      }
    } else {
      this.customEmailError = '';
    }
  }

  dynamicFormValidation() {
    this.dynamicValidationSubscription = this.channelForm
      .get('channelType')
      .valueChanges.pipe(distinctUntilChanged())
      .subscribe((value) => {
        if (value === 'PAGER_DUTY') {
          this.channelForm
            ?.get('integrationKey')
            ?.setValidators([Validators.required, Validators.pattern(/[\S]/)]);
          this.channelForm
            ?.get('channelAddress')
            ?.setValidators([
              Validators.required,
              Validators.pattern(CHANNEL_URL_REGEX),
              Validators.pattern(/[\S]/),
            ]);
        } else if (value === 'EMAIL') {
          this.channelForm?.get('integrationKey')?.clearValidators();
          this.channelForm?.get('channelAddress')?.clearValidators();
        } else {
          this.channelForm?.get('integrationKey')?.clearValidators();
          this.channelForm
            ?.get('channelAddress')
            ?.setValidators([
              Validators.required,
              Validators.pattern(CHANNEL_URL_REGEX),
              Validators.pattern(/[\S]/),
            ]);
        }
        this.channelForm?.get('integrationKey')?.updateValueAndValidity();
        this.channelForm?.get('channelAddress')?.updateValueAndValidity();
      });
  }

  handleBack = () => {
    this.router.navigate(['/settings/notification-center']);
  };

  ngOnDestroy() {
    this.dynamicValidationSubscription.unsubscribe();
  }
}
