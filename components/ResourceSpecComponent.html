import { Component, ViewChild } from '@angular/core';
import mixpanel from 'mixpanel-browser';
import { ResourceDetailsService } from '../resource-details/resource-details.service';
import { MonacoEditorConstructionOptions } from '@materia-ui/ngx-monaco-editor';
import { NbDialogService } from '@nebular/theme';
import { UiBlueprintDesignerControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params } from '@angular/router';
import { IntentDetailsService } from 'src/app/services/intent-details.service';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
import {
  BLUEPRINT_ADD_RESOURCE_FAIL,
  BLUEPRINT_EDIT_RESOURCE_SPEC_SUCCESS,
} from 'src/app/constants/analytics-track-events';

@Component({
  selector: 'app-resource-spec',
  templateUrl: './resource-spec.component.html',
  styleUrls: ['./resource-spec.component.scss'],
})
export class ResourceSpecComponent extends FacetsPage {
  blueprint: string;
  branchName: string;
  resource: any;
  schemaLink: string;
  codeChanged: boolean = false;

  resource_edit: any;
  editorOption_READ: MonacoEditorConstructionOptions = {
    theme: 'vs-light',
    language: 'json',
    readOnly: true,
    minimap: {
      enabled: true,
    },
  };
  editorOptions_WRITE: MonacoEditorConstructionOptions = {
    theme: 'vs-light',
    language: 'json',
    readOnly: false,
    minimap: {
      enabled: true,
    },
  };

  monaco: any;
  completionProvider: any;
  resourceName: string;
  resourceType: string;
  documentationURL: string = '';

  constructor(
    private uiBlueprintDesignerControllerService: UiBlueprintDesignerControllerService,
    private resourceDetailsService: ResourceDetailsService,
    private dialogService: NbDialogService,
    activatedRoute: ActivatedRoute,
    private intentDetailsService: IntentDetailsService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
    try {
      this.resource = this.resourceDetailsService.getResource();
      this.resource = JSON.parse(JSON.stringify(this.resource));
      this.schemaLink = JSON.parse(this.resource?.content)['$schema'];
    } catch (error) {}
  }

  OnPageLoad(p: Params): void {
    this.resourceName = p?.resourceName;
    this.resourceType = p?.resourceType;
    this.blueprint = p?.stackName;
    this.documentationURL = '';
    this.branchName = p?.branchName;
    this.setEditCodeContent();
    if (!this.intentDetailsService?.intentLoaded) {
      this.intentDetailsService?.updateIntentDataSubject.subscribe(
        (intentStatus) => {
          if (intentStatus) {
            this.getDocumentationUrl();
          }
        },
      );
    } else {
      this.getDocumentationUrl();
    }
  }

  getDocumentationUrl() {
    this.documentationURL = this.intentDetailsService?.getIntentDetails(
      this.resourceType,
    )?.documentation;
  }

  openDocument() {
    window.open(this.documentationURL, '_blank');
  }
  refresh: boolean = true;

  setEditCodeContent() {
    this.resource_edit = JSON.parse(JSON.stringify(this.resource));
    {
      this.isCodeValid = true;
      this.resource = JSON.parse(
        JSON.stringify(this.resourceDetailsService.getResource()),
      );
      this.refresh = false;
      setTimeout(() => {
        this.refresh = true;
      }, 10);
    }
  }

  @ViewChild('saveConfigurationDialog') saveConfigurationDialog;
  confirmSaveConfiguration() {
    this.dialogService.open(this.saveConfigurationDialog, {
      hasBackdrop: true,
      closeOnBackdropClick: false,
      context: {},
    });
  }

  updatingConfig: boolean = false;
  saveConfiguration(ref) {
    const payLoad = {
      content: this.resource_edit?.content,
      resourceName: this.resource.resourceName,
      resourceType: this.resource.resourceType,
      filename: this.resource.filename,
      directory: this.resource.directory,
    };
    this.updatingConfig = true;
    this.uiBlueprintDesignerControllerService
      .updateResourcesUsingPUT({
        branch: this.branchName,
        stackName: this.blueprint,
        resourceFileRequests: [payLoad],
      })
      .subscribe(
        (rsp) => {
          mixpanel.track(BLUEPRINT_EDIT_RESOURCE_SPEC_SUCCESS);
          this.customToastService.success(
            `Configuration saved for ${this.resource.resourceName}`,
          );
          this.refreshCodeContent();
        },
        (error) => {
          mixpanel.track(BLUEPRINT_ADD_RESOURCE_FAIL, { error });
          this.onAPIErrorService.onAPIError(
            error?.error?.message ||
              `Failed to update config for ${this.resource.resourceName}`,
            error,
          );
        },
      )
      .add(() => {
        this.updatingConfig = false;
        ref.close();
      });
  }

  refreshCodeContent() {
    this.isCodeValid = true;
    this.codeChanged = false;
    this.resourceDetailsService
      .fetchResource(
        this.blueprint,
        `${this.resourceType}:${this.resourceName}`,
        this.branchName,
      )
      .then((data) => {
        this.setEditCodeContent();
      });
  }

  closeChangesModal(ref) {
    if (ref) {
      ref.close();
    }
    this.setEditCodeContent();
  }

  isCodeValid: boolean;
  byValidityChange(event) {
    if (!event) {
      this.isCodeValid = false;
    } else {
      this.isCodeValid = true;
      this.resource_edit.content = event;
    }
  }

  byCodeChange(code) {
    this.codeChanged = this.resource?.content !== code;
  }
}
