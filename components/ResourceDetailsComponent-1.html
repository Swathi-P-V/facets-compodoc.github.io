import { Component, TemplateRef, ViewChild, OnDestroy } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import cloneDeep from 'lodash/cloneDeep';
import { Subscription } from 'rxjs';
import { map } from 'rxjs';
import { AbstractCluster } from 'src/app/cc-api/models';
import {
  UiCommonClusterControllerService,
  UiDeploymentControllerService,
  UiDropdownsControllerService,
} from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  allTabs,
  clusterStatesToEnableTab,
} from './resource-details.constants';
import * as iconMapping from 'src/app/constants/facetsIconMap.json';
import {
  hotfixResources,
  whiteListedResourcesForIngressDetails,
} from 'src/app/constants/resourceTypes';
import {
  getActionButtons,
  getResourceHeaderIcons,
} from 'src/app/utils/resource-details/header-utils';
import { NbDialogService } from '@nebular/theme';
import { ResourceDetailsService } from '../resource-details.service';
import { ResourceHistoryService } from 'src/app/services/resource-history.service';
import { HttpClient } from '@angular/common/http';
import { UiApplicationControllerService } from 'src/app/cc-api/services';
import { ActionEventEmitterService } from 'src/app/services/action-event-emitter.service';
import { BlueprintFile } from './../../../cc-api/models/blueprint-file';
import { actionEvents } from '../../../constants/action-events';
import { ContextUpdatesService } from 'src/app/services/context-updates.service';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-resource-details',
  templateUrl: './resource-details.component.html',
  styleUrls: ['./resource-details.component.scss'],
})
export class ResourceDetailsComponent extends FacetsPage implements OnDestroy {
  cluster: AbstractCluster = {};
  cloud: 'AWS' | 'AZURE' | 'LOCAL' | 'GCP' | 'KUBERNETES';
  resourceInfo: BlueprintFile;
  resourceContent: any;
  allowedFlavoursForIngress = {
    AWS: ['nlb_nginx', 'nginx_ingress_controller', 'aws_alb'],
    GCP: ['nlb_nginx', 'nginx_ingress_controller', 'gcp_alb'],
    AZURE: ['nlb_nginx', 'nginx_ingress_controller', 'azure_alb'],
  };

  finalTabs = [];
  tabsOtherResources = [];
  resourceType: string;
  resourceName: string;
  stackName: any;
  clusterId: any;
  redirectTab: string;
  routerSubscription: Subscription;
  overrideCompleteSub: Subscription;
  redirectionMap = {
    'Live release': 'live-releases',
    Configuration: 'configuration',
    Diff: 'diff',
    Domains: 'ingress-domains',
    'Grafana dashboard': 'dashboard',
    Overview: 'overview',
  };
  supportedResourceTypes: string[] = [...hotfixResources];

  headingIcons = [];
  resourceTypeIcon = {
    id: 'resource-type',
    icon: '',
    pack: 'fci',
    status: 'primary',
    iconTooltip: '',
    iconSize: '36px',
  };
  clusterState: any = '';
  headerActionButtons: any = [];
  resourceStatusProps = '';
  historyResourceStatus = '';
  pageHeadingTextAsLink = {
    link: '',
    openInNewTab: true,
    tooltip: 'View Resource in Blueprint Designer',
  };

  pageHeaderDescription: string;
  enableDisableLoader: boolean = false;
  pageLoader: boolean = false;
  isBlueGreenInPreview: boolean = false;
  promoteLoader: boolean = false;
  showHighlighterForMoreActions: boolean = false;

  private rootUrl: string = window.location.origin;
  private httpSubscriber;
  private podsSubscriber;
  private isResourceQueuedObserver;

  @ViewChild('enableDisableConfirmation')
  enableDisableConfirmation: TemplateRef<any>;

  @ViewChild('blueGreenPromoteConfirmation')
  blueGreenPromoteConfirmation: TemplateRef<any>;

  constructor(
    private _route: ActivatedRoute,
    private router: Router,
    private uiCommonClusterService: UiCommonClusterControllerService,
    private dropdownControllerService: UiDropdownsControllerService,
    private uiDeploymentControllerService: UiDeploymentControllerService,
    private dialogService: NbDialogService,
    private resourceDetailsService: ResourceDetailsService,
    private resourceHistoryService: ResourceHistoryService,
    private httpClient: HttpClient,
    private uiApplicationControllerService: UiApplicationControllerService,
    private actionEventEmitterService: ActionEventEmitterService,
    private contextUpdatesService: ContextUpdatesService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(_route);
    this.initializeSubscribers();
  }

  initializeSubscribers() {
    this.podsSubscriber = this.resourceDetailsService.podsObserver.subscribe(
      ({ isBlueGreenInPreview }) => {
        this.updatePromoteButtonStatus({ disable: !isBlueGreenInPreview });
      },
    );

    this.isResourceQueuedObserver =
      this.resourceDetailsService.resourceHistoryObserver.subscribe(
        ({ isResourceQueued: latestResourceQueued }) => {
          if (this.isResourceQueued !== latestResourceQueued) {
            this.isResourceQueued = latestResourceQueued;
            this.setHeaderButtonsAndIcons();
          }
        },
      );
  }

  OnPageLoad(params: Params): void {
    this.stackName = params?.stackName;
    this.clusterId = params?.clusterId;
    this.resourceType = params?.resourceType;
    this.resourceName = params?.resourceName;
    this.resourceStatusProps = '';
    this.historyResourceStatus = '';
    this.showHighlighterForMoreActions = false;

    this.resourceTypeIcon.icon = iconMapping[this.resourceType]
      ? this.resourceType
      : 'default';
    this.subscribeToOverrideEvent();

    this.setHeaderTextLink();
    this.getTabs();
    this.makeAllAPICalls();
    this.routerSubscription = this._route.paramMap
      .pipe(map(() => window.history.state))
      .subscribe((data) => {
        if (data) {
          if (data.redirectTo) {
            this.redirectTab = this.redirectionMap[data.redirectTo];
            this.router.navigate([this.redirectTab], {
              relativeTo: this._route,
            });
            this.routerSubscription.unsubscribe();
          }
        }
      });
    this._route.queryParams.subscribe((params) => {
      this.previousComponent = params?.previousComponent;
    });
  }
  setHeaderTextLink() {
    this.pageHeadingTextAsLink.link = `/capc/stack/${
      this.stackName
    }/blueprint-designer/${this.contextUpdatesService.getLastBranchForBlueprint(
      this.stackName,
    )}/resource/type/${this.resourceType}/name/${this.resourceName}/overview`;
  }

  makeAllAPICalls() {
    Promise.allSettled([
      this.getEnvInformation(),
      this.getResourceInfo(),
      this.getClusterState(),
      this.getResourceStatus(),
    ]).then(async () => {
      this.setHeaderButtonsAndIcons();
      this.updateTabsStatus();
      this.updateTabsWithIngressTab();
      let isPreview;
      try {
        isPreview = await this.getKubernetesPods();
      } catch (error) {
        isPreview = false;
      }
      this.updatePromoteButtonStatus({ disable: !isPreview });
    });
  }

  isBlueGreenDeployment() {
    return this.resourceInfo?.info?.strategy === 'BlueGreen';
  }

  updateTabsWithIngressTab() {
    /*
    show domains tab when
      * Resource type is ingress and cloud type is AWS
      * Resource type is ingress and cloud type is GCP or AZURE and flavour is present
    */
    if (
      whiteListedResourcesForIngressDetails?.includes(this.resourceType) &&
      !this.allowedFlavoursForIngress[this.cloud]?.includes(
        this.resourceInfo?.info?.flavour,
      )
    ) {
      this.finalTabs = this.finalTabs.filter(
        (tab) => tab?.id !== 'ingress-domains',
      );

      if (this.router.url.includes('/ingress-domains')) {
        this.goToConfigPage();
      }
    }
  }

  getEnvInformation = () => {
    return new Promise((resolve, reject) => {
      this.uiCommonClusterService
        .getClusterCommonUsingGET(this.clusterId)
        .subscribe(
          (data) => {
            this.cluster = data;
            this.cloud = this.cluster?.cloud;
            resolve(true);
          },
          (err) => {
            this.goToConfigPage();
            reject(err);
          },
        );
    });
  };

  isSelectiveReleaseSupported: boolean = false;
  getResourceInfo = () => {
    return new Promise((resolve, reject) => {
      this.dropdownControllerService
        .getResourceByClusterIdUsingGET({
          clusterId: this.clusterId,
          resourceType: this.resourceType,
          resourceName: this.resourceName,
          includeContent: true,
        })
        .subscribe(
          (data) => {
            this.resourceInfo = data;
            try {
              this.resourceContent = JSON.parse(
                this?.resourceInfo?.content || '{}',
              );
            } catch (error) {
              this.resourceContent = {};
            }

            this.isSelectiveReleaseSupported =
              this.resourceInfo.alpha ||
              this.supportedResourceTypes.includes(
                this.resourceInfo?.resourceType,
              );
            this.resourceDetailsService.setResourceInfo(data);
            this.pageHeaderDescription = this.resourceInfo?.info?.flavour
              ? `Environment Resource Details: <b>${this.resourceInfo?.resourceName}</b> ${this.resourceInfo.resourceType} with <b>${this.resourceInfo?.info?.flavour}</b> flavor implementation`
              : 'Detailed information about the resource';
            resolve(true);
          },
          (err) => {
            this.goToConfigPage();
            reject(err);
          },
        );
    });
  };

  getKubernetesPods = () => {
    return new Promise((resolve, reject) => {
      let isBlueGreenInPreview = false;
      const url = `${this.rootUrl}/cc-ui/v1/clusters/${this.clusterId}/k8s-explorer/pods?resourceType=${this.resourceType}&resourceName=${this.resourceName}`;
      this.httpSubscriber = this.httpClient.get(url).subscribe(
        (podsResponse: any) => {
          isBlueGreenInPreview = !!podsResponse?.find(
            (pod) => pod?.role === 'preview',
          );
          resolve(isBlueGreenInPreview);
        },
        (error) => {
          reject(error);
        },
      );
    });
  };

  updatePromoteButtonStatus = ({
    disable: disabledStatus,
    disableWhileLoading,
  }: {
    disable: boolean;
    disableWhileLoading?: boolean;
  }) => {
    const actionButtons = cloneDeep(this.headerActionButtons);
    const promoteButtonIndex = actionButtons?.findIndex(
      (button) => button.id === 'blue-green-promote',
    );

    if (promoteButtonIndex !== -1) {
      const promoteButton = actionButtons[promoteButtonIndex];
      const disablePromoteButton =
        disabledStatus ||
        this.resourceInfo?.info?.disabled ||
        !this.isClusterRunning;

      promoteButton.disabled = disablePromoteButton;

      if (!this.isClusterRunning) {
        promoteButton.nbTooltip =
          'Environment is not currently in running state';
      } else if (this.resourceInfo?.info?.disabled) {
        promoteButton.nbTooltip = 'Resource is in disabled state';
      } else if (disabledStatus && !disableWhileLoading) {
        promoteButton.nbTooltip = 'There are no preview changes to promote';
      }
      actionButtons[promoteButtonIndex] = promoteButton;

      this.showHighlighterForMoreActions = !disablePromoteButton;
      this.headerActionButtons = cloneDeep(actionButtons);
    }
  };

  goToConfigPage() {
    this.router.navigate([
      '/',
      this.stackName,
      'cluster',
      this.clusterId,
      'resources',
      'type',
      this.resourceType,
      'resource',
      this.resourceName,
      'configuration',
    ]);
  }

  previousComponent: string;
  handleBack = () => {
    this.router.navigate(
      this.previousComponent === 'blueprint-designer'
        ? [
            '/',
            'stack',
            this.stackName,
            'blueprint-designer',
            this.contextUpdatesService.getLastBranchForBlueprint(
              this.stackName,
            ),
            'resource',
            'type',
            `${this.resourceType}`,
            'name',
            `${this.resourceName}`,
            'overview',
          ]
        : ['/', this.stackName, 'cluster', this.clusterId, 'resources'],
    );
  };

  getTabs() {
    const tabs = cloneDeep(allTabs)?.map((tab) => ({
      ...tab,
      queryParams: { ...this._route.snapshot.queryParams },
    }));
    const finalTabs = tabs.filter(
      (tab) => tab.global || tab?.allowedFor?.includes(this.resourceType),
    );
    this.finalTabs = finalTabs;
  }

  isClusterRunning: boolean = false;
  getClusterState() {
    return new Promise((resolve, reject) => {
      this.uiDeploymentControllerService
        .getClusterStateUsingGET(this.clusterId)
        .subscribe(
          (response) => {
            this.clusterState = response;
            this.isClusterRunning = this.clusterState === 'RUNNING';
            resolve(true);
          },
          () => {},
        );
    });
  }

  setHeaderButtonsAndIcons() {
    this.headingIcons = getResourceHeaderIcons({
      clusterState: this.clusterState,
      resource: this.resourceInfo,
      type: 'environment',
      isResourceQueued: this.isResourceQueued,
      cluster: this.cluster,
    });

    this.resourceStatusProps = JSON.stringify({
      severity: this.resourceInfo?.info?.disabled ? 'disabled' : 'enabled',
      badgeClass: 'p-2 text-capitalize mb-0',
    });

    this.headerActionButtons = getActionButtons({
      clusterState: this.clusterState,
      resource: this.resourceInfo,
      isResourceQueued: this.isResourceQueued,
      historyResourceStatus: this.historyResourceStatus,
      isReleasePaused: this.cluster?.pauseReleases,
    });
  }

  subscribeToOverrideEvent() {
    this.overrideCompleteSub =
      this.actionEventEmitterService.actionEventEmiter.subscribe(
        (status) => {
          if (status === actionEvents?.ON_SAVE_ENV_OVERRIDES) {
            this.makeAllAPICalls();
          }
        },
        (err) => {
          this.overrideCompleteSub.unsubscribe();
        },
      );
  }

  updateTabsStatus = () => {
    if (this.clusterState !== 'RUNNING' || this.resourceInfo?.info?.disabled) {
      this.finalTabs = cloneDeep(this.finalTabs).map((tab) => ({
        ...tab,
        disabled: tab.enableConditionally
          ? !clusterStatesToEnableTab?.includes(this.clusterState) ||
            this.resourceInfo?.info?.disabled
          : false,
      }));
    }
  };

  handlePageActions(event: string) {
    switch (event) {
      case 'enable-resource':
      case 'disable-resource':
        this.openEnableDisableConfirmDialog(event);
        break;
      case 'release-resource':
        this.redirectToReleaseResource(event);
        break;
      case 'blue-green-promote':
        this.openPromoteConfirmation();
        break;
      case 'view-in-designer':
        this.openInDesigner();
        break;
      default:
        break;
    }
  }

  openInDesigner() {
    this.router.navigate(
      [
        'stack',
        this.stackName,
        'blueprint-designer',
        this.contextUpdatesService.getLastBranchForBlueprint(this.stackName),
        'resource',
        'type',
        `${this.resourceType}`,
        'name',
        `${this.resourceName}`,
        'overview',
      ],
      {
        queryParams: { from: 'resource-centre' },
      },
    );
  }

  redirectToReleaseResource(event) {
    const queryParams = {
      resource: `${this.resourceType}:${this.resourceName}`,
    };
    this.router.navigate(
      [`${this.stackName}/cluster/${this.clusterId}/releases`],
      {
        queryParams: this.isSelectiveReleaseSupported ? queryParams : null,
        relativeTo: null,
      },
    );
  }

  openEnableDisableConfirmDialog(buttonId) {
    this.dialogService.open(this.enableDisableConfirmation, {
      context: {
        actionText: buttonId === 'enable-resource' ? 'enable' : 'disable',
        resourceName: this.resourceName,
      },
    });
  }

  enableDisableResource(ref, actionText) {
    this.resourceStatusProps = '';
    this.enableDisableLoader = true;
    this.uiCommonClusterService
      .bulkEnableDisableResourcesUsingPUT({
        clusterId: this.clusterId,
        resourceEnableDisableRequestList: [
          {
            resourceName: this.resourceName,
            resourceType: this.resourceType,
            disabled: !this.resourceInfo?.info?.disabled,
          },
        ],
      })
      .subscribe(
        async () => {
          ref?.close();
          this.enableDisableLoader = false;
          this.pageLoader = true;
          this.customToastService.success(
            `Resource has been ${actionText}d successfully`,
          );
          try {
            await this.getResourceInfo();
            this.refreshIconsAndButtons();
            this.pageLoader = false;
          } catch (err) {
            this.pageLoader = false;
          }
        },
        (error) => {
          this.enableDisableLoader = false;
          this.onAPIErrorService.onAPIError(
            error?.error?.message || `Failed to ${actionText}d resource`,
            error,
          );
        },
      );
  }

  openPromoteConfirmation() {
    this.dialogService.open(this.blueGreenPromoteConfirmation, {
      context: {},
    });
  }

  promoteBlueGreenPreview(ref) {
    this.promoteLoader = true;
    const promoteButton =
      this.headerActionButtons?.find(
        (button) => button.id === 'blue-green-promote',
      ) || {};
    this.updatePromoteButtonStatus({
      disable: true,
      disableWhileLoading: true,
    });
    this.uiApplicationControllerService
      .promoteUsingPUT({
        clusterId: this.clusterId,
        appName: this.resourceName,
      })
      .subscribe(
        async () => {
          ref.close();
          this.promoteLoader = false;
          this.pageLoader = true;
          this.customToastService.success(
            'Promotion is successful, please wait for few secs to reflect or hit the refresh button to see the updated status.',
          );
          setTimeout(() => {
            this.pageLoader = false;
          }, 0);
          this.onPromoteSuccess();
        },
        (error) => {
          this.promoteLoader = false;
          this.pageLoader = false;
          this.onAPIErrorService.onAPIError(
            error?.error?.message ||
              error?.message ||
              'Failed to promote the preview changes.',
            error,
          );
          this.updatePromoteButtonStatus({
            disable: promoteButton?.disabled,
          });
        },
      );
  }

  async onPromoteSuccess() {
    try {
      const isPreview = await this.getKubernetesPods();
      this.updatePromoteButtonStatus({ disable: !isPreview });
      this.pageLoader = false;
    } catch (error) {
      this.pageLoader = false;
      this.updatePromoteButtonStatus({ disable: true });
    }
  }

  async refreshIconsAndButtons() {
    const copiedHeaderActionButtons = cloneDeep(this.headerActionButtons);
    this.setHeaderButtonsAndIcons();

    if (this.isBlueGreenDeployment()) {
      const promotionButton = copiedHeaderActionButtons?.find(
        (button) => button?.id === 'blue-green-promote',
      );
      this.updatePromoteButtonStatus({ disable: promotionButton?.disabled });
    }
  }

  isResourceQueued: boolean;
  resourceStatus: string;
  async getResourceStatus() {
    this.isResourceQueued = false;
    const apiResponse =
      await this.resourceHistoryService.getResourceHistoryDataToRender({
        requestParams: {
          clusterId: this.clusterId,
          resourceType: this.resourceType,
          resourceName: this.resourceName,
          blueprintName: this.stackName,
        },
        includeBlueprintHistory: true,
      });

    const {
      artifactProps,
      overrideProps,
      blueprintProps,
      historyResourceStatus,
    } = cloneDeep(apiResponse || ({} as any));
    this.historyResourceStatus = historyResourceStatus;

    this.isResourceQueued = this.resourceDetailsService.isResourceQueued({
      artifactHistory: artifactProps?.historyData?.versions,
      overridesHistory: overrideProps?.historyData?.versions,
      blueprintHistory: blueprintProps?.historyData?.versions,
    });
  }

  ngOnDestroy() {
    this.podsSubscriber?.unsubscribe();
    this.httpSubscriber?.unsubscribe();
    this.overrideCompleteSub?.unsubscribe();
    this.isResourceQueuedObserver?.unsubscribe();
  }
}
