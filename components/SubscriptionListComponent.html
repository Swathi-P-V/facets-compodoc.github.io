import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import { NotificationChannel, Subscription } from 'src/app/cc-api/models';
import {
  UiNotificationControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { SUBSCRIPTION_WRITE } from 'src/app/constants/permissions';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { convertToSentenceCase } from '../../../utils/string-util';
import { SmartTableRouterlinkComponent } from 'src/app/components/smart-table-routerlink/smart-table-routerlink.component';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

const emptyStateSubs = './assets/subs-empty.svg';
const errorIcon = './assets/error.svg';
@Component({
  selector: 'app-subscription-list',
  templateUrl: './subscription-list.component.html',
  styleUrls: ['./subscription-list.component.scss'],
})
export class SubscriptionListComponent extends FacetsPage {
  channels: Array<NotificationChannel>;
  subscriptions: Array<Subscription>;
  nc: NotificationChannel = {};
  sub: Subscription = {};
  notificationError: any = '';
  channelMap = {};
  testingResult: string;
  emptyLogoSubs = emptyStateSubs;
  errorLogo = errorIcon;
  primarySub: string = 'No subscriptions created!';
  secondarySub: string =
    'Create subscriptions for the channels created for notifications';
  showEmptySub: boolean = false;
  statusSub: 'info' | 'error' = 'info';
  showBtnSub: boolean = true;
  buttonTextSub: string = 'Add Subscription';
  tooltipSub: string = 'Add Subscription';
  subscriptionWritePermission = SUBSCRIPTION_WRITE;
  convertToSentenceCase = convertToSentenceCase;
  notificationTypeNameMap: {} = {};

  filterLabels = {
    CLUSTER_NAME: 'Environment Name',
    CLUSTER_TYPE: 'Release Stream',
    SEVERITY: 'Severity',
    ALERT_NAME: 'Alert Name',
    QASUITE_RESULT: 'QA Suite Result',
    DR_ACTION: 'DR Action',
    DR_STATUS: 'DR Status',
    DEPLOYMENT_STATUS: 'Release Status',
  };

  actionButtons = [
    {
      id: 'filter',
      icon: 'filter-outline',
      isFciPack: true,
      toolTip: 'Available filters',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      onClick: (data) => {
        this.getFilters(data?.filters, data?.stackName);
      },
    },
    {
      id: 'edit',
      icon: 'edit-outline',
      isFciPack: true,
      toolTip: 'Edit Subscription',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: 'SUBSCRIPTION_WRITE',
      onClick: (data) => {
        this.handleEditSubscription(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-outline',
      isFciPack: true,
      toolTip: 'Delete Subscription',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: 'SUBSCRIPTION_DELETE',
      onClick: (data) => {
        this.handleDeleteSubscription(data);
      },
    },
  ];

  subscriptionSettings: any = {
    columns: {
      name: {
        title: 'Subscription Name',
        filter: true,
      },
      notificationName: {
        title: 'Notification Type',
        filter: true,
      },
      channelName: {
        title: 'Channel Name',
        valuePrepareFunction: (value, row, cell) => {
          if (!value) {
            return 'Legacy Integration';
          }
          return value;
        },
      },
      stackName: {
        title: 'Blueprint',
        filter: true,
        valuePrepareFunction: (value, row, cell) => {
          // DATA FROM HERE GOES TO renderComponent
          const displayVal = {
            url: ['/', 'stack', value],
            display: value,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: SmartTableRouterlinkComponent,
        type: 'custom',
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          let displayVal;
          if (
            Object.keys(row?.filters).length === 0 ||
            (Object.keys(row?.filters).length === 1 &&
              row?.filters?.hasOwnProperty('STACK_NAME'))
          ) {
            displayVal = {
              smartTableCustomActionButtons: this.actionButtons.filter(
                (button) => button.id !== 'filter',
              ),
            };
          } else {
            displayVal = {
              smartTableCustomActionButtons: this.actionButtons,
            };
          }
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value

          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.actionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '12%',
        filter: false,
      },
    },
    pager: {
      display: false,
    },
    actions: false,
  };

  @ViewChild('deleteSubscriptionConfirmUI', { static: true })
  private deleteSubscriptionConfirmUI: TemplateRef<any>;

  @ViewChild('subscriptionFiltersPopup', { static: true })
  private subscriptionFiltersPopup: TemplateRef<any>;

  deleteInProgress: boolean;
  blueprintEnvironments: {};
  loading: boolean;

  constructor(
    private activatedRoute: ActivatedRoute,
    private uiNotificationControllerService: UiNotificationControllerService,
    private dialogService: NbDialogService,
    private router: Router,
    private stackControllerService: UiStackControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.loading = true;
    this.getAllChannels().then(
      (data) => {
        this.getAllSubscriptions().then(
          (data) => {
            this.loading = false;
          },
          (error) => {
            this.loading = false;
            this.showEmptySub = true;
            this.setErrorVars(error);
          },
        );
      },
      (error) => {
        this.loading = false;
        this.showEmptySub = true;
        this.setErrorVars(error);
      },
    );
  }

  getAllChannels() {
    return new Promise((resolve, reject) => {
      this.uiNotificationControllerService.getAllChannelsUsingGET().subscribe(
        (channels) => {
          this.populateChannels(channels);
          resolve(true);
        },
        (error) => {
          reject(error);
        },
      );
    });
  }

  getAllSubscriptions() {
    return new Promise((resolve, reject) => {
      this.uiNotificationControllerService
        .getAllSubscriptionsUsingGET()
        .subscribe(
          async (susbcriptions) => {
            if (susbcriptions?.length <= 0) {
              this.showEmptySub = true;
            } else {
              await this.getEnvironmentsMap();
              await this.getAllNotificationTypes();
              this.populateSubscriptions(susbcriptions);
            }
            resolve(true);
          },
          (error) => {
            reject(error);
          },
        );
    });
  }

  //Fetch all the Notification Types available
  getAllNotificationTypes = () => {
    return new Promise((resolve, reject) => {
      this.uiNotificationControllerService
        .getAllNotificationTypesUsingGET()
        .subscribe(
          (res) => {
            res?.forEach((notifType) => {
              this.notificationTypeNameMap[notifType?.notificationType] =
                notifType?.notificationName;
            });
            resolve(true);
          },
          (error) => {
            reject(error);
          },
        );
    });
  };

  getEnvironmentsMap() {
    return new Promise((resolve, reject) => {
      this.stackControllerService
        .getAllClustersUsingGET({
          page: 0,
          perPage: 20000,
          sortBy: 'ID',
        })
        .subscribe(
          (envs) => {
            this.blueprintEnvironments = {};
            envs.content.forEach((cluster) => {
              const stackName = cluster?.stackName;
              if (this.blueprintEnvironments.hasOwnProperty(stackName)) {
                this.blueprintEnvironments[stackName][cluster.name] =
                  cluster.id;
              } else {
                this.blueprintEnvironments[stackName] = {
                  [cluster.name]: cluster.id,
                };
              }
            });
            resolve(true);
          },
          (error) => {
            reject(error);
          },
        );
    });
  }

  setErrorVars(error: any) {
    this.primarySub = '';
    this.secondarySub =
      error?.error?.message ||
      ' There is some error fetching subscription information';
    this.showBtnSub = false;
    this.statusSub = 'error';
    this.emptyLogoSubs = this.errorLogo;
  }

  populateChannels(channels: Array<NotificationChannel>) {
    this.channels = channels;
    this.channels.forEach((channel) => {
      this.channelMap[channel.id] = channel.name;
    });
  }

  populateSubscriptions(res: Array<Subscription>) {
    this.subscriptions = res;
    this.subscriptions.forEach((subscription) => {
      subscription['name'] = subscription['name'] || 'N/A';
      subscription['notificationName'] =
        this.notificationTypeNameMap[subscription?.notificationType] || 'N/A';
      if (subscription.channelId) {
        subscription['channelName'] = this.channelMap[subscription.channelId];
      }
    });
  }

  handleCreateSubscription() {
    this.router.navigate(['../subscription/create'], {
      relativeTo: this.activatedRoute,
    });
  }

  handleEditSubscription(rowData: any) {
    this.router.navigate(['../subscription/edit/' + rowData.id], {
      relativeTo: this.activatedRoute,
    });
  }

  handleDeleteSubscription(rowData: any) {
    this.dialogService.open(this.deleteSubscriptionConfirmUI, {
      context: rowData,
    });
  }

  deleteSubscription(data, ref) {
    this.deleteInProgress = true;
    this.uiNotificationControllerService
      .deleteSubscriptionUsingDELETE(data.id)
      .subscribe(
        (response) => {
          this.populateSubscriptions(response);
          this.customToastService.success('Successfully deleted subscription');
          this.deleteInProgress = false;
          ref.close();
        },
        (error) => {
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Failed to delete the subscription',
            error,
          );
          this.deleteInProgress = false;
          ref.close();
        },
      );
  }

  getFilters(filters, stackName) {
    this.dialogService.open(this.subscriptionFiltersPopup, {
      context: this.prepareFiltersData(filters, stackName),
    });
  }

  prepareFiltersData(filters, stackName) {
    let subscriptionFilters = [];

    Object.entries(filters)
      .filter(([key, value]) => key !== 'STACK_NAME')
      .forEach(([key, value]) => {
        let obj = {
          id: key,
          label: this.filterLabels[key] || convertToSentenceCase(key),
          displayType: '',
        };
        if (key === 'CLUSTER_NAME') {
          obj.displayType = 'link';
          obj['value'] = this.getEnvironmentsAsLink(stackName, value);
        } else if (key === 'SEVERITY') {
          obj.displayType = 'badge';
          obj['value'] = value;
        } else {
          obj.displayType = 'text';
          obj['value'] = value;
        }
        subscriptionFilters.push(obj);
      });
    return subscriptionFilters;
  }

  getEnvironmentsAsLink(stackName, environments) {
    let environmentsLink = environments.map((env) => {
      return {
        display: env,
        url: [
          '/',
          stackName,
          'cluster',
          this.blueprintEnvironments[stackName][env],
        ],
        class: 'primary-text',
      };
    });
    return environmentsLink;
  }
}
