import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import {
  AbstractCluster,
  Role,
  RoleMapping,
  UserGroup,
} from 'src/app/cc-api/models';
import {
  UiCustomRoleControllerService,
  UiStackControllerService,
  UiUserGroupControllerService,
  UiResourceGroupControllerService,
  UiAccountsControllerService,
} from 'src/app/cc-api/services';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { SmartTableRouterlinkComponent } from 'src/app/components/smart-table-routerlink/smart-table-routerlink.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { SmartTableCellWithEllipsisComponent } from './../../../components/smart-table-cell-with-ellipsis/smart-table-cell-with-ellipsis.component';
import cloneDeep from 'lodash/cloneDeep';
import { Account } from 'src/app/cc-api/models';
import {
  USER_GROUP_DELETE,
  USER_GROUP_WRITE,
} from 'src/app/constants/permissions';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const noAccessIcon = './assets/alert-empty.svg';

@Component({
  selector: 'app-user-group-table',
  templateUrl: './user-group-table.component.html',
  styleUrls: ['./user-group-table.component.scss'],
})
export class UserGroupTableComponent extends FacetsPage {
  roles: Array<RoleMapping>; //Stores all roles
  allEnvs: AbstractCluster[]; //Stores all the environemnts
  environmentMap: {} = {}; //This stores map of environment id to environment name of all environments.

  userGroups;
  userGroupTableData;
  envSpecificPermissionsTableData: { role: string; clusterNames: any }[];
  loadingGroups: boolean = true;
  roleMap: {} = {};
  deleteInProgress: boolean = false;
  accountNameIdMap = {};

  emptyStateIcon = './assets/user-group-empty-state.svg';
  primaryMsg: string = 'No user groups found';
  secondaryMsg: string = 'Click below to create the first user group!';
  showBtn: boolean = true;
  status: 'info' | 'error' = 'info';
  noAccessIcon = noAccessIcon;
  userGroupWrite = USER_GROUP_WRITE;
  userSubscription: any;
  permissionDialogRef: any;

  resourceGroupIdNameMapping: {} = {};
  showResourceGroups: boolean = false;
  accountsList: Account[] = [];
  loadingGroupRelatedData: boolean = false;

  smartTableCustomActionButtons = [
    {
      id: 'edit',
      icon: 'edit-2-outline',
      toolTip: 'Edit User Group',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: USER_GROUP_WRITE,
      onClick: (data) => {
        this.editUserGroup(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete User Group',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: USER_GROUP_DELETE,
      onClick: (data) => {
        this.delete(data);
      },
    },

  userGroupTableSettings: any = {
    columns: {
      groupName: {
        title: 'User group',
        filter: true,
        type: 'text',
      },
      allRoles: {
        title: 'Roles',
        filter: true,
        type: 'custom',
        valuePrepareFunction: (cell) => {
          return JSON.stringify({ data: cell, decorate: true });
        },
        renderComponent: SmartTableCellWithEllipsisComponent,
      },
      clusterRoleBindings: {
        title: 'Permissions',

            url: null,
            display: 'View',

          instance.byLinkClick.subscribe((data) => {
            this.openViewPermissionsDialog(data);

      custom: {
        type: 'custom',

      display: false,

  envSpecificPermissionTableSettings: any = {
    columns: {
      role: {
        title: 'Role',

        valuePrepareFunction: (cell) => {
          return JSON.stringify({ data: this.roleMap[cell], decorate: true });

        renderComponent: SmartTableCellWithEllipsisComponent,
        onComponentInitFunction: (instance: any) => {
          instance?.byLinkClick?.subscribe((data: any) => {
            this.permissionDialogRef?.close();

        },
      },

    noDataMessage: 'No Permissions created',
    pager: {
      display: false,
      perPage: 10,

    mode: 'external',
    actions: false,

  @ViewChild('viewPermissionsDialog', { static: true })
  private viewPermissionsDialog: TemplateRef<any>;

  @ViewChild('deleteUserGroupConfirmUI', { static: true })
  private deleteUserGroupConfirmUI: TemplateRef<any>;

  constructor(
    private activatedRoute: ActivatedRoute,
    private userGroupControllerService: UiUserGroupControllerService,
    private dialogService: NbDialogService,
    private uiStackControllerService: UiStackControllerService,
    private router: Router,
    private customRoleService: UiCustomRoleControllerService,
    private uiResourceGroupController: UiResourceGroupControllerService,
    private accountsControllerService: UiAccountsControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.getUserGroupsTableData();
  }

  getUserGroupsTableData() {
    this.loadingGroupRelatedData = true;
    Promise.allSettled([
      this.getAllRoles(),
      this.getAllAccounts(),
      this.getAllEnvironments(),
      this.getAllResourceGroups(),
    ]).then((data) => {
      this.getAllUserGroups();
      this.loadingGroupRelatedData = false;
    });
  }

  getAllRoles() {
    return new Promise((resolve, reject) => {
      this.customRoleService.getAllRolesUsingGET().subscribe(
        (roles) => {
          this.roles = roles;
          this.roles.map((role) => {
            this.roleMap[role['roleName']] = role['label'];
          });
          resolve(true);
        },
        (error) => {
          reject(error);
        },
      );
    });
  }

  getAllAccounts() {
    return new Promise((resolve, reject) => {
      this.accountsControllerService.getAllAccountsUsingGET().subscribe(
        (response) => {
          this.accountsList = response;
          this.accountsList.forEach((account) => {
            this.accountNameIdMap[account.id] = account?.name;
          });
          resolve(true);
        },
        (error) => {
          reject(false);
        },
      );
    });
  }

  getAllEnvironments() {
    return new Promise((resolve, reject) => {
      this.uiStackControllerService
        .getAllClustersUsingGET({
          page: 0,
          perPage: 200,
          sortBy: 'ID',
        })
        .subscribe(
          (envs) => {
            let environmentMapTemp = {};
            this.allEnvs = envs.content;
            envs.content.forEach((cluster) => {
              //Create clusterId : clusterName map in a temp object and then assign it to actual variable.
              //This is to avoid exceptions in child component(ie; user-group-table) when trying to access environmentMap.
              environmentMapTemp[cluster?.id] = {
                name: cluster?.name,
                stackName: cluster?.stackName,
                envId: cluster?.id,
              };
            });
            this.environmentMap = environmentMapTemp;
            resolve(true);
          },
          (error) => {
            reject(error);
          },
        );
    });
  }

  getAllResourceGroups() {
    return new Promise((resolve, reject) => {
      this.uiResourceGroupController
        .findAllUsingGET()
        .subscribe(
          (resourceGrpInfo) => {
            resourceGrpInfo.forEach((resource) => {
              this.resourceGroupIdNameMapping[resource.id] = resource.name;
            });
            resolve(true);
          },
          (error) => {
            reject(error);
          },
        )
        .add(() => {
          resolve(true);
        });
    });
  }

  async getAllUserGroups() {
    this.loadingGroups = true;
    this.userGroupControllerService.getAllGroupUsingGET().subscribe(
      (userGroups) => {
        // console.log('User groups are ', userGroups);
        let tempUserGroupsTableData = [];
        userGroups.forEach((userGroup) => {
          this.removeDeletedEnvs(userGroup).then((userGroup: UserGroup) => {
            let group = userGroup;
            let groupDataForTable = {
              groupName: group?.groupName,
              id: group?.id,
              clusterIds: group?.clusterIds,
              stackNames: group?.stackNames,
              clusterNames: group?.clusterIds.map((id) => {
                return {
                  envId: id,
                  name: this.environmentMap[id]?.name,
                  stackName: this.environmentMap[id]?.stackName,
                };
              }),
              accounts: group?.accounts?.map((account) => {
                return {
                  accountId: account,
                  name: this.accountNameIdMap[account],
                };
              }),
              resourceGroupNames: group?.associatedResourceGroups?.map(
                (group) => {
                  return {
                    id: group,
                    name: this.resourceGroupIdNameMapping[group],
                  };
                },
              ),
              allRolesId: [group?.baseRole],
              allRoles: [this.roleMap[group?.baseRole]],
              clusterRoleBindings: group?.clusterRoleBindings,
              baseRole: this.roleMap[group?.baseRole],
            };
            tempUserGroupsTableData.push(groupDataForTable);
          });
        });
        this.userGroupTableData = tempUserGroupsTableData;
        this.loadingGroups = false;
      },
      (error) => {
        console.log(error);
        this.loadingGroups = false;
      },
    );
  }

  openViewPermissionsDialog(rowData) {
    let envSpecificPermissionsTableDataTemp = Object.entries(
      rowData.clusterRoleBindings,
    ).reduce(
      (
        arr: {
          role: string;
          clusterNames: { envId: string; stackName: string; name: string }[];
        }[],
        [clusterId, role],
      ) => {
        arr.filter((ele) => ele.role === role).length > 0
          ? arr.map((ele) => {
              if (ele.role === role) {
                ele.clusterNames.push(this.environmentMap[clusterId]);
              }
            })
          : arr.push({
              role: String(role),
              clusterNames: [this.environmentMap[clusterId]],
            });
        return arr;
      },
      [],
    );

    this.envSpecificPermissionsTableData = envSpecificPermissionsTableDataTemp;
    const commonConfig = {
      decorate: true,
      showAsLink: true,
      dataObject: true,
      defaultDisplay: 1,
      linkUnderline: false,
      noDataMessage: '--',
      removeTagTopSpacing: true,
    };
    rowData.stackEllpisisData = JSON.stringify({
      data: rowData?.stackNames.map((stack) => {
        return {
          name: stack,
          url: `/stack/${stack}`,
        };
      }),
      ...commonConfig,
    });
    rowData.resourceGroupEllipsisData = JSON.stringify({
      data: rowData?.resourceGroupNames,
      ...commonConfig,
      showAsLink: false,
    });

    rowData.clusterEllipsisData = JSON.stringify({
      data: rowData?.clusterNames.map((cluster) => {
        return {
          ...cluster,
          url: `/${cluster?.stackName}/cluster/${cluster?.envId}/overview`,
        };
      }),
      ...commonConfig,
    });

    rowData.accountsEllipsisData = JSON.stringify({
      data: rowData?.accounts,
      ...commonConfig,
    });
    this.permissionDialogRef = this.dialogService.open(
      this.viewPermissionsDialog,
      {
        context: rowData,
      },
    );
  }

  navigateToCreateUserGroup() {
    this.router.navigate(['../../user-group', 'create'], {
      relativeTo: this.activatedRoute,
    });
  }

  editUserGroup(rowdata, ref?) {
    if (ref) {
      ref.close();
    }
    let groupId = rowdata?.id;
    this.router.navigate(['../../user-group', 'edit', groupId], {
      relativeTo: this.activatedRoute,
    });
  }

  delete(rowdata) {
    this.dialogService.open(this.deleteUserGroupConfirmUI, {
      context: rowdata,
    });
  }

  deleteUserGroup(ref, userData) {
    try {
      this.deleteInProgress = true;
      this.userGroupControllerService
        .deleteUserGroupUsingDELETE(userData.id)
        .subscribe(
          (data) => {
            this.deleteInProgress = false;
            ref.close();
            this.customToastService.success('User Group successfully deleted ');
            this.getAllResourceGroups();
          },
          (error) => {
            this.deleteInProgress = false;
            ref.close();
            this.onAPIErrorService.onAPIError(
              'User group delete failed. ' + error?.error?.message,
              error,
            );
          },
        );
    } catch (err) {
      this.deleteInProgress = false;
      this.customToastService.error('User group delete failed');
      ref.close();
    }
  }

  //When any environment is deleted, the environment specific permission or accessible environments list in user group still contains those environments.
  //All such environments must be deleted.
  removeDeletedEnvs(userGroup: UserGroup) {
    return new Promise((resolve, reject) => {
      //Existing Environments list
      let envsIds = Object.keys(this.environmentMap);

      //Remove from clusterRolebindings
      let deletedRoleBindedClusters = Object.keys(
        userGroup.clusterRoleBindings,
      ).filter((key) => envsIds.indexOf(key) === -1);

      deletedRoleBindedClusters.forEach((cluster) => {
        delete userGroup.clusterRoleBindings[cluster];
      });

      //Remove from accessible clusters
      let existingAccessibleClusters = userGroup.clusterIds.filter(
        (cluster) => envsIds.indexOf(cluster) !== -1,
      );

      userGroup.clusterIds = existingAccessibleClusters;
      resolve(userGroup);
    });
  }

  detailsLinkClick(ref) {
    ref?.close();
  }
}
