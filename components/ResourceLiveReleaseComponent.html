import { Component } from '@angular/core';
import { ActivatedRoute, Router, Params } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ClusterState } from '../../../enums/cluster-state';
import { Pod } from '../../../cc-api/models/pod';
import { UiDeploymentControllerService } from '../../../cc-api/services/ui-deployment-controller.service';
import {
  UiApplicationControllerService,
  UiCommonClusterControllerService,
  UiDropdownsControllerService,
} from 'src/app/cc-api/services';
import { HorizontalPodAutoscaler } from 'src/app/cc-api/models';
const emptyStateLogo = './assets/empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-resource-live-release',
  templateUrl: './resource-live-release.component.html',
  styleUrls: ['./resource-live-release.component.scss'],
})
export class ResourceLiveReleaseComponent extends FacetsPage {
  podInfo: Array<Pod>;
  clusterId: any;
  childClusterId: any;
  clusterState: string;
  resourceName: any;
  resourceType: any;
  baseUrl = '';
  loading: boolean = true;
  showEmptyState: boolean = false;
  secondaryMsg: string;
  primaryMsg: string =
    'This page is available when environment is in running state.';
  buttonText: string = 'Manage Your Environment';
  stackName: string;
  emptyStateLogo = emptyStateLogo;
  showBtn: boolean = true;
  status: 'info' | 'error' = 'info';
  fetchingHpaInfo = false;
  hpaInfo: HorizontalPodAutoscaler;
  showK8sExplorer: boolean = false;
  cluster: any;
  fetchingCluster: boolean = false;

  constructor(
    _route: ActivatedRoute,
    private uiDeploymentControllerService: UiDeploymentControllerService,
    private router: Router,
    private uiApplicationControllerService: UiApplicationControllerService,
    private uiDropdownsControllerService: UiDropdownsControllerService,
    private commonClusterController: UiCommonClusterControllerService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.showEmptyState = false;
    this.loading = true;
    this.clusterId = p?.clusterId;
    this.resourceName = p.resourceName;
    this.stackName = p.stackName;
    this.resourceType = p?.resourceType;
    this.getEnvironmentConfig(p);
    this.getClusterState(p);
  }

  getEnvironmentConfig(p) {
    this.fetchingCluster = true;
    this.commonClusterController
      .getClusterCommonUsingGET(this.clusterId)
      .subscribe(
        (cluster) => {
          this.cluster = cluster;
          this.childClusterId = this.cluster?.baseClusterId || p?.clusterId;
          this.fetchingCluster = false;
        },
        (error) => {
          this.cluster = null;
          this.fetchingCluster = false;
        },
      );
  }

  getClusterState(p: Params) {
    this.uiDeploymentControllerService
      .getClusterStateUsingGET(p.clusterId)
      .subscribe(
        (t) => {
          this.clusterState = t;
          const clusterStates = Object.keys(ClusterState);
          if (clusterStates.includes(this.clusterState)) {
            this.showEmptyState = true;
            this.secondaryMsg =
              'Environment is in ' +
              ClusterState[this.clusterState] +
              ' state!';
          }
          this.loading = false;
        },
        (error) => {
          this.secondaryMsg =
            error?.error?.message ||
            ' There is some error fetching environment information';
          this.primaryMsg = '';
          this.showBtn = false;
          this.emptyStateLogo = errorIcon;
          this.showEmptyState = true;
          this.status = 'error';
        },
      );
  }

  navigate(event) {
    let urlToNav =
      '/' + this.stackName + '/cluster/' + this.clusterId + '/releases/';
    this.router.navigate([urlToNav]);
  }

  handleBack = () => {
    this.router.navigate([
      '/',
      this.stackName,
      'cluster',
      this.clusterId,
      'resources',
    ]);
  };
}

<ng-container *ngIf="!this.showEmptyState && !this.loading && !fetchingCluster">
  <div class="row mt-4">
    <div class="col-md-12">
      <app-app-native-charts
        [clusterID]="childClusterId"
        [resource]="resourceName"
      ></app-app-native-charts>
    </div>
  </div>
</ng-container>
<div *ngIf="this.showEmptyState || loading" class="row mt-3">
  <div class="col-sm-12">
    <nb-card [nbSpinner]="loading" class="background-grey py-5">
      <app-message
        *ngIf="this.showEmptyState && !loading"
        [primaryMsg]="primaryMsg"
        [secondaryMsg]="secondaryMsg"
        (buttonClick)="navigate($event)"
        [buttonText]="buttonText"
        [showBtn]="showBtn"
        [imgUrl]="emptyStateLogo"
        toolTip="Go to Releases"
        [status]="status"
      ></app-message>
    </nb-card>
  </div>
</div>
