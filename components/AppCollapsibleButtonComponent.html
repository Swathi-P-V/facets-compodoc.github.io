import {
  Component,
  EventEmitter,
  Output,
  Input,
  OnDestroy,
} from '@angular/core';

@Component({
  selector: 'app-collapsible-button',
  templateUrl: './collapsible-button.component.html',
  styleUrls: ['./collapsible-button.component.scss'],
})
export class AppCollapsibleButtonComponent implements OnDestroy {
  @Output() byButtonClick = new EventEmitter();
  @Output() byDragDropClick = new EventEmitter();
  @Input() buttonText: string = '';
  @Input() buttonIcon: string = '';
  @Input() color: string = '#fff';
  @Input() backgroundColor: string = '#00c2bb';
  boundary = '.scrollable-container';
  isDragging = false;
  dragTimeout: any;
  btnPosition: 'TOP' | 'BOTTOM' = 'BOTTOM';

  constructor() {}

  ngOnDestroy(): void {
    if (this.dragTimeout) {
      clearTimeout(this.dragTimeout);
    }
  }

  onDragStarted() {
    this.isDragging = true;
    this.byDragDropClick.emit(true);
  }

  onDragEnded() {
    if (this.dragTimeout) {
      clearTimeout(this.dragTimeout);
    }
    this.dragTimeout = setTimeout(() => {
      this.isDragging = false;
      this.byDragDropClick.emit(false);
    }, 50);
  }

  onButtonClick() {
    if (!this.isDragging) {
      this.byButtonClick.emit();
    }
  }

  changeBtnPosition() {
    this.btnPosition = 'TOP';
  }
}
