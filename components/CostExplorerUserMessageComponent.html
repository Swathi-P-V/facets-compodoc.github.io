import { Component, Input, TemplateRef, ViewChild } from '@angular/core';
import { NbDialogRef, NbDialogService } from '@nebular/theme';

@Component({
  selector: 'app-cost-explorer-user-message',
  templateUrl: './cost-explorer-user-message.component.html',
  styleUrls: ['./cost-explorer-user-message.component.scss'],
})
export class CostExplorerUserMessageComponent {
  @Input() additionalInfo: string[] = [];
  @Input() isAwsCostExplorerEnabled: boolean;

  @ViewChild('runbookModal', { static: true })
  private runbookModal: TemplateRef<any>;

  constructor(private dialogService: NbDialogService) {}

  showRunBook() {
    this.dialogService.open(this.runbookModal, {
      context: {},
    });
  }
}
