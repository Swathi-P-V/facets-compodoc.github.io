import { HttpClient } from '@angular/common/http';
import {
  Component,
  OnInit,
  TemplateRef,
  ViewChild,
  OnDestroy,
} from '@angular/core';
import {
  NbDialogService,
  NbMenuItem,
  NbMenuService,
  NbSidebarService,
  NbPopoverDirective,
} from '@nebular/theme';
import { ApplicationControllerService } from 'src/app/cc-api/services';
import { Subject, Subscription, takeUntil, filter } from 'rxjs';
import { SimpleOauth2User } from 'src/app/cc-api/models/simple-oauth-2user';
import {
  Router,
  NavigationEnd,
  NavigationStart,
  ActivatedRoute,
} from '@angular/router';
import { UserAuthoritiesService } from 'src/app/services/user-authorities.service';
import { SidebarStateService } from 'src/app/services/sidebar-state.service';
import { UserService } from 'src/app/services/user.service';
import { LocalStorageService } from 'src/app/services/local-storage.service';
import { CookieUtilWrapperService } from 'src/app/services/cookie-util-wrapper.service';
import { SessionContextStoreService } from 'src/app/services/session-context-store.service';
import { RedirectOnLoginService } from 'src/app/services/redirect-on-login.service';
import { FeatureFlagsService } from 'src/app/services/feature-flags.service';
import { TableStatefulFiltersService } from 'src/app/services/table-stateful-filters.service';

@Component({
  selector: 'app-layout',
  templateUrl: './app-layout.component.html',
  styleUrls: ['./app-layout.component.scss'],
})
export class AppLayoutComponent implements OnInit, OnDestroy {
  @ViewChild(NbPopoverDirective) popover: NbPopoverDirective;
  user: SimpleOauth2User; // Logged in user
  userInitials: string;
  userNameHint: string;
  sidebarOpen: boolean = true; // To show/hide the logo and the username
  icon: any; // Facets logo
  isDemo = '';
  footerItems: NbMenuItem[] = []; // Menu items for footer
  userName: any;
  showMenuItems: boolean = false;
  private destroy$ = new Subject<void>();
  reinitSidebarMenu: boolean[]; // Signal to be sent to sidebar component
  searchIcon = 'search';
  currentUrl: string;
  hideSidebar: boolean = false;
  localStorageSupported: boolean = false;
  isHovered: boolean = false;
  isHoveredHelp: boolean = false;
  searchDialog: any;
  logo: string;
  showGlobalSpinner: boolean = false;
  facetsAnimatedIcon = './assets/facets-animated.gif';
  clickupTicketUrl: string = '';
  dragging: boolean = false;
  sidebarLogoName = '';
  globalLoaderGifName = '';
  docsLink: string = '';
  topNavData: any = null;

  @ViewChild('logoutTemplate', { static: true })
  private logoutTemplate: TemplateRef<any>;
  @ViewChild('searchTemplate', { static: true })
  private searchTemplate: TemplateRef<any>;

  constructor(
    protected http: HttpClient,
    private cookieService: CookieUtilWrapperService,
    public sidebarService: NbSidebarService,
    private applicationControllerService: ApplicationControllerService,
    private menuService: NbMenuService,
    private dialogBoxService: NbDialogService,
    private router: Router,
    public userAuthoritiesService: UserAuthoritiesService,
    private sidebarStateService: SidebarStateService,
    public userService: UserService,
    private sessionContextStoreService: SessionContextStoreService,
    private localStorage: LocalStorageService,
    private redirectOnLoginService: RedirectOnLoginService,
    private featureFlagService: FeatureFlagsService,
    private tableStatefulFiltersService: TableStatefulFiltersService,
    private route: ActivatedRoute,
  ) {}

  async ngOnInit(): Promise<void> {
    this.redirectOnLoginService.handleRedirectURL(false);
    this.clickupTicketUrl = this.featureFlagService?.getClickupTicketUrl();
    this.localStorageSupported = this.localStorage.isLocalStorageSupported();
    this.getUser();
    this.getLogo();
    this.isDemo = this.cookieService.get('isDemo');
    this.setSidebarStateListeners();
    this.currentUrl = this.router.url;
    this.router.events
      .pipe(
        filter(
          (event) =>
            event instanceof NavigationEnd || event instanceof NavigationStart,
        ),
      )
      .subscribe((navEvent) => {
        if (navEvent instanceof NavigationStart) {
          this.showGlobalSpinner = true;
        }
        if (navEvent instanceof NavigationEnd) {
          this.showGlobalSpinner = false;
        }
        this.currentUrl = navEvent?.['url'];
      });

    this.sidebarStateService.sidebarState.subscribe(
      (state: 'expand' | 'collapse') => {
        this.hideSidebar = state === 'collapse';
      },
    );
    /*
      Adds event listener on document to listen to user button clicks in order to open search popup
    */
    document.addEventListener('keydown', this.handleKeyDown);
    await this.featureFlagService.getAllFeatureFlags();
    this.setFeatureFlags();
  }

  setFeatureFlags() {
    this.docsLink =
      this.featureFlagService?.getDocsLink() ||
      'https://readme.facets.cloud/docs';
    this.sidebarLogoName = this.featureFlagService.getSidebarLogoName();
    const globalLoader = this.featureFlagService.getGlobalLoaderGifName();
    if (globalLoader) {
      this.globalLoaderGifName = `./assets/${globalLoader}.gif`;
    }
    this.setFooterMenuItems();
  }

  /*
    1. Checks if meta(command key is pressed on mac) or control key is pressed on windows
    2. Checks if E key is pressed together with command or control
    3. Opens the popup on combination of above keys
  */
  handleKeyDown = (event: KeyboardEvent) => {
    if ((event?.metaKey || event?.ctrlKey) && event?.key === 'e') {
      this.openSearch();
    }
    if ((event?.metaKey || event?.ctrlKey) && event?.key === '/') {
      this.openHelpDashboard();
    }
  };

  // Catches the updated state of the sidebar and updates the sidebarOpen state accordingly
  // to show/hide the Facets logo and the username
  setSidebarStateListeners() {
    this.sidebarService.onExpand().subscribe((state) => {
      this.sidebarOpen = true;
    });
    this.sidebarService.onCompact().subscribe((state) => {
      this.sidebarOpen = false;
    });
  }

  // Logged in user's data
  getUser() {
    this.applicationControllerService
      .meUsingGET()
      .subscribe((user: SimpleOauth2User) => {
        this.user = user;
        this.userService.setUser(user);
        this.setUserName(this.user);
        //Clear the cookies.
        this.cookieService.delete('redirectTo', '/');
      });
  }

  setUserName(user) {
    const USERNAME_LIMIT: number = 18;
    this.userName = user?.attributes['name'] || user?.name;

    this.userInitials =
      this.userName?.length > USERNAME_LIMIT
        ? this.userName.substring(0, USERNAME_LIMIT) + '...'
        : this.userName;

    this.userNameHint =
      this.userName?.length > USERNAME_LIMIT ? this.userName : '';
  }

  // Facets logo for sidebar
  getLogo() {
    this.http
      .get('/public/v1/logo', { responseType: 'text' })
      .subscribe((x) => {
        this.logo = x;
      });
  }

  // Minimize/maximize sidebar
  toggleSidebar(state: 'open' | 'close') {
    let makeCompact = state === 'close';
    makeCompact
      ? this.sidebarService.compact('main')
      : this.sidebarService.expand('main');
    this.sidebarStateService.sendSidebarStateUpdate(
      makeCompact ? 'compact' : 'expand',
    );
  }

  // Asks for log out confirmation
  openLogoutDialog(): void {
    this.dialogBoxService.open(this.logoutTemplate, {
      hasBackdrop: true,
      autoFocus: false,
    });
  }

  // Logs user out
  logout() {
    const logoutUrl = this.featureFlagService.getLogoutUrl();
    this.cookieService.delete('selection', '/');
    this.tableStatefulFiltersService.flushFilterStates();
    const currentContext = this.cookieService.get('currentContext');
    this.sessionContextStoreService.setSessionContext(
      currentContext,
      this.userName,
      'false',
    );
    this.redirectOnLoginService.preserveRedirectURL();
    this.http.get(`/perform_logout`).subscribe(
      (x: SimpleOauth2User) => {},
      (error) => {
        if (logoutUrl) {
          this.router.navigate(['/sso-logout']);
        } else {
          this.router.navigate(['/login']);
        }
      },
    );
    //clear local storage after logout
    this.localStorage?.removeFacetsStore();
  }

  // Footer items for the sidebar
  setFooterMenuItems() {
    this.footerItems = [
      {
        title: 'Docs',
        icon: { icon: 'book-open-outline', pack: 'fci' },
        url: this.docsLink,
        target: '_blank',
        skipLocationChange: true,
      },
    ];
  }

  // If settings has been clicked on: then we need to reinit the mainMenu nb-menu so that click event on the other component doesnt make unexpected changes
  // If docs has been clicked on: then remove the active class from the docs anchor tag and (if needed: mark settings as the active menu item again)
  updateFooterMenuWithState() {
    // Removing all the active menu items by passing some random string to removeAllOtherActiveTabs
    // Commenting this out in the app-layout component (still runs in sidebar component)
    // this.removeAllOtherActiveTabs('_____remove-all-menu-items____');
    setTimeout(() => {
      this.menuService
        .getSelectedItem('footerMenu')
        .pipe(takeUntil(this.destroy$))
        .subscribe((menuItem) => {
          if (menuItem?.item) {
            if (menuItem.item.title.toLocaleLowerCase() == 'docs') {
              this.markDocsInactive();
              if (this.router.url.includes('settings')) {
                this.removeAllOtherActiveTabs('Settings');
              }
            } else {
              this.removeAllOtherActiveTabs('Settings');
              this.reinitSidebarMenu = [...[true]];
            }
          }
        })
        .add(() => {
          this.reinitSidebarMenu = [false];
        });
    }, 3000);
  }

  // 'Docs' menu item doesn't need to be marked as an active item as it's always opened in another tab
  markDocsInactive() {
    let allMenus: any = document.querySelectorAll('nb-menu');
    let docsATag = allMenus[1]?.getElementsByTagName('a')?.[0];
    docsATag.classList.remove('active');
  }

  // DOM manipulation because nb-menu doesn't have APIs to update the state for active menu items
  // Goes through the anchor tabs of the other menu and updates their classList
  removeAllOtherActiveTabs(currentlyActive: string) {
    let allMenus: any = document.querySelectorAll('nb-menu');
    let allAnchorTags = [
      ...allMenus[0]?.getElementsByTagName('a'),
      ...allMenus[1]?.getElementsByTagName('a'),
    ];
    for (let i = 0; i < allAnchorTags?.length; i++) {
      let item = allAnchorTags[i];
      if (
        (item?.innerText?.length && item?.innerText == currentlyActive) ||
        item?.innerHTML.includes(currentlyActive)
      ) {
        item?.classList.add('active');
      } else {
        item?.classList.remove('active');
      }
    }
  }

  openAccountSettings() {
    this.router.navigate(['/accountSettings']);
  }

  redirectToHome() {
    this.router.navigate(['/home']);
  }

  ngOnDestroy(): void {
    this.userService.resetUserData();
    this.userAuthoritiesService.resetData();
    this.featureFlagService.resetData();
  }

  openSearch() {
    this.searchDialog ? this.searchDialog.close() : null;
    this.searchDialog = this.dialogBoxService.open(this.searchTemplate, {
      hasBackdrop: true,
      autoFocus: false,
    });
  }

  openHelpDashboard() {
    if (this.clickupTicketUrl?.length) {
      window.open(
        `${this.clickupTicketUrl}?URL=${window.location.href}&Reported%20by=${this.user?.name}`,
        '_blank',
      );
    }
  }

  onDragDropClick(event) {
    this.dragging = event;
  }

  onSidebarStateChange(state) {
    this.sidebarOpen = state === 'expanded';
    const ele = document.getElementById('sidebar');
    ele.classList.remove('fixed');
  }

  onTopNavDataLoad(data) {
    this.topNavData = data;
  }
}
