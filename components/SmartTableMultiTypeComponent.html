import { Component, Input, OnInit, Output, EventEmitter } from '@angular/core';
import { ClipboardService } from 'ngx-clipboard';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';

@Component({
  selector: 'app-smart-table-multi-type',
  templateUrl: './smart-table-multi-type.component.html',
  styleUrls: ['./smart-table-multi-type.component.scss'],
})
export class SmartTableMultiTypeComponent implements OnInit {
  @Input() value: any;
  @Input() dynamicProps: any; // For non-table components, user can directly send dynamicProps as input
  @Output() byClickCellWithEllipsisLink = new EventEmitter();
  @Output() byClickCellWithRouterLink = new EventEmitter();
  childProps: any;

  constructor(
    private clipboardService: ClipboardService,
    private customToastService: CustomToastService,
  ) {}

  ngOnInit() {
    try {
      if (this.value) {
        this.dynamicProps = JSON.parse(this.value);
      }
      this.childProps = JSON.stringify(this.dynamicProps?.props);
    } catch (error) {}
  }

  onClickCopyToClipboard() {
    this.clipboardService.copy(this.dynamicProps?.props?.tooltipText);
    this.customToastService.success('Value copied to clipboad!');
  }
  onClickCellWithEllipsisLink(data) {
    this.byClickCellWithEllipsisLink.emit(data);
  }

  onClickRouterLink(event) {
    this.byClickCellWithRouterLink.emit(event);
  }
}
