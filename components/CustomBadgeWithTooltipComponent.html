import { Component, Input, Output } from '@angular/core';

@Component({
  selector: 'app-custom-badge-with-tooltip',
  templateUrl: './custom-badge-with-tooltip.component.html',
  styleUrls: ['./custom-badge-with-tooltip.component.scss'],
})
export class CustomBadgeWithTooltipComponent {
  @Input() badgeText: string = '';
  @Input() tooltipText: string = '';
  @Input() badgeSize: string = 'small';
  @Input() badgeStatus: string = 'basic';
  @Input() badgeClass: string = '';
  @Input() popover: { text?: string; urlText?: string; url?: string } = null;

  constructor() {}
}

<!-- Custom badge with tooltip -->
<nb-tag
  style="text-transform: none !important"
  appearance="filled"
  [text]="badgeText"
  [size]="badgeSize"
  [status]="badgeStatus"
  [nbTooltip]="tooltipText"
  [nbTooltipDisabled]="!tooltipText?.length"
  class="tag-restyle {{ badgeClass }}"
  *ngIf="!popover"
></nb-tag>

<!-- Custom badge with popover -->
<nb-tag
  style="text-transform: none !important"
  appearance="filled"
  [text]="badgeText"
  [size]="badgeSize"
  [status]="badgeStatus"
  class="tag-restyle cursor-pointer"
  [nbPopover]="popoverText"
  nbPopoverPlacement="top"
  nbPopoverClass="custom-badge-with-popover"
  nbPopoverTrigger="hover"
  *ngIf="popover"
></nb-tag>
