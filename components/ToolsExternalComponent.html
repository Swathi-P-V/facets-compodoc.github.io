import { Component } from '@angular/core';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params } from '@angular/router';
import {
  UiCommonClusterControllerService,
  UiApplicationControllerService,
} from 'src/app/cc-api/services';

const prometheusLogo = './assets/prometheus-icon-color.svg';

@Component({
  selector: 'app-tools-external',
  templateUrl: './tools-external.component.html',
  styleUrls: ['./tools-external.component.scss'],
})
export class ToolsExternalComponent extends FacetsPage {
  fetchingResourceDetails: boolean = false;
  fetchingTools: boolean = false;
  isToolingSet: boolean;
  wettyUrl: string;
  toolsAvailable;
  prometheusLogo = prometheusLogo;
  basicAuthPassword = '';
  clusterId: string;
  cluster: any;
  // This was used to hide k8s and grafana from this list
  // embeddedTools = ['k8s-dashboard', 'grafana'];
  embeddedTools = [];
  urlClusterId: any;
  loadingCluster: boolean = false;
  tools = {
    'k8s-dashboard': {
      label: 'k8s Dashboard',
      icon: null,
    },
    grafana: {
      label: 'Grafana',
      icon: null,
    },
    prometheus: {
      label: 'Prometheus',
      icon: prometheusLogo,
    },
    alertmanager: {
      label: 'Alert Manager',
      icon: null,
    },
    wetty: {
      label: 'Wetty',
      icon: null,
    },
  };

  constructor(
    private activatedRoute: ActivatedRoute,
    private uiCommonClusterControllerService: UiCommonClusterControllerService,
    private uiApplicationControllerService: UiApplicationControllerService,
    private commonClusterController: UiCommonClusterControllerService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(urlParams: Params): void {
    let clusterId = '';
    if (urlParams.clusterId) {
      clusterId = urlParams.clusterId;
      this.clusterId = clusterId;
      this.getEnvironmentConfig(urlParams);
      this.fetchingResourceDetails = true;
      this.toolsAvailable = [];

      //fetching all the infra rules and details from the server
      this.uiCommonClusterControllerService
        .resourceDetailsUsingGET(clusterId)
        .subscribe(
          (t) => {
            if (t?.length) {
              //filter resourceType with ingress_infra_rules from the list we get from the server
              //the list also contains basic auth password and domain name etc.
              //below code filters only ingress_infra_rules which will be used as tools from that data.
              let ingressPresent = t
                .filter((o) => o.resourceType === 'ingress_rules_infra')
                .map((tool) => {
                  const toolDetails = this.tools[tool.key];
                  return { ...toolDetails, ...tool };
                });

              //here,we are splitting the tools url from the list above. The tools url is the value key inside rules object
              //once we get the value, we split it and set first bit from it as ingressprefix.
              //The first word before the . in the url always is domain prefix for the url.
              if (ingressPresent?.length > 0) {
                ingressPresent.forEach((ingress) => {
                  let url = ingress?.value;
                  if (url?.length > 0) {
                    let params = url?.split('.');
                    ingress['ingressPrefix'] = params[0];
                  }
                });
                //separating the tools from the other ingress rules.
                this.sortTools(ingressPresent, clusterId);
              }

              this.basicAuthPassword = t.find(
                (o) => o.name === 'Basic Authentication Password',
              )?.value;
            }
            if (t?.length === 0) {
              this.isToolingSet = false;
            } else {
              this.setWettyUrl(t);
            }
            this.fetchingResourceDetails = false;
          },
          (error) => {
            this.fetchingResourceDetails = false;
            this.toolsAvailable = [];
            this.isToolingSet = false;
          },
        );
    }
  }

  getEnvironmentConfig(urlParams) {
    this.loadingCluster = true;
    this.commonClusterController
      .getClusterCommonUsingGET(this.clusterId)
      .subscribe(
        (cluster) => {
          this.cluster = cluster;
          this.urlClusterId =
            this.cluster?.baseClusterId || urlParams?.clusterId;
          this.loadingCluster = false;
        },
        (error) => {
          this.cluster = null;
          this.urlClusterId = null;
          this.loadingCluster = false;
        },
      );
  }

  setWettyUrl(t) {
    const baseInfo = t?.filter((i) => i?.key === 'tools');
    if (baseInfo?.length >= 2) {
      this.isToolingSet = true;
      const baseUrl = baseInfo?.find((i) => i?.name === 'Base Domain');
      const password = baseInfo?.find(
        (i) => i?.name === 'Basic Authentication Password',
      );
      this.wettyUrl =
        'https://toolsuser:' + password?.value + '@wetty.' + baseUrl?.value;
    }
  }

  sortTools(ingressPresent, clusterId) {
    this.fetchingTools = true;
    //fetching the ingress tools rule from server
    this.uiApplicationControllerService
      .getResourceByNameV2UsingGET({
        resourceName: 'tools',
        resourceType: 'ingress',
        clusterId: clusterId,
      })
      .subscribe(
        (res) => {
          let rules = res?.['spec']?.['rules']; //get the ingress tools rules from the response object
          let rulesList = Object.values(rules || {})?.filter(
            (rule) => rule?.['domain_prefix'] !== 'argo-rollouts',
          ); //convert object inside object into a list of objects of rules.
          let filteredToolsIngress = [];

          //here we are comparing the domain prefix name(domain_prefix) of each rule we get as above api response
          //to the each ingress rules ingressPrefix property we set earlier through url manipulation.
          //we filter the value only when both matches.
          //This ensures we filter only the ingress tools rules and filter out other application rules
          ingressPresent.forEach((ingress) => {
            if (ingress?.ingressPrefix?.length > 0) {
              rulesList.forEach((rule) => {
                if (rule?.['domain_prefix'] === ingress?.ingressPrefix) {
                  filteredToolsIngress.push(ingress);
                }
              });
            }
          });
          filteredToolsIngress.sort(function (a, b) {
            return a?.key?.localeCompare(b?.key);
          });
          this.toolsAvailable = [...filteredToolsIngress];
          this.fetchingTools = false;
        },
        (err) => {
          this.fetchingTools = false;
          this.toolsAvailable = [];
          this.isToolingSet = false;
        },
      );
  }

  getToolUrl(val) {
    let toolName = val?.split('.')[0];
    if (toolName == 'k8s') {
      return `${window.location.origin}/tunnel/${
        this.clusterId
      }/kubernetes/#/workloads?namespace=${
        this.cluster?.namespace || 'default'
      }`;
    } else if (toolName == 'grafana') {
      return `${window.location.origin}/tunnel/${this.urlClusterId}/grafana/`;
    } else {
      return `https://toolsuser:${this.basicAuthPassword}@${val}`;
    }
  }

  setIcon(icon) {
    return {
      'background-image': `url(${icon})`,
      'background-size': '2rem',
      'background-repeat': 'no-repeat',
      'background-position': 'center',
    };
  }
}

</div>
