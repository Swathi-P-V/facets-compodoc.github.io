  envTableSettings: any = {

  //Unclassified artifacts
  unclassifiedArtifactsTableSettings: any = {};

  constructor(
    private activatedRoute: ActivatedRoute,
    private uiStackController: UiStackControllerService,
    private uiArtifactCiController: UiArtifactCiControllerService,
    private promotionWorkflow: UiPromotionWorkflowControllerService,
    private router: Router,
    private dialogService: NbDialogService,
    private artifactsController: UiArtifactsControllerService,
    private formatDateService: FormatDateService,
    private uiAritifactRoutingRuleController: UiArtifactRoutingRuleControllerService,
    private uiArtifactsController: UiArtifactsControllerService,
    private contextUpdatesService: ContextUpdatesService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.updateTableSettings();
    if (!this._nestedMode) {
      this.ciId = p?.ciId || '';
      this.activatedRoute.paramMap
        .pipe(map(() => window.history.state))
        .subscribe((data) => {
          this.stackName = data?.stackName ? data?.stackName : '';
        });
    } else {
      this.ciId = this._ciID;
      this.stackName = this._blueprint;
    }
    this.getAllBlueprints();
    this.getEnvs();
    //Now fetch CI Details
  }

  getCIDetails() {
    this.fetchingCIDetails = true;
    this.status = 'info';
    this.uiArtifactCiController.getArtifactCIUsingGET(this.ciId).subscribe(
      async (res) => {
        this.artifactCI = res;
        this.ciData['registrationType'] = this.artifactCI.registrationType;
        this.ciData['stackName'] = this.artifactCI.stackName;
        //Set Ci rule name because API sends ruleID
        this.setCIRuleName(this.artifactCI.ruleId);
        //Set promotional workflow
        this.artifactCI?.registrationType === 'HYBRID'
          ? this.getAllWorkflows()
          : this.getRegSpecificWorkflows();
        //Get all artifacts related to ci
        this.getArtifacts(this.artifactCI.ciName);
        this.fetchingCIDetails = false;
      },
      (err) => {
        this.fetchingCIDetails = false;
        this.showEmptyState = true;
        this.primaryMsg = '';
        this.secondaryMsg =
          err?.error?.message ||
          'There is some error fetching CI integration details';
        this.status = 'error';
        this.emptyStateLogo = errorIcon;
      },
    );
  }

  getRegSpecificWorkflows() {
    this.fetchingWorkflows = true;
    let workflowParams = {
      registrationType: this.artifactCI?.registrationType,
      stackName: this.artifactCI?.stackName,
    };
    this.promotionWorkflow
      .getRegistrationSpecificWorkflowsUsingGET(workflowParams)
      .subscribe(
        (res) => {
          this.promotionWorkflows = res;
          this.setWorkflowDetails();
          this.fetchingWorkflows = false;
        },
        (error) => {
          this.fetchingWorkflows = false;
        },
      );
  }

  getAllWorkflows() {
    this.fetchingWorkflows = true;
    this.promotionWorkflow.getAllWorkflowsUsingGET().subscribe(
      (res) => {
        this.promotionWorkflows = res;
        this.setWorkflowDetails();
        this.fetchingWorkflows = false;
      },
      (err) => {
        this.fetchingWorkflows = false;
      },
    );
  }

  setWorkflowDetails() {
    this.selectedWorkflow = {};
    this.availableStreams = [];
    if (this.artifactCI?.promotionWorkflowId) {
      let currentHierarchy: any = [];
      this.selectedWorkflow = this.promotionWorkflows.find(
        (workflow) => workflow?.id === this.artifactCI?.promotionWorkflowId,
      );
      this.ciData['workflowName'] = this.selectedWorkflow.workflowName;
      currentHierarchy = this.selectedWorkflow.hierarchies;
      let sortedList = currentHierarchy.sort((a, b) => {
        return a.sequence - b.sequence;
      });
      sortedList.forEach((item) => {
        this.availableStreams.push(item.registrationValue);
      });
    }
  }

  getEnvs() {
    this.fetchingEnvs = true;
    this.uiStackController
      .getAllClustersUsingGET({
        page: 0,
        perPage: 10000,
        sortBy: 'ID',
      })
      .subscribe(
        (res) => {
          this.envs = res?.content;
          this.envs.forEach((env) => {
            this.envIdNameMapping[env?.id] = env?.name;
          });
          this.fetchingEnvs = false;
          this.getCIDetails();
        },
        (err) => {
          this.fetchingEnvs = false;
        },
      );
  }

  getAllBlueprints() {
    this.loadingBlueprint = true;
    this.uiStackController.getStacksUsingGET1().subscribe(
      (res) => {
        this.stacks = res;
        this.loadingBlueprint = false;
      },
      (error) => {
        this.loadingBlueprint = false;
      },
    );
  }

  getArtifacts(ciName) {
    this.fetchingArtifacts = true;
    this.classifiedArtifacts = [];
    this.unclassifiedArtifacts = [];
    this.streamArtifacts = [];
    this.envArtifacts = [];
    this.uiArtifactCiController
      .getArtifactsForCIUsingGET(ciName)
      .subscribe(
        (res) => {
          this.artifacts = res;
          //Extract classified and unclassified artifacts.
          this.artifacts.forEach((obj) => {
            obj.classified
              ? this.classifiedArtifacts.push(obj)
              : this.unclassifiedArtifacts.push(obj);
          });

          //Assign release streams and environments for classified artifacts
          this.classifiedArtifacts.forEach((artifact) => {
            if (artifact.registrationType === 'RELEASE_STREAM') {
              artifact['envs'] = this.getAllEnvs(
                artifact.registrationValue,
                this.artifactCI?.stackName,
              );
              if (artifact['envs']?.length || artifact?.artifactId) {
                this.streamArtifacts.push(artifact);
              }
            } else if (artifact.registrationType === 'ENVIRONMENT') {
              //get the name of environment
              let registeredEnv = this.envs.find(
                (env) => env.id === artifact.registrationValue,
              );
              artifact['envName'] = registeredEnv?.name;
              artifact['stackName'] = registeredEnv?.stackName;
              artifact['releaseStream'] = registeredEnv?.releaseStream;

              this.envArtifacts.push(artifact);
            }
          });

          if (this.artifacts.length <= 0) {
            this.showEmptyState = true;
          }
          this.fetchingArtifacts = false;
        },
        (err) => {
          this.fetchingArtifacts = false;
          this.showEmptyState = true;
          this.primaryMsg = '';
          this.secondaryMsg =
            err?.error?.message || 'There is some error fetching artifacts';
          this.status = 'error';
          this.emptyStateLogo = errorIcon;
        },
      )
      .add(() => {
        this._loadedTable.emit(true);
      });
  }

  //This function returns all environments which are mapped to release stream as sent in function parameter.
  //If stack name is present then environments of only that blueprint is considered else environments across all blueprints is considered
  getAllEnvs(releaseStream: string, stackName?: string) {
    let envs = [];
    if (stackName?.length > 0) {
      envs = this.envs.reduce((filteredEnvs, env) => {
        if (
          env.stackName === stackName &&
          env.releaseStream === releaseStream
        ) {
          filteredEnvs.push({
            name: env.name,
            stackName: env.stackName,
            envId: env.id,
          });
        }
        return filteredEnvs;
      }, []);
    } else {
      envs = this.envs.reduce((filteredEnvs, env) => {
        if (env.releaseStream === releaseStream) {
          filteredEnvs.push({
            name: env.name,
            stackName: env.stackName,
            envId: env.id,
          });
        }
        return filteredEnvs;
      }, []);
    }
    return envs;
  }

  setCIRuleName(ruleId) {
    this.settingCIRuleName = true;
    if (!ruleId) {
      this.ciData['ruleName'] = 'N/A';
      this.settingCIRuleName = false;
      return;
    } else {
      this.uiAritifactRoutingRuleController
        .getArtifactRoutingRuleUsingGET(ruleId)
        .subscribe(
          (res) => {
            this.ciData['ruleStackName'] = res?.stackName;
            this.ciData['ruleName'] = res?.ruleName;
            this.settingCIRuleName = false;
          },
          (error) => {
            this.settingCIRuleName = false;
          },
        )
        .add(() => {
          this._loadedTable.emit(true);
        });
    }
  }

  promoteArtifactConfirmPopup(rowData) {
    let hierarchy = this.selectedWorkflow?.hierarchies;
    let currentValue = rowData?.registrationValue;
    let nextValueIndex =
      hierarchy?.findIndex((item) => item?.registrationValue === currentValue) +
      1;
    let nextValue =
      rowData?.registrationType === 'ENVIRONMENT'
        ? this.envIdNameMapping?.[hierarchy[nextValueIndex]?.registrationValue]
        : hierarchy?.[nextValueIndex]?.registrationValue;
    rowData['nextPromote'] = nextValue;
    this.dialogService.open(this.confirmPromote, { context: rowData });
  }

  promote(data, ref) {
    if (data?.artifactId?.length > 0) {
      let artifactPromoteParams: {
        ciId: string;
        artifactId: string;
      } = {
        ciId: this.ciId,
        artifactId: data?.artifactId,
      };
      this.promotionInProgress = true;
      this.artifactsController
        .promoteArtifactUsingPOST(artifactPromoteParams)
        .subscribe(
          (res) => {
            this.getCIDetails();
            this.customToastService.success('Artifact promoted successfully');
            this.promotionInProgress = false;
            ref.close();
          },
          (error) => {
            ref.close();
            this.promotionInProgress = false;
            this.onAPIErrorService.onAPIError(
              error?.error?.message || 'Promotion of artifact failed',
              error,
            );
          },
        );
    } else {
      this.customToastService.error('Artifact id is invalid');
    }
  }

  goBack = () => {
    let currentContext = this.contextUpdatesService?.getCurrentContext();
    let stackName = currentContext?.blueprint?.length
      ? currentContext?.blueprint
      : this.stacks[0]?.name;
    this.router.navigate(['/', 'stack', stackName, 'artifacts', 'ci']);
  };

  openPushPopup(data) {
    this.ctlCommand = `facetsctl push -i <image_name_here> -a ${data?.artifactName} --registration-type ${data?.registrationType} --registration-value ${data?.registrationValue} -e <external_id_here> -d <your_build_description> --artifactory ${data?.artifactory}`;
    this.dialogService.open(this.pushArtifact, { context: data });
  }

  openArtifactHistory(data) {
    this.router.navigate(
      ['/', 'artifact-history', data?.versioningKey, 'ciId', this.ciId],
      {
        state: {
          previousComponent: 'CIDetailsComponent',
          stackName: this.stackName,
        },
      },
    );
  }

  isLoading() {
    return (
      this.fetchingArtifacts ||
      this.fetchingCIDetails ||
      this.fetchingEnvs ||
      this.fetchingWorkflows ||
      this.loadingBlueprint ||
      this.settingCIRuleName
    );
  }

  copyText() {
    navigator.clipboard
      .writeText(this.ctlCommand)
      .then((res) => {
        this.popover.show();
        setTimeout(() => {
          this.popover.hide();
        }, 2000);
      })
      .catch((e) => console.error(e));
  }

  openWorkflowDetails() {
    if (Object.keys(this.selectedWorkflow)?.length > 0) {
      this.dialogService.open(this.workflowDetails, {
        context: { rowData: this.selectedWorkflow },
      });
    }
  }

  getFormattedDate(date) {
    return this.formatDateService.getFormattedDate(date);
  }

  deleteArtifactConfirmPopup(rowData) {
    let currentValue = rowData?.registrationValue;
    let valueToShow =
      rowData?.registrationType === 'ENVIRONMENT'
        ? this.envIdNameMapping?.[currentValue]
        : currentValue;
    rowData['valueToShow'] = valueToShow;
    this.dialogService.open(this.confirmDelete, { context: rowData });
  }

  deleteArtifact(data, ref) {
    this.deleteArtifactInProgress = true;
    this.artifactsController
      .deleteArtifactUsingDELETE(data?.artifactId)
      .subscribe(
        (data) => {
          this.deleteArtifactInProgress = false;
          this.customToastService.success('Artifact deleted successfully');
          ref.close();
          this.getCIDetails();
        },
        (error) => {
          this.deleteArtifactInProgress = false;
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Artifact delete failed',
            error,
          );
          ref.close();
        },
      );
  }

  handleActions(event) {
    this.ciActionButtons?.find((button) => button.id === event)?.click();
  }

  openReclassifcationConfirmPopup() {
    this.dialogService.open(this.confirmReclassification, {
      context: this.ciData,
    });
  }

  reclassifyCIIntegration(data, ref) {
    this.reclassificationInProgress = true;
    this.uiArtifactsController
      .reclassifyArtifactsUsingPUT(this.artifactCI?.id)
      .subscribe(
        (res) => {
          this.reclassificationInProgress = false;
          mixpanel.track(CI_INTEGRATION_RECLASSIFY_SUCCESS);
          this.customToastService.success('CI successfully reclassified');
          ref.close();
          this.getArtifacts(this.artifactCI.ciName);
        },
        (error) => {
          this.reclassificationInProgress = false;
          mixpanel.track(CI_INTEGRATION_RECLASSIFY_SUCCESS, { error });
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'CI reclassification failed',
            error,
          );
          ref.close();
        },
      );
  }

  getCommonTableSettings(isUnclassified) {
    let commonSettings = {
      externalId: {
        title: 'External Id',
        filter: false,
        width: '10%',
        type: 'text',
        valuePrepareFunction: (value, row, cell) => {
          return value?.length > 0 ? value : 'N/A';
        },
      },
      artifactUri: {
        title: 'Artifact Uri',
        filter: false,
        type: 'custom',
        width: '15%',
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            uri: cell,
          });
        },
        renderComponent: ArtifactUriComponent,
      },
      buildDescription: {
        title: 'Description',
        width: '10%',
        filter: false,
        type: 'custom',
        valuePrepareFunction: (cell: any) => {
          return JSON.stringify({
            value: cell?.length > 0 ? cell : '',
            truncateLength: 15,
          });
        },
        renderComponent: SmartTableLongStringEllipsisComponent,
      },
      createdBy: {
        title: 'Registered By',
        width: '25%',
        filter: true,
        type: 'custom',
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            user: row.createdBy,
            label: row.createdOn,
          });
        },
        renderComponent: SmartTableTimeagoTooltipComponent,
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          const displayVal = {};
          let actionButtons = cloneDeep(this.smartTableCustomActionButtons);

          //Remove promote button for unclassified artifacts.
          //Else show the promote button only if Promotion workflow is attached to artifact and the stream/environment is not the last in promotion workflow.
          if (isUnclassified) {
            actionButtons = actionButtons.filter((btn) => btn.id !== 'promote');
          } else {
            if (
              !row?.artifactId ||
              !this.artifactCI?.promotionWorkflowId ||
              !(
                this.availableStreams.includes(row?.registrationValue) &&
                this.availableStreams.indexOf(row?.registrationValue) !==
                  this.availableStreams.length - 1
              )
            ) {
              actionButtons = actionButtons.filter(
                (actionButton) => actionButton.id !== 'promote',
              );
            }
          }

          //If the artifact is not present then remove delete button
          //These dummy entries are sent by backend, so that an empty entry can be shown to each release stream.
          if (!row?.artifactId) {
            actionButtons = actionButtons.filter(
              (actionButton) => actionButton.id !== 'delete',
            );
          }
          //Remove version history button if versioning key is not present for the artifact
          if (!row?.versioningKey) {
            actionButtons = actionButtons.filter(
              (actionButton) => actionButton.id !== 'history',
            );
          }
          displayVal['smartTableCustomActionButtons'] = actionButtons;
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance: any) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '20%',
        filter: false,
      },
    };
    return commonSettings;
  }

  getAdditionalSettings() {
    let addSettings = {
      noDataMessage: '',
      pager: {
        display: false,
        perPage: 10,
      },
      mode: 'external',
      actions: false,
    };
    return addSettings;
  }

  updateTableSettings() {
    //Update streamTable settings
    this.streamTableSettings.columns = {
      ...this.streamTableSettings.columns,
      ...this.getCommonTableSettings(false),
    };
    Object.assign(this.streamTableSettings, this.getAdditionalSettings());

    //Update envTable settings
    this.envTableSettings.columns = {
      ...this.envTableSettings.columns,
      ...this.getCommonTableSettings(false),
    };
    Object.assign(this.envTableSettings, this.getAdditionalSettings());

    //Update unclassifiedArtifacts table settings
    this.unclassifiedArtifactsTableSettings.columns = {
      ...this.unclassifiedArtifactsTableSettings.columns,
      ...this.getCommonTableSettings(true),
    };
    Object.assign(
      this.unclassifiedArtifactsTableSettings,
      this.getAdditionalSettings(),
    );
  }
}
