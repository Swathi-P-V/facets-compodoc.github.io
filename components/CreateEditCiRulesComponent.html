import { CdkDragDrop, moveItemInArray } from '@angular/cdk/drag-drop';
import { HttpClient } from '@angular/common/http';
import { Component, TemplateRef, ViewChild } from '@angular/core';
import {
  FormArray,
  FormBuilder,
  FormControl,
  FormGroup,
  Validators,
} from '@angular/forms';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { NbDialogService } from '@nebular/theme';
import mixpanel from 'mixpanel-browser';
import {
  UiArtifactRoutingRuleControllerService,
  UiDropdownsControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { convertToSentenceCase } from 'src/app/utils/string-util';
import { ARTIFACT_ROUTING_RULE_WRITE } from 'src/app/constants/permissions';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
import {
  CI_RULE_CREATE_FAIL,
  CI_RULE_CREATE_SUCCESS,
  CI_RULE_EDIT_FAIL,
  CI_RULE_EDIT_SUCCESS,
  CI_RULE_TEST_RULE_FAIL,
  CI_RULE_TEST_RULE_SUCCESS,
} from 'src/app/constants/analytics-track-events';

@Component({
  selector: 'app-create-edit-ci-rules',
  templateUrl: './create-edit-ci-rules.component.html',
  styleUrls: ['./create-edit-ci-rules.component.scss'],
})
export class CreateEditCiRulesComponent extends FacetsPage {
  //form
  ciRuleForm: FormGroup = new FormGroup({
    ruleName: new FormControl('', [
      Validators.required,
      Validators.minLength(4),
      Validators.maxLength(20),
      Validators.pattern(/[\S]/),
    ]),
    registrationType: new FormControl('', [Validators.required]),
    criteria: new FormArray([], [Validators.required]),
  });
  stackName: string;
  editId: any;
  mode: 'create' | 'edit';
  confirmedRegistrationType: string;
  ciRuleToBeEdited;
  //Loaders
  loadingCIRuleForm: boolean;
  createOrEditRequestInProgress: boolean = false;
  testRequestInProgress: boolean = false;
  //dropdowns
  registrationTypeOptions: { label: string; value: string }[] = [
    { label: 'Environment', value: 'ENVIRONMENT' },
    { label: 'Release Stream', value: 'RELEASE_STREAM' },
  ];
  releaseStreams: { label: string; value: string }[] = [];
  environments: { label: string; value: string }[] = [];
  operators: { label: string; value: string }[] = [];
  registrationValues;
  testRuleText: string = '';
  //Element/template references
  @ViewChild('confirmRegistrationChange', { static: true })
  private confirmRegistrationChange: TemplateRef<any>;

  @ViewChild('testRuleModal', { static: true })
  private testRuleModal: TemplateRef<any>;

  private rootUrl: string = window.location.origin;
  testRuleLoader: boolean = false;
  testGitRefName: string = '';
  artifactRoutingRuleWrite = ARTIFACT_ROUTING_RULE_WRITE;

  constructor(
    _route: ActivatedRoute,
    private router: Router,
    private fb: FormBuilder,
    private uiDropdownController: UiDropdownsControllerService,
    private uiStackController: UiStackControllerService,
    private uiArtifactRoutingRuleController: UiArtifactRoutingRuleControllerService,
    private dialogService: NbDialogService,
    private httpClient: HttpClient,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.loadingCIRuleForm = true;
    this.stackName = p?.stackName;
    this.mode = p?.mode;
    this.ciRuleForm.reset();
    this.criteria.clear();

    Promise.all([
      this.getAllReleaseStreams(),
      this.getAllEnvironments(this.stackName),
      this.getAllOperators(),
    ]).then(
      (promiseRes) => {
        [this.releaseStreams, this.environments, this.operators] = promiseRes;
        if (this.mode === 'create') {
          this.registrationType.setValue('RELEASE_STREAM');
          this.confirmedRegistrationType = 'RELEASE_STREAM';
          this.registrationValues = this.releaseStreams;
          this.criteria.push(this.createCriteriaForm());
          this.loadingCIRuleForm = false;
        } else {
          this.editId = p?.editId;
          this.loadCIRule(this.editId);
        }
      },
      (error) => {
        //If fetching any of the dropdown gives error, the form cannot be rendered
        //TODO
        this.loadingCIRuleForm = false;
      },
    );
  }

  //getters and setters
  get ruleName() {
    return this.ciRuleForm.get('ruleName');
  }

  get registrationType() {
    return this.ciRuleForm.get('registrationType');
  }

  get criteria(): FormArray {
    return <FormArray>this.ciRuleForm.get('criteria');
  }

  getAllReleaseStreams = (): Promise<{ label: string; value: string }[]> => {
    return new Promise((resolve, reject) => {
      this.uiDropdownController.getReleaseStreamsUsingGET().subscribe(
        (res) => {
          if (res?.length > 0) {
            let releaseStreams = res.map((element) => {
              return { label: element, value: element };
            });
            resolve(releaseStreams);
          }
          resolve([]);
        },
        (error) => {
          reject(error);
        },
      );
    });
  };

  getAllEnvironments = (
    stackName,
  ): Promise<{ label: string; value: string }[]> => {
    return new Promise((resolve, reject) => {
      this.uiStackController.getClustersUsingGET1(stackName).subscribe(
        (envs) => {
          if (envs?.length > 0) {
            let environments = envs.map((env) => {
              return { label: env?.name, value: env?.id };
            });
            resolve(environments);
          }
          resolve([]);
        },
        (error) => {
          reject(error);
        },
      );
    });
  };

  getAllOperators = (): Promise<{ label: string; value: string }[]> => {
    return new Promise((resolve, reject) => {
      this.uiArtifactRoutingRuleController.getAllOperatorsUsingGET().subscribe(
        (res) => {
          if (res?.length > 0) {
            let operators = res.map((element) => {
              return { label: convertToSentenceCase(element), value: element };
            });
            resolve(operators);
          }
          resolve([]);
        },
        (error) => {
          reject(error);
        },
      );
    });
  };

  loadCIRule(ruleId) {
    this.uiArtifactRoutingRuleController
      .getArtifactRoutingRuleUsingGET(ruleId)
      .subscribe(
        (data) => {
          //If user has changed stackName from edit page, navigate him back to ci rules listing page.
          if (data?.stackName !== this.stackName) {
            this.handleBack();
          }
          //Sort the criteria based on priority (say user has added criteria with priority 1 then 0 and 2 via API)
          data.criteria.sort((ele1, ele2) => {
            return ele1.priority - ele2.priority;
          });

          //Storing it in global variable so that when updating we can pick id from here.
          this.ciRuleToBeEdited = data;

          //populate form values, other than criteria
          this.ciRuleForm.patchValue({
            ruleName: data?.ruleName,
            registrationType: data?.registrationType,
          });
          //Make registration type field non-editable
          this.registrationType.disable();

          //Get the registration values dropdown based on registration selected.
          this.confirmedRegistrationType = this.registrationType.value;
          this.registrationValues =
            this.confirmedRegistrationType === 'RELEASE_STREAM'
              ? this.releaseStreams
              : this.environments;

          //Loop through criterias and push them to formArray
          this.ciRuleToBeEdited?.criteria?.forEach((ele, index) => {
            //If user has given some random priority values like 0,20,59 replace it with 0,1,2
            ele.priority = index;
            this.criteria.push(this.createCriteriaForm(ele));
          });
          this.loadingCIRuleForm = false;
        },
        (error) => {
          this.loadingCIRuleForm = false;
        },
      );
  }

  onRegistrationTypeChange(event) {
    try {
      //Check if registration value of any criteria is set
      this.isRegistrationValueSet(this.criteria.value)
        ? this.dialogService.open(this.confirmRegistrationChange, {})
        : this.onRegistrationTypeChangeConfirm(true, event);
    } catch (error) {}
  }

  onRegistrationTypeChangeConfirm(proceed: boolean, event, ref?) {
    if (proceed) {
      this.criteria.controls.forEach((element) => {
        element.patchValue({
          routeTo: null,
        });
      });
      this.confirmedRegistrationType = event;
      this.registrationValues =
        this.confirmedRegistrationType === 'RELEASE_STREAM'
          ? this.releaseStreams
          : this.environments;
    } else {
      this.registrationType.setValue(this.confirmedRegistrationType);
    }
    ref.close();
  }

  isRegistrationValueSet(criteria) {
    for (const element of criteria) {
      if (element['value'].length > 0) {
        return true;
      }
    }
    return false;
  }

  onAddCriteria() {
    //First create a form group,
    let newCriteriaFormFroup = this.createCriteriaForm();

    //Push this to form array
    this.criteria.push(newCriteriaFormFroup);
  }

  createCriteriaForm(criteria?) {
    return this.fb.group({
      operator: new FormControl(criteria?.operator || '', [
        Validators.required,
      ]),
      value: new FormControl(criteria?.value || '', [Validators.required]),
      routeTo: new FormControl(criteria?.routeTo || '', [Validators.required]),
      priority: criteria?.priority || this.criteria?.controls?.length,
      metadataKey: 'BRANCH_NAME',
    });
  }

  createOrEditCIRule() {
    this.createOrEditRequestInProgress = true;
    let payload = this.getCIRulePayload(
      this.mode,
      this.ciRuleForm.getRawValue(),
      this.stackName,
      this.ciRuleToBeEdited,
    );
    let httpReq;
    if (this.mode === 'create') {
      httpReq =
        this.uiArtifactRoutingRuleController.addArtifactRoutingRuleUsingPOST(
          payload,
        );
    } else {
      httpReq =
        this.uiArtifactRoutingRuleController.updateArtifactRoutingRuleUsingPUT(
          payload,
        );
    }
    httpReq.subscribe(
      (res) => {
        this.createOrEditRequestInProgress = false;
        this.customToastService.success(
          this.mode === 'create'
            ? `CI Rule ${this.ruleName.value} has been created. `
            : `CI Rule ${this.ruleName.value} has been updated.`,
        );
        this.handleBack();
        const eventName =
          this.mode === 'create'
            ? CI_RULE_CREATE_SUCCESS
            : CI_RULE_EDIT_SUCCESS;
        mixpanel.track(eventName);
      },
      (error) => {
        this.createOrEditRequestInProgress = false;
        const eventName =
          this.mode === 'create' ? CI_RULE_CREATE_FAIL : CI_RULE_EDIT_FAIL;
        mixpanel.track(eventName);
        this.onAPIErrorService.onAPIError(
          error?.error?.message || this.mode === 'create'
            ? `CI Rule ${this.ruleName.value} couldn't be created. `
            : `CI Rule ${this.ruleName.value} couldn't be updated.`,
          error,
        );
      },
    );
  }

  getCIRulePayload(mode, CIRuleFormValue, stackName, CIRule?) {
    let payload = { ...CIRuleFormValue, stackName };
    if (mode === 'edit') {
      payload['id'] = CIRule?.id;
    }
    return payload;
  }

  removeCriteria(index: number) {
    this.criteria.removeAt(index);
  }

  drop(event: CdkDragDrop<string[]>) {
    //arrange items in UI by changing form controls position
    moveItemInArray(
      this.criteria.controls,
      event.previousIndex,
      event.currentIndex,
    );
    //arrange items in form array model.
    moveItemInArray(
      this.criteria.value,
      event.previousIndex,
      event.currentIndex,
    );
    //Update the priorities
    this.criteria.controls.forEach((formGroup: FormGroup, index: number) => {
      formGroup.patchValue({ priority: index }); // Set index as the priority key's value
    });
  }

  handleBack = () => {
    this.router.navigate([
      '/',
      'stack',
      this.stackName,
      'artifacts',
      'ci-rules',
    ]);
  };

  openTestRuleModal = () => {
    this.testRuleText = '';
    this.testRuleLoader = false;
    this.testGitRefName = '';
    this.dialogService.open(this.testRuleModal, {});
  };

  triggerTestRule = (testGitRefName) => {
    this.testRuleText = '';
    this.testRuleLoader = true;

    const url = `${this.rootUrl}/cc-ui/v1/artifact-routing-rule/test?BRANCH_NAME=${testGitRefName}`;
    let payload = this.getCIRulePayload(
      this.mode,
      this.ciRuleForm.getRawValue(),
      this.stackName,
      this.ciRuleToBeEdited,
    );
    this.httpClient.post(url, payload.criteria).subscribe(
      (response: any) => {
        this.testRuleLoader = false;
        mixpanel.track(CI_RULE_TEST_RULE_SUCCESS);
        if (response.routeTo) {
          const registrationTypeText = convertToSentenceCase(
            this.registrationType.value,
          );
          const envName = this.environments.find(
            (env) => env.value === response.routeTo,
          )?.label;
          const routeToText =
            this.registrationType.value === 'ENVIRONMENT'
              ? envName || response?.routeTo
              : response?.routeTo;
          this.testRuleText = `Artifact will be routed to ${registrationTypeText} - ${routeToText}`;
          return;
        }
        this.testRuleText = `Not matched to any criteria`;
      },
      (error) => {
        mixpanel.track(CI_RULE_TEST_RULE_FAIL, { error });
        this.testRuleLoader = false;
        this.onAPIErrorService.onAPIError(
          error?.error?.message || 'Error while testing the rule',
          error,
        );
      },
    );
  };

  onTestGitRefChange(value) {
    this.testRuleText = '';
    this.testGitRefName = value;
  }
}
