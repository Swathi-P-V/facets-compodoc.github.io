import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import {
  UiCustomRoleControllerService,
  UiStackControllerService,
  UiUserControllerService,
  UiUserGroupControllerService,
} from 'src/app/cc-api/services';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { SmartTableCellWithEllipsisComponent } from './../../../components/smart-table-cell-with-ellipsis/smart-table-cell-with-ellipsis.component';
import { UserRendererComponent } from '../user-renderer.component';
import { User } from '../../../cc-api/models/user';
import { NbDialogService } from '@nebular/theme';
import {
  AbstractCluster,
  Role,
  RoleMapping,
  UserGroup,
} from 'src/app/cc-api/models';
import cloneDeep from 'lodash/cloneDeep';
import { USER_DELETE, USER_WRITE } from 'src/app/constants/permissions';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const noAccessIcon = './assets/alert-empty.svg';

const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-user-table',
  templateUrl: './user-table.component.html',
  styleUrls: ['./user-table.component.scss'],
})
export class UserTableComponent extends FacetsPage {
  fetchingUsers: boolean = false;
  users: Array<User>;
  errorOccurredFetchingUsers: boolean = false;
  emptyStateIcon = './assets/emptyUserListIcon.svg';
  enableCreateAndEditButton: boolean = false;
  primaryMsg: string = 'No users found';
  secondaryMsg: string = 'Click below to create the first user!';
  showBtn: boolean = true;
  status: 'info' | 'error' = 'info';
  deleteInProgress: boolean = false;
  usersTableData;
  roles: Array<RoleMapping>;
  allGroups: UserGroup[]; //Stores all groups.
  baseRoles: Array<RoleMapping>; //Extracts baseRoles from all roles
  groupMap: {} = {}; //Stores map of group id to group name.
  environmentMap: {} = {}; //This stores map of environment id to environment name of all environments.
  allEnvs: AbstractCluster[]; //Stores all the environemnts
  noAccessIcon = noAccessIcon;
  @ViewChild('deleteUserConfirmUI', { static: true })
  private deleteUserConfirmUI: TemplateRef<any>;
  userWritePermission = USER_WRITE;
  userSubscription: any;

  smartTableCustomActionButtons = [
    {
      id: 'edit',
      icon: 'edit-2-outline',
      toolTip: 'Edit User',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: USER_WRITE,
      onClick: (data) => {
        this.editUser(data);
      },
    },
    {
      id: 'password',
      icon: 'lock-outline',
      toolTip: 'Edit Password',
      status: 'primary',
      size: 'medium',
      appearance: 'ghost',
      action: USER_WRITE,
      onClick: (data) => {
        this.editPassword(data);
      },
    },
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete User',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: USER_DELETE,
      onClick: (data) => {
        this.delete(data);
      },
    },
  ];

  tableSettings: any = {
    columns: {
      userName: {
        title: 'User',
        filter: true,
        type: 'custom',
        renderComponent: UserRendererComponent,
      },
      rolesOrGroup: {
        title: 'User Roles/User Groups',
        filter: true,
        type: 'custom',
        valuePrepareFunction: (value, row, cell) => {
          if (row?.groupId) {
            return JSON.stringify({
              data: value,
              decorate: true,
              status: 'success',
            });
          } else {
            const customCell = value
              .filter(
                (c) => this.roles?.filter((r) => r?.roleName == c)?.length > 0,
              )
              .map(
                (c) => this.roles?.filter((r) => r?.roleName == c)[0]?.label,
              );
            return JSON.stringify({
              data: customCell,
              decorate: true,
            });
          }
        },
        renderComponent: SmartTableCellWithEllipsisComponent,
      },
      clusterIds: {
        title: 'Accessible Environments',
        filter: false,
        type: 'custom',
        valuePrepareFunction: (cell) => {
          let dataObject = true;
          let customCell = cell?.map((envId) => {
            return {
              name: this.environmentMap[envId],
              url:
                '/' +
                this.allEnvs.filter((env) => env?.id === envId)[0]?.stackName +
                '/cluster/' +
                envId +
                '/overview',
            };
          });
          if (!customCell?.length) {
            customCell = ['N/A'];
            dataObject = false;
          }
          return JSON.stringify({
            data: customCell,
            decorate: false,
            showAsLink: cell?.length,
            dataObject: dataObject,
            linkUnderline: true,
            defaultDisplay: 1,
          });
        },
        renderComponent: SmartTableCellWithEllipsisComponent,
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          //Considering userName as uniquie column here
          let formattedSmartTableCustomActionButtons = cloneDeep(
            this.smartTableCustomActionButtons,
          );
          formattedSmartTableCustomActionButtons.forEach((button) => {
            button['uniqueId'] = button.id + '_' + row.userName;
          });
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {
            smartTableCustomActionButtons:
              formattedSmartTableCustomActionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '17%',
        filter: false,
      },
    },
    noDataMessage: 'No users created. Who are you?',
    pager: {
      display: false,
      perPage: 10,
    },
    mode: 'external',
    actions: false,
  };
  baseRoleMap: {} = {};

  constructor(
    private activatedRoute: ActivatedRoute,
    private usersController: UiUserControllerService,
    private dialogService: NbDialogService,
    private router: Router,
    private userGroupControllerService: UiUserGroupControllerService,
    private uiStackControllerService: UiStackControllerService,
    private customRoleControllerService: UiCustomRoleControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.fetchingUsers = true;
    Promise.all([
      this.getAllRoles(),
      this.getAllGroups(),
      this.getAllEnvironments(),
    ]).then(
      (data) => {
        this.getAllUsers();
      },
      (error) => {
        console.log('Error fetching roles, groups and envs');
        this.fetchingUsers = false;
      },
    );
  }

  getAllRoles() {
    return new Promise((resolve, reject) => {
      this.customRoleControllerService.getAllRolesUsingGET().subscribe(
        (roles) => {
          this.roles = cloneDeep(roles);
          this.baseRoles = roles?.filter((r) => r?.baseRole);
          resolve(true);
        },
        (error) => {
          this.roles = [];
          this.baseRoles = [];
          reject(error);
        },
      );
    });
  }

  getAllGroups() {
    return new Promise((resolve, reject) => {
      this.userGroupControllerService.getAllGroupUsingGET().subscribe(
        (groups) => {
          this.allGroups = groups;
          let groupMapTemp = {};
          this.allGroups?.map((group) => {
            groupMapTemp[group?.id] = group?.groupName;
          });
          this.groupMap = groupMapTemp;
          resolve(true);
        },
        (error) => {
          reject(error);
        },
      );
    });
  }

  getAllEnvironments() {
    return new Promise((resolve, reject) => {
      this.uiStackControllerService
        .getAllClustersUsingGET({
          page: 0,
          perPage: 200,
          sortBy: 'ID',
        })
        .subscribe(
          (envs) => {
            let environmentMapTemp = {};
            this.allEnvs = envs.content;
            envs.content.forEach((cluster) => {
              //Create clusterId : clusterName map in a temp object and then assign it to actual variable.
              //This is to avoid exceptions in child component(ie; user-group-table) when trying to access environmentMap.
              environmentMapTemp[cluster?.id] = cluster?.name;
            });
            this.environmentMap = environmentMapTemp;
            resolve(true);
          },
          (error) => {
            reject(error);
          },
        );
    });
  }

  getAllUsers() {
    this.usersController.getAllUsersUsingGET().subscribe(
      (x) => {
        this.users = x;
        this.users?.sort((a, b) => a?.userName?.localeCompare(b?.userName));

        this.users?.map((user) => {
          //filter all deleted clusters
          if (user?.clusterIds) {
            let clusters = user?.clusterIds?.filter((clusters) =>
              this.environmentMap.hasOwnProperty(clusters),
            );
            user.clusterIds = clusters;
          }

          //Set groupName if groupId exists
          if (user?.groupId) {
            user['groupName'] = [this.groupMap[user?.groupId]];
            user['rolesOrGroup'] = [this.groupMap[user?.groupId]];
          } else {
            user['rolesOrGroup'] = user.roles;
          }
        });
        this.usersTableData = this.users;
        this.fetchingUsers = false;
      },
      (error) => {
        this.errorOccurredFetchingUsers = true;
        this.secondaryMsg =
          error?.error?.message || ' There is some error fetching users';
        this.primaryMsg = '';
        this.showBtn = false;
        this.emptyStateIcon = errorIcon;
        this.status = 'error';
        this.fetchingUsers = false;
      },
    );
  }

  createUser() {
    this.router.navigate(['../user/create'], {
      relativeTo: this.activatedRoute,
    });
  }

  editUser(rowData) {
    this.router.navigate(['../user/edit', rowData.id], {
      relativeTo: this.activatedRoute,
    });
  }

  editPassword(rowData) {
    this.router.navigate(['../user/changePassword', rowData.id], {
      relativeTo: this.activatedRoute,
    });
  }

  delete(rowdata) {
    this.dialogService.open(this.deleteUserConfirmUI, { context: rowdata });
  }

  deleteUser(ref, userData) {
    try {
      this.deleteInProgress = true;
      this.usersController.deleteUserUsingDELETE(userData.id).subscribe(
        (data) => {
          this.deleteInProgress = false;
          ref.close();
          this.customToastService.success('User successfully deleted');
          this.getAllUsers();
        },
        (error) => {
          this.deleteInProgress = false;
          ref.close();
          this.onAPIErrorService.onAPIError(
            'User delete failed. ' + error?.error?.message,
            error,
          );
        },
      );
    } catch (err) {
      this.deleteInProgress = false;
      ref.close();
      this.customToastService.error('User delete failed');
    }
  }
}
