import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { UiAccountsControllerService } from 'src/app/cc-api/services';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { AzureAccount } from 'src/app/cc-api/models';
import { getToastSubTexts } from 'src/app/utils/toast.utils';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params } from '@angular/router';

@Component({
  selector: 'app-azure-account-create-edit',
  templateUrl: './azure-account-create-edit.component.html',
  styleUrls: ['./azure-account-create-edit.component.scss'],
})
export class AzureAccountCreateEditComponent extends FacetsPage {
  @Input() mode: 'create' | 'edit' = 'create';
  @Input() passedRef: any;
  @Input() accountDetails: AzureAccount;

  @Output() afterSaveAccount: EventEmitter<any> = new EventEmitter<any>();
  azureAccountForm: FormGroup = this.formBuilder.group({
    name: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    subscriptionId: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    tenantId: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    clientId: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    clientSecret: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
  });

  readonly ACCOUNTS_WRITE = 'ACCOUNTS_WRITE';
  saveAccountInProgress: boolean = false;
  verifyInProgress: boolean = false;

  constructor(
    private activatedRoute: ActivatedRoute,
    private formBuilder: FormBuilder,
    private accountControllerService: UiAccountsControllerService,
    private customToast: CustomToastService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    if (this.mode === 'edit') {
      this.azureAccountForm.patchValue({
        name: this.accountDetails?.name,
        subscriptionId: this.accountDetails?.subscriptionId,
        tenantId: this.accountDetails?.tenantId,
        clientId: this.accountDetails?.clientId,
        clientSecret: this.accountDetails?.clientSecret,
      });
    }
  }

  get name() {
    return this.azureAccountForm.get('name');
  }
  get subscriptionId() {
    return this.azureAccountForm.get('subscriptionId');
  }
  get tenantId() {
    return this.azureAccountForm.get('tenantId');
  }
  get clientId() {
    return this.azureAccountForm.get('clientId');
  }
  get clientSecret() {
    return this.azureAccountForm.get('clientSecret');
  }

  getAzureAccountVariables() {
    let accountName =
      this.azureAccountForm?.controls?.name?.value?.trim() || '';
    let subscriptionId =
      this.azureAccountForm?.controls?.subscriptionId?.value?.trim() || '';
    let tenantId =
      this.azureAccountForm?.controls?.tenantId?.value?.trim() || '';
    let clientId =
      this.azureAccountForm?.controls?.clientId?.value?.trim() || '';
    let clientSecret =
      this.azureAccountForm?.controls?.clientSecret?.value?.trim() || '';
    return {
      accountName,
      subscriptionId,
      tenantId,
      clientId,
      clientSecret,
    };
  }

  getRequestPayload() {
    const { accountName, subscriptionId, tenantId, clientId, clientSecret } =
      this.getAzureAccountVariables();
    let accountParams: AzureAccount = {
      name: accountName,
      subscriptionId: subscriptionId,
      tenantId: tenantId,
      clientId: clientId,
      clientSecret: clientSecret,
      accountType: 'CLOUD',
      provider: 'AZURE',
    };

    return accountParams;
  }
  createEditAzureAccount() {
    let requestPayload = this.getRequestPayload();
    const isCreateMode = this.mode === 'create';
    this.saveAccountInProgress = true;

    let httpReq;
    if (isCreateMode) {
      httpReq =
        this.accountControllerService.createAzureAccountUsingPOST(
          requestPayload,
        );
    } else if (this.mode === 'edit') {
      httpReq = this.accountControllerService.updateAzureAccountUsingPUT({
        id: this.accountDetails?.id,
        azureAccountRequest: {
          ...requestPayload,
          id: this.accountDetails?.id,
          createdBy: this.accountDetails?.createdBy,
          creationDate: this.accountDetails?.creationDate,
        },
      });
    }

    let { successSubText, failureSubText } = getToastSubTexts(this.mode);
    httpReq.subscribe(
      (data) => {
        this.customToast.success(
          `Azure account ${data?.name} ${successSubText} successfully`,
        );
        this.afterSaveAccount.emit(data);
        this.passedRef.close();
        this.saveAccountInProgress = false;
        this.azureAccountForm.reset();
      },
      (err) => {
        this.saveAccountInProgress = false;
        this.customToast.error(
          err?.error?.message || ` Failed to ${failureSubText} Azure account`,
        );
      },
    );
  }

  verifyAccount() {
    this.verifyInProgress = true;
    const requestPayload = this.getRequestPayload();
    this.accountControllerService
      .validateAzureAccountUsingPOST(requestPayload)
      .subscribe(
        (data) => {
          this.verifyInProgress = false;
          this.customToast.success(`This is a verified account`);
        },
        (error) => {
          this.verifyInProgress = false;
          this.customToast.error(
            error?.error?.message || `Failed to verify account`,
          );
        },
      );
  }
}
