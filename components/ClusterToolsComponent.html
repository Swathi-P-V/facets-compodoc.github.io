import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { UiDeploymentControllerService } from '../../../cc-api/services/ui-deployment-controller.service';
import { ClusterState } from '../../../enums/cluster-state';
import {
  RUNNING,
  SCALE_DOWN_FAILED,
  SCALE_UP_FAILED,
} from '../../../constants/deployments';
import { SCALE_DOWN } from '../../../constants/releaseTypes';
const emptyStateLogo = './assets/empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-cluster-tools',
  templateUrl: './cluster-tools.component.html',
  styleUrls: ['./cluster-tools.component.scss'],
})
export class ClusterToolsComponent extends FacetsPage {
  clusterId: any;
  stackName: string;
  showEmptyState: boolean = false;
  clusterState: string;
  secondaryMsg: string;
  emptyStateLogo = emptyStateLogo;
  errorIcon = errorIcon;
  primaryMsg: string =
    'This page is available when environment is in running state.';
  buttonText: string = 'Manage Your Environment';
  loading: boolean = true;
  showBtn: boolean = true;
  status: 'info' | 'error' = 'info';

  constructor(
    private activateRoute: ActivatedRoute,
    private uiDeploymentControllerService: UiDeploymentControllerService,
    private router: Router,
  ) {
    super(activateRoute);
  }
  tabs = [
    {
      title: 'Kubernetes',
      route: ['./k8s'],
      responsive: true,
    },
    {
      title: 'Grafana',
      route: ['./grafana'],
      responsive: true,
    },
    {
      title: 'External Tools',
      route: ['./external'],
      responsive: true,
    },
  ];

  OnPageLoad(p: Params): void {
    this.loading = true;
    this.showEmptyState = false;
    this.uiDeploymentControllerService
      .getClusterStateUsingGET(p.clusterId)
      .subscribe(
        (t) => {
          this.clusterState = t;
          this.clusterId = p.clusterId;
          this.stackName = p.stackName;
          if (
            [RUNNING, SCALE_DOWN, SCALE_DOWN_FAILED, SCALE_UP_FAILED].includes(
              this.clusterState,
            )
          ) {
            this.showEmptyState = false;
          } else {
            this.showEmptyState = true;
            this.secondaryMsg =
              'Environment is in ' +
              (ClusterState[this.clusterState] || this.clusterState) +
              ' state!';
          }
          this.loading = false;
        },
        (error) => {
          this.loading = false;
          this.secondaryMsg =
            error?.error?.message || 'Error in fetching the environment';
          this.primaryMsg = '';
          this.showBtn = false;
          this.loading = false;
          this.emptyStateLogo = errorIcon;
          this.showEmptyState = true;
          this.status = 'error';
        },
      );
  }

  navigate() {
    let urlToNav =
      '/' + this.stackName + '/cluster/' + this.clusterId + '/releases/';
    this.router.navigate([urlToNav]);
  }
}
