  getDeliveryPipelineInfo() {

      status: 'success',

    return new Promise((resolve, reject) => {

        (envs) => {
          let allEnvs = envs;
          this.deliveryPipelineService
            .getDeliveryPipelineUsingGET(this.stackName)
            .subscribe(
              (pipelineData: any) => {
                this.fetchingDeliveryPipeline = false;
                if (pipelineData?.length > 0) {
                  let envIdNameMapping = {};
                  pipelineData.forEach((pipeline) => {
                    if (pipeline?.clusterName) {
                      envIdNameMapping[pipeline?.clusterId] =
                        pipeline?.clusterName;
                    }
                  });
                  this.setParentEnvContent(
                    pipelineData,
                    allEnvs,
                    envIdNameMapping,
                  );
                  this.setChildEnvContent(
                    pipelineData,
                    allEnvs,
                    envIdNameMapping,
                    tableCellEllipsisData,
                  );
                }
                resolve(true);
              },
              (error: any) => {
                this.fetchingDeliveryPipeline = false;
                reject(false);
                tableCellEllipsisData['data'] = [];
                tableCellEllipsisData['extraData'] = [];
                this.childEnvData = JSON.stringify(tableCellEllipsisData);
              },
            );
        },
        (err) => {
          this.fetchingDeliveryPipeline = false;
          reject(false);
        },
      );
    });

          id: pipeline?.clusterId,
          url:
            '/' +
            this.stackName +
            '/cluster/' +
            pipeline?.clusterId +
            '/releases',
          disableLink: allEnvs.find((env) => env.id === pipeline?.clusterId)
            ? false
            : true,
          linkTooltip: 'Access is forbidden',

      }
    });

  alphaResourcesSet: Set<string>;
  createResourceOptions() {

      (resource) =>
        (this.supportedResourceTypes.includes(resource?.resourceType) ||
          resource?.alpha) &&
        resource?.resourceType !== 'UNKNOWN',
    );
    this.alphaResourcesSet = new Set(
      filteredResources.map(
        (resource) => resource.resourceType + ':' + resource.resourceName,
      ),
    );
    filteredResources.forEach((resource) => {
      if (Object.keys(tempResourcesObj).includes(resource?.resourceType)) {
        tempResourcesObj[resource?.resourceType].push(resource?.resourceName);
      } else {
        tempResourcesObj[resource?.resourceType] = [resource?.resourceName];
      }
    });
    if (Object.keys(tempResourcesObj)?.length > 0) {
      Object.keys(tempResourcesObj).forEach((key) => {
        if (tempResourcesObj[key]?.length > 0) {
          this.resources.push({
            label: key,
            value: key,
            expand: true,
            checked: false,
            children: this.sortOptionsService
              .sortList(tempResourcesObj[key])
              .map((resource) => {
                return {
                  label: resource,
                  value: resource,
                  checked: false,
                };
              }),
          });
          //this.resources[key] = this.sortOptionsService.sortList(res[key]);
        }
      });
      if (this.resources?.length > 0) {
        this.resources = this.sortOptionsService.sortList(
          this.resources,
          'label',
        );
      }
    }
  }

  //Add the compareUrl and commit path to each deployment
  //CompareUrl is used compare the blueprint version of the successful deployments with the latest signedoff deployment.
  //So if environment does not have current signed off version nothing to compare with. So set compareUrl to null

  addCompareUrlToDeployments({ deployments }) {
    return deployments.map((deployment) => {
      if (deployment['allowSignoff'] && this.currentSignedOffDeployment) {
        if (this.stack?.vcs === 'GITHUB' || this.stack?.vcs === 'GITLAB') {
          deployment['compareUrl'] =
            this.stack.vcsUrl.replace('.git', '') +
            '/compare/' +
            this.currentSignedOffDeployment?.stackVersion +
            '...' +
            deployment.stackVersion;
        } else if (this.stack?.vcs === 'BITBUCKET') {
          deployment['compareUrl'] =
            this.stack.vcsUrl.replace('.git', '') +
            '/compare/' +
            deployment.stackVersion +
            '%0' +
            this.currentSignedOffDeployment?.stackVersion;
        }
      } else {
        deployment['compareUrl'] = null;
      }
      const commitPath =
        this.stack?.vcs === 'GITHUB' || this.stack?.vcs === 'GITLAB'
          ? 'commit'
          : 'commits';
      deployment['stackUrl'] =
        this.stack.vcsUrl.replace('.git', '') +
        '/' +
        commitPath +
        '/' +
        deployment.stackVersion;

      return deployment;
    });
  }
  parseDeploymentsData(deployments) {
    let clonedDeployments = cloneDeep(deployments);

    //If environment has a current signed off deployment then allowSignOff is set to true if deployment
    //is successful, has blueprint and IaC version, is not already signedOff, is of type RELEASE and it is a deployment after the current signed off deployment.
    //Further if environment does not have a current signedOff deployment(ex: a recently launched env)
    //then allowSignOff is set to true for all successful deployment of type RELEASE
    clonedDeployments.forEach((deployment) => {
      if (this.currentSignedOffDeployment) {
        deployment['allowSignoff'] =
          deployment.status === 'SUCCEEDED' &&
          deployment.stackVersion?.length > 0 &&
          deployment.tfVersion?.length > 0 &&
          !deployment.signedOff &&
          deployment.releaseType === 'RELEASE' &&
          deployment.createdOn > this.currentSignedOffDeployment?.createdOn;
      } else {
        deployment['allowSignoff'] =
          deployment.status === 'SUCCEEDED' &&
          deployment.stackVersion?.length > 0 &&
          deployment.tfVersion?.length > 0 &&
          !deployment.signedOff &&
          deployment.releaseType === 'RELEASE';
      }
    });

    clonedDeployments = this.addCompareUrlToDeployments({
      deployments: clonedDeployments,
    });
    return clonedDeployments;
  }

          const createdOn = moment(d?.createdOn);

          diffHours =
            this.duration?.hours() < 10
              ? '0' + this.duration?.hours()
              : this.duration?.hours();

            this.duration?.minutes() < 10
              ? '0' + this.duration?.minutes()
              : this.duration?.minutes();
          diffSeconds =
            this.duration?.seconds() < 10
              ? '0' + this.duration?.seconds()
              : this.duration?.seconds();
        }
        this.dialogService.open(dialog, {
          context: {
            deploymentId,
            changes: d.changesApplied,
            appDeployments: d.appDeployments,
            errors: d.errorLogs,
            overrideBuildSteps: d.overrideBuildSteps,
            response: d,
            diffMins: diffMins,
            diffSeconds: diffSeconds,

          ref.close();
          this.initializePage();
        },
        (error) => {
          this.serverRequest = false;
          this.onAPIErrorService.onAPIError(
            error?.error?.message || 'Failed to delete release schedule',
            error,
          );
          ref.close();
        },
      );
  }

      return false;

      this.fetchingClusterState ||
      this.fetchingDeployments ||

      this.settingButtonState ||
      this.fetchingDeliveryPipeline
    );
  }

  urlBuilder(base, vcs, branch, folder, fileName) {
    let finalUrl: string = base;
    if (vcs === 'GITHUB') {
      finalUrl += '/tree/' + branch;
    } else if (vcs === 'BITBUCKET') {
      finalUrl += '/src/' + branch;
    } else if (vcs === 'GITLAB') {
      finalUrl += '/-/tree/' + branch;
    }
    if (folder) {
      finalUrl += '/' + folder;
    }
    if (fileName) {
      finalUrl += '/' + fileName;
    }
    return finalUrl;
  }

  navigate(err) {
    this.linkLoading = true;
    this.stackService.getStackUsingGET(this.stackName).subscribe((stack) => {
      let baseUrl;
      if (stack?.vcs === 'GITHUB') {
        baseUrl = stack?.vcsUrl?.substring(0, stack?.vcsUrl?.length - 4);
      } else {
        baseUrl = stack?.vcsUrl;
      }
      let navigateUrl = this.urlBuilder(
        baseUrl,
        stack?.vcs,
        stack?.branch,
        err.directory,
        err.filename,
      );
      this.linkLoading = false;
      window.open(navigateUrl, '_blank');
    });
  }

    window.open(url, '_blank').focus();

    });
  }

  pauseOrResumeReleases(isPaused: boolean, ref?) {
    this.pauseOrResumeInProgress = true;
    this.commonClusterController
      .pauseReleaseUsingPOST({
        clusterId: this.clusterId,
        pauseReleaseRequest: {
          clusterId: this.clusterId,
          pauseReleases: isPaused,
        },
      })
      .subscribe(
        (data) => {
          this.pauseOrResumeInProgress = false;
          ref?.close();
          if (isPaused) {
            this.customToastService.success(
              'All releases for this environment is paused successfully',
            );
            this.isReleasePaused = true;
            this.setActionButtonsStatus();
          } else {
            this.customToastService.success(
              'Releases for this environment is resumed successfully',
            );
            this.isReleasePaused = false;
            this.setActionButtonsStatus();
          }
        },
        (error) => {
          this.errorHandler(error);
          this.pauseOrResumeInProgress = false;
          ref?.close();
        },
      );
  }

  setActionButtonsStatus() {
    this.setReleaseDisabledStatus();
    const actionsStatusMap = getReleasePageActionsStatusMap({
      clusterState: this.clusterState,
      clusterDetails: this.clusterDetails,
      isReleasePaused: this.isReleasePaused,
      tfVersionMap: this.tfVersionMap,
      latestRelease: this.latestRelease,
    });
    this.settingButtonState = this.showReleaseLoader;

    let buttons = cloneDeep(this.releaseActionButtons);
    const pageButtons = buttons.filter(
      (button) =>
        ['secondary', 'primary'].includes(button?.buttonType) &&
        button?.id !== 'destroy' &&
        actionsStatusMap[button?.id]?.visible,
    );
    const buttonIdTooltipMap = {
      'scale-up':
        'Scales up deployments and statefulsets, and enables cronjobs in the cluster.',
      'scale-down':
        'Scales down deployments and statefulsets, and disables cronjobs in the cluster',
    };
    this.releaseActionButtons = buttons?.map(
      ({ buttonType, nbTooltip, ...button }) => {
        if (button.id === 'destroy') {
          buttonType = pageButtons?.length >= 2 ? 'tertiary' : 'secondary';
        }
        if (['scale-up', 'scale-down'].includes(button.id)) {
          nbTooltip = this.clusterDetails?.baseClusterId
            ? 'Action not permitted in dependent environment'
            : buttonIdTooltipMap[button.id];
        }
        return {
          ...button,
          hidden: !actionsStatusMap[button?.id]?.visible,
          disabled: actionsStatusMap[button?.id]?.disable,
          nbTooltip,
          buttonType,
        };
      },
    );
    this.settingButtonState = false;
  }

  }

  latestReleaseLoader: boolean = false;
  countdownCompleted(field) {
    this.latestReleaseLoader = true;
    if (
      this.tableLatestRelease.length &&
      [IN_PROGRESS, STARTED].includes(this.tableLatestRelease[0]?.status)
    ) {
      this.latestReleaseLoader = false;
    }
    this.releaseTriggeredFor.add(field?.executionContext);
    Promise.allSettled([
      this.getClusterState(this.clusterId),
      this.getClusterDetails(this.clusterId),
      this.getDeploymentOverview(),
    ]).then((data) => {
      this.clusterExecutionStates = processExecutionTime(
        this.executionTime,
        this.clusterState,
      );
      this.latestReleaseLoader = false;
    });
  }

  isResourceService(stackName: string, resourceName: string) {
    let resourceMap =
      this.contextUpdatesService.getResourceMapForBlueprint(stackName);

    );
  }

    this.releaseScheduleCrudService

        this.releaseScheduleData?.[0],
        this.clusterId,
        ref,
      )
      .then(
        (res: any) => {
          this.pauseResumeScheduleInProgress = false;
          this.initializePage();
        },
        (error) => {
          this.pauseResumeScheduleInProgress = false;
        },
      );
  }

  getReleaseScheduleData() {
    return new Promise((resolve, reject) => {
      this.commonClusterController
        .getClusterSchedulesUsingGET(this.clusterId)
        .subscribe(
          (data) => {
            //Filter out only RELEASE releasetype object: Returns an array
            this.releaseScheduleData = data.filter(
              (schedule) => schedule.releaseType === 'RELEASE',
            );
            resolve(true);
          },
          (error) => {
            resolve(false);
          },
        );
    });
  }

  }
