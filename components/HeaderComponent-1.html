import {
  Component,
  Input,
  Output,
  EventEmitter,
  OnInit,
  ViewChild,
  TemplateRef,
  OnChanges,
  SimpleChanges,
} from '@angular/core';
import { ActivatedRoute, Params, Router, NavigationEnd } from '@angular/router';
import { FacetsPage } from '../../../facets/facets-page/facets-page.component';
import {
  UiAccountsControllerService,
  UiCoderControllerService,
} from 'src/app/cc-api/services';
import { NbDialogService } from '@nebular/theme';
@Component({
  selector: 'app-blueprint-header',
  templateUrl: './header.component.html',
  styleUrls: ['./header.component.scss'],
})
export class HeaderComponent extends FacetsPage implements OnChanges {
  @ViewChild('coderDialog') coderDialog: TemplateRef<any>;
  @Input() blueprintName: string = '';
  @Input() zoomValue: number;
  @Input() mode: 'GRAPH' | 'TABLE' = 'GRAPH';
  @Input() blueprintVscType: 'GITHUB' | 'GITLAB' | 'BITBUCKET';
  @Input() primaryCloud: 'AWS' | 'AZURE' | 'GCP' | 'KUBERNETES';

  @Output() byZoom = new EventEmitter();
  @Output() byActionButtonClick = new EventEmitter();
  @Output() byModeChange = new EventEmitter();
  url: string = '';
  resourcePages: boolean = true;

  private blueprint: string = '';
  showCoderButton: boolean = false;

  coderButtonDisabledMessage: string;
  loadingCoderButtonStatus: boolean = false;

  constructor(
    private router: Router,
    private activatedRoute: ActivatedRoute,
    private uiAccountsControllerService: UiAccountsControllerService,
    private dialogService: NbDialogService,
    private uiCoderControllerService: UiCoderControllerService,
  ) {
    super(activatedRoute);
  }

  onZoomAction(zoomType) {
    this.byZoom.emit(zoomType);
  }

  OnPageInit(): void {
    this.url = this.router.url;
    this.resourcePages = this.url.includes('resource-list');
    this.router.events.subscribe((val) => {
      // see also
      if (val instanceof NavigationEnd) {
        this.url = val?.urlAfterRedirects;
        this.resourcePages = this.url.includes('resource-list');
      }
    });
  }
  OnPageLoad(params: Params) {
    this.blueprint = params?.stackName;
  }

  ngOnChanges(changes: SimpleChanges): void {
    if (changes?.blueprintVscType?.currentValue) {
      this.setCoderVisibilityStatus();
    }
  }

  setCoderVisibilityStatus = () => {
    this.showCoderButton = false;
    this.loadingCoderButtonStatus = true;
    this.uiAccountsControllerService
      .getAccountsByTypeUsingGET('CODER')
      .subscribe(
        (response) => {
          if (response?.length && this.blueprintVscType === 'GITHUB') {
            this.showCoderButton = true;
            this.setCoderDisabledStatus();
          }
        },
        () => {
          this.showCoderButton = false;
          this.loadingCoderButtonStatus = false;
        },
      );
  };

  setCoderDisabledStatus = () => {
    this.coderButtonDisabledMessage = '';
    this.uiCoderControllerService
      .identifyCoderLaunchEligibleUsingGET(this.blueprint)
      .subscribe(
        ({ coderAccountPresent, userVCSTokenPresent }) => {
          if (!coderAccountPresent) {
            this.coderButtonDisabledMessage =
              'Please link your coder.com account in Account Management under Settings.';
          } else if (!userVCSTokenPresent) {
            this.coderButtonDisabledMessage =
              'Kindly map your GitHub Personal Access Token in the profile settings.';
          }
          this.loadingCoderButtonStatus = false;
        },
        () => {
          this.loadingCoderButtonStatus = false;
          this.coderButtonDisabledMessage =
            'Unable to fetch coder account details.';
        },
      );
  };

  onActionButtonClick(action: 'addedResource' | 'bulkUpload', extras?: any) {
    this.byActionButtonClick.emit({ action: action, extras: extras });
  }

  onModeChange(mode: 'GRAPH' | 'TABLE') {
    if (mode !== this.mode) {
      this.mode = mode;
      this.byModeChange.emit(mode);
    }
  }

  openCoder() {
    this.dialogService.open(this.coderDialog);
  }
}
