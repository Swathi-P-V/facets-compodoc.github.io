  constructor(
    private _route: ActivatedRoute,
    private router: Router,
    private cloudCostExplorer: UiCloudCostExplorerControllerService,
    private stackController: UiStackControllerService,
    private customToastService: CustomToastService,

    super(_route);
  }

  // Getters
  get selectedEnvironments() {
    return this.filterForm.get('selectedEnvironments');
  }

  get selectedDateRange() {
    return this.filterForm.get('selectedDateRange');
  }

  OnPageLoad(p: Params): void {
    this.stackName = p?.stackName;
    this.showCostExplorer = false;
    this.errorMessage = '';
    this.asyncOnPageLoad();
  }

  resetPropsOnPageLoad() {
    this.filterForm.patchValue({
      selectedDateRange: 7,
    });
    this.mode = 'TOTAL_COST';
    this.breakdownChartOptions = {};
    this.trendChartOptions = {};
    this.costExplorerActionButtons.forEach((btn) => {
      btn.disabled = true;
    });
    this.serviceLevelCostData = [];
  }

  async asyncOnPageLoad() {
    this.fetchingIfCostExplorerEnabled = true;
    await this.checkIfAwsCostExplorerEnabled();
    await this.getEnvironmentsByStack(this.stackName);

    if (!this.isAwsCostExplorerEnabled) {
      const azureEnvs = this.environmentDropdown.filter(
        (environment) => environment?.cloud === 'AZURE',
      );
      this.showCostExplorer = azureEnvs?.length > 0;
      this.fetchingIfCostExplorerEnabled = false;
    } else {
      this.showCostExplorer = this.isAwsCostExplorerEnabled;
      this.fetchingIfCostExplorerEnabled = false;
    }

    if (this.showCostExplorer) {
      this.fetchingBreakdownData = true;
      this.fetchingTrendData = true;
      this.fetchingServiceCost = true;
      this.costExplorerActionButtons = [...this.getCostExplorerActionButton()];
      this.resetPropsOnPageLoad();

      //For selected dateRange get start and end date
      if (this.supportedEnvsList?.length > 0) {
        const { startDate, endDate } = this.getStartAndEndDate(
          this.selectedDateRange.value,
        );

        const data = await this.getCloudCost(
          this.stackName,
          startDate,
          endDate,
          this.selectedEnvironments?.value,
        );
        // Cloud Cost explorer - stack and environment level
        if (data['dailyCost']?.length > 0) {
          this.costExplorerActionButtons.forEach((btn) => {
            btn.disabled = false;
          });
          this.constructDataForTotalTrend({
            apiResponse: data,
            startDate,
            endDate,
            range: this.selectedDateRange.value,
          });
          this.constructDataForBreakdownChart(data);
        } else {
          this.fetchingBreakdownData = false;
          this.fetchingTrendData = false;
        }
        // Cloud Cost explorer - Resources
        this.setServiceCostTableData(data);
      } else {
        this.fetchingBreakdownData = false;
        this.fetchingTrendData = false;
        this.fetchingServiceCost = false;
      }
    }
  }

  checkIfAwsCostExplorerEnabled(): Promise<boolean> {
    this.isAwsCostExplorerEnabled = false;
    return new Promise((resolve, reject) => {
      this.cloudCostExplorer.isAwsCostExplorerEnabledUsingGET().subscribe(
        (data) => {
          this.isAwsCostExplorerEnabled = true;
          resolve(data);
        },
        (error) => {
          this.isAwsCostExplorerEnabled = false;
          resolve(false);
        },
      );
    });
  }

    return [

        buttonType: 'primary',
        icon: 'download-outline',

  handleActions = (event) => {
    this.costExplorerActionButtons?.find((btn) => btn?.id === event)?.onclick();
  };

  downloadCostBreakdownAsCSV() {
    const { startDate, endDate } = this.getStartAndEndDate(
      this.selectedDateRange.value,
    );
    let filename = constructFileName(this.stackName, startDate, endDate);
    let clusterBreakdownData = this.cloudCostRawData?.dailyCost?.map(
      (cluster) => {
        return {
          ...cluster,
          stackName: this.cloudCostRawData.stackName,
          unit: this.cloudCostRawData.unit,
        };
      },
    );

    let headerList = [
      'SL.No',
      'Environment Name',
      'Environment ID',
      'Blueprint',
      'Total Cost',
      'Unit',
    ];
    let jsonKeys = ['clusterName', 'clusterId', 'stackName', 'total', 'unit'];

    const csvData = ConvertToCSV(clusterBreakdownData, headerList, jsonKeys);

    generateCSVFile(csvData, filename);
  }

  getEnvironmentsByStack(stackName) {
    this.fetchingEnvironments = true;
    //Reset dropdown and dropdown selected values
    this.environmentDropdown = [];
    this.filterForm.patchValue({
      selectedEnvironments: [],
    });
    this.supportedEnvsList = [];
    return new Promise((resolve, reject) => {
      this.stackController.getClustersUsingGET1(stackName).subscribe(
        (data) => {
          // If environments are present then construct dropdown and get list of supported cloud environments.
          if (data?.length > 0) {
            this.environmentDropdown = data?.map((cluster) => {
              if (this.supportedClouds?.includes(cluster?.cloud)) {
                this.clusterList.push(cluster);
                this.supportedEnvsList.push(cluster?.id);
              }
              const isCloudSupported = this.supportedClouds?.includes(
                cluster?.cloud,
              );
              return {
                value: cluster?.id,
                label: cluster?.name,
                disabled: !isCloudSupported,
                cloud: cluster?.cloud,
              };
            });
            // Sort the environments by name
            this.environmentDropdown.sort((env1, env2) =>
              env1?.label?.localeCompare(env2?.label),
            );
            this.additionalInfo = pushDependentNote(
              this.clusterList,
              this.additionalInfo,
              this.dependentEnvNote,
            );
            this.supportedEnvsList?.length > 0
              ? this.filterForm.patchValue({
                  selectedEnvironments: this.supportedEnvsList,
                })
              : this.setErrorOrEmptyState('empty');
          } else {
            this.setErrorOrEmptyState('empty');
          }
          this.fetchingEnvironments = false;
          resolve(true);
        },
        (error) => {
          this.setErrorOrEmptyState('error', error?.error?.message);
          this.fetchingEnvironments = false;
          resolve(false);
        },
      );
    });
  }

  getCloudCost(stackName, startDate?, endDate?, clusterIds?) {
    this.cloudCostRawData = {};
    this.cloudCostRoundedData = {};
    this.percentageChange = null;
    return new Promise((resolve, reject) => {
      this.cloudCostExplorer
        .getDailyCloudCostUsingGET({
          start: startDate,
          stackName: stackName,
          end: endDate,
          clusterIds: clusterIds,
        })
        .subscribe(
          (response) => {
            this.cloudCostRawData = response;
            this.cloudCostRoundedData = this.getRoundedTotalCost(response);
            this.percentageChange = this.calculatePercentageChangeForDuration(
              this.cloudCostRoundedData?.total,
              this.cloudCostRoundedData?.previousPeriodTotal,
            );
            setTotalCostUnit(this.cloudCostRawData?.unit);
            resolve(response);
          },
          (error) => {
            //TODO - Show error state
            this.setErrorOrEmptyState(
              'error',
              error?.error?.message || 'Some error while fetching data',
            );
            reject(error);
          },
        );
    });
  }

  getRoundedTotalCost(actualCostData) {
    return {
      total: parseFloat(actualCostData?.total?.toFixed(DECIMAL_PLACES)),
      previousPeriodTotal: parseFloat(
        actualCostData?.previousPeriodTotal?.toFixed(DECIMAL_PLACES),
      ),
    };
  }

  onDateRangeChange(event) {
    this.refreshChartDataOnFilterChange();
  }

  onEnvironmentChange(event) {
    this.refreshChartDataOnFilterChange();
  }

  onModeChange(mode) {
    this.mode = mode;
    this.refreshChartDataOnModeChange();
  }

  async refreshChartDataOnFilterChange() {
    // Initiate both chart widgets loader and table loader
    this.fetchingTrendData = true;
    this.fetchingBreakdownData = true;
    this.fetchingServiceCost = true;
    // Reset both chart options
    this.breakdownChartOptions = {};
    this.trendChartOptions = {};
    let filteredEnvs = this.clusterList.filter((env) =>
      this.selectedEnvironments?.value?.includes(env?.id),
    );
    this.additionalInfo = pushDependentNote(
      filteredEnvs,
      this.additionalInfo,
      this.dependentEnvNote,
    );
    //For selected dateRange get start and end date
    const { startDate, endDate } = this.getStartAndEndDate(
      this.selectedDateRange.value,
    );

    // Fetch the graph data for selected filter values.
    let apiResponse = await this.getCloudCost(
      this.stackName,
      startDate,
      endDate,
      this.selectedEnvironments?.value,
    );
    //Construct the breakdown chart data
    if (apiResponse['dailyCost']?.length > 0) {
      this.constructDataForBreakdownChart(apiResponse);

      // Construct the trend chart data based on selected mode
      let args = {
        apiResponse,
        startDate,
        endDate,
        range: this.selectedDateRange?.value,
      };
      this.mode === 'TOTAL_COST'
        ? this.constructDataForTotalTrend(args)
        : this.constructDataForBreakdownTrend(args);
    } else {
      this.costExplorerActionButtons.forEach((btn) => {
        btn.disabled = true;
      });
      this.fetchingBreakdownData = false;
      this.fetchingTrendData = false;
    }

    // Cost exporer - services level
    this.setServiceCostTableData(apiResponse);
  }

  refreshChartDataOnModeChange() {
    // Initiate chart widgets loader
    this.fetchingTrendData = true;

    // Reset chart options
    this.trendChartOptions = {};

    if (this.cloudCostRawData?.dailyCost?.length > 0) {
      //For selected dateRange get start and end date
      const { startDate, endDate } = this.getStartAndEndDate(
        this.selectedDateRange.value,
      );

      let args = {
        apiResponse: this.cloudCostRawData,
        startDate,
        endDate,
        range: this.selectedDateRange?.value,
      };
      this.mode === 'TOTAL_COST'
        ? this.constructDataForTotalTrend(args)
        : this.constructDataForBreakdownTrend(args);
    } else {
      this.fetchingTrendData = false;
    }
  }

  getStartAndEndDate(range: number): { startDate: string; endDate: string } {
    const endDate = moment().subtract(2, 'days');
    const startDate = moment().subtract(2 + range, 'days');

    return {
      startDate: startDate.format('YYYY-MM-DD'),
      endDate: endDate.format('YYYY-MM-DD'),
    };
  }

  aggregateData(dailyCost) {
    const aggregatedData = {};
    dailyCost.forEach((item) => {
      const date = item?.date;
      const amount = item?.amount;
      aggregatedData[date]
        ? (aggregatedData[date] += amount)
        : (aggregatedData[date] = amount);
    });

    let aggregatedGraphData = Object.entries(aggregatedData).map(
      ([date, totalCost]: [string, number]) => ({
        date: this.convertToOffsetDate(date),
        totalCost: parseFloat(totalCost.toFixed(DECIMAL_PLACES)),
      }),
    );

    // Sort the aggregated graph data by date in ascending order
    aggregatedGraphData.sort((a, b) => a.date.getTime() - b.date.getTime());

    return aggregatedGraphData;
  }

  //Once we get API data, there are two ways of processing it based on total cost or breakdown.
  constructDataForTotalTrend({ apiResponse, startDate, endDate, range }) {
    // this.fetchingTrendData = true;

    // Combine daily costs of all clusters
    let selectedClusterDailyCostsEntries = [];
    apiResponse?.dailyCost?.forEach((cluster) => {
      selectedClusterDailyCostsEntries = [
        ...selectedClusterDailyCostsEntries,
        ...cluster?.costs,
      ];
    });

    let trendGraphTotalData = this.aggregateData(
      selectedClusterDailyCostsEntries,
    );

    this.trendChartOptions = TimeSeriesOptions(
      startDate,
      endDate,
      range,
      this.stackName,
      this.cloudCostRawData?.unit,
      trendGraphTotalData,
    );

    this.fetchingTrendData = false;
  }

  constructDataForBreakdownTrend({ apiResponse, startDate, endDate, range }) {
    this.fetchingTrendData = true;
    let selectedClusterDailyCost = [];
    let trendBreakdownChartSeries = [];

    apiResponse?.dailyCost.forEach((cluster) => {
      selectedClusterDailyCost = cluster?.costs?.map((cost) => ({
        [`${cluster?.clusterName}_amount`]: parseFloat(
          cost?.amount?.toFixed(DECIMAL_PLACES),
        ),
        date: this.convertToOffsetDate(cost?.date),
      }));
      trendBreakdownChartSeries.push(
        BreakdownChartSeries(
          'date',
          `${cluster?.clusterName}_amount`,
          cluster?.clusterName,
          selectedClusterDailyCost,
        ),
      );
    });

    this.trendChartOptions = TimeSeriesOptions(
      startDate,
      endDate,
      range,
      this.stackName,
      this.cloudCostRawData?.unit,
    );
    this.trendChartOptions.series = trendBreakdownChartSeries;

    this.fetchingTrendData = false;
  }

  convertToOffsetDate(inputDate) {
    const inputMoment = moment
      .utc(inputDate)
      .subtract(moment().utcOffset(), 'minutes');
    return new Date(inputMoment.toISOString());
  }

  constructDataForBreakdownChart(apiResponse) {
    this.fetchingBreakdownData = true;

    // Prepare data for the pie chart
    const pieChartData = apiResponse?.dailyCost?.map((cluster) => ({
      label: cluster?.clusterName,
      value: parseFloat(cluster?.total?.toFixed(DECIMAL_PLACES)),
    }));

    // Filter out data with zero values
    const filteredChartData = pieChartData.filter(
      (entry) => entry?.value !== 0,
    );

    // Update the breakdown chart options with the filtered data
    this.breakdownChartOptions = PieChartOptions(
      filteredChartData,
      PieChartTooltipRenderer,
    );

    this.fetchingBreakdownData = false;
  }

  calculatePercentageChangeForDuration(
    totalCost: number,
    previousPeriodTotalCost: number,
  ): number {
    if (totalCost !== null && previousPeriodTotalCost !== null) {
      if (totalCost === 0 && previousPeriodTotalCost === 0) {
        return 0;
      }
      if (totalCost === 0) {
        return -100;
      }
      if (previousPeriodTotalCost === 0) {
        return 100;
      }

      const percentageChange =
        ((totalCost - previousPeriodTotalCost) / previousPeriodTotalCost) * 100;

      return parseFloat(percentageChange.toFixed(DECIMAL_PLACES));
    }
    return null;
  }

  setErrorOrEmptyState(state, errorMessage?) {
    if (state === 'empty') {
      this.errorMessage = errorMessage || 'No AWS or Azure environments found';
      this.emptyErrorStateIcon = this.alertsEmptyIcon;
      this.status = 'info';
      return;
    }
    this.errorMessage = errorMessage || '';
    this.emptyErrorStateIcon = this.alertsEmptyIcon;
    this.status = 'error';
  }

  syncCostExplorer() {
    let syncBtn = this.costExplorerActionButtons.find(
      (btn) => btn?.id === 'cost-explorer-sync',
    );
    syncBtn.loading = true;
    this.cloudCostExplorer.syncCloudCostUsingGET().subscribe(
      (data) => {
        syncBtn.loading = false;
        this.customToastService.success(
          'Cloud cost data updated successfully.',
        );
        this.refreshChartDataOnFilterChange();
      },
      (error) => {
        syncBtn.loading = false;
        this.customToastService.error('Cloud cost data update failed.');
      },
    );
  }

  getServiceList(serviceCostList) {
    let servicesList = serviceCostList?.map((service) => ({
      value: service?.serviceName,
      title: service?.serviceName,
    }));
    return servicesList || [];
  }

  setServiceCostTableData(data) {
    this.fetchingServiceCost = true;
    if (data['serviceCost']?.length > 0) {
      this.tableSettings.columns.cost.title = `Current Cost (${this.cloudCostRawData.unit})`;
      this.tableSettings.columns.previousCost.title = `Previous Period Cost (${this.cloudCostRawData.unit})`;
      this.tableSettings.columns.serviceName.filter.config.list =
        this.getServiceList(data['serviceCost']);
      let serviceLevelRoundedCost: [] = data['serviceCost']?.map((each) => ({
        ...each,
        cost: parseFloat(each?.cost?.toFixed(DECIMAL_PLACES)),
        previousCost: parseFloat(each?.previousCost?.toFixed(DECIMAL_PLACES)),
        trend: this.calculatePercentageChangeForDuration(
          +each?.cost,
          +each?.previousCost,
        ),
      }));
      this.serviceLevelCostData = [...serviceLevelRoundedCost];
      this.fetchingServiceCost = false;
    } else {
      this.tableSettings.columns.cost.title = `Current Cost`;
      this.tableSettings.columns.previousCost.title = `Previous Period Cost`;
      this.serviceLevelCostData = [];
      this.fetchingServiceCost = false;
    }
  }

  downloadServiceLevelCostAsCSV() {
    const { startDate, endDate } = this.getStartAndEndDate(
      this.selectedDateRange.value,
    );
    let filename = `Cloud Resources Cost of ${this.stackName} (${startDate}...${endDate})`;
    let serviceLevelCostCSVData = this.serviceLevelCostData?.map(
      (service: any) => {
        return {
          ...service,
          trend: service?.trend + '%',
          unit: this.cloudCostRawData?.unit,
        };
      },
    );
    let headerList = [
      'SL.No',
      'Resource Name',
      'Current Cost',
      'Previous Period Cost',
      'Trend',
      'Unit',
    ];
    let jsonKeys = ['serviceName', 'cost', 'previousCost', 'trend', 'unit'];

    const csvData = ConvertToCSV(serviceLevelCostCSVData, headerList, jsonKeys);

    generateCSVFile(csvData, filename);
  }
}
