import { Component, OnInit, TemplateRef, ViewChild } from '@angular/core';
import { UiUserControllerService } from '../../../cc-api/services/ui-user-controller.service';
import { UserAccessToken } from '../../../cc-api/models/user-access-token';
import { NbDialogService } from '@nebular/theme';
import { v4 as uuid } from 'uuid';
import { ClipboardService } from 'ngx-clipboard';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { FormatDateService } from 'src/app/services/format-date.service';
import { SmartTableTimeagoTooltipComponent } from 'src/app/components/smart-table-timeago-tooltip/smart-table-timeago-tooltip.component';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { FormControl } from '@angular/forms';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
const emptyStateLogo = './assets/token-empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-user-access-token',
  templateUrl: './user-access-token.component.html',
  styleUrls: ['./user-access-token.component.scss'],
})
export class UserAccessTokenComponent implements OnInit {
  tokens: Array<UserAccessToken>;
  token: UserAccessToken;
  whiteSpaceInputPattern = whiteSpaceInputPattern;
  isSuccess = false;
  secondaryMsg: string = 'Generate personal tokens to access Facets API & CLI';
  emptyStateLogo = emptyStateLogo;
  primaryMsg: string = 'No personal tokens found!';
  showEmptyState: boolean = false;
  loading: boolean = true;
  buttonText: string = 'Generate Token';
  showBtn: boolean = true;
  status: 'info' | 'error' = 'info';
  smartTableCustomActionButtons = [
    {
      id: 'delete',
      icon: 'trash-2-outline',
      toolTip: 'Delete Token',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: undefined,
      onClick: (data) => {
        this.openDeletePopup(data);
      },
    },
  ];
  tokenName: FormControl = new FormControl('');
  saveTokenInProgress: boolean = false;
  deleteInProgress: boolean = false;

  @ViewChild('deletePopup', { static: true })
  private deletePopup: TemplateRef<any>;

  tableSettings: any = {
    columns: {
      name: {
        title: 'Token Name',
        filter: false,
      },
      createdOn: {
        title: 'Created On',
        filter: false,
        type: 'custom',
        valuePrepareFunction: (date) => {
          if (date) {
            let labelToSend = '';
            return JSON.stringify({
              label: date,
              tooltipText: this.formatDateService.getFormattedDate(date),
            });
          } else {
            return null;
          }
        },
        renderComponent: SmartTableTimeagoTooltipComponent,
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          // DATA FROM HERE GOES TO renderComponent. rowData property holds row data and any custom data sent will reside in value
          const displayVal = {
            smartTableCustomActionButtons: this.smartTableCustomActionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '12%',
        filter: false,
      },
    },
    noDataMessage:
      'No personal tokens found! Generate personal tokens to access Facets API & CLI.',
    pager: {
      display: false,
      perPage: 15,
    },
    mode: 'external',
    actions: false,
  };
  error: any;

  constructor(
    private userController: UiUserControllerService,
    private dialogService: NbDialogService,
    private clipboardService: ClipboardService,
    private formatDateService: FormatDateService,
    private customToastService: CustomToastService,
  ) {}

  ngOnInit(): void {
    this.getAllTokens();
  }

  getAllTokens() {
    this.loading = true;
    this.userController.getTokenUsingGET().subscribe(
      (tokens) => {
        this.tokens = tokens;
        if (this.tokens?.length > 0) {
          this.showEmptyState = false;
        } else {
          this.showEmptyState = true;
        }
        this.loading = false;
      },
      (error) => {
        this.loading = false;
        this.showEmptyState = true;
        this.secondaryMsg =
          error?.error?.message || ' There is some error fetching tokens';
        this.primaryMsg = '';
        this.showBtn = false;
        this.emptyStateLogo = errorIcon;
        this.status = 'error';
      },
    );
  }

  openCreatePopup(createToken: any) {
    this.isSuccess = false;
    this.token = {};
    this.token['token'] = uuid();
    this.tokenName.reset();
    this.dialogService.open(createToken, { context: { edit: false } });
  }

  createTokenCall(ref: any) {
    this.saveTokenInProgress = true;
    this.userController
      .createTokenUsingPOST({
        ...this.token,
        name: this.tokenName?.value?.trim(),
      })
      .subscribe(
        (tokens) => {
          this.saveTokenInProgress = false;
          this.tokens = tokens;
          this.isSuccess = true;
          this.getAllTokens();
        },
        (error) => {
          this.saveTokenInProgress = false;
          this.error = error?.error?.message || error?.message;
          this.customToastService.error(
            this.error || 'Failed to create token!',
          );
        },
      );
  }

  openDeletePopup(data) {
    this.dialogService.open(this.deletePopup, {
      context: { data },
    });
  }

  deleteToken(data, ref) {
    this.deleteInProgress = true;
    this.userController.deleteTokenUsingDELETE(data?.tokenId).subscribe(
      (tokens) => {
        this.tokens = tokens;
        this.isSuccess = true;
        this.deleteInProgress = false;
        ref.close();
        this.customToastService.success(
          `Token ${data?.name} has been deleted successfully`,
        );
        this.getAllTokens();
      },
      (err) => {
        this.deleteInProgress = false;
        this.customToastService.error(
          err?.error?.message ||
            err?.message ||
            `Failed to delete token ${data?.name}`,
        );
      },
    );
  }

  copyToken(token: string) {
    this.clipboardService.copy(token);
  }
}
