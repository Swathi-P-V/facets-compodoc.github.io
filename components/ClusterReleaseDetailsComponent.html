    const defaultProps = {
      type: 'default',
      props: {
        value: cell || 'N/A',
      },
    };
    if (!cell) {
      return defaultProps;
    }
    switch (row?.attribute) {
      case 'commitId': {
        return {
          type: 'hyper-link',
          props: {
            url: this.getCommitUrl(cell, row),
            displayText: cell,
          },
        };
      }
      case 'artifactUrl': {
        let label = '';
        if (cell?.length > 0) {
          let params = cell?.split('/');
          if (params?.length > 1) {
            label = params[params?.length - 1];
          } else if (params?.length === 1) {
            label = params?.[0];
          }
        }
        return {
          type: 'text-tooltip',
          allowCopyClipboard: true,
          props: {
            label,
            tooltipText: cell,
          },
        };
      }
      default: {
        return defaultProps;
      }
    }

    const resourceTypes = new Set();
    this.releaseChangesData?.changes?.forEach((change) => {

  getDeploymentDetails(deploymentId) {
    this.loading = true;
    this.deploymentService
      .getDeploymentUsingGET({ deploymentId, clusterId: this.clusterId })
      .subscribe(
        (d) => {
          this.dLog = d;
          this.setStatusValue(this.dLog);
          if (this.dLog?.appDeployments?.length === 0) {
            if (this.dLog?.status !== 'IN_PROGRESS') {
              this.deploymentPrimary = 'No applications deployed';
            } else {
              this.deploymentPrimary =
                'A release is in progress. Changes related to it will reflect soon here.';
            }
          }
          if (this.dLog?.tfVersion?.length > 0) {
            let labelToSend = this.dLog?.tfVersion?.split('.zip')[0]
              ? this.dLog?.tfVersion?.split('.zip')[0]
              : 'N/A';
            this.longStringTooltipConfig = JSON.stringify({
              label: labelToSend || 'N/A',
              tooltipText: this.dLog?.tfVersion ? this.dLog?.tfVersion : 'N/A',
            });
          }
          if (this.dLog?.createdOn && this.dLog?.finishedOn) {
            const createdOn = moment(this.dLog?.createdOn);
            const finishedOn = moment(this.dLog?.finishedOn);
            const diff = finishedOn.diff(createdOn);
            this.duration = moment.duration(diff);
            this.diffHours =
              this.duration?.hours() < 10
                ? '0' + this.duration?.hours()
                : this.duration?.hours();
            this.diffMins =
              this.duration?.minutes() < 10
                ? '0' + this.duration?.minutes()
                : this.duration?.minutes();
            this.diffSeconds =
              this.duration?.seconds() < 10
                ? '0' + this.duration?.seconds()
                : this.duration?.seconds();
          }
          this.loading = false;
          if (this.dLog?.changesApplied?.length > 0) {
            this.dLog.changesApplied.forEach((change) => {
              if (change.type === 'Creation') {
                this.createdChanges += 1;
                this.created.push(change);
              } else if (change.type === 'Destruction') {
                this.destroyedChanges += 1;
                this.destroyed.push(change);
              } else if (change.type === 'Modifications') {
                this.modifiedChanges += 1;
                this.modified.push(change);
              }
            });
            let largestChange = Math.max(
              this.createdChanges,
              this.destroyedChanges,
              this.modifiedChanges,
            );
            if (largestChange === this.createdChanges) {
              this.createdVal = 100;
              this.destroyedVal = Math.floor(
                (this.destroyedChanges / this.createdChanges) * 100,
              );
              this.modifiedVal = Math.floor(
                (this.modifiedChanges / this.createdChanges) * 100,
              );
            } else if (largestChange === this.destroyedChanges) {
              this.destroyedVal = 100;
              this.modifiedVal = Math.floor(
                (this.modifiedChanges / this.destroyedChanges) * 100,
              );
              this.createdVal = Math.floor(
                (this.createdChanges / this.destroyedChanges) * 100,
              );
            } else if (largestChange === this.modifiedChanges) {
              this.modifiedVal = 100;
              this.destroyedVal = Math.floor(
                (this.destroyedChanges / this.modifiedChanges) * 100,
              );
              this.createdVal = Math.floor(
                (this.createdChanges / this.modifiedChanges) * 100,
              );
            }
          }
        },
        (err) => {
          this.loading = false;
          this.showEmptyState = true;
          this.detailSecondary =
            err?.error?.message ||
            'There is some error fetching deployment information';
          this.emptyStatus = 'error';
        },
      );
  }

  searchError(event, titleOne, childOne, titleTwo = '', childTwo = '') {
    const key = event?.target?.value?.toLowerCase();

    let allTitleParent = Array.from(document.querySelectorAll('.' + titleOne));
    let allTitleItems = Array.from(document.querySelectorAll('.' + childOne));
    allTitleParent.forEach((logItem, index) => {
      const text = new DOMParser().parseFromString(
        allTitleItems[index].innerHTML?.toLowerCase(),
        'text/html',
      ).documentElement.textContent;
      let HTMLLowerCase: string = text?.toLowerCase();
      // @ts-ignore
      let newHTML = HTMLLowerCase.replaceAll(
        new RegExp(key, 'ig'),
        '<span class="highlight__">' + key + '</span>',
      );
      let newSpan = document.createElement('span');
      newSpan.classList.add(childOne);
      newSpan.classList.add('weight-600');
      newSpan.innerHTML = newHTML;
      logItem.replaceChild(newSpan, allTitleItems[index]);
    });
    if (titleTwo?.length > 0 && childTwo?.length > 0) {
      let allStackParent = Array.from(
        document.querySelectorAll('.' + titleTwo),
      );
      let allStackItems = Array.from(document.querySelectorAll('.' + childTwo));
      allStackParent.forEach((logItem, index) => {
        const text = new DOMParser().parseFromString(
          allStackItems[index].innerHTML?.toLowerCase(),
          'text/html',
        ).documentElement.textContent;
        let HTMLLowerCase: string = text?.toLowerCase();
        // @ts-ignore
        let newHTML = HTMLLowerCase.replaceAll(
          new RegExp(key, 'ig'),
          '<span class="highlight__">' + key + '</span>',
        );
        let newSpan = document.createElement('span');
        newSpan.classList.add(childTwo);
        newSpan.classList.add('weight-600');
        newSpan.innerHTML = newHTML;
        logItem.replaceChild(newSpan, allStackItems[index]);
      });
    }
  }

  getFormattedHTML(stack) {
    if (stack) {
      let stackString = this.convert.toHtml(stack);
      const convertSpan = document.createElement('span');
      convertSpan.innerHTML = stackString;
      return (
        convertSpan?.textContent?.toLowerCase() ||
        convertSpan?.innerText?.toLowerCase()
      );
    }
    return '';
  }
