import { Component } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import {
  UiCommonClusterControllerService,
  UiDeploymentControllerService,
} from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { GrafanaServiceService } from 'src/app/services/grafana-service.service';
import {
  RUNNING,
  SCALE_DOWN_FAILED,
  SCALE_UP_FAILED,
} from '../../../constants/deployments';
import { SCALE_DOWN, SCALE_UP } from '../../../constants/releaseTypes';
const emptyIcon = './assets/empty.svg';
const errorIcon = './assets/error.svg';
@Component({
  selector: 'app-cluster-alerts-manager',
  templateUrl: './cluster-alerts-manager.component.html',
  styleUrls: ['./cluster-alerts-manager.component.scss'],
})
export class ClusterAlertsManagerComponent extends FacetsPage {
  clusterId: any;
  stackName: any;
  clusterState: string;
  loading: boolean = true;

  alertGroups: any;
  alertChartData: any;
  alertChartError: any;
  alertDataError: boolean = false;

  //Variables to manage error and empty state
  messageObject: any = {
    messageIcon: '',
    primaryMsg: '',
    secondaryMsg: '',
    messageStatus: 'none',
  };

  //Icons
  emptyIcon = emptyIcon;
  errorIcon = errorIcon;

  cluster: any;
  loadingEnvConfig: boolean = false;

  constructor(
    activatedRoute: ActivatedRoute,
    private uiDeploymentControllerService: UiDeploymentControllerService,
    private commonClusterControllerService: UiCommonClusterControllerService,
    private grafanaService: GrafanaServiceService,
    private uiCommonClusterControllerService: UiCommonClusterControllerService,
    private router: Router,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.loading = true;
    this.cluster = null;
    this.clusterId = p.clusterId;
    this.stackName = p.stackName;

    this.initializeMessgeObject();
    this.getEnvironmentConfig(this.clusterId);
    this.getAlertsInfo();
  }

  getEnvironmentConfig(clusterId, setClusterInfo = false) {
    this.loadingEnvConfig = true;
    return new Promise((resolve, reject) => {
      this.uiCommonClusterControllerService
        .getClusterCommonUsingGET(clusterId)
        .subscribe(
          (cluster) => {
            this.loadingEnvConfig = false;
            if (!setClusterInfo) {
              this.cluster = cluster;
              resolve(true);
            } else {
              resolve(cluster);
            }
          },
          (error) => {
            this.cluster = null;
            this.loadingEnvConfig = false;
            reject(false);
          },
        );
    });
  }

  getAlertsInfo() {
    this.loading = true;
    this.getMonthData();
    this.resolveClusterState().then(
      (data) => {
        //If cluster state is RUNNING, get Alerts.
        this.getAlerts().then(
          (data) => {
            this.loading = false;
          },
          (error) => {
            this.loading = false;
          },
        );
      },
      (error) => {
        this.loading = false;
      },
    );
  }

  getMonthData() {
    this.alertDataError = false;
    this.grafanaService
      .getAlertsSummaryBy(this.clusterId, 'alerts_summary_by_severity')
      .subscribe(
        (res: any) => {
          if (res?.status === 'success') {
            this.alertChartData = res?.data?.result;
          } else if (res?.status === 'error') {
            this.alertChartError = res;
            this.alertDataError = true;
          }
        },
        (err) => {
          this.alertChartError = err;
          this.alertDataError = true;
        },
      );
  }

  resolveClusterState() {
    return new Promise((resolve, reject) => {
      this.uiDeploymentControllerService
        .getClusterStateUsingGET(this.clusterId)
        .subscribe(
          (state) => {
            this.clusterState = state;
            if (
              [
                RUNNING,
                SCALE_DOWN,
                SCALE_DOWN_FAILED,
                SCALE_UP_FAILED,
              ]?.includes(this.clusterState)
            ) {
              resolve(true);
            } else {
              this.messageObject.messageStatus = 'info';
              this.messageObject.messageIcon = this.emptyIcon;
              this.messageObject.primaryMsg =
                'This page is available when environment is in running state.';
              this.messageObject.secondaryMsg =
                'Environment is in ' + this.clusterState + ' state!';
              reject(false);
            }
          },
          (error) => {
            //Do something nice
            this.messageObject.messageStatus = 'error';
            this.messageObject.messageIcon = this.errorIcon;
            this.messageObject.secondaryMsg =
              error?.error?.message ||
              'There is error fetching environment state';
            reject(false);
          },
        );
    });
  }

  onSilenceEvent(event) {
    this.getAlertsInfo();
  }

  getAlerts() {
    return new Promise((resolve, reject) => {
      this.commonClusterControllerService
        .getAlertsUsingGET(this.clusterId)
        .subscribe(
          (alertList) => {
            if (alertList['status'] === 'success') {
              this.alertGroups = alertList['data']['groups'];
              this.messageObject = null;
              resolve(true);
            } else {
              this.messageObject.messageStatus = 'error';
              this.messageObject.messageIcon = this.errorIcon;
              this.messageObject.secondaryMsg = String(alertList['reason']);
              reject(false);
            }
          },
          (error) => {
            this.messageObject.messageStatus = 'error';
            this.messageObject.messageIcon = this.errorIcon;
            this.messageObject.secondaryMsg =
              error?.error?.message ||
              'There is error fetching alerts for this environment';
            reject(false);
          },
        );
    });
  }

  navigate() {
    let urlToNav =
      '/' + this.stackName + '/cluster/' + this.clusterId + '/releases/';
    this.router.navigate([urlToNav]);
  }

  async getParentEnvDetails() {
    const parentCluster = await this.getEnvironmentConfig(
      this.cluster?.baseClusterId,
      true,
    );
    this.goToParentAlerts(parentCluster);
  }

  goToParentAlerts(cluster) {
    let urlToAlerts =
      '/' + cluster?.stackName + '/cluster/' + cluster?.id + '/alerts';
    this.router.navigate([urlToAlerts]);
  }

  initializeMessgeObject() {
    this.messageObject = {
      messageIcon: '',
      primaryMsg: '',
      secondaryMsg: '',
      messageStatus: 'none',
    };
  }
}

  </nb-card-body>
