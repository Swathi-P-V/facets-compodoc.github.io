import { Component, EventEmitter, Input, OnInit, Output } from '@angular/core';

import { ViewCell } from 'ng2-smart-table';

@Component({
  template: `
    <div
      [ngClass]="renderValue?.length > 0 ? setClass : ''"
      [nbTooltip]="toolTip"
      [nbTooltipDisabled]="!showTooltip || renderValue?.length <= 0"
    >
      {{ renderValue }}
    </div>
  `,
  styles: [
    `
      .tablet {
        width: 1.5rem;
        height: 1.5rem;
        border-radius: 50%;
        display: flex;
        text-align: center;
        border: 2px solid var(--color-primary-active);
        color: var(--color-primary-active);
        align-items: center;
        justify-content: center;
        font-weight: 600;
        cursor: default;
      }
    `,
  ],
})
export class TextCircleTooltipComponent implements ViewCell, OnInit {
  @Input() value: any;
  @Input() rowData: any;
  showTooltip: boolean = false;
  setClass: string = '';
  renderValue: any = '';
  toolTip: string = '';

  ngOnInit() {
    try {
      this.renderValue = JSON.parse(this.value)?.value || '';
      this.toolTip = JSON.parse(this.value)?.toolTip || '';
      this.setClass = JSON.parse(this.value)?.setClass || '';
      this.showTooltip = JSON.parse(this.value)?.showTooltip || false;
    } catch (error) {}
  }
}
