  constructor(
    private activatedRoute: ActivatedRoute,
    private uiStackControllerService: UiStackControllerService,
    private router: Router,
    private dialogService: NbDialogService,
    public userAuthoritiesService: UserAuthoritiesService,
    private commonClusterController: UiCommonClusterControllerService,
    private contextUpdatesService: ContextUpdatesService,
    private uiCommonClusterControllerService: UiCommonClusterControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.stackName = p?.stackName;
    this.resetTableData();
    this.activatedRoute.paramMap
      .pipe(map(() => window.history.state))
      .subscribe((data) => {
        if (data?.redirectToEdit) {
          this.redirectToEdit = data.redirectToEdit;
        }
        if (data?.redirectEnvID) {
          this.redirectEnvID = data.redirectEnvID;
        }
      });

    this.callStackClusterApis();
  }

  resetTableData() {
    this.tableData = [];
    this.overViewList = [];
  }

  callStackClusterApis() {
    this.fetchingClusters = true;
    Promise.allSettled([
      this.getStackDetails(),
      this.getStackRelatedClusters(),
    ]).then(
      (data) => {
        this.setTableData();
        this.fetchingClusters = false;
      },
      (error) => {
        this.fetchingClusters = false;
      },
    );
  }

  getStackDetails() {
    return new Promise((resolve, reject) => {
      this.uiStackControllerService.getStackUsingGET(this.stackName).subscribe(
        (s) => {
          this.stack = s;
          this.subHeading = `List of all environments of ${this.stack?.name} Blueprint.
          <a href="${this.stack?.vcsUrl}" target="_blank">
            Repository
          </a>`;
          this.pauseReleases = s.pauseReleases;
          resolve(true);
        },
        (error) => {
          this.primaryMsg = `Blueprint doesn't exist`;
          this.secondaryMsg = error?.error?.message;
          this.showBtn = false;
          this.emptyStateLogo = this.errorIcon;
          reject(error);
        },
      );
    });
  }

  getStackRelatedClusters() {
    return new Promise((resolve, reject) => {
      this.uiStackControllerService
        .getClustersOverviewUsingGET(this.stackName)
        .subscribe(
          (res) => {
            let clusterList = res?.map((cluster) => cluster.cluster);
            if (this.redirectToEdit) {
              let redirectionEnv = clusterList.find(
                (env) => env.id == this.redirectEnvID,
              );
              this.edit(redirectionEnv);
              return;
            }
            this.overViewList = res;
            if (res?.length > 0) {
              this.clusterId = res[0]?.cluster?.id;
            }
            resolve(true);
          },
          (error) => {
            this.secondaryMsg =
              error?.error?.message ||
              ' There is some error fetching environment information';
            this.tableData = null;
            this.emptyStateLogo = this.errorIcon;
            this.primaryMsg = '';
            this.showBtn = false;
            this.status = 'error';
            reject(error);
          },
        );
    });
  }

  setTableData() {
    this.fetchingClusters = true;
    let tempData = [];
    if (this.overViewList.length > 0) {
      this.overViewList.forEach((info) => {
        let newData = {};
        const baseClusterId = info?.cluster?.baseClusterId;
        newData['name'] = info?.['cluster']['name']
          ? info?.['cluster']['name']
          : '--';
        newData['cloud'] = info?.['cluster']?.['cloud'] || '--';
        newData['releaseStream'] = info?.['cluster']?.['releaseStream'] || '--';
        newData['vpcCIDR'] = info?.['cluster']?.['vpcCIDR'] || '--';
        newData['lastReleaseId'] = info?.['lastRelease']?.['id'] || '';
        newData['lastRelease'] = info?.['lastRelease']?.['createdOn']
          ? moment(info?.['lastRelease']?.['createdOn']).fromNow()
          : '--';
        newData['baseClusterId'] = baseClusterId;
        newData['baseClusterName'] = info?.cluster?.baseClusterName || null;
        newData['baseCluster'] = info?.baseCluster;
        newData['infraType'] = baseClusterId
          ? 'Dependent Environment'
          : 'Base Environment';
        newData['releaseStatus'] = info?.['lastRelease']?.['status'] || '--';
        newData['clusterState'] =
          this.clusterStateFormatted[info['clusterState']] || '--';
        newData['nextRelease'] = info['timeToNextRelease'] || '--';
        newData['cluster'] = info['cluster'];
        newData['pauseReleases'] = info['cluster']['pauseReleases'];
        tempData.push(newData);
      });
    }
    this.tableData = tempData;
    this.fetchingClusters = false;
  }

  createCluster(): void {
    // this.router.navigate([
    //   '/',
    //   this.stack.name,
    //   'chooseClusterCreate',
    //   'clusterCreate',
    //   'aws',
    // ]);
    this.router.navigate(['/', this.stack.name, 'chooseClusterCreate']);
  }

  newToggleClick(pauseReleases): void {
    const status: string = pauseReleases ? 'Enabled' : 'Disabled';
    if (!this.userAuthoritiesService.isActionDisabled('STACK_CONFIGURE')) {
      this.dialogService
        .open(PauseReleaseDialogComponent, {
          context: { status: pauseReleases ? 'enable' : 'disable' },
        })
        .onClose.subscribe((proceed) => {
          console.log('received value from pause dialog ' + proceed);
          if (proceed) {
            this.uiStackControllerService
              .toggleReleaseUsingPOST1({
                toggleRelease: {
                  stackName: this.stack.name,
                  pauseReleases: !pauseReleases,
                },
                stackName: this.stack.name,
              })
              .subscribe(
                (s) => {
                  console.log(
                    'Prod Release %s for the stack %s',
                    status,
                    this.stack.name,
                  );
                  this.customToastService.success(
                    status.concat(' Prod Release'),
                  );
                },
                (error) => {
                  this.pauseReleases = pauseReleases;
                  this.onAPIErrorService.onAPIError(
                    error?.error?.message || 'Pause release failed',
                    error,
                  );
                },
              );
          } else {
            this.pauseReleases = pauseReleases;
          }
        });
    }
  }

  refreshStack() {
    this.addOverrideSpinner = true;
    this.setButtonLoadingStatus(true);
    try {
      this.uiStackControllerService
        .reloadStackUsingGET1(this.stack?.name)
        .subscribe(
          (stack) => {
            this.setButtonLoadingStatus(false);
            this.customToastService.success('Synced with git');
            //location.reload();
          },
          (err) => {
            this.setButtonLoadingStatus(false);
            this.onAPIErrorService.onAPIError(
              err?.error?.message || 'Syncing with git failed',
              err,
            );
          },
        );
    } catch (err) {
      console.log(err);
      this.addOverrideSpinner = false;
    }
  }

  setButtonLoadingStatus(status) {
    this.blueprintEnvironmentsActionButtons.forEach((button) => {
      if (button?.id === 'sync-with-git-button') {
        button.loading = status;
      }
    });
  }

  edit(rowData) {
    this.router.navigate(
      [
        '/',
        rowData?.cluster?.stackName,
        'cluster',
        rowData?.cluster?.id,
        'edit',
      ],
      {
        state: {
          previousComponent: 'blueprintOverview',
        },
      },
    );
  }

  async delete(rowData) {
    let dependentEnvs: any[] = [];
    if (!rowData?.cluster?.baseClusterId) {
      dependentEnvs = await this.getEnvironmentOverview(rowData?.cluster?.id);
    }
    dependentEnvs?.length
      ? this.dialogService.open(this.dependentEnvsInfoPopup, {
          context: dependentEnvs,
        })
      : this.dialogService.open(this.deleteEnvConfirmUI, {
          context: rowData.cluster,
        });
  }

  // deleteEnvironment(ref, forceDelete?);
  deleteEnvironment(ref, rowData) {
    try {
      this.deleteInProgress = true;
      // var httpReq = forceDelete
      //   ? this.CommonClusterController.deleteClusterForceUsingDELETE(
      //       this.rowData.id,
      //     )
      //   : this.CommonClusterController.deleteClusterUsingDELETE1(
      //       this.rowData.id,
      //     );
      this.commonClusterController
        .deleteClusterUsingDELETE1(rowData.id)
        .subscribe(
          (data) => {
            this.customToastService.success(
              'Environment successfully deleted ',
            );
            this.deleteInProgress = false;
            ref.close();
            this.contextUpdatesService.requestContextSwitcherRefresh();
            //Get clusters for this stack
            this.callStackClusterApis();
          },
          (error) => {
            this.deleteInProgress = false;
            ref.close();
            try {
              error = JSON.parse(error?.error);
            } catch (error) {
              error = {};
            }
            this.onAPIErrorService.onAPIError(
              'Environment delete failed. ' + error?.message,
              error,
            );
          },
        );
    } catch (err) {
      this.deleteInProgress = false;
      this.customToastService.error('Environment delete failed');
      ref.close();
      // this.cancelDeleteEnvironment(ref);
    }
  }

  handleActions(event) {
    this.blueprintEnvironmentsActionButtons
      ?.find((button) => button.id === event)
      ?.click();
  }

    return new Promise((resolve, reject) => {
      this.uiCommonClusterControllerService
        .getClusterInfoUsingGET(clusterId)
        .subscribe(
          (response) => {
            resolve(response?.dependentClusters);
          },
          (err) => {
            resolve([]);
          },
        );
    });
  }
