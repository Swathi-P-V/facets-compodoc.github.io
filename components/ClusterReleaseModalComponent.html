import {
  Component,
  Output,
  EventEmitter,
  Input,
  SimpleChanges,
  OnChanges,
} from '@angular/core';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { ActivatedRoute, Params, Router } from '@angular/router';
import mixpanel from 'mixpanel-browser';
import { UiDeploymentControllerService } from 'src/app/cc-api/services';
import {
  AwsCluster,
  DeploymentRequest,
  HotfixDeploymentRecipe,
  Resource,
} from 'src/app/cc-api/models';
import { FormControl, FormGroup } from '@angular/forms';
import { isEqual } from 'lodash';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import {
  APPLY_PLANNED_RELEASE_TRIGGER_FAIL,
  APPLY_PLANNED_RELEASE_TRIGGER_SUCCESS,
  CUSTOM_RELEASE_TRIGGER_FAIL,
  CUSTOM_RELEASE_TRIGGER_SUCCESS,
  FULL_PLAN_RELEASE_TRIGGER_FAIL,
  FULL_PLAN_RELEASE_TRIGGER_SUCCESS,
  FULL_RELEASE_TRIGGER_FAIL,
  FULL_RELEASE_TRIGGER_SUCCESS,
  SELECTIVE_PLAN_RELEASE_TRIGGER_FAIL,
  SELECTIVE_PLAN_RELEASE_TRIGGER_SUCCESS,
  SELECTIVE_RELEASE_TRIGGER_FAIL,
  SELECTIVE_RELEASE_TRIGGER_SUCCESS,
  USER_CREATE_FAIL,
} from 'src/app/constants/analytics-track-events';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
import { er } from '@antv/x6/lib/registry/router/main';

@Component({
  selector: 'app-cluster-release-modal',
  templateUrl: './cluster-release-modal.component.html',
  styleUrls: ['./cluster-release-modal.component.scss'],
})
export class ClusterReleaseModalComponent
  extends FacetsPage
  implements OnChanges
{
  @Output() releaseCloseClickEvent = new EventEmitter<any>();
  @Output() pageResetEvent = new EventEmitter<any>();
  @Input() resources: any = {};
  @Input() resourcesByResGroups: any = {};
  @Input() showSpinner: boolean = false;
  @Input() tfVersionInfo: any = {};
  @Input() deploymentDialogConfig: any;
  @Input() plannedRelease: boolean = false;
  @Input() retryRepeatResources: any = {};
  @Input() planType: 'PLAN' | 'HOTFIX_PLAN';
  @Input() lastReleaseData: any = null;
  @Input() releaseDisabledStatus: {
    FULL: boolean;
    SELECTIVE: boolean;
    CUSTOM: boolean;
    FULL_DESTRUCTION: boolean;
    SELECTIVE_DESTRUCTION: boolean;
    CUSTOM_DESTRUCTION: boolean;
  };
  @Input() selectiveFromResourceCentre: boolean = false;

  allowDestructDisableTooltip: string = 'Access is forbidden';
  retryReleaseMode: boolean = false;
  clusterSettingsForm: FormGroup;
  cluster: AwsCluster;
  clusterId: string;
  stackName: any;
  withRefresh: FormControl = new FormControl(false);
  forceRelease: FormControl = new FormControl(false);
  allowDestruct: FormControl = new FormControl(false);
  comments: FormControl = new FormControl('');
  command: FormControl = new FormControl('');
  commandFormGroup: FormGroup = new FormGroup({});
  showAdvance: boolean = false;
  commandList: string[] = [];
  editIndexList: any[] = [];
  tempResourceList: { title: string; value: string }[] = [];
  resourceList: Array<Resource> = [];
  resourceMap = [];
  releaseTypeDisabledTooltip: string = 'Access is forbidden';
  releaseRadioValues: any = [
    { value: 'FULL', label: 'Full Release', checked: true, disabled: false },
    {
      value: 'SELECTIVE',
      label: 'Selective Release',
      checked: false,
      disabled: false,
    },
    {
      value: 'CUSTOM',
      label: 'Custom Release',
      checked: false,
      disabled: false,
    },
  ];
  groupByOptions = [
    { value: 'resourceType', label: 'Resource Type' },
    { value: 'resourceGroup', label: 'Resource Group' },
  ];
  groupByMode: FormControl = new FormControl('resourceType');
  releaseType: FormControl = new FormControl('FULL');
  applyRequest: boolean = false;
  advanceOpenLogo: string = './assets/advance-open.svg';
  advanceCloseLogo: string = './assets/advance-close.svg';
  fetchingResourceGroups: boolean = false;
  fetchingError: boolean = false;
  allResourceGroups: [];
  //Separate  loaders for apply and plan
  applyReleaseInProgress: boolean = false;
  planReleaseInProgress: boolean = false;
  retryOrRepeat: string = '';

  objectKeys = Object.keys;
  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private deploymentService: UiDeploymentControllerService,
    private customToasterService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.stackName = p.stackName;
    this.clusterId = p.clusterId;
    this.groupByMode.setValue(this.groupByOptions[0].value);
  }

  emitCloseClick() {
    this.releaseCloseClickEvent.emit();
  }

  releaseOptionChange(releaseType) {
    this.setDefaultValues(false, releaseType);
  }

  refreshSelectedList: boolean = false;
  updateReleaseType(event) {
    this.refreshSelectedList = true;
    const resources =
      event === 'resourceGroup' ? this.resources : this.resourcesByResGroups;
    resources.forEach((resGrp) => {
      resGrp.checked = false;
      resGrp.children.forEach((res) => {
        res.checked = false;
      });
    });
    setTimeout(() => {
      this.refreshSelectedList = false;
    }, 0);
  }

  ngOnChanges(changes: SimpleChanges) {
    if (changes?.deploymentDialogConfig?.currentValue) {
      changes?.deploymentDialogConfig?.currentValue.onClose.subscribe(() => {
        this.setDefaultValues(true);
      });
    }
    if (changes?.plannedRelease?.currentValue) {
      this.releaseType.setValue('APPLYPLAN');
    }
    if (changes?.releaseDisabledStatus?.currentValue) {
      //only fire the condition when permission value changes
      //this is to avoid multiple renders of release buttons conditions even if no value changes in disabled states

      if (
        !isEqual(
          changes?.releaseDisabledStatus?.currentValue,
          changes?.releaseDisabledStatus?.previousValue,
        )
      ) {
        const disabledStateValues =
          changes?.releaseDisabledStatus?.currentValue;
        this.setReleaseButtonDisabledState(disabledStateValues);
      }
    }
    if (changes?.retryRepeatResources?.currentValue) {
      let resourcesList = changes?.retryRepeatResources?.currentValue;
      this.setRetryResources(resourcesList);
    }
    if (
      changes?.selectiveFromResourceCentre &&
      !changes?.selectiveFromResourceCentre?.currentValue
    ) {
      this.selectiveFromResourceCentre = false;
    }
  }

  setReleaseButtonDisabledState(disabledStateValues) {
    let releaseTypeSet = false;
    let tempReleaseType = '';
    this.releaseRadioValues.forEach((radioValue) => {
      const upperCaseRadioValue = radioValue?.value?.toUpperCase();
      radioValue.disabled = disabledStateValues[upperCaseRadioValue];
      if (!radioValue.disabled && !releaseTypeSet) {
        tempReleaseType = upperCaseRadioValue;
        releaseTypeSet = true;
      }
    });
    if (this.releaseType?.value !== 'APPLYPLAN') {
      this.releaseType.setValue(tempReleaseType);
    }
    if (tempReleaseType?.length > 0) {
      this.allowDestruct.setValue(false);
      disabledStateValues[tempReleaseType + '_DESTRUCTION']
        ? this.allowDestruct.disable()
        : this.allowDestruct.enable();
    }
  }

  setRetryResources(resourcesList) {
    this.tempResourceList = [];
    resourcesList.forEach((resource) => {
      this.tempResourceList.push({
        title: resource.resourceType,
        value: resource.resourceName,
      });
      this.resources.forEach((parentOption) => {
        parentOption.children.forEach((childOption) => {
          if (
            parentOption.value === resource.resourceType &&
            childOption.value === resource.resourceName
          ) {
            childOption.checked = true;
          }
        });
      });
    });
    this.retryOrRepeat =
      this.lastReleaseData?.status === 'SUCCEEDED' ? 'Repeat' : 'Retry';
    this.retryReleaseMode = true;
    this.releaseType.setValue('SELECTIVE');
  }

  setDefaultValues(setDefaultType, releaseType = '') {
    this.allowDestruct.enable();
    if (
      releaseType?.length > 0 &&
      this.releaseDisabledStatus[releaseType + '_DESTRUCTION']
    ) {
      this.allowDestruct.setValue(false);
      this.allowDestruct.disable();
    }
    this.allowDestruct.setValue(false);
    this.withRefresh.setValue(false);
    this.forceRelease.setValue(false);
    this.comments.setValue('');
    this.showAdvance = false;
    this.command.setValue('');
    this.commandList = [];
    if (setDefaultType) {
      this.releaseType.setValue('FULL');
    }
    this.commandFormGroup = new FormGroup({});
  }

  openAdvance() {
    this.showAdvance = !this.showAdvance;
  }

  selectedOption(event) {
    this.tempResourceList = [];
    if (event?.selectedOptionsList?.length > 0) {
      this.tempResourceList = event.selectedOptionsList.flatMap((parent) => {
        return parent.children.map((child) => {
          return {
            title: parent.value,
            value: child.value,
          };
        });
      });
    }
  }

  addCommand() {
    if (this.command?.value?.length > 0) {
      this.commandList.push(this.command.value);
      this.commandFormGroup.addControl(
        'command' + (this.commandList.length - 1),
        new FormControl(this.command.value),
      );
      this.command.reset();
    }
  }

  editCommand(i) {
    this.editIndexList.push(i);
  }

  deleteCommand(i) {
    let newCommandList = [];
    this.commandList.forEach((command, index) => {
      if (index !== i) {
        newCommandList.push(command);
      }
    });
    this.commandList = [...newCommandList];
    let newIndexList = [];
    this.editIndexList.forEach((elem, index) => {
      if (elem !== i) {
        newIndexList.push(elem);
      }
    });
    this.commandFormGroup = new FormGroup({});
    this.commandList.forEach((command, index) => {
      this.commandFormGroup.addControl(
        'command' + index,
        new FormControl(command),
      );
    });
    this.editIndexList = [...newIndexList];
  }

  saveEditChanges(i) {
    let editedCommand = this.commandFormGroup.get('command' + i).value;
    let newCommandList = [];
    this.commandList.forEach((command, index) => {
      if (index === i) {
        newCommandList.push(editedCommand);
      } else {
        newCommandList.push(command);
      }
    });
    this.commandList = [...newCommandList];
    let newIndexList = [];
    this.editIndexList.forEach((elem, index) => {
      if (elem !== i) {
        newIndexList.push(elem);
      }
    });
    this.editIndexList = [...newIndexList];
  }

  cancelEditChanges(i) {
    let newIndexList = [];
    this.editIndexList.forEach((elem, index) => {
      if (elem !== i) {
        newIndexList.push(elem);
      }
    });
    this.commandFormGroup.get('command' + i).setValue(this.commandList[i]);
    this.editIndexList = [...newIndexList];
  }

  removeResource(resource) {
    this.resourceMap = this.resourceMap.filter((item) => item !== resource);
  }

  getCommandValue(command) {
    return JSON.stringify({
      value: command,
      truncateLength: 70,
    });
  }

  triggerRelease() {
    switch (this.releaseType?.value) {
      case 'SELECTIVE':
        this.triggerSelectiveRelease();
        break;
      case 'FULL':
        this.triggerFullRelease();
        break;
      case 'CUSTOM':
        this.triggerCustomRelease();
        break;
      default:
        break;
    }
  }

  triggerCustomRelease() {
    this.applyReleaseInProgress = true;
    const deploymentParams: DeploymentRequest = {
      allowDestroy: this.allowDestruct.value,
      extraEnv: {},
      forceRelease: true,
      overrideBuildSteps: this.commandList,
      releaseComment: this.comments.value,
      releaseType: 'CUSTOM',
      withRefresh: this.withRefresh.value,
      canQueue: true,
    };
    this.deploymentService
      .createDeploymentUsingPOST({
        clusterId: this.clusterId,
        deploymentRequest: deploymentParams,
      })
      .subscribe(
        (response) => {
          mixpanel.track(CUSTOM_RELEASE_TRIGGER_SUCCESS);
          const successMessage =
            response?.status === 'QUEUED'
              ? 'Custom release has been successfully added to queue.'
              : 'Successfully triggered custom release';
          this.handleAPISuccessResponse(successMessage);
        },
        (err) => {
          mixpanel.track(CUSTOM_RELEASE_TRIGGER_FAIL, { error: err });
          this.errorHandler(err);
        },
      )
      .add(() => {
        this.applyReleaseInProgress = false;
      });
  }

  //function to perform selective(or hotfix) release of applications in any environment
  triggerSelectiveRelease(plan = false) {
    plan === true
      ? (this.planReleaseInProgress = true)
      : (this.applyReleaseInProgress = true);
    this.resourceList = [];
    const recipe: HotfixDeploymentRecipe = {};
    this.tempResourceList?.forEach((resource: any) => {
      if (this.groupByMode.value === 'resourceType') {
        this.resourceList.push({
          resourceType: resource?.title,
          resourceName: resource?.value,
        });
      } else {
        const res = resource?.value?.split(':');
        this.resourceList.push({
          resourceType: res[0],
          resourceName: res[1],
        });
      }
    });
    recipe.resourceList = this.resourceList;
    try {
      this.deploymentService
        .runHotfixDeploymentRecipeUsingPOST({
          clusterId: this.clusterId,
          deploymentRecipe: recipe,
          comment: this.comments.value,
          withRefresh: this.withRefresh.value,
          allowDestroy: this.allowDestruct.value,
          forceRelease: this.forceRelease.value,
          isPlan: plan,
          canQueue: true,
        })
        .subscribe(
          (response) => {
            mixpanel.track(
              plan
                ? SELECTIVE_PLAN_RELEASE_TRIGGER_SUCCESS
                : SELECTIVE_RELEASE_TRIGGER_SUCCESS,
            );
            const successMessage =
              response?.status === 'QUEUED'
                ? `Selective ${
                    plan ? 'plan' : ''
                  } release has been successfully added to queue`
                : `Successfully triggered selective ${
                    plan ? 'plan' : ''
                  } release`;
            this.handleAPISuccessResponse(successMessage);
          },
          (err) => {
            mixpanel.track(
              plan
                ? SELECTIVE_PLAN_RELEASE_TRIGGER_FAIL
                : SELECTIVE_RELEASE_TRIGGER_FAIL,
              { error: err },
            );
            this.errorHandler(err);
          },
        )
        .add(() => {
          plan === true
            ? (this.planReleaseInProgress = false)
            : (this.applyReleaseInProgress = false);
        });
    } catch (err) {
      plan === true
        ? (this.planReleaseInProgress = false)
        : (this.applyReleaseInProgress = false);
      this.releaseCloseClickEvent.emit(true);
      this.customToasterService.error('Trigger hotfix failed');
    }
    this.resourceMap = [];
  }

  //function to trigger full release of any environment
  triggerFullRelease() {
    this.applyReleaseInProgress = true;
    this.deploymentService
      .releaseUsingPUT({
        clusterId: this.clusterId,
        withRefresh: this.withRefresh.value,
        allowDestroy: this.allowDestruct.value,
        forceRelease: this.forceRelease.value,
        comment: this.comments.value,
        canQueue: true,
      })
      .subscribe(
        (response) => {
          mixpanel.track(FULL_RELEASE_TRIGGER_SUCCESS);
          const successMessage =
            response?.status === 'QUEUED'
              ? 'Full release has been successfully added to queue'
              : 'Successfully triggered full release';
          this.handleAPISuccessResponse(successMessage);
        },
        (err) => {
          mixpanel.track(FULL_RELEASE_TRIGGER_FAIL, { error: err });
          this.errorHandler(err);
        },
      )
      .add(() => {
        this.applyReleaseInProgress = false;
      });
  }

  triggerPlannedRelease() {
    this.releaseType?.value === 'FULL'
      ? this.triggerPlannedFullRelease()
      : this.triggerSelectiveRelease(true);
  }

  triggerPlannedFullRelease() {
    this.planReleaseInProgress = true;
    let params: any = {
      releaseType: 'PLAN',
      clusterId: this.clusterId,
      withRefresh: this.withRefresh.value,
      allowDestroy: this.allowDestruct.value,
      forceRelease: this.forceRelease.value,
      comment: this.comments.value,
      canQueue: true,
    };
    this.deploymentService
      .releaseV2UsingPUT(params)
      .subscribe(
        (res) => {
          mixpanel.track(FULL_PLAN_RELEASE_TRIGGER_SUCCESS);
          const successMessage =
            res?.status === 'QUEUED'
              ? 'Plan full release has been successfully added to queue'
              : 'Successfully triggered plan full release';
          this.handleAPISuccessResponse(successMessage);
        },
        (err) => {
          mixpanel.track(FULL_PLAN_RELEASE_TRIGGER_FAIL, { error: err });
          this.errorHandler(err);
        },
      )
      .add(() => {
        this.planReleaseInProgress = false;
      });
  }

  triggerApplyRelease() {
    let params: any = {
      releaseType:
        this.planType === 'PLAN' ? 'APPLY_PLAN' : 'APPLY_HOTFIX_PLAN',
      clusterId: this.clusterId,
      withRefresh: this.withRefresh.value,
      allowDestroy: this.allowDestruct.value,
      forceRelease: this.forceRelease.value,
      comment: this.comments.value,
      canQueue: false,
    };
    this.applyRequest = true;
    this.deploymentService
      .releaseV2UsingPUT(params)
      .subscribe(
        (res) => {
          mixpanel.track(APPLY_PLANNED_RELEASE_TRIGGER_SUCCESS);
          const releaseType =
            this.planType === 'HOTFIX_PLAN' ? 'selective' : 'full';
          const successMessage =
            res?.status === 'QUEUED'
              ? `Apply ${releaseType} release has been successfully added to queue`
              : `Successfully triggered apply ${releaseType} release`;

          this.handleAPISuccessResponse(successMessage);
        },
        (err) => {
          mixpanel.track(APPLY_PLANNED_RELEASE_TRIGGER_FAIL, { error: err });
          this.errorHandler(err);
        },
      )
      .add(() => {
        this.applyRequest = false;
      });
  }

  errorHandler(error) {
    this.onAPIErrorService.onAPIError(error?.error?.message, error);
  }

  handleAPISuccessResponse(toasterMsg) {
    this.customToasterService.success(toasterMsg);
    this.setDefaultValues(true);
    this.pageResetEvent.emit();
    this.releaseCloseClickEvent.emit(true);
  }
}
