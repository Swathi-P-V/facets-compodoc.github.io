import { Component, Input, OnInit, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'app-link-with-ellipsis',
  templateUrl: './link-with-ellipsis.component.html',
  styleUrls: ['./link-with-ellipsis.component.scss'],
})
export class LinkWithEllipsisComponent implements OnInit {
  @Input() value: any; // Value which comes in from the parent component
  rowData: any; // Data for row
  rowValue: { display: string[]; hide: string[] } = { display: [], hide: [] }; // value stores some values to display and others to hide
  decorate: boolean = false; // If decorate is true: items would go inside badge-pills
  isRowList: boolean = false;
  truncateAfter: number = 15;
  status: string = 'primary'; //Primary will be set by default
  @Output() optionClickEvent = new EventEmitter<string>();

  // Parse the input value and initialize rowData and decorate flag
  ngOnInit() {
    this.decorate = this.value?.decorate;
    this.rowData = this.value?.data;
    this.truncateAfter = this.value?.truncateAfter;
    this.isRowList = Array.isArray(this.rowData);
    this.isRowList
      ? this.buildRows()
      : this.buildSingleEntry(this.truncateAfter);
    this.status = this.value?.status || 'primary';
  }

  // Puts first X entries in display and others in hidden
  // [Can add host listener in future and update defaultDisplay value with changing width of screen]
  buildRows(defaultDisplay = 3) {
    for (let i = 0; i < this.rowData?.length; i++) {
      if (i < defaultDisplay) {
        if (
          !this.decorate &&
          i != defaultDisplay - 1 &&
          i != this.rowData?.length - 1
        ) {
          this.rowData[i] += ', ';
        }
        this.rowValue.display.push(this.rowData[i]);
      } else {
        if (!this.decorate && i != this.rowData?.length - 1) {
          this.rowData[i] += ', ';
        }
        this.rowValue.hide.push(this.rowData[i]);
      }
    }
  }

  buildSingleEntry(truncateAfter = 15) {
    if (this.rowData) {
      this.rowValue.display = [
        (this.rowData as string).substring(0, truncateAfter),
      ];
      if (this.rowData?.length > truncateAfter) {
        this.rowValue.hide = [(this.rowData as string).substring(0)];
      } else {
        this.rowValue.hide = [''];
      }
    }
  }

  optionClick(option) {
    this.optionClickEvent.emit(option);
  }
}
