import { Component, Input } from '@angular/core';
import { BlueprintDesignerUtilService } from '../../blueprint-designer.service';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { ResourceDetailsService } from '../../resource-details/resource-details.service';
import { ContextUpdatesService } from 'src/app/services/context-updates.service';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';

@Component({
  selector: 'app-resource-node',
  templateUrl: './resource-node.component.html',
  styleUrls: ['./resource-node.component.scss'],
})
export class ResourceNodeComponent extends FacetsPage {
  @Input() nodeInfo: any; // Node data received from the parent component
  @Input() environmentMode: boolean = false;
  @Input() resourceSpecificGraph: boolean = false;
  blueprint: string; // Name of current blueprint
  branchName: string; // Name of current branch

  highlight: boolean = false; // If the node's configuration wrapper is open then the node should be in its highlighted state
  highlightAsNeighbor: boolean = false; // // If the node is a neighbor of the node in focus, then highlight it differently

  constructor(
    public blueprintDesignerUtilService: BlueprintDesignerUtilService,
    private resourceDetailsService: ResourceDetailsService,
    private router: Router,
    private activatedRoute: ActivatedRoute,
    private contextUpdatesService: ContextUpdatesService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.blueprint = p?.stackName;
    this.branchName = p?.branchName;
  }

  // Should tell the service that it needs to open the
  // configuration tool for this particular node
  // This function should also remove the highlight and close the previously open configuration if any
  // This function should also highlight the incoming & outgoing edges to and from this node
  openConfigurationWrapper() {
    if (!this.resourceSpecificGraph) {
      this.blueprintDesignerUtilService.setActiveNode(this.nodeInfo);
      this.blueprintDesignerUtilService.onClickNode(this.nodeInfo);
    }
  }

  // These 2 subscriptions are currently not in use.
  // the highlightsMap inside the service should carry the state and angular CD should capture it from there
  // subscribeToConfigWrapperState() {
  //   this.blueprintDesignerUtilService.configurationWrapperState.subscribe(
  //     (nodeInfo) => {
  //       if (
  //         nodeInfo === 'closed' ||
  //         this.blueprintDesignerUtilService.getResourceID(nodeInfo) !=
  //           this.blueprintDesignerUtilService.getResourceID(this.nodeInfo)
  //       ) {
  //         this.highlight = false;
  //       }
  //     },
  //   );
  // }

  // subscribeToConnectionHighlights() {
  //   this.blueprintDesignerUtilService.connectionHighlights.subscribe(
  //     (highlights) => {
  //       this.highlightAsNeighbor = highlights?.nodes?.has(this.nodeInfo.id);
  //     },
  //   );
  // }

  onClickResource(event, node) {
    event?.stopPropagation();
    if (!this.resourceSpecificGraph) {
      this.router.navigate(
        [
          '/',
          'stack',
          node?.stackName || this.blueprint,
          'blueprint-designer',
          this.branchName,
          'resource',
          'type',
          node?.resourceType,
          'name',
          node?.resourceName,
          'overview',
        ],
        {
          relativeTo: this.activatedRoute,
          state: {
            mode: 'GRAPH',
          },
        },
      );
    } else {
      const currentResource = this.resourceDetailsService.getResource();
      const currentBranchName =
        this.activatedRoute.snapshot.params['branchName'];
      const prevURL = `stack/${currentResource.stackName}/blueprint-designer/${currentBranchName}/resource/type/${currentResource.resourceType}/name/${currentResource.resourceName}/overview`;
      this.resourceDetailsService.pushContext(prevURL);
      setTimeout(() => {
        this.router.navigate(
          [
            `stack/${node.stackName}/blueprint-designer/${currentBranchName}/resource/type/${node.resourceType}/name/${node.resourceName}/overview`,
          ],
          {
            state: {
              mode: 'GRAPH',
            },
          },
        );
      }, 10);
      this.router.navigate([
        `stack/${node.stackName}/blueprint-designer/${currentBranchName}/${currentResource.resourceType}/${currentResource.resourceName}/loading`,
      ]);
    }
  }
}
