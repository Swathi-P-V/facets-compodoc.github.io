import { Component, TemplateRef, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { map } from 'rxjs';
import { Version } from 'src/app/cc-api/models/version';
import cloneDeep from 'lodash/cloneDeep';
import { SmartTableLongStringEllipsisComponent } from 'src/app/components/smart-table-long-string-ellipsis/smart-table-long-string-ellipsis.component';
import { AbstractCluster } from 'src/app/cc-api/models';
import {
  UiStackControllerService,
  UiVersioningControllerService,
} from 'src/app/cc-api/services';
import { SmartTableTimeagoTooltipComponent } from '../../../components/smart-table-timeago-tooltip/smart-table-timeago-tooltip.component';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { NbDialogService } from '@nebular/theme';
import { ArtifactUriComponent } from 'src/app/components/artifact-uri/artifact-uri.component';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-artifact-history',
  templateUrl: './artifact-history.component.html',
  styleUrls: ['./artifact-history.component.scss'],
})
export class ArtifactHistoryComponent extends FacetsPage {
  versioningKey: string = ''; // versioning key to fetch artifact versions
  stackName: string = ''; //blueprint name
  noOfHistory: number = 100; // number of history versions to fetch.
  allVersions: Version[] = [];
  tableData: any = [];
  loading: boolean = true;
  showEmptyState: boolean = false;
  emptyStateLogo = './assets/artifact-empty.svg';
  primaryMsg: string = ''; //
  secondaryMsg: string = '';
  status: 'error' | 'info' = 'info';
  registrationType: string = '';
  envs: AbstractCluster[] = [];
  ciId: string = '';
  streamTableSettings: any;

  constructor(
    private activatedRoute: ActivatedRoute,
    private versioningService: UiVersioningControllerService,
    private router: Router,
    private uiStackController: UiStackControllerService,
    private dialogService: NbDialogService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    if (p?.versioningKey) {
      this.versioningKey = p?.versioningKey;
      this.getEnvs();
    }
    if (p?.ciId) {
      this.ciId = p?.ciId;
    }
    this.activatedRoute.paramMap
      .pipe(map(() => window.history.state))
      .subscribe((data) => {
        this.stackName = data?.stackName ? data?.stackName : '';
      });
  }

  getVersions(key) {
    this.loading = true;
    this.versioningService
      .getVersionsUsingGET({
        versioningKey: this.versioningKey,
        perPage: this.noOfHistory,
      })
      .subscribe(
        (versions) => {
          this.allVersions = cloneDeep(versions);
          let tempTableData = [];
          this.allVersions.forEach((version) => {
            if (version?.['entity']?.['registrationType'] === 'ENVIRONMENT') {
              this.envs.forEach((env) => {
                if (env.id === version?.['entity']?.['clusterId']) {
                  version['entity']['clusterName'] = env.name;
                }
              });
            }
            let tempVersion = {};
            tempVersion = { ...version?.entity };
            tempVersion['version'] = version?.version;
            tempVersion['restoreID'] = version?.id;
            this.registrationType = version?.entity?.['registrationType'];
            tempTableData.push(tempVersion);
          });
          this.setSettings(this.registrationType);
          this.tableData = [...tempTableData];
          if (!this.tableData?.length) {
            this.showEmptyState = true;
            this.primaryMsg =
              'No version history available for selected artifact';
            this.secondaryMsg = '';
            this.status = 'info';
          }
          this.loading = false;
        },
        (err) => {
          this.loading = false;
          this.showEmptyState = true;
          this.primaryMsg = '';
          this.secondaryMsg =
            err?.error?.message ||
            'There is some error fetching artifact version history';
          this.status = 'error';
          this.emptyStateLogo = errorIcon;
        },
      );
  }

  getEnvs() {
    this.loading = true;
    this.uiStackController
      .getAllClustersUsingGET({
        page: 0,
        perPage: 10000,
        sortBy: 'ID',
      })
      .subscribe(
        (res) => {
          this.envs = res.content;
          this.getVersions(this.versioningKey);
        },
        (err) => {
          this.loading = false;
        },
      );
  }

  setSettings(registrationType) {
    let newSettings = {};
    let settingsColumn = {};
    settingsColumn['version'] = {
      title: 'Version',
      width: '10%',
      type: 'text',
      filter: false,
    };
    settingsColumn['releaseStream'] = {
      title: 'Release Stream',
      filter: true,
      width: '15%',
      type: 'text',
      valuePrepareFunction: (releaseStream, row) => {
        if (releaseStream) {
          return releaseStream;
        } else {
          return 'N/A';
        }
      },
    };
    if (this.registrationType === 'ENVIRONMENT') {
      settingsColumn['clusterName'] = {
        title: 'Environment',
        filter: true,
        width: '10%',
        type: 'text',
        valuePrepareFunction: (env, row) => {
          if (env) {
            return env;
          } else {
            return 'N/A';
          }
        },
      };
    }
    settingsColumn['buildId'] = {
      title: 'Build Id',
      width: '10%',
      type: 'text',
      filter: false,
      valuePrepareFunction: (buildId, row) => {
        if (buildId) {
          return buildId;
        } else {
          return 'N/A';
        }
      },
    };
    settingsColumn['artifactUri'] = {
      title: 'Artifact URI',
      filter: false,
      type: 'custom',
      width: '20%',
      valuePrepareFunction: (cell, row) => {
        return JSON.stringify({
          uri: cell,
        });
      },
      renderComponent: ArtifactUriComponent,
    };
    settingsColumn['buildDescription'] = {
      title: 'Description',
      width: '15%',
      filter: false,
      type: 'custom',
      valuePrepareFunction: (cell: any) => {
        return JSON.stringify({
          value: cell?.length > 0 ? cell : '-',
          truncateLength: 15,
        });
      },
      renderComponent: SmartTableLongStringEllipsisComponent,
    };
    settingsColumn['lastModifiedBy'] = {
      title: 'Modified By',
      width: '20%',
      filter: true,
      type: 'custom',
      valuePrepareFunction: (cell, row) => {
        return JSON.stringify({
          user: row.lastModifiedBy,
          label: row.lastModifiedDate,
        });
      },
      renderComponent: SmartTableTimeagoTooltipComponent,
    };
    let actionButtons = [
      {
        id: 'rollback',
        icon: 'refresh-outline',
        toolTip: 'Artifact rollback',
        status: 'primary',
        size: 'medium',
        appearance: 'ghost',
        action: 'ARTIFACTS_WRITE',
        onClick: (data) => {
          this.openRollBackDialog(data);
        },
      },
    ];
    settingsColumn['custom'] = {
      type: 'custom',
      title: 'Actions',
      valuePrepareFunction: (value, row, cell) => {
        const displayVal = {
          smartTableCustomActionButtons: actionButtons,
        };
        return JSON.stringify(displayVal);
      },
      renderComponent: CustomActionsComponent,
      onComponentInitFunction: (instance) => {
        instance.byCustomActionButtonClick.subscribe((data) => {
          let customActionButton = actionButtons.filter((actionButton) => {
            return actionButton.id === data.actionButton.id;
          })[0];
          customActionButton.onClick(data.rowData);
        });
      },
      width: '12%',
      filter: false,
    };
    newSettings['columns'] = settingsColumn;
    newSettings['noDataMessage'] = 'No history present for given artifact';
    newSettings['mode'] = 'external';
    newSettings['actions'] = 'false';
    newSettings['pager'] = {
      display: false,
      perPage: 10,
    };
    this.streamTableSettings = newSettings;
  }

  goBack = () => {
    this.router.navigate(['/', 'artifact-ci-details', this.ciId]);
  };

  versionRollbackInProgress: boolean = false;
  @ViewChild('openRollbackConfirmPopup', { static: true })
  private openRollbackConfirmPopup: TemplateRef<any>;
  openRollBackDialog(artifact) {
    this.dialogService.open(this.openRollbackConfirmPopup, {
      context: {
        restoreId: artifact?.restoreID,
        version: artifact?.version,
        buildId: artifact?.buildId,
      },
    });
  }

  versionRollback(versiondetails, ref) {
    this.versionRollbackInProgress = true;
    this.versioningService
      .restoreUsingPOST(versiondetails?.restoreId)
      .subscribe(
        (data) => {
          this.versionRollbackInProgress = false;
          this.customToastService.success(`Artifact Rollback was successful`);
          ref.close();
          this.getEnvs();
        },
        (error) => {
          this.versionRollbackInProgress = false;
          ref.close();
          this.onAPIErrorService.onAPIError(
            error?.error?.message || `Artifact Rollback failed `,
            error,
          );
        },
      );
  }
}
