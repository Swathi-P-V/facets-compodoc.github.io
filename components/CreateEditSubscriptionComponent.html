@Component({
  selector: 'app-create-edit-subscription',
  templateUrl: './create-edit-subscription.component.html',
  styleUrls: ['./create-edit-subscription.component.scss'],
})
export class CreateEditSubscriptionComponent
  extends FacetsPage
  implements OnDestroy
{
  headingOptions = {
    create: 'Create Subscription',
    edit: 'Edit Subscription',
  };
  mode: any;
  sub: Subscription = {};
  stackOptions = [];
  notificationTypeOptions = [];
  channelOptions = [];
  channelNameMap = {};
  channelAddressMap = {};
  integrationKeyMap = {};
  optionalFilters: {
    id: string;
    label: string;
    tooltip: string;
    multiSelect: boolean;
    selectedValues: [];
    selectValues: [];
  }[] = [];
  presetOptionalFilters: {
    // Preserves the state of the updated form fields of this component while going back & forth with the add-subscription-payload component
    id: string;
    label: string;
    tooltip: string;
    multiSelect: boolean;
    selectedValues: [];
    selectValues: [];
  }[] = [];
  loadingFilters: boolean;
  errorMessage: string = '';
  subscriptionActionInProgress: boolean;
  editId: string;
  initialising: any;
  currentChannelType: string; // To track if the channel type ever becomes 'WEBHOOK'
  addPayloadMode: boolean = false; // (Currently not in use) state to let the template know that it only needs to render the router outlet content
  payload: any; // This payload will come back from the add-subscription-payload component
  skipPopulateForEdit: boolean = false; // Once the form fields have been updated on this page
  // We would not want them to be set back to default in the event of routing to the
  // add-subscription-payload and then coming back, this flag prevents the
  // form field data to be reverted back
  @ViewChild('currentPayloadDialog', { static: true })
  private currentPayloadDialog: TemplateRef<any>; // Toggles the dialog box with current payload
  editorOptions: MonacoEditorConstructionOptions = {
    // Editor options for the monaco editor
    theme: 'vs-light',
    language: 'json',
    readOnly: true,
    minimap: {
      enabled: false,
    },
  };
  cdkOverlayContainerNode: any;
  optionalFiltersMutationObserver: MutationObserver;
  isOptionalFilterDrodpdownClosed = new BehaviorSubject(true);
  currentObservingOptionalFilterId: any = null;
  previousStateOfOptionalFilters: any = null;
  testingResult: string;
  notificationError: string;
  subscriptionWrite = SUBSCRIPTION_WRITE;
  testNotificationInprogress: boolean = false;

  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    private uiStackControllerService: UiStackControllerService,
    private uiNotificationControllerService: UiNotificationControllerService,
    private dialogBoxService: NbDialogService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
    this.setRouterStateListener();
  }

  //Disconnects mutation observer and behavior subject that tracks dropdown open and close state.
  ngOnDestroy(): void {
    this.optionalFiltersMutationObserver?.disconnect();
    this.isOptionalFilterDrodpdownClosed?.unsubscribe();
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    if (p) {
      this.mode = p.mode;
      this.editId = p.editId;
      this.initialising = true;
      //Fetch all the Blueprints available for the control plane
      this.uiStackControllerService.getStacksUsingGET1().subscribe((res) => {
        res.forEach((stack) => {
          var option = {};
          option['value'] = stack.name;
          option['label'] = stack.name;
          this.stackOptions.push(option);
        });
      });

      //Fetch all the Notification Types available
      this.uiNotificationControllerService
        .getAllNotificationTypesUsingGET()
        .subscribe((res) => {
          res.forEach((type) => {
            var option = {};
            option['value'] = type.key;
            option['label'] = type.notificationName;
            this.notificationTypeOptions.push(option);
          });
          if (this.mode === 'edit' && this.editId) {
            this.populateValuesForEdit();
          }
        });

      //Fetch all channels
      this.uiNotificationControllerService
        .getAllChannelsUsingGET()
        .subscribe((res) => {
          res.forEach((channel) => {
            this.channelNameMap[channel.id] = channel.name;
            this.channelAddressMap[channel.id] = channel.channelAddress;
            this.integrationKeyMap[channel.id] = channel.integrationKey;
            var option = {};
            option['value'] = channel.id;
            option['label'] = channel.name + ' - ' + channel.channelType;
            option['type'] = channel.channelType;
            this.channelOptions.push(option);
            if (this.mode === 'create') {
              this.initialising = false;
            }
          });
        });
    }
  }

  //This is called when a selected value from any of the optional filter changes.
  //This function sets a mutation observer on .cdk-overlay-container (cdk-overlay-container gets rendered when a dropdown is openend).
  //The 'cdkOverlayContainerNode' value and 'optionalFiltersMutationObserver' is set only for the first selection and persisted till the dropdown is closed.
  //Once the drodpdown closes(ie; nodes below .cdk-overlay-container is zero), the mutation observer is disconnected.
  //A behavior subject 'isOptionalFilterDrodpdownClosed' is used to reload the optional filters only on dropdown close.

  setMutationObserver(filterList, id) {
    //A global variable is set to track the optional filter which is currently being observed.
    this.currentObservingOptionalFilterId = id;

    //Set isOptionalFilterDrodpdownClosed value as false as dropdown is open.
    //This prevents reloading filters whenever select is changed and dropdown is still open.
    this.isOptionalFilterDrodpdownClosed.next(false);

    if (!this.cdkOverlayContainerNode) {
      this.cdkOverlayContainerNode = document.querySelector(
        '.cdk-overlay-container',
      );
    }

    if (!this.optionalFiltersMutationObserver) {
      this.optionalFiltersMutationObserver = new MutationObserver(
        (mutations) => {
          mutations.forEach((mutation) => {
            if (mutation.addedNodes.length == 0) {
              //Set isOptionalFilterDrodpdownClosed as true when dropdown closes.
              this.isOptionalFilterDrodpdownClosed.next(true);
              this.optionalFiltersMutationObserver.disconnect();
            }
          });
        },
      );
    }

    //Connects (for the first time and after every disconnect) to optionalFiltersMutationObserver
    this.optionalFiltersMutationObserver.observe(this.cdkOverlayContainerNode, {
      attributes: true,
      childList: true,
      characterData: true,
    });

    //Subscribe to 'isOptionalFilterDrodpdownClosed' Behavior subject to listen to optional filter dropdown open or close events.
    //If closed then check if the dropdown selected value is changed. If selectedValue is changed then reload the optional filters.
    this.isOptionalFilterDrodpdownClosed.subscribe(async (value) => {
      if (value) {
        await this.isPreviousAndCurrentSelectedValueDifferent(
          this.previousStateOfOptionalFilters,
          this.optionalFilters,
          this.currentObservingOptionalFilterId,
        ).then((data) => {
          if (data) {
            this.getFilterDropdown(
              this.optionalFilters,
              this.currentObservingOptionalFilterId,
            );
          }
        });
      }
    });
  }

  isPreviousAndCurrentSelectedValueDifferent(previousList, currentList, id) {
    return new Promise((resolve, reject) => {
      if (
        this.getFilterValue(previousList, id)?.sort()?.toString() ==
        this.getFilterValue(currentList, id)?.sort()?.toString()
      ) {
        return resolve(false);
      } else {
        return resolve(true);
      }
    });
  }

  //This function returns the selected value of filter whose id is equal to param id.
  getFilterValue(filterList, id) {
    let selectedValue = [];
    if (filterList) {
      selectedValue = filterList.filter((element) => element.id === id)[0]
        .selectedValues;
    }
    return selectedValue;
  }

  // Data is send back & forth to the add-subscription-payload component
  // so that the state is preserved (If optionalFilters is present, the populateForEdit API call is skipped)
  // The payload is also sent back to this component through the state object
  setRouterStateListener() {
    this.activatedRoute.paramMap
      .pipe(map(() => window.history.state))
      .subscribe((state) => {
        // console.log('State from Create-edit-subscription => ', state);
        if (state.optionalFilters?.length) {
          this.presetOptionalFilters = [...state.optionalFilters];
          this.skipPopulateForEdit = true;
        }
        if (state.subscription) {
          this.sub = state.subscription;
          this.currentChannelType = this.sub.channelType;
          this.getNotificationTags(false);
        }
        if (state.payload != this.sub.payloadJson) {
          this.sub.payloadJson = state.payload;
          this.customToastService.success(`Payload updated successfully`);
        }
      });
  }

  populateValuesForEdit() {
    let subscriptionsRes = [];
    this.uiNotificationControllerService
      .getAllSubscriptionsUsingGET()
      .subscribe((res) => {
        if (res && res.length > 0) {
          subscriptionsRes = res;
        }
        subscriptionsRes.forEach((s) => {
          if (this.editId === s.id) {
            if (s.channelId) {
              s['channelName'] = this.channelNameMap[s.channelId];
            }
            if (!this.skipPopulateForEdit) {
              this.sub = s;
              this.updateCurrentChannelType();
            }
            this.initialising = false;
            if (!this.skipPopulateForEdit) {
              this.loadingFilters = true;
              this.uiNotificationControllerService
                .getNotificationTagsForNotificationTypeUsingGET(
                  this.sub.notificationType,
                )
                .subscribe(
                  (data) => {
                    if (this.presetOptionalFilters?.length) {
                      data = this.presetOptionalFilters as any;
                    }
                    if (data?.length !== 0) {
                      data.forEach((filter) => {
                        filter['selectedValues'] = this.sub.filters[filter.id];
                      });
                      this.getFilterDropdown(data);
                    } else {
                      this.loadingFilters = false;
                    }
                  },
                  (error) => {
                    console.log('Error', error);
                    this.errorMessage = error?.error?.message;
                    this.loadingFilters = false;
                  },
                );
            } else {
              this.addCachedOptionalFilters();
            }
          }
        });
      });
  }

  addCachedOptionalFilters() {
    let cachedData: NotificationTagsForTypeResult[] = [];
    (this.presetOptionalFilters as any[]).forEach((filter) => {
      let data = {} as NotificationTagsForTypeResult;
      data['id'] = filter.id;
      data['label'] = filter.label;
      data['selectedValues'] = filter.selectedValues;
      data['tooltip'] = filter.tooltip;
      cachedData.push(data);
    });
    this.getFilterDropdown(cachedData);
  }

  getNotificationTags(newListNeeded) {
    if (newListNeeded) {
      this.skipPopulateForEdit = false;
    }

    //get Notification Tags only when the user has selected both blueprint and notification type fields
    if (this.sub.stackName && this.sub.notificationType) {
      if (this.skipPopulateForEdit) {
        this.addCachedOptionalFilters();
        return;
      }
      this.optionalFilters = [];
      this.sub.filters = {};
      this.loadingFilters = true;

      this.uiNotificationControllerService
        .getNotificationTagsForNotificationTypeUsingGET(
          this.sub.notificationType,
        )
        .subscribe(
          (data) => {
            if (data?.length !== 0) {
              this.getFilterDropdown(data);
            } else {
              this.loadingFilters = false;
            }
          },
          (error) => {
            console.log('Error', error);
            this.errorMessage = error?.error?.message;
            this.loadingFilters = false;
          },
        );
    }
  }

  getFilterDropdown(filterList, id?) {
    filterList.sort((a, b) => (a.id > b.id ? 1 : -1));
    let count = 0;
    this.loadingFilters = true;
    this.optionalFilters = [];
    let duplicateOptionalFilters = [];

    let selectedFilterObject = {};
    selectedFilterObject['STACK_NAME'] = [this.sub.stackName];

    //Contruct filter object with already selected values.
    filterList.forEach((filter) => {
      if (filter?.selectedValues?.length !== 0) {
        selectedFilterObject[filter.id] = filter?.selectedValues;
      }
    });

    //For each optional filter available, if value is already selected don't fetch the dropdown list again
    //else fetch the dropdown values by sending all the filters selected so far
    filterList.forEach(async (filter) => {
      if (id && filter.id === id) {
        duplicateOptionalFilters.push(filter);
        count++;
      } else {
        await this.getDropdownValues(filter, selectedFilterObject).then(
          (eachFilter) => {
            duplicateOptionalFilters.push(eachFilter);
            count++;
          },
          (error) => {
            filter.selectValues = undefined;
            filter.selectedValues = undefined;
            duplicateOptionalFilters.push(filter);
            count++;
          },
        );
      }
      // Once all the dropdowns are fetched re render the optional filters
      // If presetOptionalFilters are present, it means that we need to initialise the form fields
      // with the data we already had (to preserve the state and not reintialise this with either
      // blanks or the data which will be sent back from the backend).
      if (count === filterList?.length) {
        if (this.presetOptionalFilters && this.presetOptionalFilters?.length) {
          this.optionalFilters = this.presetOptionalFilters;
          this.presetOptionalFilters = null;
        } else {
          duplicateOptionalFilters.sort((a, b) => (a.id > b.id ? 1 : -1));
          this.optionalFilters = duplicateOptionalFilters;
        }
        this.sub.filters = selectedFilterObject;
        this.loadingFilters = false;
        //Once the dropdown closes, save a clone of currentState of optional filters.
        //This will be used to compare on next change of any dropdown value.
        this.previousStateOfOptionalFilters = cloneDeep(this.optionalFilters);
      }
    });
  }

  getDropdownValues(filter, qp) {
    return new Promise((resolve, reject) => {
      this.uiNotificationControllerService
        .getFiltersForSubscriptionsUsingPOST({
          tagName: filter.id,
          notificationType: this.sub.notificationType,
          filterParams: qp,
        })
        .subscribe(
          (data) => {
            data['values'].sort((a, b) => (a.label > b.label ? 1 : -1));

            let obj = {};
            obj['id'] = filter.id;
            obj['label'] = filter.label;
            obj['tooltip'] = filter.tooltip;
            obj['multiSelect'] = data['isMultiSelect'];
            obj['selectValues'] = data['values'];
            obj['selectedValues'] = filter['selectedValues']
              ? [...filter['selectedValues']]
              : [];
            resolve(obj);
          },
          (error) => {
            if (error?.status === 500 || error?.status === 404) {
              reject(false);
            }
          },
        );
    });
  }

  private populateChannels(res: Array<NotificationChannel>) {
    this.channelOptions = [];
    res.forEach((channel) => {
      this.channelNameMap[channel.id] = channel.name;
      var option = {};
      option['value'] = channel.id;
      option['label'] = channel.name + ' - ' + channel.channelType;
      this.channelOptions.push(option);
    });
  }

  createOrEditSubscription() {
    this.sub.filters = this.sub.filters || (this.optionalFilters as any) || {};
    this.subscriptionActionInProgress = true;
    let httpReq;

    if (this.mode === 'create') {
      httpReq =
        this.uiNotificationControllerService.createSubscriptionUsingPOST(
          this.sub,
        );
    } else if (this.mode === 'edit') {
      httpReq = this.uiNotificationControllerService.editSubscriptionUsingPUT({
        subscription: this.sub,
        subscriptionId: this.sub.id,
      });
    }
    const { successSubText, failureSubText } = getToastSubTexts(this.mode);
    httpReq.subscribe(
      (response) => {
        this.customToastService.success(
          `Successfully ${successSubText} subscription`,
        );
        this.subscriptionActionInProgress = false;
        this.handleBack();
      },
      (error) => {
        this.onAPIErrorService.onAPIError(
          error?.error?.message || `Failed to ${failureSubText} subscription`,
          error,
        );
        this.subscriptionActionInProgress = false;
        this.handleBack();
      },
    );
  }

  handleBack = () => {
    this.router.navigate(['/settings/notification-center/subscriptions']);
  };

  // Used for checking if the channel type ever becomes 'WEBHOOK'
  updateCurrentChannelType() {
    this.currentChannelType = this.channelOptions.find(
      (channel) => channel.value === this.sub.channelId,
    )?.type;
    // console.log('Current channel type => ', this.currentChannelType);
    this.sub.channelType = this.currentChannelType as any;
  }

  // Opens another component to add/edit payload for WEBHOOK and sends state of current form fields to let it stay cached
  addPayloadConfig() {
    let url = this.editId
      ? `/settings/notification-center/subscription/edit/${this.editId}/edit/payload`
      : `/settings/notification-center/subscription/create/payload`;
    this.addPayloadMode = true;
    this.sub.channelAddress = this.channelAddressMap[this.sub.channelId];
    this.router.navigate([url], {
      state: {
        subscription: this.sub,
        channelName: this.channelNameMap[this.sub.channelId],
        optionalFilters: this.optionalFilters,
        previousComponent: this.editId
          ? 'edit-subscription'
          : 'create-subscription',
      },
    });
  }

  removePayload() {
    this.sub.payloadJson = null;
  }

  // Displays the current payload JSON inside a dialog box
  showCurrentPayload() {
    this.dialogBoxService.open(this.currentPayloadDialog, {
      hasBackdrop: true,
      autoFocus: false,
    });
  }

  testNotification() {
    this.testingResult = '';
    this.notificationError = '';
    this.testNotificationInprogress = true;
    let payload: TestNotificationRequest = {
      channelAddress: this.channelAddressMap[this.sub.channelId],
      channelType: this.sub.channelType,
      integrationKey:
        this.currentChannelType == 'PAGER_DUTY'
          ? this.integrationKeyMap[this.sub.channelId]
          : '',
      notificationType: this.sub?.notificationType
        ?.toUpperCase()
        ?.replace(' ', '_') as any,
      payload:
        this.currentChannelType == 'WEBHOOK' ? this.sub.payloadJson : null,
    };

    this.uiNotificationControllerService
      .testNotificationChannelUsingPOSTResponse(payload)
      .subscribe(
        (response) => {
          if (response) {
            this.testingResult = 'Test Notification succeeded';
          }
          this.testNotificationInprogress = false;
        },
        (error) => {
          try {
            error = JSON.parse(error.error);
          } catch (error) {
            error = {};
          }
          this.testNotificationInprogress = false;
          this.notificationError =
            'Error: ' + (error?.message || 'Test Notification failed');
        },
      );
  }
}
