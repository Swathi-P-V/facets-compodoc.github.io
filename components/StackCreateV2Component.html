  createStack() {
    try {
      this.callInProgress = true;
      let requestBodyV1: Stack = {
        name: this.name?.value,
        versionControlAccountId: this.accountId?.value,
        template: this.templatise?.value,
        description: this.description?.value,
        relativePath: this.relativePath?.value,
        primaryCloud: this.primaryCloud?.value,

        branch: this.branch?.value,
        vcs: this.vcs?.value,
        templateId:
          this.creationType?.value === 'NEW' && !this.templateId?.value
            ? 'emptystack'
            : undefined,
        templateName:
          this.creationType?.value === 'EXISTING'
            ? null
            : this.templateId?.value || undefined,
      };

      let requestBodyV2: CreateBlueprintRequest = {
        name: this.name?.value,
        githubAccountId: this.accountId?.value,
        template: this.templatise?.value,
        templateName: this.templateId?.value,
        description: this.description?.value,
        relativePath: this.relativePath?.value,
        primaryCloud: this.primaryCloud?.value,
        githubOrganisationName: this.organisationName?.value,
        monoRepo: this.isCreate
          ? this.featureFlagsService.getIsMonoRepo()
          : this.stack?.monoRepo,
      };

      const httpRequest =
        this.vcs?.value !== 'GITHUB' || this.creationType?.value === 'EXISTING'
          ? this.uiStackControllerService.createStackUsingPOST1(requestBodyV1)
          : this.uiStackControllerService.createStackV2UsingPOST(requestBodyV2);

      httpRequest.subscribe(
        (stack) => {
          this.onStackCreateSuccess(stack);
        },
        (error) => {
          this.onStackCreateFailure(error);
        },
      );
    } catch (err) {

      this.callInProgress = false;
    }
  }

  updateVcsOrgNameValidators(creationType) {
    const validators = [
      Validators.pattern(whiteSpaceInputPattern),
      Validators.pattern(urlDetectionPattern),
    ];
    if (creationType === 'EXISTING') {
      this.stackForm
        .get('vcsUrl')
        .setValidators([...validators, Validators.required]);
      this.stackForm?.get('organisationName').setValidators([]);
    } else if (creationType === 'NEW') {
      if (this.vcs?.value && this.vcs?.value !== 'GITHUB') {
        this.stackForm
          .get('vcsUrl')
          .setValidators([...validators, Validators.required]);
        this.stackForm?.get('organisationName').setValidators([]);
      } else {
        this.stackForm.get('vcsUrl').setValidators([...validators]);
        this.stackForm
          ?.get('organisationName')
          .setValidators(
            this.templatise?.value
              ? []
              : [
                  Validators.required,
                  Validators.pattern(whiteSpaceInputPattern),
                ],
          );
      }
    }
    this.stackForm.get('organisationName').updateValueAndValidity();
  }

  replaceElem(key, item, logItem, classNameList) {
    const text = new DOMParser().parseFromString(
      item?.innerHTML?.toLowerCase(),
      'text/html',
    ).documentElement?.textContent;
    let HTMLLowerCase: string = text?.toLowerCase();
    // @ts-ignore
    let newHTML = HTMLLowerCase.replaceAll(
      new RegExp(key, 'ig'),
      '<span class="highlight__">' + key + '</span>',
    );
    let newSpan = document.createElement('span');
    newSpan.classList.add(...classNameList);
    newSpan.innerHTML = newHTML;
    logItem.replaceChild(newSpan, item);
  }

  search(event) {
    const key = event?.target?.value?.toLowerCase();
    this.secondaryTemplates = cloneDeep(this.originalSecondaryTemplates).filter(

    ref.close();
