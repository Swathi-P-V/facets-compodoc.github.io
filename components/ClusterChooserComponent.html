import { Component, OnInit } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CPCloudService } from 'src/app/services/cp-cloud.service';

@Component({
  selector: 'app-cluster-chooser',
  templateUrl: './cluster-chooser.component.html',
  styleUrls: ['./cluster-chooser.component.scss'],
})
export class ClusterChooserComponent extends FacetsPage {
  stackName: any;
  cloudProviders = [
    {
      cloud: 'aws',
      name: 'AWS',
      title: 'Public Cloud',
      pictureURL:
        'https://images.squarespace-cdn.com/content/v1/52ca3b73e4b04a45ef2c5cb6/1551884861331-C9U2RHJQLOPL9F332X5O/ke17ZwdGBToddI8pDm48kK6x8IOhzX_rTtQGKY1qp-hZw-zPPgdn4jUwVcJE1ZvWhcwhEtWJXoshNdA9f1qD7XaGPskfQtVutuSA2B-JW6wkqqV024XP-059DP6wRYqYVUGMzsBxTo2UlkZq10GTrg/AWS_blog_01.PNG?format=750w',
      description:
        'Launch your blueprint in an AWS account. You should have Admin privileges to this account in order to complete the creation process.',
    },
    {
      cloud: 'azure',
      name: 'Azure',
      title: 'Public Cloud',
      pictureURL:
        'https://cdn.icon-icons.com/icons2/2407/PNG/512/azure_icon_146223.png',
      description:
        'Launch your blueprint in an Azure account. You should have Admin privileges to this account in order to complete the creation process.',
    },
    {
      cloud: 'gcp',
      name: 'GCP',
      title: 'Public Cloud',
      pictureURL:
        'https://icon-library.com/images/google-cloud-platform-icon/google-cloud-platform-icon-9.jpg',
      description:
        'Launch your blueprint in an GCP account. You should have Admin privileges to this account in order to complete the creation process.',
    },
    {
      cloud: 'kubernetes',
      name: 'Kubernetes',
      title: '',
      pictureURL:
        'https://cdn.icon-icons.com/icons2/2699/PNG/512/kubernetes_logo_icon_168359.png',
      description:
        'Launch your blueprint in any kubernetes cluster. You should have Admin privileges to this cluster in order to complete the creation process.',
    },
  ];

  constructor(
    private activatedRoute: ActivatedRoute,
    private router: Router,
    public cPCloudService: CPCloudService,
  ) {
    super(activatedRoute);
  }

  OnPageInit(): void {}

  OnPageLoad(p: Params): void {
    this.stackName = p.stackName;
  }

  navigateToCreateCluster(stackName: string, cloud: string) {
    this.router.navigate([
      stackName,
      'chooseClusterCreate',
      'clusterCreate',
      cloud,
    ]);
  }

  handleBack = () => {
    this.router.navigate(['stack', this.stackName]);
  };
}

  </ng-container>
</div>
