import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { cloneDeep } from 'lodash';
import {
  OpaPolicy,
  OpaPolicyExecutionResult,
  OpaPolicyRequest,
} from 'src/app/cc-api/models';
import {
  UiDropdownsControllerService,
  UiOpaControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { OPA_WRITE } from 'src/app/constants/permissions';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';

@Component({
  selector: 'app-create-guardrails',
  templateUrl: './create-guardrails.component.html',
  styleUrls: ['./create-guardrails.component.scss'],
})
export class CreateGuardrailsComponent extends FacetsPage {
  readonly OPA_WRITE = OPA_WRITE;
  sampleRegoCode =
    'package <package_name>\ndefault allow = false\nallow {\n\t// your code\n}\nelse = {"error": "<your_error>"}';
  mode: 'create' | 'edit' = 'create';
  // Form
  policyForm = new FormGroup({
    policyName: new FormControl('', [
      Validators.required,
      Validators.minLength(3),
      Validators.maxLength(40),
      Validators.pattern('^[a-zA-Z_][a-zA-Z0-9_]*$'),
    ]),
    description: new FormControl(''),
    stackName: new FormControl('', [Validators.required]),
    clusterIds: new FormControl([]),
    resourceTypes: new FormControl([]),
    resourceTypeAndResourceNamesMap: new FormControl({}),
    policyType: new FormControl('RAW'),
    rawPolicy: new FormControl(
      this.mode === 'create' ? this.sampleRegoCode : '',
      [Validators.required],
    ),
    packageName: new FormControl(''),
    parameters: new FormControl({}),
    policyTemplateName: new FormControl(''),
    severity: new FormControl('WARNING', [Validators.required]),
    disabled: new FormControl(false),
  });
  // Dropdown options
  blueprintsDropdown = [];
  environmentsDropdown = [];
  resourceTypesDropdown = [];
  resourceNamesDropdown = [];
  resourceTypesDropdownForSelectedStack: any[];
  // Map
  blueprintEnvironmentsMap: {};
  stackSpecificResourcesMap: {};
  clusterSpecificResourcesMap: {};
  selectedClusterResourcesMap: {};
  //  Loaders
  loading: boolean = true;
  policyRequestInProgress: boolean = false;
  fetchingEnvsDropdown: boolean = false;
  fetchingResTypeDropdown: boolean = false;
  fetchingResNamesDropdown: boolean = false;
  fetchingEditDetails: boolean;

  errorList: string[] = [];
  clonedPolicyName: any;

  constructor(
    private router: Router,
    private _route: ActivatedRoute,
    private uiStackControllerService: UiStackControllerService,
    private uiDropdownsControllerService: UiDropdownsControllerService,
    private uiOpaControllerService: UiOpaControllerService,
    private customToastService: CustomToastService,
  ) {
    super(_route);
  }

  get policyName() {
    return this.policyForm.get('policyName');
  }

  get description() {
    return this.policyForm.get('description');
  }

  get stackName() {
    return this.policyForm.get('stackName');
  }

  get clusterIds() {
    return this.policyForm.get('clusterIds');
  }

  get resourceTypeAndResourceNamesMap() {
    return this.policyForm.get('resourceTypeAndResourceNamesMap');
  }

  get resourceTypes() {
    return this.policyForm.get('resourceTypes');
  }

  get rawPolicy() {
    return this.policyForm.get('rawPolicy');
  }

  OnPageLoad(p: Params): void {
    this.route.queryParams.subscribe((params) => {
      this.clonedPolicyName = params['clonePolicy'];
    });
    this.asyncOnPageLoad(p);
  }

  async asyncOnPageLoad(p: Params) {
    // Get the blueprints and environments
    await this.getClusterAndBlueprintMapping();
    if (this.clonedPolicyName?.length > 0) {
      await this.getPolicyDetails(this.clonedPolicyName);
      // Need to change the name
      this.policyForm.patchValue({
        policyName: this.policyName.value + '_copy',
      });
    }
    if (p?.policyName) {
      this.mode = 'edit';
      await this.getPolicyDetails(p?.policyName);
    }
    this.loading = false;
  }

  handleBack = () => {
    this.router.navigate(['settings/guardrails']);
  };

  // When blueprint is selected or changed, fetch environments and resourceTypes dropdown
  // And reset environments, resourceTypes and resources fields.
  async onBlueprintSelection() {
    if (!this.fetchingEditDetails) {
      this.fetchingEnvsDropdown = true;
      this.fetchingResTypeDropdown = true;

      await this.getStackLevelResourceTypeAndResourcesMapping(
        this.stackName.value,
      );
      // Need not call this API on every select of cluster as this is stack specific
      await this.getClusterLevelResourceTypeAndResourcesMap(
        this.stackName.value,
      );

      this.environmentsDropdown = this.constructEnvironmentsDropdown(
        this.stackName.value,
        this.blueprintEnvironmentsMap,
      );
      this.resourceTypesDropdown = cloneDeep(
        this.resourceTypesDropdownForSelectedStack,
      );

      this.policyForm.patchValue({
        clusterIds: [],
        resourceTypeAndResourceNamesMap: {},
        resourceTypes: [],
      });

      this.fetchingEnvsDropdown = false;
      this.fetchingResTypeDropdown = false;
    }
  }

  async onEnvironmentSelection(selectedEnvs) {
    // First set the selected values to formControl and reset value for resourceTypes and resourceNames
    this.fetchingResTypeDropdown = true;
    let environments = selectedEnvs.map((env) => env?.value);
    this.policyForm.patchValue({
      clusterIds: environments,
      resourceTypes: [],
      resourceTypeAndResourceNamesMap: {},
    });

    // If last environment is deselected then fetch resourceType based on stackName
    if (this.clusterIds.value?.length === 0 && this.stackName.value) {
      this.resourceTypesDropdown = cloneDeep(
        this.resourceTypesDropdownForSelectedStack,
      );
      this.resourceNamesDropdown = [];
      this.fetchingResTypeDropdown = false;
      return;
    }

    // Now for these clusters get the resourceTypes and resources
    this.resourceTypesDropdown = Object.keys(
      this.constructResourcesMapForSelectedClusters(this.clusterIds.value),
    )?.map((resourceType) => ({
      label: resourceType,
      value: resourceType,
      checked: false,
    }));

    //  As no resourceType is selected, clear the resources dropdown
    this.resourceNamesDropdown = [];
    this.fetchingResTypeDropdown = false;

    // TODO - Retain all the selected resourceTypes and resourceNames
  }

  onResourceTypeSelection(selectedResTypes) {
    this.fetchingResNamesDropdown = true;
    let removedResourceType;
    // First check if resourceType was removed or added
    let resourceTypes = selectedResTypes.map((resType) => resType?.value);
    if (resourceTypes?.length < this.resourceTypes.value.length) {
      removedResourceType = this.findDeletedItems(
        this.resourceTypes.value,
        resourceTypes,
      );
    }
    this.policyForm.patchValue({
      resourceTypes: resourceTypes,
    });

    // There could be previously selected resourceNamesDropdown
    const selectedResources = this.resourceTypeAndResourceNamesMap.value;

    // If clusters are selected then fetch resourceType at cluster level
    const resourcesMap =
      this.clusterIds.value?.length > 0
        ? this.selectedClusterResourcesMap
        : this.stackSpecificResourcesMap;

    this.resourceNamesDropdown = this.constructResourcesDropdown(
      this.resourceTypes.value,
      resourcesMap,
    );

    if (Object.keys(selectedResources).length > 0) {
      const updatedResources = selectedResources;
      delete updatedResources[removedResourceType];
      this.resourceNamesDropdown = this.markSelectedOptionsAsCheckedForGrouped(
        this.resourceNamesDropdown,
        updatedResources,
      );
      this.policyForm.patchValue({
        resourceTypeAndResourceNamesMap: updatedResources,
      });
    }
    this.fetchingResNamesDropdown = false;
  }

  onResourceSelection(selectedResources) {
    let resources = {};
    selectedResources?.selectedOptionsList?.forEach((group) => {
      if (group?.children) {
        group?.children?.forEach((res) => {
          const resourceType = group?.value;
          const resourceName = res?.value;
          if (resources?.hasOwnProperty(resourceType)) {
            resources[resourceType].push(resourceName);
          } else {
            resources[resourceType] = [resourceName];
          }
        });
      }
    });
    this.policyForm.patchValue({
      resourceTypeAndResourceNamesMap: resources,
    });
  }

  // Make an API to get all the clusters, create blueprint->cluster map, construct blueprint dropdown
  getClusterAndBlueprintMapping() {
    return new Promise((resolve) => {
      this.uiStackControllerService
        .getAllClustersUsingGET({
          page: 0,
          perPage: 20000,
          sortBy: 'ID',
        })
        .subscribe(async (envs) => {
          this.blueprintEnvironmentsMap = {};
          envs.content.forEach((cluster) => {
            const stackName = cluster?.stackName;
            if (this.blueprintEnvironmentsMap.hasOwnProperty(stackName)) {
              this.blueprintEnvironmentsMap[stackName].push(cluster);
            } else {
              this.blueprintEnvironmentsMap[stackName] = [cluster];
            }
          });

          this.blueprintsDropdown = this.getBlueprintsDropdown(
            this.blueprintEnvironmentsMap,
          );
          resolve(true);
        });
    });
  }

  getStackLevelResourceTypeAndResourcesMapping(stackName) {
    return new Promise((resolve, reject) => {
      this.resourceNamesDropdown = [];
      this.resourceTypesDropdown = [];
      this.stackSpecificResourcesMap = {};

      let params = {
        stackName: stackName,
      };

      this.uiDropdownsControllerService
        .getAllResourcesByStackUsingGET(params)
        .subscribe(
          (resources) => {
            resources.forEach((resource) => {
              const resourceType = resource?.resourceType;
              if (resourceType !== 'UNKNOWN') {
                if (
                  this.stackSpecificResourcesMap.hasOwnProperty(resourceType)
                ) {
                  this.stackSpecificResourcesMap[resourceType].push(
                    resource?.resourceName,
                  );
                } else {
                  this.stackSpecificResourcesMap[resourceType] = [
                    resource?.resourceName,
                  ];
                }
              }
            });

            this.resourceTypesDropdownForSelectedStack = Object.keys(
              this.stackSpecificResourcesMap || {},
            ).map((resourceType) => ({
              label: resourceType,
              value: resourceType,
              checked: false,
            }));
            resolve(true);
          },
          (error) => {
            reject(error);
          },
        );
    });
  }

  getClusterLevelResourceTypeAndResourcesMap(stackName) {
    return new Promise((resolve, reject) => {
      this.uiDropdownsControllerService
        .getAllResourcesForAllClusterOfStackUsingGET(stackName)
        .subscribe(
          (clusters) => {
            this.clusterSpecificResourcesMap = {};
            let resultMap = {};

            // Iterate over the API response to create the map
            clusters?.forEach((cluster) => {
              const clusterId = cluster?.clusterId;
              resultMap[clusterId] = {};

              cluster?.resources?.forEach((resource) => {
                const { resourceType, resourceName } = resource;
                if (resultMap[clusterId]?.hasOwnProperty(resourceType)) {
                  resultMap[clusterId][resourceType]?.push(resourceName);
                } else {
                  resultMap[clusterId][resourceType] = [resourceName];
                }
              });
            });

            this.clusterSpecificResourcesMap = resultMap;
            resolve(true);
          },
          (error) => {
            reject(error);
          },
        );
    });
  }

  constructResourcesMapForSelectedClusters(selectedClusters) {
    const resourceTypeToResourcesMap = {};

    // Iterate through the selected clusters
    selectedClusters?.forEach((clusterId) => {
      const clusterData = this.clusterSpecificResourcesMap[clusterId];
      for (const resourceType in clusterData) {
        if (resourceType !== 'UNKNOWN') {
          if (resourceTypeToResourcesMap?.hasOwnProperty(resourceType)) {
            // Use a Set to ensure unique resources for each resource type
            const uniqueResources = new Set(
              resourceTypeToResourcesMap[resourceType],
            );

            // Add resources from the current cluster
            clusterData[resourceType]?.forEach((resource) => {
              uniqueResources?.add(resource);
            });

            // Convert the Set back to an array
            resourceTypeToResourcesMap[resourceType] = [...uniqueResources];
          } else {
            resourceTypeToResourcesMap[resourceType] = [
              ...clusterData[resourceType],
            ];
          }
        }
      }
    });

    this.selectedClusterResourcesMap = resourceTypeToResourcesMap;
    return resourceTypeToResourcesMap;
  }

  getBlueprintsDropdown(blueprintEnvironmentsMap) {
    return Object.keys(blueprintEnvironmentsMap || {})?.map((stackName) => {
      return { label: stackName, value: stackName };
    });
  }

  markSelectedOptionsAsChecked(dropdown, selectedValues) {
    return dropdown?.map((option) => ({
      ...option,
      checked: selectedValues?.includes(option?.value),
    }));
  }

  markSelectedOptionsAsCheckedForGrouped(dropdown, selectedValues) {
    return dropdown?.map((group) => ({
      ...group,
      children: group?.children?.map((option) => ({
        ...option,
        checked: selectedValues?.[group.label]?.includes(option.value),
      })),
    }));
  }

  clearDropdownValues(dropdown) {
    return dropdown?.map((option) => ({
      ...option,
      checked: false,
    }));
  }

  constructResourcesDropdown(
    selectedResourceTypes,
    resourceTypesResourceNamesMap,
  ) {
    const resourceTypesArray = [];

    selectedResourceTypes.forEach((resourceType) => {
      if (resourceTypesResourceNamesMap?.hasOwnProperty(resourceType)) {
        const resourceNames = resourceTypesResourceNamesMap[resourceType];
        const children = resourceNames?.map((resourceName) => ({
          label: resourceName,
          value: resourceName,
        }));
        resourceTypesArray.push({
          label: resourceType,
          value: resourceType,
          expand: true,
          children: children,
        });
      }
    });

    return resourceTypesArray;
  }

  constructEnvironmentsDropdown(stackName, blueprintEnvironmentsMap) {
    if (stackName in blueprintEnvironmentsMap) {
      return blueprintEnvironmentsMap[stackName]?.map((cluster) => ({
        checked: false,
        label: cluster?.name,
        value: cluster?.id,
      }));
    } else {
      return []; // Return an empty array if the stackName is not found in the map
    }
  }

  findDeletedItems(originalArray, updatedArray) {
    return originalArray?.filter((item) => !updatedArray?.includes(item));
  }

  getPolicyDetails(policyName) {
    this.fetchingEditDetails = true;
    return new Promise((resolve, reject) => {
      // Reset environments, resourceTypeOptions and resources
      this.environmentsDropdown = [];
      this.resourceTypesDropdown = [];
      this.resourceNamesDropdown = [];

      this.uiOpaControllerService
        .getPolicyUsingGET(policyName)
        .subscribe(async (details: any) => {
          const {
            policyName,
            stackName,
            policyCode,
            description = '',
            clusterIds,
            resourceTypeAndResourceNamesMap,
            resourceTypes,
            packageName,
            severity,
            disabled,
          } = details;

          this.policyForm.patchValue({
            policyName,
            stackName,
            rawPolicy: policyCode,
            description,
            clusterIds,
            resourceTypeAndResourceNamesMap,
            resourceTypes,
            packageName,
            severity: severity || 'WARNING',
            disabled,
          });
          // Once stackName is set, fetch environmentsDropdown dropdown
          this.environmentsDropdown = this.constructEnvironmentsDropdown(
            this.stackName.value,
            this.blueprintEnvironmentsMap,
          );

          //If only stack is set, then fetch resourceType dropdown options at stack level and cluster level
          if (
            !clusterIds?.length &&
            !resourceTypes?.length &&
            !Object.keys(resourceTypeAndResourceNamesMap).length
          ) {
            await this.getStackLevelResourceTypeAndResourcesMapping(stackName);
            await this.getClusterLevelResourceTypeAndResourcesMap(stackName);
            this.resourceTypesDropdown = cloneDeep(
              this.resourceTypesDropdownForSelectedStack,
            );
            this.fetchingEditDetails = false;
            resolve(true);
            return;
          }

          // If clusterIds are present then manually mark all those checkboxes
          if (clusterIds?.length > 0) {
            await this.getStackLevelResourceTypeAndResourcesMapping(stackName);
            await this.getClusterLevelResourceTypeAndResourcesMap(stackName);
            this.environmentsDropdown = this.markSelectedOptionsAsChecked(
              this.environmentsDropdown,
              clusterIds,
            );

            // Construct resources map for selected cluster
            this.resourceTypesDropdown = Object.keys(
              this.constructResourcesMapForSelectedClusters(
                this.clusterIds.value,
              ),
            ).map((resourceType) => ({
              label: resourceType,
              value: resourceType,
              checked: false,
            }));

            //If resourceType is selected, then mark those checkboxes and construct resources dropdown.
            if (resourceTypes?.length) {
              this.resourceTypesDropdown = this.markSelectedOptionsAsChecked(
                this.resourceTypesDropdown,
                resourceTypes,
              );

              // Based on selected resourceTypes construct resources dropdown
              this.resourceNamesDropdown = this.constructResourcesDropdown(
                resourceTypes,
                this.selectedClusterResourcesMap,
              );

              if (Object.keys(resourceTypeAndResourceNamesMap)?.length) {
                this.resourceNamesDropdown =
                  this.markSelectedOptionsAsCheckedForGrouped(
                    this.resourceNamesDropdown,
                    resourceTypeAndResourceNamesMap,
                  );
              }
            }
            this.fetchingEditDetails = false;
            resolve(true);
            return;
          }

          if (resourceTypes?.length) {
            await this.getStackLevelResourceTypeAndResourcesMapping(stackName);
            await this.getClusterLevelResourceTypeAndResourcesMap(stackName);
            this.resourceTypesDropdown = cloneDeep(
              this.resourceTypesDropdownForSelectedStack,
            );

            this.resourceTypesDropdown = this.markSelectedOptionsAsChecked(
              this.resourceTypesDropdown,
              resourceTypes,
            );

            // Based on selected resourceTypes construct resources dropdown
            this.resourceNamesDropdown = this.constructResourcesDropdown(
              resourceTypes,
              this.stackSpecificResourcesMap,
            );

            if (Object.keys(resourceTypeAndResourceNamesMap)?.length) {
              this.resourceNamesDropdown =
                this.markSelectedOptionsAsCheckedForGrouped(
                  this.resourceNamesDropdown,
                  resourceTypeAndResourceNamesMap,
                );
            }
            this.fetchingEditDetails = false;
            resolve(true);
            return;
          }
        });
    });
  }

  createOrEditPolicy() {
    this.policyRequestInProgress = true;
    this.errorList = [];
    let params = {
      policyName: this.policyName?.value,
      opaRequest: this.policyForm.value,
    };
    let httpReq =
      this.mode === 'create'
        ? this.uiOpaControllerService.createPolicyUsingPOST(params)
        : this.uiOpaControllerService.editPolicyUsingPUT(params);
    httpReq.subscribe(
      (response) => {
        this.policyRequestInProgress = false;
        this.customToastService.success(
          this.mode === 'create'
            ? "The policy has been created successfully. To enforce the policy right away, 'Run Validations' on the releases page."
            : "The policy has been updated successfully. To enforce the policy changes right away, 'Run Validations' on the releases page.",
        );
        this.handleBack();
      },
      (error) => {
        this.policyRequestInProgress = false;
        if (error?.error?.code === 400) {
          let parsedError = this.parseErrorMessage(error?.error?.message);
          this.errorList = this.generateErrorList(parsedError);
        } else {
          this.customToastService.error(
            this.mode === 'create'
              ? error?.error?.message || "Policy couldn't be created."
              : error?.error?.message || "Policy couldn't be updated.",
          );
        }
      },
    );
  }

  parseErrorMessage(errorMessage) {
    const startIndex = errorMessage.indexOf('{');
    const endIndex = errorMessage.lastIndexOf('}');

    if (startIndex !== -1 && endIndex !== -1) {
      const jsonString = errorMessage.substring(startIndex, endIndex + 1);
      try {
        const errorObject = JSON.parse(jsonString);
        return errorObject;
      } catch (error) {
        // Handle JSON parsing error
        console.error('Error parsing JSON:', error);
      }
    }

    return null; // Return null if no valid JSON object found
  }

  generateErrorList(parsedError): string[] {
    let errorList: string[] = [];
    parsedError?.errors?.forEach((error, index) => {
      errorList.push(
        `Error ${index + 1} at row: ${error?.location?.row} col: ${
          error?.location?.col
        } - ${error?.message}.`,
      );
    });
    return errorList;
  }
}
