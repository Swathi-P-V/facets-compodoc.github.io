import { Component } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { forkJoin } from 'rxjs';
import {
  UiApplicationControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { getEffectiveValue, getUserVars } from 'src/app/utils/app-config-utils';
const emptyEnvIcon = './assets/empty-env.svg';

@Component({
  selector: 'app-app-diff-compare',
  templateUrl: './app-diff-compare.component.html',
  styleUrls: ['./app-diff-compare.component.scss'],
})
export class AppDiffCompareComponent extends FacetsPage {
  clusterId: any;
  appName: any;
  stackName: any;
  environmentsToCompare: Array<string>;
  keys = Object.keys;
  appJsonVersion2 = true;
  appJson: any;
  environments: Array<any>;
  configData: any = {};
  variableData: any = {};
  fetchingEnvs = false;
  fetchingApplicationData = false;
  fetchingOverrides = {};
  overrideJson: any = {};
  diffData: any = {};
  resourceType: any;
  primaryMsg: string;
  showBtn: boolean = true;
  status: 'info' | 'error' = 'info';
  pageHeading: string = '';
  columnWidth: string = '';
  icon = emptyEnvIcon;
  noDiff: boolean = true;
  envNames = {};
  configKeyPaths = {
    disabled: { paths: ['disabled'], defaultValue: false },
    size: { paths: ['size', 'spec.runtime.size.value'], defaultValue: 'NA' },
    resourceAllocationStrategy: {
      paths: ['resourceAllocationStrategy', 'spec.runtime.size.namespace'],
      defaultValue: 'NA',
    },
    hpaMinReplicas: {
      paths: ['scaling.hpaMinReplicas', 'spec.runtime.autoscaling.min'],
      defaultValue: 'NA',
    },
    hpaMaxReplicas: {
      paths: ['scaling.hpaMaxReplicas', 'spec.runtime.autoscaling.max'],
      defaultValue: 'NA',
    },
    hpaMetricThreshold: {
      paths: [
        'scaling.hpaMetricThreshold',
        'spec.runtime.autoscaling.cpuThreshold',
      ],
      defaultValue: 'NA',
    },
  };

  diffStructure = [
    {
      cardLabel: 'Configuration Variables',
      diffExists: false,
      containers: [
        {
          containerLabel: 'Application Status',
          key: 'disabled',
          pill: true,
          valuePrepare: (val) => (val ? 'Disabled' : 'Enabled'),
          displayPrepare: (val) => (val ? 'danger' : 'success'),
        },
        { containerLabel: 'Size', key: 'size' },
        {
          containerLabel: 'Size Namespace',
          key: 'resourceAllocationStrategy',
        },
      ],
    },
    {
      cardLabel: 'Autoscaling Pods',
      diffExists: false,
      containers: [
        { containerLabel: 'Minimum Available', key: 'hpaMinReplicas' },
        { containerLabel: 'Maximum Available', key: 'hpaMaxReplicas' },
        {
          containerLabel: 'Autoscaling CPU Trigger',
          key: 'hpaMetricThreshold',
        },
      ],
    },
    {
      cardLabel: 'User-defined Variables',
      diffExists: false,
      containers: [],
    },
  ];
  variableList: string[];

  constructor(
    route: ActivatedRoute,
    private router: Router,
    private uiStackControllerService: UiStackControllerService,
    private uiApplicationControllerService: UiApplicationControllerService,
  ) {
    super(route);
  }

  OnPageLoad(p: Params): void {
    this.clusterId = p.clusterId;
    this.appName = p.resourceName;
    this.resourceType = p.resourceType;
    this.stackName = p.stackName;
    this.fetchingEnvs = true;
    this.fetchingApplicationData = true;
    const toCompare = p.envs?.split('__') || [];
    this.environmentsToCompare = [p.clusterId, ...toCompare];
    if (this.environmentsToCompare?.length < 2) {
      this.primaryMsg = 'Please select at least 1 environment to compare!';
    } else {
      this.primaryMsg = 'Selected environments have no diff!';
    }

    this.uiStackControllerService
      .getClustersUsingGET1(this.stackName)
      .subscribe(
        (json) => {
          this.environments = json;
          this.setPageHeading(this.environments);
          this.setColumnWidth(this.environmentsToCompare);
          this.setEnvironmentName(this.environmentsToCompare);
          this.fetchingEnvs = false;
        },
        () => (this.fetchingEnvs = false),
      );
    this.uiApplicationControllerService
      .getResourceByNameUsingGET({
        appName: this.appName,
        clusterId: this.clusterId,
        resourceType: this.resourceType,
      })
      .subscribe(
        (json: any) => {
          this.appJson = json;
          if ('v2' !== this.appJson?.apiVersion) {
            this.appJsonVersion2 = false;
          }
          this.fetchingApplicationData = false;
          for (const envId of this.environmentsToCompare) {
            this.fetchingOverrides[envId] = true;
            this.configData[envId] = {};
            this.diffData[envId] = {};
            this.variableData[envId] = {};
            this.uiApplicationControllerService
              .getResourceOverrideObjectUsingGET({
                resourceName: this.appName,
                resourceType: this.resourceType,
                clusterId: envId,
              })
              .subscribe(
                (json) => {
                  this.overrideJson[envId] = json?.overrides || {};
                  this.fetchingOverrides[envId] = false;
                  this.prepareConfigData(
                    this.appJson,
                    this.overrideJson[envId],
                    envId,
                  );
                  this.prepareVariableData(
                    this.appJson,
                    this.overrideJson[envId],
                    envId,
                  );
                  this.populateDiff(this.configData, this.variableData);
                },
                () => {
                  this.fetchingOverrides[envId] = false;
                  this.prepareConfigData(
                    this.appJson,
                    this.overrideJson[envId],
                    envId,
                  );
                  this.prepareVariableData(
                    this.appJson,
                    this.overrideJson[envId],
                    envId,
                  );
                  this.populateDiff(this.configData, this.variableData);
                },
              );
          }
        },
        () => (this.fetchingApplicationData = false),
      );
  }

  prepareConfigData(json: any, overrides: any, envId: string) {
    Object.keys(this.configKeyPaths).forEach((key) => {
      this.configData[envId][key] = getEffectiveValue(
        this.configKeyPaths[key].paths,
        json,
        overrides,
        this.configKeyPaths[key].defaultValue,
      );
    });
  }

  prepareVariableData(json: any, overrides: any, envId: string) {
    const { varsMap = {}, variableList = [] } = getUserVars(
      json,
      overrides,
      this.appJsonVersion2,
    );
    this.variableData[envId] = varsMap;
    variableList.forEach((variable) => {
      this.configData[envId][variable] = varsMap[variable];
    });
    this.variableList = variableList;
  }

  markDiffExists(cardLabel: string) {
    const index = this.diffStructure.findIndex(
      (o) => o.cardLabel === cardLabel,
    );
    this.diffStructure[index].diffExists = true;
  }

  populateDiff(envData: any, varsData: any) {
    if (Object.values(this.fetchingOverrides).indexOf(true) === -1) {
      Object.keys(this.configKeyPaths).forEach((key) => {
        this.environmentsToCompare.forEach((env) => {
          if (
            envData[env][key]?.effectiveValue !==
            envData[this.clusterId][key]?.effectiveValue
          ) {
            this.diffData[env][key] = envData[env][key]?.effectiveValue;
            // set the value for base cluster as well
            this.diffData[this.clusterId][key] =
              envData[this.clusterId][key]?.effectiveValue;
            if (
              ['disabled', 'size', 'resourceAllocationStrategy'].indexOf(
                key,
              ) !== -1
            ) {
              this.markDiffExists('Configuration Variables');
            }
            if (
              [
                'hpaMinReplicas',
                'hpaMaxReplicas',
                'hpaMetricThreshold',
              ].indexOf(key) !== -1
            ) {
              this.markDiffExists('Autoscaling Pods');
            }
            this.noDiff = false;
          }
        });
      });
      const varContainers = [];
      const varsPushed = [];
      // console.log({ varsData });
      this.variableList.forEach((variable) => {
        this.environmentsToCompare.forEach((env) => {
          if (
            varsData[env][variable]?.effectiveValue !==
            varsData[this.clusterId][variable]?.effectiveValue
          ) {
            if (varsPushed.indexOf(variable) === -1) {
              varContainers.push({ containerLabel: variable, key: variable });
              varsPushed.push(variable);
            }
            this.diffData[env][variable] =
              this.variableData[env][variable]?.effectiveValue;
            // // set the value for base cluster as well
            this.diffData[this.clusterId][variable] =
              this.variableData[this.clusterId][variable]?.effectiveValue;
          }
        });
      });
      const varIndex = this.diffStructure.findIndex(
        (o) => o.cardLabel === 'User-defined Variables',
      );
      if (varContainers.length) {
        this.diffStructure[varIndex].containers = varContainers;
        this.markDiffExists('User-defined Variables');
      }
    }
  }

  setEnvironmentName(envIds: string[]) {
    envIds.forEach((envId) => {
      this.envNames[envId] = this.getEnvironmentName(envId);
    });
  }

  getEnvironmentName(id: string) {
    return this.environments
      ? this.environments.find((env) => env.id === id)?.name
      : '';
  }

  getLoadingState() {
    return (
      this.fetchingApplicationData ||
      this.fetchingEnvs ||
      Object.values(this.fetchingOverrides).indexOf(true) !== -1
    );
  }

  setColumnWidth(envs: any[]) {
    this.columnWidth = envs?.length === 2 ? 'col-md-2' : 'col-md-2';
  }

  handleBack = () => {
    this.router.navigate([
      '/',
      this.stackName,
      'cluster',
      this.clusterId,
      'resources',
      'type',
      this.resourceType,
      'resource',
      this.appName,
      'diff',
    ]);
  };

  setPageHeading(envs: any[]) {
    let heading = `Comparing `;
    if (envs?.length) {
      this.environmentsToCompare.forEach((envId, index) => {
        heading += `${index > 1 ? 'and ' : ''} ${this.getEnvironmentName(
          envId,
        )} ${index === 0 ? 'with ' : ''}`;
      });
    }
    this.pageHeading = heading;
  }
}
