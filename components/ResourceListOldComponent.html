import { Component, ViewChild } from '@angular/core';
import { ActivatedRoute, Params, Router } from '@angular/router';
import { UiDropdownsControllerService } from 'src/app/cc-api/services/ui-dropdowns-controller.service';
import { UiStackControllerService } from 'src/app/cc-api/services';
import { whiteListedResources } from 'src/app/constants/resourceTypes';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { CustomToastService } from '../../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';
const emptyIcon = './assets/resource-empty.svg';
const errorIcon = './assets/error.svg';

@Component({
  selector: 'app-resource-list-old',
  templateUrl: './resource-list-old.component.html',
  styleUrls: ['./resource-list-old.component.scss'],
})
export class ResourceListOldComponent extends FacetsPage {
  keys = Object.keys;
  values = Object.values;
  resources: any;
  stackName: any;
  clusterId: string;
  @ViewChild('autoInput') input;
  fetchingResources = false;
  emptyIcon = emptyIcon;
  filteredResources: any = {};
  resourceDropdown = {};
  selectedFilter: any;
  primaryMsg: string = 'No resources found!';
  secondaryMsg: string;
  showBtn: boolean = false;
  status: 'info' | 'error' = 'info';
  selectOptionsList: { label: string; value: string }[] = [];
  actionButtons = [
    {
      id: 'sync-git',
      title: 'Sync with Git',
      buttonType: 'secondary',
      status: 'control',
      nbTooltip:
        'Not seeing a newly defined application in the blueprint? Click here',
      icon: 'sync-outline',
      loading: false,
      hidden: false,
      onClick: () => {
        this.refreshStack();
      },
    },
  ];

  constructor(
    private _route: ActivatedRoute,
    private uiDropdownsControllerService: UiDropdownsControllerService,
    private router: Router,
    private stackService: UiStackControllerService,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(_route);
  }

  OnPageLoad(p: Params): void {
    this.filteredResources = {};
    this.selectOptionsList = [];
    if (p?.stackName && p?.clusterId) {
      this.stackName = p?.stackName;
      this.secondaryMsg = `Please add resources for ${this.stackName} blueprint`;
      this.clusterId = p?.clusterId;
      this.resources = {};
      this.filteredResources = {};
      this.getAllResources(this.clusterId);
    }
  }

  getAllResources(clusterId) {
    this.fetchingResources = true;
    let params = {
      clusterId: clusterId,
      excludeSubstackResources: false,
    };
    this.uiDropdownsControllerService.getAllResourcesUsingGET(params).subscribe(
      (resources) => {
        this.resources = resources;
        let resourceKeys = [];
        if (Object.keys(this.resources).length > 0) {
          Object.keys(this.resources).forEach((resource) => {
            resourceKeys.push(resource);
          });
          resourceKeys.sort();
          resourceKeys.forEach((resource) => {
            this.selectOptionsList.push({
              label: resource,
              value: resource,
            });
          });
        }
        this.filteredResources = resources;
        this.fetchingResources = false;
      },
      (error) => {
        this.filteredResources = null;
        this.fetchingResources = false;
        this.secondaryMsg =
          error?.error?.message || ' There is some error fetching resources';
        this.primaryMsg = '';
        this.emptyIcon = errorIcon;
        this.status = 'error';
      },
    );
  }

  private filter(value: string): any {
    const filteredResources = {};
    let filteredByResourcesTypeList = {};
    if (this.selectedFilter) {
      filteredByResourcesTypeList[this.selectedFilter] =
        this.resources[this.selectedFilter];
    } else {
      filteredByResourcesTypeList = this.resources;
    }
    const filterValue = value.split(/[-]/).join('')?.toLowerCase();
    this.keys(filteredByResourcesTypeList).forEach((rType) => {
      const tempResources = this.resources[rType];
      const filtered = tempResources.filter(
        (a) =>
          a.split(/[-]/).join('')?.toLowerCase().indexOf(filterValue) !== -1,
      );
      if (filtered?.length) {
        filteredResources[rType] = filtered;
      }
    });
    return filteredResources;
  }

  onChange() {
    this.filteredResources = this.filter(this.input.nativeElement.value);
  }

  onResourceChange(event) {
    if (event === 'clear') {
      this.selectedFilter = null;
      this.filteredResources = this.filter(this.input.nativeElement.value);
    } else if (event) {
      this.selectedFilter = event;
      this.filteredResources = this.filter(this.input.nativeElement.value);
    }
  }

  resourcesExist() {
    if (!this.resources) return false;
    let count = 0;
    this.keys(this.resources).forEach((rType) => {
      count += this.resources[rType]?.length;
    });
    return count !== 0;
  }

  getLink(resource, app) {
    const url = `type/${resource}/resource/${app}/${
      whiteListedResources.indexOf(resource) === -1
        ? 'configuration'
        : 'overview'
    }`;
    this.router.navigate([url], { relativeTo: this._route });
  }

  listenToActionButtonEvents(event) {
    let clickedButton = this.actionButtons.filter((button) => {
      return button?.id === event;
    })[0];
    clickedButton?.onClick();
  }

  refreshStack() {
    let gitSyncButton = this.actionButtons[0];
    gitSyncButton.loading = true;
    this.stackService.reloadStackUsingGET1(this.stackName).subscribe(
      (stack) => {
        gitSyncButton.loading = false;
        this.customToastService.success('Synced with git');
        this.getAllResources(this.clusterId);
      },
      (error) => {
        gitSyncButton.loading = false;
        this.onAPIErrorService.onAPIError(
          error?.error?.message || 'Syncing with git failed',
          error,
        );
      },
    );
  }
}
