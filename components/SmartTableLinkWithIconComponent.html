import { Component, Input, Output, EventEmitter } from '@angular/core';
import { ActivatedRoute, Params } from '@angular/router';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';

@Component({
  selector: 'app-smart-table-link-with-icon',
  templateUrl: './smart-table-link-with-icon.component.html',
  styleUrls: ['./smart-table-link-with-icon.component.scss'],
})
export class SmartTableLinkWithIconComponent extends FacetsPage {
  @Input() value: any; // Value which comes in from the parent component, needs to be parsed.
  @Input() rowData: any;
  @Output() byTitleClick = new EventEmitter();
  @Output() byImgClick = new EventEmitter();

  parsedValue: any;
  isImgLink: boolean = false;
  imageSVG;
  textContent: string = '';
  toolTipText: string = '';
  imgClass;
  //By default text is placed first and image next. To change the order set is_reverse to true
  is_reverse: boolean = false;
  noTooltip: boolean = false;
  badgeClass: string = '';
  imgHeight: any;
  imgWidth: any;
  showNbIcon: boolean = false;
  nbIconLink: string = '';
  labelLink: boolean = false;
  reverseClass: string = 'ml-2';
  url: string | undefined;

  constructor(activatedRoute: ActivatedRoute) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    try {
      this.parsedValue = JSON.parse(this.value);
      this.imageSVG = this.parsedValue.imageSVG;
      this.textContent = this.parsedValue.textContent;
      this.toolTipText = this.parsedValue.toolTipText;
      if (this.parsedValue?.url) {
        this.url = this.parsedValue.url;
      }
      if (this.parsedValue?.is_reverse) {
        this.is_reverse = this.parsedValue.is_reverse;
      }
      if (this.parsedValue?.noTooltip) {
        this.noTooltip = this.parsedValue.noTooltip;
      }
      if (this.parsedValue?.badgeClass) {
        this.badgeClass = this.parsedValue.badgeClass;
      }
      if (this.parsedValue?.imgHeight) {
        this.imgHeight = this.parsedValue.imgHeight;
      }
      if (this.parsedValue?.imgWidth) {
        this.imgWidth = this.parsedValue.imgWidth;
      }
      if (this.parsedValue?.imgClass) {
        this.imgClass = this.parsedValue.imgClass;
      }
      if (this.parsedValue?.labelLink) {
        this.labelLink = this.parsedValue.labelLink;
      }
      if (this.parsedValue?.showNbIcon) {
        this.showNbIcon = this.parsedValue.showNbIcon;
      }
      if (this.parsedValue?.nbIconLink) {
        this.nbIconLink = this.parsedValue.nbIconLink;
      }
      if (this.parsedValue?.isImgLink) {
        this.isImgLink = this.parsedValue?.isImgLink;
      }
      if (this.parsedValue?.reverseClass) {
        this.reverseClass = this.parsedValue?.reverseClass;
      }
    } catch (error) {}
  }

  emitTitleButtonClickEvent() {
    this.byTitleClick.emit({
      rowData: this.rowData,
    });
  }

  emitImgClickEvent() {
    if (this.isImgLink) {
      this.byImgClick.emit({
        rowData: this.rowData,
      });
    }
  }
}

<div [ngClass]="{ reverse: is_reverse, 'no-reverse': !is_reverse }">
  <a
    *ngIf="url"
    [routerLink]="url"
    class="cursor-pointer pl-2"
    [ngStyle]="{ 'text-decoration': 'none' }"
  >
    {{ textContent }}
  </a>
  <label
    *ngIf="textContent?.length > 0 && !url"
    class="{{ badgeClass }}"
    [ngClass]="{ 'mr-1': !is_reverse, 'text-label': true }"
    class="{{ is_reverse ? reverseClass : '' }}"
    (click)="emitTitleButtonClickEvent()"
  >
    {{ textContent }}
  </label>
  <nb-icon
    *ngIf="showNbIcon && toolTipText?.length > 0"
    icon="{{ nbIconLink }}"
    class="mr-2 {{ imgClass }}"
    [nbTooltip]="toolTipText"
    (click)="emitImgClickEvent()"
  ></nb-icon>
  <nb-icon
    *ngIf="showNbIcon && !toolTipText"
    icon="{{ nbIconLink }}"
    class="mr-2 {{ imgClass }}"
    (click)="emitImgClickEvent()"
  ></nb-icon>
  <img
    class="{{ imgClass }}"
    *ngIf="!noTooltip && !showNbIcon"
    [src]="imageSVG"
    [nbTooltip]="toolTipText"
    (click)="emitImgClickEvent()"
  />
  <img
    class="{{ imgClass }}"
    [ngStyle]="{ height: imgHeight, width: imgWidth }"
    *ngIf="noTooltip && !showNbIcon"
    [src]="imageSVG"
  />
</div>

@mixin custom-badge() {
  $badge-types: 'warning', 'stop', 'primary';

  @each $value in $badge-types {
    .#{$value} {
      border-radius: 0.4rem;
      font-size: 0.9rem;
      @if $value == warning {
        color: var(--color-warning);
      } @else if $value == stop {
        color: var(--color-danger);
      } @else if $value == primary {
        color: var(--color-success);
      }
    }
  }
}
