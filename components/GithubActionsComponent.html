import { Component } from '@angular/core';
import { NbMenuService } from '@nebular/theme';
import { FormControl } from '@angular/forms';
import { ActivatedRoute, Router, Params } from '@angular/router';
import moment from 'moment';
import { SmartTableIconSubheadingComponent } from './../../../components/smart-table-icon-subheading/smart-table-icon-subheading.component';
import { ActionTimingComponent } from './action-timing/action-timing.component';
import { SmartTableLinkComponent } from './../../cluster-pages/cluster-releases-new-v2/smart-table-link-component';
import { CustomServerDataSource } from 'src/app/services/custom-server-data-source';
import { FormatDateService } from 'src/app/services/format-date.service';
import { HttpClient } from '@angular/common/http';
import {
  UiBlueprintDesignerControllerService,
  UiStackControllerService,
} from 'src/app/cc-api/services';
import { Stack } from 'src/app/cc-api/models';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { WorkflowRun } from './../../../cc-api/models/workflow-run';
const emptyStateLogo = './assets/artifact-empty.svg';
const errorLogo = './assets/error.svg';

@Component({
  selector: 'app-github-actions',
  templateUrl: './github-actions.component.html',
  styleUrls: ['./github-actions.component.scss'],
})
export class GithubActionsComponent extends FacetsPage {
  workflowNames: {
    label: string;
    value: string;
  }[] = [];
  workflowFormControl: FormControl = new FormControl('ALL');
  emptyStateLogo = emptyStateLogo;
  fetchingPR: boolean = false;
  workflowEmptyIcon: string = emptyStateLogo;
  secondaryMsg: string = 'No workflow runs present for the blueprint';
  stackName: string = '';
  workflowData: any = [];
  loadingStackDetails: boolean = false;
  stack: Stack;
  duration: any;
  currentActionsResponse: [] = [];
  resourceType: string;
  resourceName: string;
  tableSettings: any = {
    columns: {
      displayTitle: {
        title: 'Workflow Name',
        type: 'custom',
        filter: false,
        valuePrepareFunction: (value, row, cell) => {
          const commitId = row?.headCommit?.id.slice(0, 6);
          let icon = '';
          if (!row?.conclusion) {
            icon = 'clock';
          } else if (row?.conclusion === 'success') {
            icon = 'success-fill';
          } else {
            icon = 'failure-fill';
          }
          return JSON.stringify({
            icon: icon,
            tooltipText: `Run ${row?.conclusion ? row?.conclusion : 'Pending'}`,
            pack: 'fci',
            status: 'primary',
            title: value,
            maxWidth: '100',
            subtitle: `${row?.name} #${row?.runNumber}: Commit ${commitId} pushed by ${row?.triggeringActor?.login}`,
          });
        },
        renderComponent: SmartTableIconSubheadingComponent,
        onComponentInitFunction: (instance) => {
          instance?.byLinkClick?.subscribe((data) => {
            window.open(data?.htmlUrl, '_blank');
          });
        },
        width: '40%',
      },
      headBranch: {
        title: 'Branch Name',
        type: 'custom',
        filter: false,
        valuePrepareFunction: (value, row, cell) => {
          return JSON.stringify({
            display: value,
            showImg: false,
            isRouterLink: true,
            iconName: '',
            iconTooltip: null,
          });
        },
        renderComponent: SmartTableLinkComponent,
        onComponentInitFunction: (instance) => {
          instance?.byLinkClick?.subscribe((data) => {
            const workflowUrl = data?.htmlUrl;
            const branchUrl = `${workflowUrl.split('/actions')?.[0]}/tree/${
              data?.headBranch
            }`;
            window.open(branchUrl, '_blank');
          });
        },
      },
      status: {
        title: 'Run Time',
        type: 'custom',
        filter: false,
        valuePrepareFunction: (cell, row) => {
          const runStartTime = this.formatDateService.getRelativeTime(
            row?.runStartedAt,
          );
          const createdAt = row?.createdAt;
          const updatedAt = row?.updatedAt;
          let diffMins: string | number = '--';
          let diffSeconds: string | number = '--';
          let diffHours: string | number = '--';
          if (createdAt && updatedAt) {
            const createdOn = moment(createdAt);
            const finishedOn = moment(updatedAt);
            const diff = finishedOn.diff(createdOn);
            this.duration = moment.duration(diff);
            diffHours =
              this.duration?.hours() < 10
                ? '0' + this.duration?.hours()
                : this.duration?.hours();
            diffMins =
              this.duration?.minutes() < 10
                ? '0' + this.duration?.minutes()
                : this.duration?.minutes();
            diffSeconds =
              this.duration?.seconds() < 10
                ? '0' + this.duration?.seconds()
                : this.duration?.seconds();
          }
          return JSON.stringify({
            primaryIcon: 'calendar-outline',
            primaryText: runStartTime,
            secondaryIcon: 'clock-primary',
            secondaryText: `${Number(diffHours) > 0 ? diffHours + 'h ' : ''}${
              Number(diffMins) > 0 ? diffMins + 'm ' : ''
            }${Number(diffSeconds) > 0 ? diffSeconds + 's ' : ''}`,
          });
        },
        renderComponent: ActionTimingComponent,
      },
      triggeringActor: {
        title: 'Started By',
        type: 'custom',
        filter: false,
        valuePrepareFunction: (value, row, cell) => {
          return JSON.stringify({
            display: row?.triggeringActor?.login,
            showImg: false,
            isRouterLink: true,
            iconName: '',
            iconTooltip: null,
          });
        },
        renderComponent: SmartTableLinkComponent,
        onComponentInitFunction: (instance) => {
          instance?.byLinkClick?.subscribe((data) => {
            window.open(data?.triggeringActor?.htmlUrl, '_blank');
          });
        },
      },
    },
    noDataMessage: '',
    pager: {
      display: true,
      perPage: 10,
    },
    mode: 'external',
    actions: false,
  };
  loadingWorkflows: boolean = false;
  level: 'BLUEPRINT' | 'RESOURCE';
  githubActionsError;

  constructor(
    private menuService: NbMenuService,
    private formatDateService: FormatDateService,
    private blueprintDesignerService: UiBlueprintDesignerControllerService,
    private uiStackControllerService: UiStackControllerService,
    private http: HttpClient,
    private activatedroute: ActivatedRoute,
    private router: Router,
  ) {
    super(activatedroute);
  }

  OnPageLoad(p: Params) {
    this.stackName = p?.stackName;
    this.resourceType = p?.resourceType;
    this.resourceName = p?.resourceName;
    this.level =
      this.resourceName && this.resourceType ? 'RESOURCE' : 'BLUEPRINT';
    this.resetWorkflowFields();
    this.getWorkflowRunsData();
    this.getWorkflowDetails();
    this.getStackDetails();
  }

  resetWorkflowFields() {
    this.workflowNames = [];
    this.workflowData = null;
  }

  getWorkflowDetails() {
    const params = {
      stackName: this.stackName,
      resourceType: this.resourceType,
      resourceName: this.resourceName,
    };
    let httpReq;
    switch (this.level) {
      case 'BLUEPRINT':
        httpReq = this.blueprintDesignerService.getWorkflowsUsingGET1(
          this.stackName,
        );
        break;
      case 'RESOURCE':
        httpReq = this.blueprintDesignerService.getWorkflowsUsingGET(params);
        break;
    }
    httpReq.subscribe((data) => {
      data.forEach((workflow) => {
        this.workflowNames.push({
          label: workflow?.name,
          value: workflow?.id,
        });
      });
    });
  }

  getStackDetails() {
    this.loadingStackDetails = true;
    this.uiStackControllerService.getStackUsingGET(this.stackName).subscribe(
      (s) => {
        this.stack = s;
        this.loadingStackDetails = false;
      },
      (err) => {
        this.loadingStackDetails = false;
      },
    );
  }

  getWorkflowRunsData(workflowId = null) {
    this.loadingWorkflows = true;
    let customServerUrl = '';
    if (this.resourceType?.length) {
      customServerUrl = `${
        window.location.origin
      }/cc-ui/v1/designer/${encodeURIComponent(
        String(this.stackName),
      )}/resourceType/${encodeURIComponent(
        String(this.resourceType),
      )}/resourceName/${encodeURIComponent(
        String(this.resourceName),
      )}/workflow-runs`;
    } else {
      customServerUrl = `${
        window.location.origin
      }/cc-ui/v1/designer/${encodeURIComponent(
        String(this.stackName),
      )}/workflow-runs`;
    }
    this.workflowData = new CustomServerDataSource(
      this.http,
      {
        endPoint: customServerUrl,
        dataKey: 'data.workflowRuns', // path in the response which holds array of deployments
        totalKey: 'data.totalCount', // this key from response holds total number of deployments present, smart table shows the pagination numbers based on this
        pagerLimitKey: 'pageSize', // will be passed as queryparam to the api, set 15 in table settings config
        pagerPageKey: 'pageNumber',
        filterFieldKey: '#field#', // Takes the field name dynamically for filtering purpose
      },
      {
        customQueryParams:
          workflowId !== 'ALL'
            ? {
                workflowId: workflowId,
              }
            : null,
      },
      (workflowRuns: [] = []) => {
        this.githubActionsError = null;
        this.loadingWorkflows = false;
        this.currentActionsResponse = [...workflowRuns];
        return workflowRuns;
      },
      true,
      (error) => {
        this.githubActionsError = error;
        this.loadingWorkflows = false;
      },
    );
  }

  parseRunsData(workflowRuns) {
    this.workflowData = workflowRuns;
  }

  onWorkflowChange(event) {
    this.getWorkflowRunsData(event);
  }
}
