import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { PublicApIsService } from '../../cc-api/services/public-ap-is.service';
import { CustomOAuth2ClientRegistration } from '../../cc-api/models/custom-oauth-2client-registration';
import { ActivatedRoute, Router } from '@angular/router';
import { ApplicationControllerService } from 'src/app/cc-api/services';
import { SimpleOauth2User } from '../../cc-api/models/simple-oauth-2user';
import {
  FormBuilder,
  FormControl,
  FormGroup,
  NgForm,
  Validators,
} from '@angular/forms';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { RedirectOnLoginService } from 'src/app/services/redirect-on-login.service';

@Component({
  selector: 'app-login-page',
  templateUrl: './login-page.component.html',
  styleUrls: ['./login-page.component.scss'],
})
export class LoginPageComponent implements OnInit {
  errorMsg: string = '';
  logo: any;
  loginOptions: Array<CustomOAuth2ClientRegistration>;
  isCustomIntegrationPresent: boolean = false;
  loginInProgress: boolean = false;
  show: boolean = false;
  redirectURL: string = '';
  loginForm: FormGroup = new FormGroup({
    username: new FormControl('', [
      Validators.required,
      Validators.email,
      Validators.pattern(whiteSpaceInputPattern),
    ]),
    password: new FormControl('', [Validators.required]),
    rememberMe: new FormControl(false),
  });

  constructor(
    private http: HttpClient,
    private publicApIsService: PublicApIsService,
    private applicationControllerService: ApplicationControllerService,
    private route: ActivatedRoute,
    private router: Router,
    private readonly fb: FormBuilder,
    private redirectOnLoginService: RedirectOnLoginService,
  ) {}

  async ngOnInit(): Promise<void> {
    //If the user is already logged-in then redirect to home page or redirectURL
    this.applicationControllerService.meUsingGET().subscribe(
      (x: SimpleOauth2User) => {
        this.redirectOnLoginService.handleRedirectURL();
      },
      (error) => {},
    );

    this.http
      .get('/public/v1/logo', { responseType: 'text' })
      .subscribe((x) => {
        this.logo = x;
      });

    this.publicApIsService.getLoginOptionsUsingGET().subscribe((response) => {
      this.loginOptions = response;
      this.isCustomIntegrationPresent =
        this.loginOptions.filter((x) => !x.isSystemConfigured)?.length > 0;
    });
  }

  get username() {
    return this.loginForm.get('username');
  }

  get password() {
    return this.loginForm.get('password');
  }

  get rememberMe() {
    return this.loginForm.get('rememberMe');
  }

  oauthLogin(registrationId: string) {
    window.location.href =
      '/oauth2/authorization/' +
      registrationId +
      '?state=' +
      window.location.href;
  }

  formLogin() {
    this.errorMsg = '';
    this.loginInProgress = true;
    let data = new FormData();
    data.append('username', this.username.value);
    data.append('password', this.password.value);

    this.http.post('/perform_login', data, { responseType: 'text' }).subscribe(
      () => {
        this.loginInProgress = false;
        this.redirectOnLoginService.handleRedirectURL();
      },
      (err) => {
        this.loginInProgress = false;
        if (err.status == 404) {
          this.router.navigate(['/home']);
          return;
        }
        this.errorMsg = [503, 504]?.includes(err.status)
          ? 'Server is down, please try again after sometime!'
          : 'Incorrect email address or password';
      },
    );
  }

  togglePassword() {
    this.show = !this.show;
  }
}
