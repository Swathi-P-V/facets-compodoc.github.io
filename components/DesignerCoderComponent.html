import { CustomToastService } from 'src/app/pages/custom-toast/custom-toast.service';
import { Component, Input, TemplateRef, ViewChild } from '@angular/core';
import { NbDialogService } from '@nebular/theme';
import { ActivatedRoute, Params, Router } from '@angular/router';
import {
  FormBuilder,
  FormGroup,
  Validators,
  FormControl,
} from '@angular/forms';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import {
  whiteSpaceInputPattern,
  whiteSpaceAllPattern,
  coderNamePattern,
} from 'src/app/constants/inputPatterns';
import {
  UiBlueprintDesignerControllerService,
  UiCoderControllerService,
} from 'src/app/cc-api/services';
import { ToastLink } from '../../custom-toast/toast-interfaces';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-designer-coder',
  templateUrl: './designer-coder.component.html',
  styleUrls: ['./designer-coder.component.scss'],
})
export class DesignerCoderComponent extends FacetsPage {
  @ViewChild('coderDialog') coderDialog: TemplateRef<any>;
  stackName: string = '';
  resourceType: string;
  resourceName: string;
  @Input() ref: any;
  @Input() level: 'BLUEPRINT' | 'RESOURCE';
  toasterLinkSubscription: any;
  pageStepperCount: number = 0;
  title: string = 'Launch Workspace';
  existingWorkspaceControl: FormControl = new FormControl(
    null,
    Validators.required,
  );
  workspaceTypeControl: FormControl = new FormControl('NEW', [
    Validators.required,
  ]);
  newWorkspaceForm: FormGroup = this.fb.group({
    name: [
      null,
      [
        Validators.required,
        Validators.pattern(whiteSpaceInputPattern),
        Validators.pattern(coderNamePattern),
      ],
    ],
    branchType: ['NEW', [Validators.required]],
    sourceBranch: [null, [Validators.required]],
    featureBranch: [
      null,
      [Validators.required, Validators.pattern(whiteSpaceAllPattern)],
    ],
    existingBranch: [null],
  });

  workspaceTypeOptions: {
    value: string;
    label: string;
    description: string;
  }[] = [
    {
      value: 'NEW',
      label: 'Create New Workspace',
      description: 'Create a new workspace from scratch and launch the coder',
    },
    {
      value: 'EXISTING',
      label: 'Choose Existing Workspace',
      description:
        'Choose from existing set of workspaces and launch the coder',
    },
  ];
  existingWorkspaceOptions: {
    label: string;
    value: string;
    checked: boolean;
  }[] = [];
  branchOptions: {
    label: string;
    value: string;
    checked: boolean;
  }[] = [];
  loadingBranch: boolean = false;
  loadingWorkspaces: boolean = false;
  branchTypeOptions: {
    label: string;
    value: string;
  }[] = [
    { label: 'New Branch', value: 'NEW' },
    { label: 'Existing Branch', value: 'EXISTING' },
  ];
  toastersLinks: ToastLink[] = [
    {
      id: 'new-workspace',
      label: 'Open Workspace',
      icon: 'external-link',
      url: '',
      click: (data) => {
        window.open(data?.link?.url, '_blank');
      },
    },
    {
      id: 'github-creds',
      label: 'Map Now',
      url: '',
      click: () => {
        this.router.navigate(['/', 'accountSettings', 'credentials']);
      },
    },
  ];
  launchingInProgress: boolean = false;
  constructor(
    private dialogService: NbDialogService,
    private fb: FormBuilder,
    route: ActivatedRoute,
    private router: Router,
    private uiBlueprintDesignerControllerService: UiBlueprintDesignerControllerService,
    private uiCoderControllerService: UiCoderControllerService,
    private customToast: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(route);
  }

  get name() {
    return this.newWorkspaceForm.get('name');
  }

  get branchType() {
    return this.newWorkspaceForm.get('branchType');
  }

  get sourceBranch() {
    return this.newWorkspaceForm.get('sourceBranch');
  }

  get featureBranch() {
    return this.newWorkspaceForm.get('featureBranch');
  }

  get existingBranch() {
    return this.newWorkspaceForm.get('existingBranch');
  }

  OnPageInit(): void {
    this.toasterLinkSubscription = this.customToast.buttonClick$.subscribe(
      (data) => {
        this.toastersLinks
          .find((link) => link?.id === data?.link?.id)
          ?.click(data);
      },
    );
  }
  OnPageLoad(p: Params) {
    this.stackName = p?.stackName;
    this.resourceType = p?.resourceType;
    this.resourceName = p?.resourceName;
    this.getAllBranches();
    this.getAllWorkspaces();
  }

  getAllWorkspaces() {
    this.loadingWorkspaces = true;
    let params = {
      stackName: this.stackName,
      resourceName: this.resourceName,
      resourceType: this.resourceType,
    };
    let httpReq;
    switch (this.level) {
      case 'BLUEPRINT':
        httpReq = this.uiCoderControllerService.getAllUsingGET2(this.stackName);
        break;
      case 'RESOURCE':
        httpReq = this.uiCoderControllerService.getAllUsingGET1(params);
        break;
    }
    httpReq.subscribe(
      (data) => {
        this.existingWorkspaceOptions = data.map((workspace) => {
          return {
            label: workspace?.name,
            value: workspace?.url,
            checked: false,
          };
        });
        this.loadingWorkspaces = false;
      },
      (error) => {
        this.loadingWorkspaces = false;
      },
    );
  }

  getAllBranches() {
    this.loadingBranch = true;
    let httpReq;
    let params = {
      stackName: this.stackName,
      resourceName: this.resourceName,
      resourceType: this.resourceType,
    };
    switch (this.level) {
      case 'BLUEPRINT':
        httpReq =
          this.uiBlueprintDesignerControllerService.listBranchesUsingGET(
            this.stackName,
          );
        break;
      case 'RESOURCE':
        httpReq =
          this.uiBlueprintDesignerControllerService.listBranchesUsingGET1(
            params,
          );
        break;
    }
    httpReq.subscribe(
      (data) => {
        let branches = this.level === 'BLUEPRINT' ? data?.branches : data;
        this.branchOptions = branches?.map((branch) => {
          return {
            label: branch,
            value: branch,
            checked: false,
          };
        });
        this.loadingBranch = false;
      },
      (error) => {
        this.loadingBranch = false;
      },
    );
  }

  OnDestroy() {
    this.toasterLinkSubscription?.unsubscribe();
  }

  resetFormValues() {
    this.existingWorkspaceControl.setValue(null);
    this.name.reset();
    this.branchType.setValue('NEW');
    this.sourceBranch.reset();
    this.existingBranch.reset();
    this.existingWorkspaceControl.reset();
    this.featureBranch.reset();
    this.existingWorkspaceOptions.forEach((option) => (option.checked = false));
    this.branchOptions.forEach((option) => (option.checked = false));
  }

  onClickLaunch(ref) {
    this.workspaceTypeControl?.value === 'NEW'
      ? this.createNewWorkspace(ref)
      : this.launchExistingWorkspace(ref);
  }

  launchExistingWorkspace(ref) {
    window.open(this.existingWorkspaceControl?.value, '_blank');
    ref.close();
  }

  createNewWorkspace(ref) {
    this.branchType?.value === 'NEW'
      ? this.createNewBranchWorkspace(ref)
      : this.createExistingBranchWorkspace(ref);
  }

  createNewBranchWorkspace(ref) {
    this.launchingInProgress = true;
    let params: any = {
      stackName: this.stackName,
      createWorkspaceNewBranchRequest: {
        newBranch: this.featureBranch?.value,
        sourceBranch: this.sourceBranch?.value,
        workspaceName: this.name?.value,
      },
    };
    if (this.level === 'RESOURCE') {
      params['resourceName'] = this.resourceName;
      params['resourceType'] = this.resourceType;
    }
    let httpReq;
    switch (this.level) {
      case 'BLUEPRINT':
        httpReq =
          this.uiCoderControllerService.createWorkspaceWithNewBranchUsingPOST(
            params,
          );
        break;
      case 'RESOURCE':
        httpReq =
          this.uiCoderControllerService.createWorkspaceWithNewBranchUsingPOST1(
            params,
          );
        break;
    }
    this.createWorkspace(httpReq, ref);
  }

  showLaunchFailToaster(error) {
    this.onAPIErrorService.onAPIError(
      error?.error?.message ||
        'Map credentials with respective host name to enable workspace creation.',
      error,
    );
    this.launchingInProgress = false;
    this.resetFormValues();
  }

  createExistingBranchWorkspace(ref) {
    this.launchingInProgress = true;
    let params: any = {
      stackName: this.stackName,
      createWorkspaceExistingBranchRequest: {
        existingBranch: this.existingBranch?.value,
        workspaceName: this.name?.value,
      },
    };
    if (this.level === 'RESOURCE') {
      params['resourceName'] = this.resourceName;
      params['resourceType'] = this.resourceType;
    }
    let httpReq;
    switch (this.level) {
      case 'BLUEPRINT':
        httpReq =
          this.uiCoderControllerService.createWorkspaceWithExistingBranchUsingPOST(
            params,
          );
        break;
      case 'RESOURCE':
        httpReq =
          this.uiCoderControllerService.createWorkspaceWithExistingBranchUsingPOST1(
            params,
          );
        break;
    }
    this.createWorkspace(httpReq, ref);
  }

  createWorkspace(httpReq, ref) {
    httpReq.subscribe(
      (data) => {
        let coderUrl = data?.url;
        this.toastersLinks[0].url = coderUrl;
        this.openSuccessLaunchToaster(ref, coderUrl);
      },
      (error) => {
        this.showLaunchFailToaster(error);
        ref.close();
      },
    );
  }

  openSuccessLaunchToaster(ref, coderUrl) {
    this.customToast.show({
      title: 'Success',
      content:
        'Workspace launch is successful! Go to the created workspace to start your work.',
      type: 'success',
      duration: 5000,
      link: this.toastersLinks[0],
    });
    ref.close();
    this.resetFormValues();
    window.open(coderUrl, '_blank');
    this.launchingInProgress = false;
  }

  changeBranchType(branchType) {
    this.existingBranch.setValue(null);
    this.sourceBranch.setValue(null);
    this.featureBranch.setValue(null);
    this.branchOptions.forEach((option) => (option.checked = false));
    if (branchType === 'NEW') {
      this.existingBranch.clearValidators();
      this.sourceBranch.setValidators([Validators.required]);
      this.featureBranch.setValidators([Validators.required]);
    } else {
      this.existingBranch.setValidators([Validators.required]);
      this.sourceBranch.clearValidators();
      this.featureBranch.clearValidators();
    }
    this.existingBranch.updateValueAndValidity();
    this.sourceBranch.updateValueAndValidity();
    this.featureBranch.updateValueAndValidity();
  }

  workspaceTypeChange(type) {
    this.resetFormValues();
  }

  openCoder() {
    this.getAllBranches();
    this.getAllWorkspaces();
    this.dialogService.open(this.coderDialog).onClose.subscribe((result) => {
      this.resetFormValues();
    });
  }

  closePopup() {
    this.ref.close();
    this.resetFormValues();
  }
}
