import { Component, TemplateRef, ViewChild } from '@angular/core';
import { Params, ActivatedRoute } from '@angular/router';
import { UiReleaseStreamControllerService } from 'src/app/cc-api/services';
import { FacetsPage } from 'src/app/facets/facets-page/facets-page.component';
import { NbDialogService } from '@nebular/theme';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { whiteSpaceInputPattern } from 'src/app/constants/inputPatterns';
import { SmartTableTimeagoTooltipComponent } from 'src/app/components/smart-table-timeago-tooltip/smart-table-timeago-tooltip.component';
import { SmartTableCellWithEllipsisComponent } from 'src/app/components/smart-table-cell-with-ellipsis/smart-table-cell-with-ellipsis.component';
import { CustomActionsComponent } from 'src/app/components/custom-actions/custom-actions.component';
import { cloneDeep } from 'lodash';
import { SmartTableMultiTypeComponent } from 'src/app/components/smart-table-multi-type/smart-table-multi-type.component';
import {
  RELEASE_STREAM_DELETE,
  RELEASE_STREAM_WRITE,
} from 'src/app/constants/permissions';
import { CustomToastService } from '../custom-toast/custom-toast.service';
import { OnAPIErrorService } from 'src/app/services/on-api-error.service';

@Component({
  selector: 'app-release-streams',
  templateUrl: './release-streams.component.html',
  styleUrls: ['./release-streams.component.scss'],
})
export class ReleaseStreamsComponent extends FacetsPage {
  @ViewChild('addReleaseStreamModal', { static: true })
  private addReleaseStreamModal: TemplateRef<any>;

  @ViewChild('deleteReleaseStreamConfirmation', { static: true })
  private deleteReleaseStreamConfirmation: TemplateRef<any>;

  releaseStreamForm: FormGroup = this.releaseStreamFormBuilder.group({
    name: [
      '',
      [Validators.required, Validators.pattern(whiteSpaceInputPattern)],
    ],
    description: ['', [Validators.pattern(whiteSpaceInputPattern)]],
    productionReleaseStream: [],
  });

  getAllReleaseStreamsLoader: boolean = false;
  singleActionLoader: boolean = false;

  releaseStreams = [];
  releaseStreamWritePermission = RELEASE_STREAM_WRITE;
  readonly RELEASE_STREAM_DELETE = RELEASE_STREAM_DELETE;
  smartTableCustomActionButtons = [
    {
      id: 'delete',
      icon: 'trash-outline',
      isFciPack: true,
      toolTip: 'Delete release stream',
      status: 'danger',
      size: 'medium',
      appearance: 'ghost',
      action: RELEASE_STREAM_DELETE,
      onClick: (data) => {
        this.openDeleteConfirmPopup(data);
      },
    },
  ];
  tableSettings: any = {
    columns: {
      name: {
        title: 'Release Stream',
        filter: true,
      },
      description: {
        title: 'Description',
        filter: false,
      },
      builtIn: {
        title: 'Type',
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [
              { value: true, title: 'System defined' },
              { value: false, title: 'User defined' },
            ],
          },
        },
        width: '13%',
        valuePrepareFunction: (cell) => {
          return cell ? 'System defined' : 'User defined';
        },
      },
      prod: {
        title: 'Production',
        width: '10%',
        editable: false,
        type: 'custom',
        renderComponent: SmartTableMultiTypeComponent,
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            type: 'icon',
            props: {
              icon: cell ? 'checkmark-outline' : '',
            },
          });
        },
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [
              { value: true, title: 'True' },
              { value: false, title: 'False' },
            ],
          },
        },
      },
      mappedClusterNames: {
        title: 'Environments',
        type: 'custom',
        filter: {
          type: 'list',
          config: {
            selectText: 'All',
            list: [],
          },
        },
        valuePrepareFunction: (value) => {
          return JSON.stringify({
            data: value,
            decorate: true,
            noDataMessage: 'N/A',
          });
        },
        renderComponent: SmartTableCellWithEllipsisComponent,
      },
      createdBy: {
        title: 'Created By',
        filter: false,
        type: 'custom',
        renderComponent: SmartTableTimeagoTooltipComponent,
        valuePrepareFunction: (cell, row) => {
          return JSON.stringify({
            user: row.createdBy === 'System' ? 'Deployer' : row.createdBy,
            label: row.creationDate,
          });
        },
      },
      custom: {
        type: 'custom',
        title: 'Actions',
        valuePrepareFunction: (value, row, cell) => {
          let tableActionButtons = cloneDeep(
            this.smartTableCustomActionButtons,
          );
          if (row.builtIn) {
            tableActionButtons = [];
          }

          const displayVal = {
            smartTableCustomActionButtons: tableActionButtons,
          };
          return JSON.stringify(displayVal);
        },
        renderComponent: CustomActionsComponent,
        onComponentInitFunction: (instance: any) => {
          instance.byCustomActionButtonClick.subscribe((data) => {
            let customActionButton = this.smartTableCustomActionButtons.filter(
              (actionButton) => {
                return actionButton.id === data.actionButton.id;
              },
            )[0];
            customActionButton.onClick(data.rowData);
          });
        },
        width: '5%',
        filter: false,
      },
    },
    noDataMessage: '',
    pager: {
      display: false,
    },
    mode: 'external',
    actions: false,
  };

  RELEASE_STREAM_NAME_MIN_LENGTH = 2;
  RELEASE_STREAM_NAME_MAX_LENGTH = 10;
  constructor(
    private activatedRoute: ActivatedRoute,
    private uiReleaseStreamControllerService: UiReleaseStreamControllerService,
    private dialogService: NbDialogService,
    private releaseStreamFormBuilder: FormBuilder,
    private customToastService: CustomToastService,
    private onAPIErrorService: OnAPIErrorService,
  ) {
    super(activatedRoute);
  }

  OnPageLoad(p: Params): void {
    this.getAllReleaseStreams();
  }
  getAllReleaseStreams() {
    this.getAllReleaseStreamsLoader = true;
    this.uiReleaseStreamControllerService.getAllUsingGET3().subscribe(
      (apiResponse) => {
        this.getAllReleaseStreamsLoader = false;
        this.releaseStreams = apiResponse;
        this.buildEnvironmentsFilter(apiResponse);
      },
      () => {
        this.getAllReleaseStreamsLoader = false;
      },
    );
  }
  get name() {
    return this.releaseStreamForm.get('name');
  }
  get description() {
    return this.releaseStreamForm.get('description');
  }
  get productionReleaseStream() {
    return this.releaseStreamForm.get('productionReleaseStream');
  }
  buildEnvironmentsFilter(releaseStreamData) {
    let allEnvironments = [];
    releaseStreamData?.forEach((releaseStream) => {
      allEnvironments?.push(...releaseStream?.mappedClusterNames);
    });
    allEnvironments = allEnvironments?.map((environment) => ({
      title: environment,
      value: environment,
    }));
    this.tableSettings.columns['mappedClusterNames'].filter.config.list =
      allEnvironments;
    this.tableSettings = { ...this.tableSettings };
  }
  onClickAddReleaseStream() {
    this.releaseStreamForm.reset();
    this.dialogService.open(this.addReleaseStreamModal, {
      context: {},
    });
  }
  onClickCreateReleaseStream(ref) {
    this.singleActionLoader = true;
    this.uiReleaseStreamControllerService
      .addUsingPOST({
        name: this.name?.value?.trim(),
        prod: this.productionReleaseStream?.value,
        description: this.description?.value?.trim(),
      })
      .subscribe(
        () => {
          this.singleActionLoader = false;
          this.customToastService.success(
            'Release Stream Created Successfully!',
          );
          ref?.close();
          this.releaseStreamForm.reset();
          this.getAllReleaseStreams();
        },
        (error) => {
          this.singleActionLoader = false;
          this.onAPIErrorService.onAPIError(
            error?.error?.message ||
              error?.message ||
              'Failed to create Release Stream!',
            error,
          );
        },
      );
  }
  openDeleteConfirmPopup(data) {
    this.dialogService.open(this.deleteReleaseStreamConfirmation, {
      context: data,
    });
  }
  deleteReleaseStream(row, ref) {
    this.singleActionLoader = true;
    this.uiReleaseStreamControllerService
      .deleteUsingDELETE(row?.name)
      .subscribe(
        () => {
          this.singleActionLoader = false;
          ref.close();
          this.customToastService.success(
            `Release stream ${row?.name} deleted successfully`,
          );
          this.getAllReleaseStreams();
        },
        (error) => {
          this.singleActionLoader = false;
          this.onAPIErrorService.onAPIError(
            error?.error?.message ||
              `Failed to delete release stream ${row?.name}`,
            error,
          );
        },
      );
  }
}
